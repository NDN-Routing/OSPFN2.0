00347 {

00350 
00351   /* Allocate new route map inex. */
00352   index = route_map_index_new ();
00353   index->map = map;
00354   index->type = type;
00355   index->pref = pref;

00357   /* Compare preference. */
00358   for (point = map->head; point; point = point->next)
00359     if (point->pref >= pref)
00360       break;

00364       map->head = map->tail = index;
00365     }

00373     {
00374       index->next = map->head;
00375       map->head->prev = index;
00376       map->head = index;
00377     }
00378   else
00379     {
00380       index->next = point;
00381       index->prev = point->prev;
00382       if (point->prev)
00383     point->prev->next = index;
00384       point->prev = index;
00385     }
00386 
00387   /* Execute event hook. */
00388   if (route_map_master.event_hook)

00530   struct route_map_rule *next;
00531   struct route_map_rule_cmd *cmd;
00532   void *compile;
00533   int replaced = 0;
00534 
00535   /* First lookup rule for add match statement. */
00536   cmd = route_map_lookup_match (match_name);

00538     return RMAP_RULE_MISSING;
00539 
00540   /* Next call compile function for this match statement. */

00542     {
00543       compile= (*cmd->func_compile)(match_arg);
00544       if (compile == NULL)

00551   for (rule = index->match_list.head; rule; rule = next)

00553       next = rule->next;

00555     {   
00556       route_map_rule_delete (&index->match_list, rule);
00557       replaced = 1;
00558     }
00559     }

00561   /* Add new route map match rule. */
00562   rule = route_map_rule_new ();

00702               Match   |   No Match
00703                       |
00704     permit    action  |     cont
00705                       |
00706     ------------------+---------------
00707                       |
00708     deny      deny    |     cont
00709                       |
00710   
00711    action)
00712       -Apply Set statements, accept route
00713       -If Call statement is present jump to the specified route-map, if it
00714          denies the route we finish.
00715       -If NEXT is specified, goto NEXT statement
00716       -If GOTO is specified, goto the first clause where pref > nextpref
00717       -If nothing is specified, do as Cisco and finish
00718    deny)
00719       -Route is denied by route-map.
00720    cont)
00721       -Goto Next index
00722   
00723    If we get no matches after we've processed all updates, then the route
00724    is dropped too.
00725   
00726    Some notes on the new "CALL", "NEXT" and "GOTO"
00727      call WORD        - If this clause is matched, then the set statements
00728                         are executed and then we jump to route-map 'WORD'. If
00729                         this route-map denies the route, we finish, in other case we
00730                         do whatever the exit policy (EXIT, NEXT or GOTO) tells.
00731      on-match next    - If this clause is matched, then the set statements
00732                         are executed and then we drop through to the next clause
00733      on-match goto n  - If this clause is matched, then the set statments
00734                         are executed and then we goto the nth clause, or the
00735                         first clause greater than this. In order to ensure
00736                         route-maps *always* exit, you cannot jump backwards.
00737                         Sorry ;)
00738   
00739    We need to make sure our route-map processing matches the above
00740 */
00741 
00742 static route_map_result_t
00743 route_map_apply_match (struct route_map_rule_list *match_list,
00744                        struct prefix *prefix, route_map_object_t type,
00745                        void *object)
00746 {
00747   route_map_result_t ret = RMAP_NOMATCH;
00748   struct route_map_rule *match;
00749 
00750 
00751   /* Check all match rule and if there is no match rule, go to the
00752      set statement. */
00753   if (!match_list->head)
00754     ret = RMAP_MATCH;
00755   else
00756     {
00757       for (match = match_list->head; match; match = match->next)
00758         {
00759           /* Try each match statement in turn, If any do not return
00760              RMAP_MATCH, return, otherwise continue on to next match 
00761              statement. All match statements must match for end-result
00762              to be a match. */
00763           ret = (*match->cmd->func_apply) (match->value, prefix,
00764                                            type, object);
00765           if (ret != RMAP_MATCH)
00766             return ret;
00767         }
00768     }
00769   return ret;
00770 }

00772 /* Apply route map to the object. */
00773 route_map_result_t
00774 route_map_apply (struct route_map *map, struct prefix *prefix,
00775                  route_map_object_t type, void *object)
00776 {
00777   static int recursion = 0;
00778   int ret = 0;
00779   struct route_map_index *index;
00780   struct route_map_rule *set;
00781 
00782   if (recursion > RMAP_RECURSION_LIMIT)
00783     {
00784       zlog (NULL, LOG_WARNING,
00785             "route-map recursion limit (%d) reached, discarding route",
00786             RMAP_RECURSION_LIMIT);
00787       recursion = 0;
00788       return RMAP_DENYMATCH;
00789     }
00790 
00791   if (map == NULL)
00792     return RMAP_DENYMATCH;
00793 
00794   for (index = map->head; index; index = index->next)
00795     {
00796       /* Apply this index. */
00797       ret = route_map_apply_match (&index->match_list, prefix, type, object);
00798 
00799       /* Now we apply the matrix from above */
00800       if (ret == RMAP_NOMATCH)
00801         /* 'cont' from matrix - continue to next route-map sequence */
00802         continue;
00803       else if (ret == RMAP_MATCH)
00804         {
00805           if (index->type == RMAP_PERMIT)
00806             /* 'action' */
00807             {
00808               /* permit+match must execute sets */
00809               for (set = index->set_list.head; set; set = set->next)
00810                 ret = (*set->cmd->func_apply) (set->value, prefix,
00811                                                type, object);
00812 
00813               /* Call another route-map if available */
00814               if (index->nextrm)
00815                 {
00816                   struct route_map *nextrm =
00817                                     route_map_lookup_by_name (index->nextrm);
00818 
00819                   if (nextrm) /* Target route-map found, jump to it */
00820                     {
00821                       recursion++;
00822                       ret = route_map_apply (nextrm, prefix, type, object);
00823                       recursion--;
00824                     }
00825 
00826                   /* If nextrm returned 'deny', finish. */
00827                   if (ret == RMAP_DENYMATCH)
00828                     return ret;
00829                 }
00830                 
00831               switch (index->exitpolicy)
00832                 {
00833                   case RMAP_EXIT:
00834                     return ret;
00835                   case RMAP_NEXT:
00836                     continue;
00837                   case RMAP_GOTO:
00838                     {
00839                       /* Find the next clause to jump to */
00840                       struct route_map_index *next = index->next;
00841                       int nextpref = index->nextpref;
00842 
00843                       while (next && next->pref < nextpref)
00844                         {
00845                           index = next;
00846                           next = next->next;
00847                         }
00848                       if (next == NULL)
00849                         {
00850                           /* No clauses match! */
00851                           return ret;
00852                         }
00853                     }
00854                 }
00855             }
00856           else if (index->type == RMAP_DENY)
00857             /* 'deny' */
00858             {
00859                 return RMAP_DENYMATCH;
00860             }
00861         }
00862     }
00863   /* Finally route-map does not match at all. */
00864   return RMAP_DENYMATCH;
00865 }

01244 /* Configuration write function. */
01245 static int
01246 route_map_config_write (struct vty *vty)
01247 {
01248   struct route_map *map;
01249   struct route_map_index *index;
01250   struct route_map_rule *rule;
01251   int first = 1;
01252   int write = 0;
01253 
01254   for (map = route_map_master.head; map; map = map->next)
01255     for (index = map->head; index; index = index->next)
01256       {
01257     if (!first)
01258       vty_out (vty, "!%s", VTY_NEWLINE);
01259     else
01260       first = 0;
01261 
01262     vty_out (vty, "route-map %s %s %d%s", 
01263          map->name,
01264          route_map_type_str (index->type),
01265          index->pref, VTY_NEWLINE);
01266 
01267     if (index->description)
01268       vty_out (vty, " description %s%s", index->description, VTY_NEWLINE);
01269 
01270     for (rule = index->match_list.head; rule; rule = rule->next)
01271       vty_out (vty, " match %s %s%s", rule->cmd->str, 
01272            rule->rule_str ? rule->rule_str : "",
01273            VTY_NEWLINE);
01274 
01275     for (rule = index->set_list.head; rule; rule = rule->next)
01276       vty_out (vty, " set %s %s%s", rule->cmd->str,
01277            rule->rule_str ? rule->rule_str : "",
01278            VTY_NEWLINE);
01279    if (index->nextrm)
01280      vty_out (vty, " call %s%s", index->nextrm, VTY_NEWLINE);
01281     if (index->exitpolicy == RMAP_GOTO)
01282       vty_out (vty, " on-match goto %d%s", index->nextpref, VTY_NEWLINE);
01283     if (index->exitpolicy == RMAP_NEXT)
01284       vty_out (vty," on-match next%s", VTY_NEWLINE);
01285     
01286     write++;
01287       }
01288   return write;
