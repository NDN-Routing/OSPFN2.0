00178     {

00195             /* calculate the appropriate broadcast address */

00994 /* Update interface status. */
00995 void
00996 rip_enable_apply (struct interface *ifp)
00997 {
00998   int ret;
00999   struct rip_interface *ri = NULL;
01000 
01001   /* Check interface. */
01002   if (! if_is_operative (ifp))
01003     return;
01004 
01005   ri = ifp->info;
01006 
01007   /* Check network configuration. */
01008   ret = rip_enable_network_lookup_if (ifp);
01009 
01010   /* If the interface is matched. */
01011   if (ret > 0)
01012     ri->enable_network = 1;
01013   else
01014     ri->enable_network = 0;
01015 
01016   /* Check interface name configuration. */
01017   ret = rip_enable_if_lookup (ifp->name);
01018   if (ret >= 0)
01019     ri->enable_interface = 1;
01020   else
01021     ri->enable_interface = 0;
01022 
01023   /* any interface MUST have an IPv4 address */
01024   if ( ! rip_if_ipv4_address_check (ifp) )
01025     {
01026       ri->enable_network = 0;
01027       ri->enable_interface = 0;
01028     }
01029 
01030   /* Update running status of the interface. */
01031   if (ri->enable_network || ri->enable_interface)
01032     {
01033     {
01034       if (IS_RIP_DEBUG_EVENT)
01035         zlog_debug ("turn on %s", ifp->name);
01036 
01037       /* Add interface wake up thread. */
01038       if (! ri->t_wakeup)
01039         ri->t_wakeup = thread_add_timer (master, rip_interface_wakeup,
01040                          ifp, 1);
01041           rip_connect_set (ifp, 1);
01042     }
01043     }
01044   else
01045     {
01046       if (ri->running)
01047     {
01048       /* Might as well clean up the route table as well
01049        * rip_if_down sets to 0 ri->running, and displays "turn off %s"
01050        **/ 
01051       rip_if_down(ifp);
01052 
01053           rip_connect_set (ifp, 0);
01054     }
01055     }
01056 }
