00159 /* Dump attribute. */
00160 int
00161 bgp_dump_attr (struct peer *peer, struct attr *attr, char *buf, size_t size)
00162 {
00163   if (! attr)
00164     return 0;
00165 
00166   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_NEXT_HOP)))
00167     snprintf (buf, size, "nexthop %s", inet_ntoa (attr->nexthop));
00168 
00169   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_ORIGIN)))
00170     snprintf (buf + strlen (buf), size - strlen (buf), ", origin %s",
00171           bgp_origin_str[attr->origin]);
00172 
00173 #ifdef HAVE_IPV6
00174   if (attr->extra)
00175     {
00176       char addrbuf[BUFSIZ];
00177 
00178       /* Add MP case. */
00179       if (attr->extra->mp_nexthop_len == 16 
00180           || attr->extra->mp_nexthop_len == 32)
00181         snprintf (buf + strlen (buf), size - strlen (buf), ", mp_nexthop %s",
00182                   inet_ntop (AF_INET6, &attr->extra->mp_nexthop_global, 
00183                              addrbuf, BUFSIZ));
00184 
00185       if (attr->extra->mp_nexthop_len == 32)
00186         snprintf (buf + strlen (buf), size - strlen (buf), "(%s)",
00187                   inet_ntop (AF_INET6, &attr->extra->mp_nexthop_local, 
00188                              addrbuf, BUFSIZ));
00189     }
00190 #endif /* HAVE_IPV6 */
00191 
00192   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_LOCAL_PREF)))
00193     snprintf (buf + strlen (buf), size - strlen (buf), ", localpref %d",
00194           attr->local_pref);
00195 
00196   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_MULTI_EXIT_DISC))) 
00197     snprintf (buf + strlen (buf), size - strlen (buf), ", metric %d",
00198           attr->med);
00199 
00200   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_COMMUNITIES))) 
00201     snprintf (buf + strlen (buf), size - strlen (buf), ", community %s",
00202           community_str (attr->community));
00203 
00204   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_ATOMIC_AGGREGATE)))
00205     snprintf (buf + strlen (buf), size - strlen (buf), ", atomic-aggregate");
00206 
00207   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_AGGREGATOR)))
00208     snprintf (buf + strlen (buf), size - strlen (buf), ", aggregated by %u %s",
00209           attr->extra->aggregator_as,
00210           inet_ntoa (attr->extra->aggregator_addr));
00211 
00212   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_ORIGINATOR_ID)))
00213     snprintf (buf + strlen (buf), size - strlen (buf), ", originator %s",
00214           inet_ntoa (attr->extra->originator_id));
00215 
00216   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_CLUSTER_LIST)))
00217     {
00218       int i;
00219 
00220       snprintf (buf + strlen (buf), size - strlen (buf), ", clusterlist");
00221       for (i = 0; i < attr->extra->cluster->length / 4; i++)
00222     snprintf (buf + strlen (buf), size - strlen (buf), " %s",
00223           inet_ntoa (attr->extra->cluster->list[i]));
00224     }
00225 
00226   if (CHECK_FLAG (attr->flag, ATTR_FLAG_BIT (BGP_ATTR_AS_PATH))) 
00227     snprintf (buf + strlen (buf), size - strlen (buf), ", path %s",
00228           aspath_print (attr->aspath));
00229 
00230   if (strlen (buf) > 1)
00231     return 1;
00232   else
00233     return 0;
00234 }

00236 /* dump notify packet */
00237 void
00238 bgp_notify_print(struct peer *peer, struct bgp_notify *bgp_notify, 
00239                  const char *direct)
00240 {
00241   const char *subcode_str;
00242 
00243   subcode_str = "";
00244 
00245   switch (bgp_notify->code) 
00246     {
00247     case BGP_NOTIFY_HEADER_ERR:
00248       subcode_str = LOOKUP (bgp_notify_head_msg, bgp_notify->subcode);
00249       break;
00250     case BGP_NOTIFY_OPEN_ERR:
00251       subcode_str = LOOKUP (bgp_notify_open_msg, bgp_notify->subcode);
00252       break;
00253     case BGP_NOTIFY_UPDATE_ERR:
00254       subcode_str = LOOKUP (bgp_notify_update_msg, bgp_notify->subcode);
00255       break;
00256     case BGP_NOTIFY_HOLD_ERR:
00257       subcode_str = "";
00258       break;
00259     case BGP_NOTIFY_FSM_ERR:
00260       subcode_str = "";
00261       break;
00262     case BGP_NOTIFY_CEASE:
00263       subcode_str = LOOKUP (bgp_notify_cease_msg, bgp_notify->subcode);
00264       break;
00265     case BGP_NOTIFY_CAPABILITY_ERR:
00266       subcode_str = LOOKUP (bgp_notify_capability_msg, bgp_notify->subcode);
00267       break;
00268     }
00269 
00270   if (bgp_flag_check (peer->bgp, BGP_FLAG_LOG_NEIGHBOR_CHANGES))
00271     zlog_info ("%%NOTIFICATION: %s neighbor %s %d/%d (%s%s) %d bytes %s",
00272               strcmp (direct, "received") == 0 ? "received from" : "sent to",
00273               peer->host, bgp_notify->code, bgp_notify->subcode,
00274                LOOKUP (bgp_notify_msg, bgp_notify->code),
00275               subcode_str, bgp_notify->length,
00276               bgp_notify->data ? bgp_notify->data : "");
00277   else if (BGP_DEBUG (normal, NORMAL))
00278     plog_debug (peer->log, "%s %s NOTIFICATION %d/%d (%s%s) %d bytes %s",
00279            peer ? peer->host : "",
00280            direct, bgp_notify->code, bgp_notify->subcode,
00281            LOOKUP (bgp_notify_msg, bgp_notify->code),
00282            subcode_str, bgp_notify->length,
00283            bgp_notify->data ? bgp_notify->data : "");
00284 }
