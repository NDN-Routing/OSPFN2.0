00177 ospf6_lsa_age_set (struct ospf6_lsa *lsa)
00178 {
00179   struct timeval now;

00183   if (quagga_gettime (QUAGGA_CLK_MONOTONIC, &now) < 0)
00184     zlog_warn ("LSA: quagga_gettime failed, may fail LSA AGEs: %s",
00185                safe_strerror (errno));

00193 /* this function calculates current age from its birth,
00194    then update age field of LSA header. return value is current age */
00195 u_int16_t
00196 ospf6_lsa_age_current (struct ospf6_lsa *lsa)
00197 {
00198   struct timeval now;
00199   u_int32_t ulage;
00200   u_int16_t age;
00201 
00202   assert (lsa);
00203   assert (lsa->header);
00204 
00205   /* current time */
00206   if (quagga_gettime (QUAGGA_CLK_MONOTONIC, &now) < 0)
00207     zlog_warn ("LSA: quagga_gettime failed, may fail LSA AGEs: %s",
00208                safe_strerror (errno));
00209 
00210   if (ntohs (lsa->header->age) >= MAXAGE)
00211     {
00212       /* ospf6_lsa_premature_aging () sets age to MAXAGE; when using
00213          relative time, we cannot compare against lsa birth time, so
00214          we catch this special case here. */
00215       lsa->header->age = htons (MAXAGE);
00216       return MAXAGE;
00217     }
00218   /* calculate age */
00219   ulage = now.tv_sec - lsa->birth.tv_sec;
00220 
00221   /* if over MAXAGE, set to it */
00222   age = (ulage > MAXAGE ? MAXAGE : ulage);
00223 
00224   lsa->header->age = htons (age);
00225   return age;
00226 }

00254 /* check which is more recent. if a is more recent, return -1;
00255    if the same, return 0; otherwise(b is more recent), return 1 */
00256 int
00257 ospf6_lsa_compare (struct ospf6_lsa *a, struct ospf6_lsa *b)
00258 {
00259   int seqnuma, seqnumb;
00260   u_int16_t cksuma, cksumb;
00261   u_int16_t agea, ageb;
00262 
00263   assert (a && a->header);
00264   assert (b && b->header);
00265   assert (OSPF6_LSA_IS_SAME (a, b));
00266 
00267   seqnuma = (int) ntohl (a->header->seqnum);
00268   seqnumb = (int) ntohl (b->header->seqnum);
00269 
00270   /* compare by sequence number */
00271   if (seqnuma > seqnumb)
00272     return -1;
00273   if (seqnuma < seqnumb)
00274     return 1;
00275 
00276   /* Checksum */
00277   cksuma = ntohs (a->header->checksum);
00278   cksumb = ntohs (b->header->checksum);
00279   if (cksuma > cksumb)
00280     return -1;
00281   if (cksuma < cksumb)
00282     return 0;
00283 
00284   /* Update Age */
00285   agea = ospf6_lsa_age_current (a);
00286   ageb = ospf6_lsa_age_current (b);
00287 
00288   /* MaxAge check */
00289   if (agea == MAXAGE && ageb != MAXAGE)
00290     return -1;
00291   else if (agea != MAXAGE && ageb == MAXAGE)
00292     return 1;
00293 
00294   /* Age check */
00295   if (agea > ageb && agea - ageb >= MAX_AGE_DIFF)
00296     return 1;
00297   else if (agea < ageb && ageb - agea >= MAX_AGE_DIFF)
00298     return -1;
00299 
00300   /* neither recent */
00301   return 0;
00302 }

00665 /* enhanced Fletcher checksum algorithm, RFC1008 7.2 */
00666 #define MODX                4102
00667 #define LSA_CHECKSUM_OFFSET   15
00668 
00669 unsigned short
00670 ospf6_lsa_checksum (struct ospf6_lsa_header *lsa_header)
00671 {
00672   u_char *sp, *ep, *p, *q;
00673   int c0 = 0, c1 = 0;
00674   int x, y;
00675   u_int16_t length;
00676 
00677   lsa_header->checksum = 0;
00678   length = ntohs (lsa_header->length) - 2;
00679   sp = (u_char *) &lsa_header->type;
00680 
00681   for (ep = sp + length; sp < ep; sp = q)
00682     {
00683       q = sp + MODX;
00684       if (q > ep)
00685         q = ep;
00686       for (p = sp; p < q; p++)
00687         {
00688           c0 += *p;
00689           c1 += c0;
00690         }
00691       c0 %= 255;
00692       c1 %= 255;
00693     }
00694 
00695   /* r = (c1 << 8) + c0; */
00696   x = ((length - LSA_CHECKSUM_OFFSET) * c0 - c1) % 255;
00697   if (x <= 0)
00698     x += 255;
00699   y = 510 - c0 - x;
00700   if (y > 255)
00701     y -= 255;
00702 
00703   lsa_header->checksum = htons ((x << 8) + y);
00704 
00705   return (lsa_header->checksum);
00706 }
