00077     list_delete (tlvs->lsp_entries);
00078   if (tlvs->lan_neighs)
00079     list_delete (tlvs->lan_neighs);
00080   if (tlvs->prefix_neighs)
00081     list_delete (tlvs->prefix_neighs);
00082   if (tlvs->ipv4_addrs)
00083     list_delete (tlvs->ipv4_addrs);
00084   if (tlvs->ipv4_int_reachs)
00085     list_delete (tlvs->ipv4_int_reachs);
00086   if (tlvs->ipv4_ext_reachs)
00087     list_delete (tlvs->ipv4_ext_reachs);
00088   if (tlvs->te_ipv4_reachs)

01101 int
01102 tlv_add_padding (struct stream *stream)
01103 {
01104   int fullpads, i, left;
01105 
01106   /*
01107    * How many times can we add full padding ?
01108    */
01109   fullpads = (STREAM_SIZE (stream) - stream_get_endp (stream)) / 257;
01110   for (i = 0; i < fullpads; i++)
01111     {
01112       if (!stream_putc (stream, (u_char) PADDING))  /* TAG */
01113     goto err;
01114       if (!stream_putc (stream, (u_char) 255))  /* LENGHT */
01115     goto err;
01116       stream_put (stream, NULL, 255);       /* zero padding */
01117     }
01118 
01119   left = STREAM_SIZE (stream) - stream_get_endp (stream);
01120 
01121   if (left < 2)
01122     return ISIS_OK;
01123 
01124   if (left == 2)
01125     {
01126       stream_putc (stream, PADDING);
01127       stream_putc (stream, 0);
01128       return ISIS_OK;
01129     }
01130 
01131   stream_putc (stream, PADDING);
01132   stream_putc (stream, left - 2);
01133   stream_put (stream, NULL, left-2);
01134 
01135   return ISIS_OK;
01136 
01137 err:
01138   zlog_warn ("tlv_add_padding(): no room for tlv");
01139   return ISIS_WARNING;
01140 }
