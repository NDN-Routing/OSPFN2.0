00072 {
00073   struct ospf6_area *oa;

00075   struct listnode *node, *nnode;
00076 
00077   for (ALL_LIST_ELEMENTS (area->ospf6->area_list, node, nnode, oa))
00078     {
00079       /* update B bit for each area */
00080       OSPF6_ROUTER_LSA_SCHEDULE (oa);
00081 
00082       /* install other area's configured address range */
00083       if (oa != area)
00084         {
00085           for (ro = ospf6_route_head (oa->range_table); ro;
00086                ro = ospf6_route_next (ro))
00087             {
00088               if (CHECK_FLAG (ro->flag, OSPF6_ROUTE_ACTIVE_SUMMARY))
00089                 ospf6_abr_originate_summary_to_area (ro, area);
00090             }
00091         }
00092     }
00093 
00094   /* install calculated routes to border routers */
00095   for (ro = ospf6_route_head (area->ospf6->brouter_table); ro;
00096        ro = ospf6_route_next (ro))
00097     ospf6_abr_originate_summary_to_area (ro, area);
00098 
00099   /* install calculated routes to network (may be rejected by ranges) */
00100   for (ro = ospf6_route_head (area->ospf6->route_table); ro;
00101        ro = ospf6_route_next (ro))

00482

00485        ro; ro = ospf6_route_match_next (&range->prefix, ro))
00486     {
00487       if (ro->path.area_id == range->path.area_id &&
00488           ! CHECK_FLAG (ro->flag, OSPF6_ROUTE_REMOVE))
00489         cost = MAX (cost, ro->path.cost);

00491

00493     {
00494       range->path.cost = cost;
00495 
00496       if (range->path.cost)
00497         SET_FLAG (range->flag, OSPF6_ROUTE_ACTIVE_SUMMARY);
00498       else
00499         UNSET_FLAG (range->flag, OSPF6_ROUTE_ACTIVE_SUMMARY);
00500 
00501       ospf6_abr_originate_summary (range);
00502     }

00566       struct ospf6_inter_router_lsa *router_lsa;
00567 
00568       if (IS_OSPF6_DEBUG_EXAMIN (INTER_ROUTER))
00569         {
00570           is_debug++;
00571           zlog_debug ("Examin %s in area %s", lsa->name, oa->name);
00572         }
00573 
00574       router_lsa = (struct ospf6_inter_router_lsa *)
00575         OSPF6_LSA_HEADER_END (lsa->header);
00576       ospf6_linkstate_prefix (router_lsa->router_id, htonl (0), &prefix);
00577       inet_ntop (AF_INET, &router_lsa->router_id, buf, sizeof (buf));

00581       options[1] = router_lsa->options[1];

00584       SET_FLAG (router_bits, OSPF6_ROUTER_BIT_E);
00585     }
00586   else
00587     assert (0);

00591   if (route)
00592     ospf6_route_lock (route);
00593   while (route && ospf6_route_is_prefix (&prefix, route))
00594     {
00595       if (route->path.area_id == oa->area_id &&
00596           route->path.origin.type == lsa->header->type &&
00597           route->path.origin.id == lsa->header->id &&
00598           route->path.origin.adv_router == lsa->header->adv_router &&
00599           ! CHECK_FLAG (route->flag, OSPF6_ROUTE_WAS_REMOVED))
00600         old = route;
00601       route = ospf6_route_next (route);
00602     }
00603 
00604   /* (1) if cost == LSInfinity or if the LSA is MaxAge */
00605   if (cost == LS_INFINITY)
00606     {
00607       if (is_debug)
00608         zlog_debug ("cost is LS_INFINITY, ignore");
00609       if (old)
00610         ospf6_route_remove (old, table);
00611       return;
00612     }
00613   if (OSPF6_LSA_IS_MAXAGE (lsa))
00614     {
00615       if (is_debug)
00616         zlog_debug ("LSA is MaxAge, ignore");
00617       if (old)
00618         ospf6_route_remove (old, table);
00619       return;
00620     }
00621 
00622   /* (2) if the LSA is self-originated, ignore */
00623   if (lsa->header->adv_router == oa->ospf6->router_id)
00624     {
00625       if (is_debug)
00626         zlog_debug ("LSA is self-originated, ignore");
00627       if (old)
00628         ospf6_route_remove (old, table);
00629       return;
00630     }
00631 
00632   /* (3) if the prefix is equal to an active configured address range */
00633   if (lsa->header->type == htons (OSPF6_LSTYPE_INTER_PREFIX))
00634     {
00635       range = ospf6_route_lookup (&prefix, oa->range_table);
00636       if (range)
00637         {
00638           if (is_debug)
00639             zlog_debug ("Prefix is equal to address range, ignore");
00640           if (old)
00641             ospf6_route_remove (old, table);
00642           return;
00643         }
00644     }
00645 
00646   /* (4) if the routing table entry for the ABR does not exist */
00647   ospf6_linkstate_prefix (lsa->header->adv_router, htonl (0), &abr_prefix);
00648   abr_entry = ospf6_route_lookup (&abr_prefix, oa->ospf6->brouter_table);
00649   if (abr_entry == NULL || abr_entry->path.area_id != oa->area_id ||
00650       CHECK_FLAG (abr_entry->flag, OSPF6_ROUTE_REMOVE) ||
00651       ! CHECK_FLAG (abr_entry->path.router_bits, OSPF6_ROUTER_BIT_B))
00652     {
00653       if (is_debug)
00654         zlog_debug ("ABR router entry does not exist, ignore");
00655       if (old)
00656         ospf6_route_remove (old, table);
00657       return;
00658     }
00659 
00660   /* Check import list */
00661   if (IMPORT_NAME (oa))
00662     {
00663       if (IMPORT_LIST (oa) == NULL)
00664         IMPORT_LIST (oa) = access_list_lookup (AFI_IP6, IMPORT_NAME (oa));
00665 
00666       if (IMPORT_LIST (oa))
00667         if (access_list_apply (IMPORT_LIST (oa), &prefix) == FILTER_DENY)
00668           {
00669             if (is_debug)
00670               zlog_debug ("Prefix was denied by import-list");
00671             if (old)
00672               ospf6_route_remove (old, table);
00673             return;
00674           }
00675     }
00676 
00677   /* Check input prefix-list */
00678   if (PREFIX_NAME_IN (oa))
00679     {
00680       if (PREFIX_LIST_IN (oa) == NULL)
00681         PREFIX_LIST_IN (oa) = prefix_list_lookup (AFI_IP6, PREFIX_NAME_IN (oa));
00682 
00683       if (PREFIX_LIST_IN (oa))
00684         if (prefix_list_apply (PREFIX_LIST_IN (oa), &prefix) != PREFIX_PERMIT)
00685           {
00686             if (is_debug)
00687               zlog_debug ("Prefix was denied by prefix-list");
00688             if (old)
00689               ospf6_route_remove (old, table);
00690             return;
00691           }
00692     }
00693 
00694   /* (5),(6),(7) the path preference is handled by the sorting
00695      in the routing table. Always install the path by substituting
00696      old route (if any). */
00697   if (old)
00698     route = ospf6_route_copy (old);
00699   else
00700     route = ospf6_route_create ();
00701 
00702   route->type = type;
00703   route->prefix = prefix;
00704   route->path.origin.type = lsa->header->type;
00705   route->path.origin.id = lsa->header->id;
00706   route->path.origin.adv_router = lsa->header->adv_router;
00707   route->path.router_bits = router_bits;
00708   route->path.options[0] = options[0];
00709   route->path.options[1] = options[1];
00710   route->path.options[2] = options[2];
00711   route->path.prefix_options = prefix_options;
00712   route->path.area_id = oa->area_id;
00713   route->path.type = OSPF6_PATH_TYPE_INTER;
00714   route->path.cost = abr_entry->path.cost + cost;
00715   for (i = 0; i < OSPF6_MULTI_PATH_LIMIT; i++)
00716     route->nexthop[i] = abr_entry->nexthop[i];
00717 
00718   if (is_debug)
00719     zlog_debug ("Install route: %s", buf);
00720   ospf6_route_add (route, table);
00721 }
