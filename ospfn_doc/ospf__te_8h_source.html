00027 /*
00028  * Opaque LSA's link state ID for Traffic Engineering is
00029  * structured as follows.
00030  *
00031  *        24       16        8        0
00032  * +--------+--------+--------+--------+
00033  * |    1   |  MBZ   |........|........|
00034  * +--------+--------+--------+--------+
00035  * |<-Type->|<Resv'd>|<-- Instance --->|
00036  *
00037  *
00038  * Type:      IANA has assigned '1' for Traffic Engineering.
00039  * MBZ:       Reserved, must be set to zero.
00040  * Instance:  User may select an arbitrary 16-bit value.
00041  *
00042  */
00043 
00044 #define LEGAL_TE_INSTANCE_RANGE(i)  (0 <= (i) && (i) <= 0xffff)
00045 
00046 /*
00047  *        24       16        8        0
00048  * +--------+--------+--------+--------+ ---
00049  * |   LS age        |Options |   10   |  A
00050  * +--------+--------+--------+--------+  |
00051  * |    1   |   0    |    Instance     |  |
00052  * +--------+--------+--------+--------+  |
00053  * |        Advertising router         |  |  Standard (Opaque) LSA header;
00054  * +--------+--------+--------+--------+  |  Only type-10 is used.
00055  * |        LS sequence number         |  |
00056  * +--------+--------+--------+--------+  |
00057  * |   LS checksum   |     Length      |  V
00058  * +--------+--------+--------+--------+ ---
00059  * |      Type       |     Length      |  A
00060  * +--------+--------+--------+--------+  |  TLV part for TE; Values might be
00061  * |              Values ...           |  V  structured as a set of sub-TLVs.
00062  * +--------+--------+--------+--------+ ---
00063  */
00064 
00065 /*
00066  * Following section defines TLV (tag, length, value) structures,
00067  * used for Traffic Engineering.
00068  */
00069 struct te_tlv_header
00070 {
00071   u_int16_t type;           /* TE_TLV_XXX (see below) */
00072   u_int16_t length;         /* Value portion only, in octets */
00073 };
00074 
00075 #define TLV_HDR_SIZE \
00076     (sizeof (struct te_tlv_header))
00077 
00078 #define TLV_BODY_SIZE(tlvh) \
00079     (ROUNDUP (ntohs ((tlvh)->length), sizeof (u_int32_t)))
00080 
00081 #define TLV_SIZE(tlvh) \
00082     (TLV_HDR_SIZE + TLV_BODY_SIZE(tlvh))
00083 
00084 #define TLV_HDR_TOP(lsah) \
00085     (struct te_tlv_header *)((char *)(lsah) + OSPF_LSA_HEADER_SIZE)
00086 
00087 #define TLV_HDR_NEXT(tlvh) \
00088     (struct te_tlv_header *)((char *)(tlvh) + TLV_SIZE(tlvh))
