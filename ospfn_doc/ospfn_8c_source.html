00915

00917         zprivs_init (&ospfd_privs);
00918         master = thread_master_create ();
00919 
00920         init(); 
00921         pid_create(getpid()); 
00922     readConfigFile(config_file,1);
00923 
00924         if(isLoggingEnabled)
