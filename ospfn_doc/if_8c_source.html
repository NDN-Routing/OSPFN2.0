00049 /* Compare interface names, returning an integer greater than, equal to, or
00050  * less than 0, (following the strcmp convention), according to the
00051  * relationship between ifp1 and ifp2.  Interface names consist of an
00052  * alphabetic prefix and a numeric suffix.  The primary sort key is
00053  * lexicographic by name, and then numeric by number.  No number sorts
00054  * before all numbers.  Examples: de0 < de1, de100 < fxp0 < xl0, devpty <
00055  * devpty0, de0 < del0
00056  */         
00057 int
00058 if_cmp_func (struct interface *ifp1, struct interface *ifp2)
00059 {
00060   unsigned int l1, l2;
00061   long int x1, x2;
00062   char *p1, *p2;
00063   int res;
00064 
00065   p1 = ifp1->name;
00066   p2 = ifp2->name;
00067 
00068   while (*p1 && *p2) {
00069     /* look up to any number */
00070     l1 = strcspn(p1, "0123456789");
00071     l2 = strcspn(p2, "0123456789");
00072 
00073     /* name lengths are different -> compare names */
00074     if (l1 != l2)
00075       return (strcmp(p1, p2));
00076 
00077     /* Note that this relies on all numbers being less than all letters, so
00078      * that de0 < del0.
00079      */
00080     res = strncmp(p1, p2, l1);
00081 
00082     /* names are different -> compare them */
00083     if (res)
00084       return res;
00085 
00086     /* with identical name part, go to numeric part */
00087     p1 += l1;
00088     p2 += l1;
00089 
00090     if (!*p1) 
00091       return -1;
00092     if (!*p2) 
00093       return 1;
00094 
00095     x1 = strtol(p1, &p1, 10);
00096     x2 = strtol(p2, &p2, 10);
00097 
00098     /* let's compare numbers now */
00099     if (x1 < x2)
00100       return -1;
00101     if (x1 > x2)
00102       return 1;
00103 
00104     /* numbers were equal, lets do it again..
00105     (it happens with name like "eth123.456:789") */
00106   }
00107   if (*p1)
00108     return 1;
00109   if (*p2)
00110     return -1;
00111   return 0;
00112 }

00301     }

00494 /* Need to handle upgrade from SUNWzebra to Quagga. SUNWzebra created
00495  * a seperate struct interface for each logical interface, so config
00496  * file may be full of 'interface fooX:Y'. Solaris however does not
00497  * expose logical interfaces via PF_ROUTE, so trying to track logical
00498  * interfaces can be fruitless, for that reason Quagga only tracks
00499  * the primary IP interface.
00500  *
00501  * We try accomodate SUNWzebra by:
00502  * - looking up the interface name, to see whether it exists, if so
00503  *   its useable
00504  *   - for protocol daemons, this could only because zebra told us of
00505  *     the interface
00506  *   - for zebra, only because it learnt from kernel
00507  * - if not:
00508  *   - search the name to see if it contains a sub-ipif / logical interface
00509  *     seperator, the ':' char. If it does:
00510  *     - text up to that char must be the primary name - get that name.
00511  *     if not:
00512  *     - no idea, just get the name in its entirety.
00513  */
00514 static struct interface *
00515 if_sunwzebra_get (const char *name, size_t nlen)
00516 {
00517   struct interface *ifp;
00518   size_t seppos = 0;
00519 
00520   if ( (ifp = if_lookup_by_name_len(name, nlen)) != NULL)
00521     return ifp;
00522   
00523   /* hunt the primary interface name... */
00524   while (seppos < nlen && name[seppos] != ':')
00525     seppos++;
00526   
00527   /* Wont catch seperator as last char, e.g. 'foo0:' but thats invalid */
00528   if (seppos < nlen)
00529     return if_get_by_name_len (name, seppos);
00530   else
00531     return if_get_by_name_len (name, nlen);
00532 }
00533 #endif /* SUNOS_5 */
