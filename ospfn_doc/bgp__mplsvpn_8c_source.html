00035 static u_int16_t
00036 decode_rd_type (u_char *pnt)
00037 {
00038   u_int16_t v;
00039   
00040   v = ((u_int16_t) *pnt++ << 8);
00041   v |= (u_int16_t) *pnt;
00042   return v;

00045 u_int32_t
00046 decode_label (u_char *pnt)
00047 {
00048   u_int32_t l;
00049 
00050   l = ((u_int32_t) *pnt++ << 12);
00051   l |= (u_int32_t) *pnt++ << 4;
00052   l |= (u_int32_t) ((*pnt & 0xf0) >> 4);
00053   return l;
00054 }

00056 static void
00057 decode_rd_as (u_char *pnt, struct rd_as *rd_as)
00058 {
00059   rd_as->as = (u_int16_t) *pnt++ << 8;
00060   rd_as->as |= (u_int16_t) *pnt++;
00061   
00062   rd_as->val = ((u_int32_t) *pnt++ << 24);
00063   rd_as->val |= ((u_int32_t) *pnt++ << 16);
00064   rd_as->val |= ((u_int32_t) *pnt++ << 8);
00065   rd_as->val |= (u_int32_t) *pnt;
00066 }

00068 static void
00069 decode_rd_ip (u_char *pnt, struct rd_ip *rd_ip)
00070 {
00071   memcpy (&rd_ip->ip, pnt, 4);
00072   pnt += 4;
00073   
00074   rd_ip->val = ((u_int16_t) *pnt++ << 8);
00075   rd_ip->val |= (u_int16_t) *pnt;
00076 }

00078 int
00079 bgp_nlri_parse_vpnv4 (struct peer *peer, struct attr *attr, 
00080               struct bgp_nlri *packet)
00081 {
00082   u_char *pnt;
00083   u_char *lim;
00084   struct prefix p;
00085   int psize;
00086   int prefixlen;
00087   u_int32_t label;
00088   u_int16_t type;
00089   struct rd_as rd_as;
00090   struct rd_ip rd_ip;
00091   struct prefix_rd prd;
00092   u_char *tagpnt;
00093 
00094   /* Check peer status. */
00095   if (peer->status != Established)
00096     return 0;
00097   
00098   /* Make prefix_rd */
00099   prd.family = AF_UNSPEC;
00100   prd.prefixlen = 64;
00101 
00102   pnt = packet->nlri;
00103   lim = pnt + packet->length;
00104 
00105   for (; pnt < lim; pnt += psize)
00106     {
00107       /* Clear prefix structure. */
00108       memset (&p, 0, sizeof (struct prefix));
00109 
00110       /* Fetch prefix length. */
00111       prefixlen = *pnt++;
00112       p.family = AF_INET;
00113       psize = PSIZE (prefixlen);
00114 
00115       if (prefixlen < 88)
00116     {
00117       zlog_err ("prefix length is less than 88: %d", prefixlen);
00118       return -1;
00119     }
00120 
00121       label = decode_label (pnt);
00122 
00123       /* Copyr label to prefix. */
00124       tagpnt = pnt;;
00125 
00126       /* Copy routing distinguisher to rd. */
00127       memcpy (&prd.val, pnt + 3, 8);
00128 
00129       /* Decode RD type. */
00130       type = decode_rd_type (pnt + 3);
00131 
00132       /* Decode RD value. */
00133       if (type == RD_TYPE_AS)
00134     decode_rd_as (pnt + 5, &rd_as);
00135       else if (type == RD_TYPE_IP)
00136     decode_rd_ip (pnt + 5, &rd_ip);
00137       else
00138     {
00139       zlog_err ("Invalid RD type %d", type);
00140       return -1;
00141     }
00142 
00143       p.prefixlen = prefixlen - 88;
00144       memcpy (&p.u.prefix, pnt + 11, psize - 11);
00145 
00146 #if 0
00147       if (type == RD_TYPE_AS)
00148     zlog_info ("prefix %ld:%ld:%ld:%s/%d", label, rd_as.as, rd_as.val,
00149            inet_ntoa (p.u.prefix4), p.prefixlen);
00150       else if (type == RD_TYPE_IP)
00151     zlog_info ("prefix %ld:%s:%ld:%s/%d", label, inet_ntoa (rd_ip.ip),
00152            rd_ip.val, inet_ntoa (p.u.prefix4), p.prefixlen);
00153 #endif /* 0 */
00154 
00155       if (pnt + psize > lim)
00156     return -1;
00157 
00158       if (attr)
00159     bgp_update (peer, &p, attr, AFI_IP, SAFI_MPLS_VPN,
00160             ZEBRA_ROUTE_BGP, BGP_ROUTE_NORMAL, &prd, tagpnt, 0);
00161       else
00162     bgp_withdraw (peer, &p, attr, AFI_IP, SAFI_MPLS_VPN,
00163               ZEBRA_ROUTE_BGP, BGP_ROUTE_NORMAL, &prd, tagpnt);
00164     }
00165 
00166   /* Packet length consistency check. */
00167   if (pnt != lim)

00173 int
00174 str2prefix_rd (const char *str, struct prefix_rd *prd)
00175 {
00176   int ret;
00177   char *p;
00178   char *p2;
00179   struct stream *s;
00180   char *half;
00181   struct in_addr addr;
00182 
00183   s = stream_new (8);
00184 
00185   prd->family = AF_UNSPEC;
00186   prd->prefixlen = 64;
00187 
00188   p = strchr (str, ':');
00189   if (! p)
00190     return 0;
00191 
00192   if (! all_digit (p + 1))
00193     return 0;
00194 
00195   half = XMALLOC (MTYPE_TMP, (p - str) + 1);
00196   memcpy (half, str, (p - str));
00197   half[p - str] = '\0';
00198 
00199   p2 = strchr (str, '.');
00200 
00201   if (! p2)
00202     {
00203       if (! all_digit (half))
00204     {
00205       XFREE (MTYPE_TMP, half);
00206       return 0;
00207     }
00208       stream_putw (s, RD_TYPE_AS);
00209       stream_putw (s, atoi (half));
00210       stream_putl (s, atol (p + 1));
00211     }
00212   else
00213     {
00214       ret = inet_aton (half, &addr);
00215       if (! ret)
00216     {
00217       XFREE (MTYPE_TMP, half);
00218       return 0;
00219     }
00220       stream_putw (s, RD_TYPE_IP);
00221       stream_put_in_addr (s, &addr);
00222       stream_putw (s, atol (p + 1));
00223     }
00224   memcpy (prd->val, s->data, 8);
00225 
00226   return 1;
00227 }

00229 int
00230 str2tag (const char *str, u_char *tag)
00231 {
00232   unsigned long l;
00233   char *endptr;
00234   u_int32_t t;
00235 
00236   l = strtoul (str, &endptr, 10);
00237   
00238   if (*endptr == '\0' || l == ULONG_MAX || l > UINT32_MAX)
00239     return 0;
00240 
00241   t = (u_int32_t) l;
00242   
00243   tag[0] = (u_char)(t >> 12);
00244   tag[1] = (u_char)(t >> 4);
00245   tag[2] = (u_char)(t << 4);
00246 
00247   return 1;
