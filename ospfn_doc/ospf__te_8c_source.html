00281 ospf_mpls_te_foreach_area (
00282   void (*func)(struct mpls_te_link *lp, enum sched_opcode),
00283   enum sched_opcode sched_opcode)
00284 {
00285   struct listnode *node, *nnode; 
00286   struct listnode *node2;
00287   struct mpls_te_link *lp;
00288   struct ospf_area *area;
00289 
00290   for (ALL_LIST_ELEMENTS (OspfMplsTE.iflist, node, nnode, lp))
00291     {
00292       if ((area = lp->area) == NULL)
00293         continue;
00294       if (lp->flags & LPFLG_LOOKUP_DONE)
00295         continue;
00296 
00297       if (func != NULL)
00298         (* func)(lp, sched_opcode);
00299 
00300       for (node2 = listnextnode (node); node2; node2 = listnextnode (node2))
00301         if ((lp = listgetdata (node2)) != NULL)
00302           if (lp->area != NULL)
00303             if (IPV4_ADDR_SAME (&lp->area->area_id, &area->area_id))
00304               lp->flags |= LPFLG_LOOKUP_DONE;
00305     }
00306 
00307   for (ALL_LIST_ELEMENTS_RO (OspfMplsTE.iflist, node, lp))
00308     if (lp->area != NULL)
00309       lp->flags &= ~LPFLG_LOOKUP_DONE;
00310 
00311   return;

00396 static void
00397 set_linkparams_link_id (struct ospf_interface *oi, struct mpls_te_link *lp)
00398 {
00399   struct ospf_neighbor *nbr;
00400   int done = 0;
00401 
00402   lp->link_id.header.type   = htons (TE_LINK_SUBTLV_LINK_ID);
00403   lp->link_id.header.length = htons (sizeof (lp->link_id.value));
00404 
00405   /*
00406    * The Link ID is identical to the contents of the Link ID field
00407    * in the Router LSA for these link types.
00408    */
00409   switch (oi->type)
00410     {
00411     case OSPF_IFTYPE_POINTOPOINT:
00412       /* Take the router ID of the neighbor. */
00413       if ((nbr = ospf_nbr_lookup_ptop (oi))
00414       && nbr->state == NSM_Full)
00415         {
00416           lp->link_id.value = nbr->router_id;
00417           done = 1;
00418         }
00419       break;
00420     case OSPF_IFTYPE_BROADCAST:
00421     case OSPF_IFTYPE_NBMA:
00422       /* Take the interface address of the designated router. */
00423       if ((nbr = ospf_nbr_lookup_by_addr (oi->nbrs, &DR (oi))) == NULL)
00424         break;
00425 
00426       if (nbr->state == NSM_Full
00427       || (IPV4_ADDR_SAME (&oi->address->u.prefix4, &DR (oi))
00428       &&  ospf_nbr_count (oi, NSM_Full) > 0))
00429         {
00430           lp->link_id.value = DR (oi);
00431           done = 1;
00432         }
00433       break;
00434     default:
00435       /* Not supported yet. *//* XXX */
00436       lp->link_id.header.type = htons (0);
00437       break;
00438     }
00439 
00440   if (! done)
00441     {
00442       struct in_addr mask;
00443       masklen2ip (oi->address->prefixlen, &mask);

01037   if (IS_LSA_MAXAGE (lsa))

01078                            enum sched_opcode opcode)
01079 {
01080   struct ospf_lsa lsa;

01091   lsah.id.s_addr = htonl (tmp);
01092

01097           OSPF_OPAQUE_AREA_LSA, OPAQUE_TYPE_TRAFFIC_ENGINEERING_LSA);
