00058   if (a(3) > b(3))

00061     return 1;
00062   else if ((a(3) == b(3)) && (a(2) == b(2)) && (a(1) > b(1)))
00063     return 1;
00064   else if ((a(3) == b(3)) && (a(2) == b(2)) && (a(1) == b(1)) && (a(0) > b(0)))
00065     return 1;
00066 
00067   if ((a(3) == b(3)) && (a(2) == b(2)) && (a(1) == b(1)) && (a(0) == b(0)))
00068     return 0;
00069
