00059 /* Join to the all rip routers multicast group. */
00060 static int
00061 ripng_multicast_join (struct interface *ifp)
00062 {
00063   int ret;
00064   struct ipv6_mreq mreq;
00065   int save_errno;
00066 
00067   if (if_is_up (ifp) && if_is_multicast (ifp)) {
00068     memset (&mreq, 0, sizeof (mreq));
00069     inet_pton(AF_INET6, RIPNG_GROUP, &mreq.ipv6mr_multiaddr);
00070     mreq.ipv6mr_interface = ifp->ifindex;
00071 
00072     /*
00073      * NetBSD 1.6.2 requires root to join groups on gif(4).
00074      * While this is bogus, privs are available and easy to use
00075      * for this call as a workaround.
00076      */
00077     if (ripngd_privs.change (ZPRIVS_RAISE))
00078       zlog_err ("ripng_multicast_join: could not raise privs");
00079 
00080     ret = setsockopt (ripng->sock, IPPROTO_IPV6, IPV6_JOIN_GROUP,
00081               (char *) &mreq, sizeof (mreq));
00082     save_errno = errno;
00083 
00084     if (ripngd_privs.change (ZPRIVS_LOWER))
00085       zlog_err ("ripng_multicast_join: could not lower privs");
00086 
00087     if (ret < 0 && save_errno == EADDRINUSE)
00088       {
00089     /*
00090      * Group is already joined.  This occurs due to sloppy group
00091      * management, in particular declining to leave the group on
00092      * an interface that has just gone down.
00093      */
00094     zlog_warn ("ripng join on %s EADDRINUSE (ignoring)\n", ifp->name);
00095     return 0;       /* not an error */
00096       }
00097 
00098     if (ret < 0)
00099       zlog_warn ("can't setsockopt IPV6_JOIN_GROUP: %s",
00100              safe_strerror (save_errno));
00101 
00102     if (IS_RIPNG_DEBUG_EVENT)
00103       zlog_debug ("RIPng %s join to all-rip-routers multicast group", ifp->name);
00104 
00105     if (ret < 0)
00106       return -1;
00107   }
00108   return 0;
00109 }

00169       for (rp = route_top (ripng->table); rp; rp = route_next (rp))
00170     if ((rinfo = rp->info) != NULL)
00171       {
00172         /* Routes got through this interface. */
00173         if (rinfo->ifindex == ifp->ifindex
00174         && rinfo->type == ZEBRA_ROUTE_RIPNG
00175         && rinfo->sub_type == RIPNG_ROUTE_RTE)
00176           {
00177         ripng_zebra_ipv6_delete ((struct prefix_ipv6 *) &rp->p,
00178                      &rinfo->nexthop,
00179                      rinfo->ifindex);

00181         ripng_redistribute_delete (rinfo->type, rinfo->sub_type,
00182                        (struct prefix_ipv6 *)&rp->p,
00183                        rinfo->ifindex);
00184           }
00185         else
00186           {
00187         /* All redistributed routes got through this interface,
00188          * but the static and system ones are kept. */
00189         if ((rinfo->ifindex == ifp->ifindex) &&
00190             (rinfo->type != ZEBRA_ROUTE_STATIC) &&
00191             (rinfo->type != ZEBRA_ROUTE_SYSTEM))
00192           ripng_redistribute_delete (rinfo->type, rinfo->sub_type,
00193                          (struct prefix_ipv6 *) &rp->p,
00194                          rinfo->ifindex);
00195           }
00196       }
00197     }
00198 
00199   ri = ifp->info;
00200   
00201   if (ri->running)
00202    {
00203      if (IS_RIPNG_DEBUG_EVENT)
00204        zlog_debug ("turn off %s", ifp->name);
00205 
00206      /* Leave from multicast group. */

00738 /* Check RIPng is enabed on this interface. */
00739 void
00740 ripng_enable_apply (struct interface *ifp)
00741 {
00742   int ret;
00743   struct ripng_interface *ri = NULL;
00744 
00745   /* Check interface. */
00746   if (! if_is_up (ifp))
00747     return;
00748   
00749   ri = ifp->info;
00750 
00751   /* Is this interface a candidate for RIPng ? */
00752   ret = ripng_enable_network_lookup_if (ifp);
00753 
00754   /* If the interface is matched. */
00755   if (ret > 0)
00756     ri->enable_network = 1;
00757   else
00758     ri->enable_network = 0;
00759 
00760   /* Check interface name configuration. */
00761   ret = ripng_enable_if_lookup (ifp->name);
00762   if (ret >= 0)
00763     ri->enable_interface = 1;
00764   else
00765     ri->enable_interface = 0;
00766 
00767   /* any candidate interface MUST have a link-local IPv6 address */
00768   if ((! ripng_if_ipv6_lladdress_check (ifp)) &&
00769       (ri->enable_network || ri->enable_interface)) {
00770     ri->enable_network = 0;
00771     ri->enable_interface = 0;
00772     zlog_warn("Interface %s does not have any link-local address",
00773               ifp->name);
00774   }
00775 
00776   /* Update running status of the interface. */
00777   if (ri->enable_network || ri->enable_interface)
00778     {
00779     {
00780       if (IS_RIPNG_DEBUG_EVENT)
00781         zlog_debug ("RIPng turn on %s", ifp->name);
00782 
00783       /* Add interface wake up thread. */
00784       if (! ri->t_wakeup)
00785         ri->t_wakeup = thread_add_timer (master, ripng_interface_wakeup,
00786                          ifp, 1);
00787 
00788       ripng_connect_set (ifp, 1);
00789     }
00790     }
00791   else
00792     {
00793       if (ri->running)
00794     {
00795       /* Might as well clean up the route table as well
00796        * ripng_if_down sets to 0 ri->running, and displays "turn off %s"
00797        **/
00798       ripng_if_down(ifp);
00799 
00800       ripng_connect_set (ifp, 0);
00801     }
00802     }
00803 }

01141 interface_config_write (struct vty *vty)
01142 {
01143   struct listnode *node;
01144   struct interface *ifp;
01145   struct ripng_interface *ri;

01147 
01148   for (ALL_LIST_ELEMENTS_RO (iflist, node, ifp))

01165       /* Split horizon. */

01169           case RIPNG_SPLIT_HORIZON:
01170             vty_out (vty, " ipv6 ripng split-horizon%s", VTY_NEWLINE);
01171             break;
01172           case RIPNG_SPLIT_HORIZON_POISONED_REVERSE:
01173             vty_out (vty, " ipv6 ripng split-horizon poisoned-reverse%s",
01174                           VTY_NEWLINE);

01178             vty_out (vty, " no ipv6 ripng split-horizon%s", VTY_NEWLINE);
