00181 struct bgp_advertise *
00182 bgp_advertise_clean (struct peer *peer, struct bgp_adj_out *adj,
00183              afi_t afi, safi_t safi)
00184 {
00185   struct bgp_advertise *adv;
00186   struct bgp_advertise_attr *baa;
00187   struct bgp_advertise *next;
00188 
00189   adv = adj->adv;
00190   baa = adv->baa;
00191   next = NULL;
00192 
00193   if (baa)
00194     {
00195       /* Unlink myself from advertise attribute FIFO.  */
00196       bgp_advertise_delete (baa, adv);
00197 
00198       /* Fetch next advertise candidate. */
00199       next = baa->adv;
00200 
00201       /* Unintern BGP advertise attribute.  */
00202       bgp_advertise_unintern (peer->hash[afi][safi], baa);
00203     }
00204 
00205   /* Unlink myself from advertisement FIFO.  */
00206   FIFO_DEL (adv);
00207 
00208   /* Free memory.  */
00209   bgp_advertise_free (adj->adv);
00210   adj->adv = NULL;
00211 
00212   return next;
00213 }

00293       /* We need advertisement structure.  */
00294       adj->adv = bgp_advertise_new ();
