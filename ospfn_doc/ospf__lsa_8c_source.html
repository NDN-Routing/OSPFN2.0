00180 ospf_lsa_checksum (struct lsa_header *lsa)
00181 {
00182   u_char *buffer = (u_char *) &lsa->options;
00183   int options_offset = buffer - (u_char *) &lsa->ls_age; /* should be 2 */
00184 
00185   /* Skip the AGE field */

00191

00471 /* Set a link information. */
00472 static char
00473 link_info_set (struct stream *s, struct in_addr id,
00474            struct in_addr data, u_char type, u_char tos, u_int16_t cost)
00475 {
00476   /* LSA stream is initially allocated to OSPF_MAX_LSA_SIZE, suits
00477    * vast majority of cases. Some rare routers with lots of links need more.
00478    * we try accomodate those here.
00479    */
00480   if (STREAM_WRITEABLE(s) < OSPF_ROUTER_LSA_LINK_SIZE)
00481     {
00482       size_t ret = OSPF_MAX_LSA_SIZE;
00483       
00484       /* Can we enlarge the stream still? */
00485       if (STREAM_SIZE(s) == OSPF_MAX_LSA_SIZE)
00486         {
00487           /* we futz the size here for simplicity, really we need to account
00488            * for just:
00489            * IP Header - (sizeof (struct ip))
00490            * OSPF Header - OSPF_HEADER_SIZE
00491            * LSA Header - OSPF_LSA_HEADER_SIZE
00492            * MD5 auth data, if MD5 is configured - OSPF_AUTH_MD5_SIZE.
00493            *
00494            * Simpler just to subtract OSPF_MAX_LSA_SIZE though.
00495            */
00496           ret = stream_resize (s, OSPF_MAX_PACKET_SIZE - OSPF_MAX_LSA_SIZE);
00497         }
00498       
00499       if (ret == OSPF_MAX_LSA_SIZE)
00500         {
00501           zlog_warn ("%s: Out of space in LSA stream, left %zd, size %zd",
00502                      __func__, STREAM_REMAIN (s), STREAM_SIZE (s));
00503           return 0;
00504         }
00505     }
00506   
00507   /* TOS based routing is not supported. */
00508   stream_put_ipv4 (s, id.s_addr);       /* Link ID. */
00509   stream_put_ipv4 (s, data.s_addr);     /* Link Data. */
00510   stream_putc (s, type);            /* Link Type. */
00511   stream_putc (s, tos);             /* TOS = 0. */
00512   stream_putw (s, cost);            /* Link Cost. */
00513   
00514   return 1;
00515 }

01733 /* As Type-7 */
01734 static void
01735 ospf_install_flood_nssa (struct ospf *ospf, 
01736              struct ospf_lsa *lsa, struct external_info *ei)
01737 {
01738   struct ospf_lsa *new;
01739   struct as_external_lsa *extlsa;
01740   struct ospf_area *area;
01741   struct listnode *node, *nnode;
01742 
01743   /* LSA may be a Type-5 originated via translation of a Type-7 LSA
01744    * which originated from an NSSA area. In which case it should not be 
01745    * flooded back to NSSA areas.
01746    */
01747   if (CHECK_FLAG (lsa->flags, OSPF_LSA_LOCAL_XLT))
01748     return;
01749     
01750   /* NSSA Originate or Refresh (If anyNSSA)
01751 
01752   LSA is self-originated. And just installed as Type-5.
01753   Additionally, install as Type-7 LSDB for every attached NSSA.
01754 
01755   P-Bit controls which ABR performs translation to outside world; If
01756   we are an ABR....do not set the P-bit, because we send the Type-5,
01757   not as the ABR Translator, but as the ASBR owner within the AS!
01758 
01759   If we are NOT ABR, Flood through NSSA as Type-7 w/P-bit set.  The
01760   elected ABR Translator will see the P-bit, Translate, and re-flood.
01761 
01762   Later, ABR_TASK and P-bit will scan Type-7 LSDB and translate to
01763   Type-5's to non-NSSA Areas.  (it will also attempt a re-install) */
01764 
01765   for (ALL_LIST_ELEMENTS (ospf->areas, node, nnode, area))
01766     {
01767       /* Don't install Type-7 LSA's into nonNSSA area */
01768       if (area->external_routing != OSPF_AREA_NSSA)
01769         continue;
01770 
01771       /* make lsa duplicate, lock=1 */
01772       new = ospf_lsa_dup (lsa);
01773       new->area = area;
01774       new->data->type = OSPF_AS_NSSA_LSA;
01775 
01776       /* set P-bit if not ABR */
01777       if (! IS_OSPF_ABR (ospf))
01778         {
01779       SET_FLAG(new->data->options, OSPF_OPTION_NP);
01780        
01781       /* set non-zero FWD ADDR
01782        
01783       draft-ietf-ospf-nssa-update-09.txt
01784        
01785       if the network between the NSSA AS boundary router and the
01786       adjacent AS is advertised into OSPF as an internal OSPF route,
01787       the forwarding address should be the next op address as is cu
01788       currently done with type-5 LSAs.  If the intervening network is
01789       not adversited into OSPF as an internal OSPF route and the
01790       type-7 LSA's P-bit is set a forwarding address should be
01791       selected from one of the router's active OSPF inteface addresses
01792       which belong to the NSSA.  If no such addresses exist, then
01793       no type-7 LSA's with the P-bit set should originate from this
01794       router.   */
01795        
01796       /* kevinm: not updating lsa anymore, just new */
01797       extlsa = (struct as_external_lsa *)(new->data);
01798        
01799       if (extlsa->e[0].fwd_addr.s_addr == 0)
01800         extlsa->e[0].fwd_addr = ospf_get_nssa_ip(area); /* this NSSA area in ifp */
01801 
01802       if (extlsa->e[0].fwd_addr.s_addr == 0) 
01803       {
01804         if (IS_DEBUG_OSPF_NSSA)
01805           zlog_debug ("LSA[Type-7]: Could not build FWD-ADDR");
01806         ospf_lsa_discard (new);
01807         return;
01808       }
01809     }
01810 
01811       /* install also as Type-7 */
01812       ospf_lsa_install (ospf, NULL, new);   /* Remove Old, Lock New = 2 */
01813 
01814       /* will send each copy, lock=2+n */
01815       ospf_flood_through_as (ospf, NULL, new); /* all attached NSSA's, no AS/STUBs */
01816     }

02028 /* Originate an AS-external-LSA, install and flood. */
02029 struct ospf_lsa *
02030 ospf_external_lsa_originate (struct ospf *ospf, struct external_info *ei)
02031 {
02032   struct ospf_lsa *new;
02033 
02034   /* Added for NSSA project....
02035 
02036        External LSAs are originated in ASBRs as usual, but for NSSA systems.
02037      there is the global Type-5 LSDB and a Type-7 LSDB installed for
02038      every area.  The Type-7's are flooded to every IR and every ABR; We
02039      install the Type-5 LSDB so that the normal "refresh" code operates
02040      as usual, and flag them as not used during ASE calculations.  The
02041      Type-7 LSDB is used for calculations.  Each Type-7 has a Forwarding
02042      Address of non-zero.
02043 
02044      If an ABR is the elected NSSA translator, following SPF and during
02045      the ABR task it will translate all the scanned Type-7's, with P-bit
02046      ON and not-self generated, and translate to Type-5's throughout the
02047      non-NSSA/STUB AS.
02048 
02049      A difference in operation depends whether this ASBR is an ABR
02050      or not.  If not an ABR, the P-bit is ON, to indicate that any
02051      elected NSSA-ABR can perform its translation.
02052 
02053      If an ABR, the P-bit is OFF;  No ABR will perform translation and
02054      this ASBR will flood the Type-5 LSA as usual.
02055 
02056      For the case where this ASBR is not an ABR, the ASE calculations
02057      are based on the Type-5 LSDB;  The Type-7 LSDB exists just to
02058      demonstrate to the user that there are LSA's that belong to any
02059      attached NSSA.
02060 
02061      Finally, it just so happens that when the ABR is translating every
02062      Type-7 into Type-5, it installs it into the Type-5 LSDB as an
02063      approved Type-5 (translated from Type-7);  at the end of translation
02064      if any Translated Type-5's remain unapproved, then they must be
02065      flushed from the AS.
02066 
02067      */
02068   
02069   /* Check the AS-external-LSA should be originated. */
02070   if (!ospf_redistribute_check (ospf, ei, NULL))
02071     return NULL;
02072   
02073   /* Create new AS-external-LSA instance. */
02074   if ((new = ospf_external_lsa_new (ospf, ei, NULL)) == NULL)
02075     {
02076       if (IS_DEBUG_OSPF_EVENT)
02077     zlog_debug ("LSA[Type5:%s]: Could not originate AS-external-LSA",
02078            inet_ntoa (ei->p.prefix));
02079       return NULL;
02080     }
02081 
02082   /* Install newly created LSA into Type-5 LSDB, lock = 1. */
02083   ospf_lsa_install (ospf, NULL, new);
02084 
02085   /* Update LSA origination count. */
02086   ospf->lsa_originate_count++;
02087 
02088   /* Flooding new LSA. only to AS (non-NSSA/STUB) */
02089   ospf_flood_through_as (ospf, NULL, new);
02090 
02091   /* If there is any attached NSSA, do special handling */
02092   if (ospf->anyNSSA &&
02093       /* stay away from translated LSAs! */
02094       !(CHECK_FLAG (new->flags, OSPF_LSA_LOCAL_XLT)))
02095     ospf_install_flood_nssa (ospf, new, ei); /* Install/Flood Type-7 to all NSSAs */
02096 
02097   /* Debug logging. */
02098   if (IS_DEBUG_OSPF (lsa, LSA_GENERATE))
02099     {
02100       zlog_debug ("LSA[Type%d:%s]: Originate AS-external-LSA %p",
02101          new->data->type, inet_ntoa (new->data->id), new);
02102       ospf_lsa_header_dump (new->data);
02103     }
02104 
02105   return new;
02106 }

02414      the shortest path calculations for each area (not just the
02415      area whose link-state database has changed). 
02416   */

02418   if (IS_LSA_SELF (new))

02420

02422        * If LSA was received by flooding, the RECEIVED flag is set so do
02423        * not link the LSA */
02424       if (CHECK_FLAG (new->flags, OSPF_LSA_RECEIVED))
02425     return new; /* ignore stale LSA */

02428       OSPF_TIMER_OFF (area->t_router_lsa_self);
02429       OSPF_AREA_TIMER_ON (area->t_router_lsa_self,
02430                           ospf_router_lsa_timer, OSPF_LS_REFRESH_TIME);

02435 
02436     }
02437   if (rt_recalc)
02438     ospf_spf_calculate_schedule (ospf);
02439 
02440   return new;
02441 }

02638 struct ospf_lsa *
02639 ospf_lsa_install (struct ospf *ospf, struct ospf_interface *oi,
02640           struct ospf_lsa *lsa)
02641 {
02642   struct ospf_lsa *new = NULL;
02643   struct ospf_lsa *old = NULL;
02644   struct ospf_lsdb *lsdb = NULL;
02645   int rt_recalc;
02646 
02647   /* Set LSDB. */
02648   switch (lsa->data->type)
02649     {
02650       /* kevinm */
02651     case OSPF_AS_NSSA_LSA:
02652       if (lsa->area)
02653     lsdb = lsa->area->lsdb;
02654       else
02655     lsdb = ospf->lsdb;
02656       break;
02657     case OSPF_AS_EXTERNAL_LSA:
02658 #ifdef HAVE_OPAQUE_LSA
02659     case OSPF_OPAQUE_AS_LSA:
02660 #endif /* HAVE_OPAQUE_LSA */
02661       lsdb = ospf->lsdb;
02662       break;
02663     default:
02664       lsdb = lsa->area->lsdb;
02665       break;
02666     }
02667 
02668   assert (lsdb);
02669 
02670   /*  RFC 2328 13.2.  Installing LSAs in the database
02671 
02672         Installing a new LSA in the database, either as the result of
02673         flooding or a newly self-originated LSA, may cause the OSPF
02674         routing table structure to be recalculated.  The contents of the
02675         new LSA should be compared to the old instance, if present.  If
02676         there is no difference, there is no need to recalculate the
02677         routing table. When comparing an LSA to its previous instance,
02678         the following are all considered to be differences in contents:
02679 
02680             o   The LSA's Options field has changed.
02681 
02682             o   One of the LSA instances has LS age set to MaxAge, and
02683                 the other does not.
02684 
02685             o   The length field in the LSA header has changed.
02686 
02687             o   The body of the LSA (i.e., anything outside the 20-byte
02688                 LSA header) has changed. Note that this excludes changes
02689                 in LS Sequence Number and LS Checksum.
02690 
02691   */
02692   /* Look up old LSA and determine if any SPF calculation or incremental
02693      update is needed */
02694   old = ospf_lsdb_lookup (lsdb, lsa);
02695 
02696   /* Do comparision and record if recalc needed. */
02697   rt_recalc = 0;
02698   if (  old == NULL || ospf_lsa_different(old, lsa))
02699     rt_recalc = 1;
02700 
02701   /*
02702      Sequence number check (Section 14.1 of rfc 2328)
02703      "Premature aging is used when it is time for a self-originated
02704       LSA's sequence number field to wrap.  At this point, the current
02705       LSA instance (having LS sequence number MaxSequenceNumber) must
02706       be prematurely aged and flushed from the routing domain before a
02707       new instance with sequence number equal to InitialSequenceNumber
02708       can be originated. "
02709    */
02710 
02711   if (ntohl(lsa->data->ls_seqnum) - 1 == OSPF_MAX_SEQUENCE_NUMBER)
02712     {
02713       if (ospf_lsa_is_self_originated(ospf, lsa))
02714         {
02715           lsa->data->ls_seqnum = htonl(OSPF_MAX_SEQUENCE_NUMBER);
02716           
02717           if (!IS_LSA_MAXAGE(lsa))
02718             lsa->flags |= OSPF_LSA_PREMATURE_AGE;
02719           lsa->data->ls_age = htons (OSPF_LSA_MAXAGE);
02720         
02721           if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
02722             {
02723               zlog_debug ("ospf_lsa_install() Premature Aging "
02724                  "lsa 0x%lx", (u_long)lsa);
02725               ospf_lsa_header_dump (lsa->data);
02726             }
02727         }
02728       else
02729         {
02730           if (IS_DEBUG_OSPF (lsa, LSA_GENERATE))
02731             {
02732               zlog_debug ("ospf_lsa_install() got an lsa with seq 0x80000000 "
02733                  "that was not self originated. Ignoring\n");
02734               ospf_lsa_header_dump (lsa->data);
02735             }
02736       return old;
02737         }
02738     }
02739 
02740   /* discard old LSA from LSDB */
02741   if (old != NULL)
02742     ospf_discard_from_db (ospf, lsdb, lsa);
02743 
02744   /* Calculate Checksum if self-originated?. */
02745   if (IS_LSA_SELF (lsa))
02746     ospf_lsa_checksum (lsa->data);
02747 
02748   /* Insert LSA to LSDB. */
02749   ospf_lsdb_add (lsdb, lsa);
02750   lsa->lsdb = lsdb;
02751 
02752   /* Do LSA specific installation process. */
02753   switch (lsa->data->type)
02754     {
02755     case OSPF_ROUTER_LSA:
02756       new = ospf_router_lsa_install (ospf, lsa, rt_recalc);
02757       break;
02758     case OSPF_NETWORK_LSA:
02759       assert (oi);
02760       new = ospf_network_lsa_install (ospf, oi, lsa, rt_recalc);
02761       break;
02762     case OSPF_SUMMARY_LSA:
02763       new = ospf_summary_lsa_install (ospf, lsa, rt_recalc);
02764       break;
02765     case OSPF_ASBR_SUMMARY_LSA:
02766       new = ospf_summary_asbr_lsa_install (ospf, lsa, rt_recalc);
02767       break;
02768     case OSPF_AS_EXTERNAL_LSA:
02769       new = ospf_external_lsa_install (ospf, lsa, rt_recalc);
02770       break;
02771 #ifdef HAVE_OPAQUE_LSA
02772     case OSPF_OPAQUE_LINK_LSA:
02773       if (IS_LSA_SELF (lsa))
02774     lsa->oi = oi; /* Specify outgoing ospf-interface for this LSA. */
02775       else
02776     ; /* Incoming "oi" for this LSA has set at LSUpd reception. */
02777       /* Fallthrough */
02778     case OSPF_OPAQUE_AREA_LSA:
02779     case OSPF_OPAQUE_AS_LSA:
02780       new = ospf_opaque_lsa_install (lsa, rt_recalc);
02781       break;
02782 #endif /* HAVE_OPAQUE_LSA */
02783     case OSPF_AS_NSSA_LSA:
02784       new = ospf_external_lsa_install (ospf, lsa, rt_recalc);
02785     default: /* type-6,8,9....nothing special */
02786       break;
02787     }
02788 
02789   if (new == NULL)
02790     return new;  /* Installation failed, cannot proceed further -- endo. */
02791 
02792   /* Debug logs. */
02793   if (IS_DEBUG_OSPF (lsa, LSA_INSTALL))
02794     {
02795       char area_str[INET_ADDRSTRLEN];
02796 
02797       switch (lsa->data->type)
02798         {
02799         case OSPF_AS_EXTERNAL_LSA:
02800 #ifdef HAVE_OPAQUE_LSA
02801         case OSPF_OPAQUE_AS_LSA:
02802 #endif /* HAVE_OPAQUE_LSA */
02803         case OSPF_AS_NSSA_LSA:
02804           zlog_debug ("LSA[%s]: Install %s",
02805                  dump_lsa_key (new),
02806                  LOOKUP (ospf_lsa_type_msg, new->data->type));
02807           break;
02808         default:
02809       strcpy (area_str, inet_ntoa (new->area->area_id));
02810           zlog_debug ("LSA[%s]: Install %s to Area %s",
02811                  dump_lsa_key (new),
02812                  LOOKUP (ospf_lsa_type_msg, new->data->type), area_str);
02813           break;
02814         }
02815     }
02816 
02817   /* 
02818      If received LSA' ls_age is MaxAge, or lsa is being prematurely aged
02819      (it's getting flushed out of the area), set LSA on MaxAge LSA list. 
02820    */
02821   if ((lsa->flags & OSPF_LSA_PREMATURE_AGE) ||
02822       (IS_LSA_MAXAGE (new) && !IS_LSA_SELF (new)))
02823     {
02824       if (IS_DEBUG_OSPF (lsa, LSA_INSTALL))
02825         zlog_debug ("LSA[Type%d:%s]: Install LSA 0x%p, MaxAge",
02826                    new->data->type, 
02827                    inet_ntoa (new->data->id), 
02828                    lsa);
02829       ospf_lsa_maxage (ospf, lsa);
02830     }
02831 
02832   return new;
02833 }

02891 static int
02892 ospf_maxage_lsa_remover (struct thread *thread)
02893 {
02894   struct ospf *ospf = THREAD_ARG (thread);
02895   struct ospf_lsa *lsa;
02896   struct listnode *node, *nnode;
02897   int reschedule = 0;
02898 
02899   ospf->t_maxage = NULL;
02900 
02901   if (IS_DEBUG_OSPF (lsa, LSA_FLOODING))
02902     zlog_debug ("LSA[MaxAge]: remover Start");
02903 
02904   reschedule = !ospf_check_nbr_status (ospf);
02905 
02906   if (!reschedule)
02907     for (ALL_LIST_ELEMENTS (ospf->maxage_lsa, node, nnode, lsa))
02908       {
02909         if (lsa->retransmit_counter > 0)
02910           {
02911             reschedule = 1;
02912             continue;
02913           }
02914 
02915         /* Remove LSA from the LSDB */
02916         if (CHECK_FLAG (lsa->flags, OSPF_LSA_SELF))
02917           if (IS_DEBUG_OSPF (lsa, LSA_FLOODING))
02918             zlog_debug ("LSA[Type%d:%s]: LSA 0x%lx is self-oririnated: ",
02919                        lsa->data->type, inet_ntoa (lsa->data->id), (u_long)lsa);
02920 
02921         if (IS_DEBUG_OSPF (lsa, LSA_FLOODING))
02922           zlog_debug ("LSA[Type%d:%s]: MaxAge LSA removed from list",
02923                      lsa->data->type, inet_ntoa (lsa->data->id));
02924 
02925     /* Flood max age LSA. */
02926 #ifdef ORIGINAL_CODING
02927     ospf_maxage_flood (lsa);
02928 #else /* ORIGINAL_CODING */
02929         ospf_flood_through (ospf, NULL, lsa);
02930 #endif /* ORIGINAL_CODING */
02931 
02932     if (lsa->flags & OSPF_LSA_PREMATURE_AGE)  
02933           {
02934             if (IS_DEBUG_OSPF (lsa, LSA_FLOODING))
02935               zlog_debug ("originating new router lsa for lsa 0x%lx \n", 
02936                          (u_long)lsa);
02937             ospf_router_lsa_originate(lsa->area);
02938           }
02939 
02940     /* Remove from lsdb. */
02941     if (lsa->lsdb)
02942       {
02943         ospf_discard_from_db (ospf, lsa->lsdb, lsa);
02944         ospf_lsdb_delete (lsa->lsdb, lsa);
02945           }

02958   return 0;
02959 }

02978      and schedule the MaxAge LSA remover. */
02979   if (CHECK_FLAG(lsa->flags, OSPF_LSA_IN_MAXAGE))
02980     {

02990   if (IS_DEBUG_OSPF (lsa, LSA_FLOODING))
02991     zlog_debug ("LSA[%s]: MaxAge LSA remover scheduled.", dump_lsa_key (lsa));

02993   OSPF_TIMER_ON (ospf->t_maxage, ospf_maxage_lsa_remover, 2);

02996 static int
02997 ospf_lsa_maxage_walker_remover (struct ospf *ospf, struct ospf_lsa *lsa)
02998 {
02999   /* Stay away from any Local Translated Type-7 LSAs */
03000   if (CHECK_FLAG (lsa->flags, OSPF_LSA_LOCAL_XLT))
03001     return 0;
03002 
03003   if (IS_LSA_MAXAGE (lsa))
03004     /* Self-originated LSAs should NOT time-out instead,
03005        they're flushed and submitted to the max_age list explicitly. */
03006     if (!ospf_lsa_is_self_originated (ospf, lsa))
03007       {
03008     if (IS_DEBUG_OSPF (lsa, LSA_FLOODING))
03009       zlog_debug("LSA[%s]: is MaxAge", dump_lsa_key (lsa));
03010 
03011         switch (lsa->data->type)
03012           {
03013 #ifdef HAVE_OPAQUE_LSA
03014           case OSPF_OPAQUE_LINK_LSA:
03015           case OSPF_OPAQUE_AREA_LSA:
03016           case OSPF_OPAQUE_AS_LSA:
03017             /*
03018              * As a general rule, whenever network topology has changed
03019              * (due to an LSA removal in this case), routing recalculation
03020              * should be triggered. However, this is not true for opaque
03021              * LSAs. Even if an opaque LSA instance is going to be removed
03022              * from the routing domain, it does not mean a change in network
03023              * topology, and thus, routing recalculation is not needed here.
03024              */
03025             break;
03026 #endif /* HAVE_OPAQUE_LSA */
03027           case OSPF_AS_EXTERNAL_LSA:
03028           case OSPF_AS_NSSA_LSA:
03029         ospf_ase_incremental_update (ospf, lsa);
03030             break;
03031           default:
03032         ospf_spf_calculate_schedule (ospf);
03033             break;
03034           }
03035     ospf_lsa_maxage (ospf, lsa);
03036       }
03037 
03038   return 0;
03039 }

03216 /* return +n, l1 is more recent.
03217    return -n, l2 is more recent.
03218    return 0, l1 and l2 is identical. */
03219 int
03220 ospf_lsa_more_recent (struct ospf_lsa *l1, struct ospf_lsa *l2)
03221 {
03222   int r;
03223   int x, y;
03224 
03225   if (l1 == NULL && l2 == NULL)
03226     return 0;
03227   if (l1 == NULL)
03228     return -1;
03229   if (l2 == NULL)
03230     return 1;
03231 
03232   /* compare LS sequence number. */
03233   x = (int) ntohl (l1->data->ls_seqnum);
03234   y = (int) ntohl (l2->data->ls_seqnum);
03235   if (x > y)
03236     return 1;
03237   if (x < y)
03238     return -1;
03239 
03240   /* compare LS checksum. */
03241   r = ntohs (l1->data->checksum) - ntohs (l2->data->checksum);
03242   if (r)
03243     return r;
03244 
03245   /* compare LS age. */
03246   if (IS_LSA_MAXAGE (l1) && !IS_LSA_MAXAGE (l2))
03247     return 1;
03248   else if (!IS_LSA_MAXAGE (l1) && IS_LSA_MAXAGE (l2))
03249     return -1;
03250 
03251   /* compare LS age with MaxAgeDiff. */
03252   if (LS_AGE (l1) - LS_AGE (l2) > OSPF_LSA_MAXAGE_DIFF)
03253     return -1;
03254   else if (LS_AGE (l2) - LS_AGE (l1) > OSPF_LSA_MAXAGE_DIFF)
03255     return 1;
03256 
03257   /* LSAs are identical. */
03258   return 0;
03259 }

03261 /* If two LSAs are different, return 1, otherwise return 0. */
03262 int
03263 ospf_lsa_different (struct ospf_lsa *l1, struct ospf_lsa *l2)
03264 {
03265   char *p1, *p2;
03266   assert (l1);
03267   assert (l2);
03268   assert (l1->data);
03269   assert (l2->data);
03270 
03271   if (l1->data->options != l2->data->options)
03272     return 1;
03273 
03274   if (IS_LSA_MAXAGE (l1) && !IS_LSA_MAXAGE (l2))
03275     return 1;
03276 
03277   if (IS_LSA_MAXAGE (l2) && !IS_LSA_MAXAGE (l1))
03278     return 1;
03279 
03280   if (l1->data->length != l2->data->length)
03281     return 1;
03282 
03283   if (l1->data->length ==  0)
03284     return 1;
03285 
03286   if (CHECK_FLAG ((l1->flags ^ l2->flags), OSPF_LSA_RECEIVED))
03287     return 1; /* May be a stale LSA in the LSBD */
03288 
03289   assert ( ntohs(l1->data->length) > OSPF_LSA_HEADER_SIZE);
03290 
03291   p1 = (char *) l1->data;
03292   p2 = (char *) l2->data;
03293 
03294   if (memcmp (p1 + OSPF_LSA_HEADER_SIZE, p2 + OSPF_LSA_HEADER_SIZE,
03295               ntohs( l1->data->length ) - OSPF_LSA_HEADER_SIZE) != 0)
03296     return 1;
03297 
03298   return 0;
03299 }

03301 #ifdef ORIGINAL_CODING
03302 void
03303 ospf_lsa_flush_self_originated (struct ospf_neighbor *nbr,
03304                                 struct ospf_lsa *self,
03305                                 struct ospf_lsa *new)
03306 {
03307   u_int32_t seqnum;
03308 
03309   /* Adjust LS Sequence Number. */
03310   seqnum = ntohl (new->data->ls_seqnum) + 1;
03311   self->data->ls_seqnum = htonl (seqnum);
03312 
03313   /* Recalculate LSA checksum. */
03314   ospf_lsa_checksum (self->data);
03315 
03316   /* Reflooding LSA. */
03317   /*  RFC2328  Section 13.3
03318         On non-broadcast networks, separate Link State Update
03319         packets must be sent, as unicasts, to each adjacent neighbor
03320         (i.e., those in state Exchange or greater).  The destination
03321         IP addresses for these packets are the neighbors' IP
03322         addresses.   */
03323   if (nbr->oi->type == OSPF_IFTYPE_NBMA)
03324     {
03325       struct route_node *rn;
03326       struct ospf_neighbor *onbr;
03327 
03328       for (rn = route_top (nbr->oi->nbrs); rn; rn = route_next (rn))
03329     if ((onbr = rn->info) != NULL)
03330       if (onbr != nbr->oi->nbr_self && onbr->status >= NSM_Exchange)
03331         ospf_ls_upd_send_lsa (onbr, self, OSPF_SEND_PACKET_DIRECT);
03332     }
03333   else
03334   ospf_ls_upd_send_lsa (nbr, self, OSPF_SEND_PACKET_INDIRECT);
03335 
03336   if (IS_DEBUG_OSPF (lsa, LSA_GENERATE))
03337     zlog_debug ("LSA[Type%d:%s]: Flush self-originated LSA",
03338            self->data->type, inet_ntoa (self->data->id));
03339 }

03342 ospf_lsa_flush_schedule (struct ospf *ospf, struct ospf_lsa *lsa)

03345     return 0;
03346

03348     zlog_debug ("LSA[Type%d:%s]: Schedule self-originated LSA to FLUSH", lsa->data->type, inet_ntoa (lsa->data->id));
03349 
03350   /* Force given lsa's age to MaxAge. */
03351   lsa->data->ls_age = htons (OSPF_LSA_MAXAGE);

03357     case OSPF_OPAQUE_AREA_LSA:

03359       ospf_opaque_lsa_refresh (lsa);

03361 #endif /* HAVE_OPAQUE_LSA */
03362     default:
03363       ospf_lsa_maxage (ospf, lsa);
03364       break;
03365     }
03366

03450 /* If there is self-originated LSA, then return 1, otherwise return 0. */
03451 /* An interface-independent version of ospf_lsa_is_self_originated */
03452 int 
03453 ospf_lsa_is_self_originated (struct ospf *ospf, struct ospf_lsa *lsa)
03454 {
03455   struct listnode *node;
03456   struct ospf_interface *oi;
03457 
03458   /* This LSA is already checked. */
03459   if (CHECK_FLAG (lsa->flags, OSPF_LSA_SELF_CHECKED))
03460     return CHECK_FLAG (lsa->flags, OSPF_LSA_SELF);
03461 
03462   /* Make sure LSA is self-checked. */
03463   SET_FLAG (lsa->flags, OSPF_LSA_SELF_CHECKED);
03464 
03465   /* AdvRouter and Router ID is the same. */
03466   if (IPV4_ADDR_SAME (&lsa->data->adv_router, &ospf->router_id))
03467     SET_FLAG (lsa->flags, OSPF_LSA_SELF);
03468 
03469   /* LSA is router-LSA. */
03470   else if (lsa->data->type == OSPF_ROUTER_LSA &&
03471       IPV4_ADDR_SAME (&lsa->data->id, &ospf->router_id))
03472     SET_FLAG (lsa->flags, OSPF_LSA_SELF);
03473 
03474   /* LSA is network-LSA.  Compare Link ID with all interfaces. */
03475   else if (lsa->data->type == OSPF_NETWORK_LSA)
03476     for (ALL_LIST_ELEMENTS_RO (ospf->oiflist, node, oi))
03477       {
03478     /* Ignore virtual link. */
03479         if (oi->type != OSPF_IFTYPE_VIRTUALLINK)
03480       if (oi->address->family == AF_INET)
03481         if (IPV4_ADDR_SAME (&lsa->data->id, &oi->address->u.prefix4))
03482           {
03483         /* to make it easier later */
03484         SET_FLAG (lsa->flags, OSPF_LSA_SELF);
03485         return CHECK_FLAG (lsa->flags, OSPF_LSA_SELF);
03486           }
03487       }
03488 
03489   return CHECK_FLAG (lsa->flags, OSPF_LSA_SELF);
03490 }

03492 /* Get unique Link State ID. */
03493 struct in_addr
03494 ospf_lsa_unique_id (struct ospf *ospf,
03495             struct ospf_lsdb *lsdb, u_char type, struct prefix_ipv4 *p)
03496 {
03497   struct ospf_lsa *lsa;
03498   struct in_addr mask, id;
03499 
03500   id = p->prefix;
03501 
03502   /* Check existence of LSA instance. */
03503   lsa = ospf_lsdb_lookup_by_id (lsdb, type, id, ospf->router_id);
03504   if (lsa)
03505     {
03506       struct as_external_lsa *al = (struct as_external_lsa *) lsa->data;
03507       if (ip_masklen (al->mask) == p->prefixlen)
03508     {
03509       if (IS_DEBUG_OSPF (lsa, LSA_GENERATE))
03510         zlog_debug ("ospf_lsa_unique_id(): "
03511                "Can't get Link State ID for %s/%d",
03512                inet_ntoa (p->prefix), p->prefixlen);
03513       /*      id.s_addr = 0; */
03514       id.s_addr = 0xffffffff;
03515       return id;
03516     }
03517       /* Masklen differs, then apply wildcard mask to Link State ID. */
03518       else
03519     {
03520       masklen2ip (p->prefixlen, &mask);
03521 
03522       id.s_addr = p->prefix.s_addr | (~mask.s_addr);
03523       lsa = ospf_lsdb_lookup_by_id (ospf->lsdb, type,
03524                        id, ospf->router_id);
03525       if (lsa)
03526         {
03527           if (IS_DEBUG_OSPF (lsa, LSA_GENERATE))
03528         zlog_debug ("ospf_lsa_unique_id(): "
03529                "Can't get Link State ID for %s/%d",
03530                inet_ntoa (p->prefix), p->prefixlen);
03531           /*          id.s_addr = 0; */
03532           id.s_addr = 0xffffffff;
03533           return id;
03534         }
03535     }
03536     }

03648 void
03649 ospf_refresher_register_lsa (struct ospf *ospf, struct ospf_lsa *lsa)
03650 {
03651   u_int16_t index, current_index;
03652   
03653   assert (CHECK_FLAG (lsa->flags, OSPF_LSA_SELF));
03654 
03655   if (lsa->refresh_list < 0)
03656     {
03657       int delay;
03658 
03659       if (LS_AGE (lsa) == 0 &&
03660       ntohl (lsa->data->ls_seqnum) == OSPF_INITIAL_SEQUENCE_NUMBER)
03661     /* Randomize first update by  OSPF_LS_REFRESH_SHIFT factor */ 
03662     delay = OSPF_LS_REFRESH_SHIFT + (random () % OSPF_LS_REFRESH_TIME);
03663       else
03664     /* Randomize another updates by +-OSPF_LS_REFRESH_JITTER factor */
03665     delay = OSPF_LS_REFRESH_TIME - LS_AGE (lsa) - OSPF_LS_REFRESH_JITTER
03666       + (random () % (2*OSPF_LS_REFRESH_JITTER)); 
03667 
03668       if (delay < 0)
03669     delay = 0;
03670 
03671       current_index = ospf->lsa_refresh_queue.index +
03672     (quagga_time (NULL) - ospf->lsa_refresher_started)/OSPF_LSA_REFRESHER_GRANULARITY;
03673       
03674       index = (current_index + delay/OSPF_LSA_REFRESHER_GRANULARITY)
03675     % (OSPF_LSA_REFRESHER_SLOTS);
03676 
03677       if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03678     zlog_debug ("LSA[Refresh]: lsa %s with age %d added to index %d",
03679            inet_ntoa (lsa->data->id), LS_AGE (lsa), index);
03680       if (!ospf->lsa_refresh_queue.qs[index])
03681     ospf->lsa_refresh_queue.qs[index] = list_new ();
03682       listnode_add (ospf->lsa_refresh_queue.qs[index],
03683                     ospf_lsa_lock (lsa)); /* lsa_refresh_queue */
03684       lsa->refresh_list = index;
03685       if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03686         zlog_debug ("LSA[Refresh:%s]: ospf_refresher_register_lsa(): "
03687                    "setting refresh_list on lsa %p (slod %d)", 
03688                    inet_ntoa (lsa->data->id), lsa, index);
03689     }
03690 }

03710 int
03711 ospf_lsa_refresh_walker (struct thread *t)
03712 {
03713   struct list *refresh_list;
03714   struct listnode *node, *nnode;
03715   struct ospf *ospf = THREAD_ARG (t);
03716   struct ospf_lsa *lsa;
03717   int i;
03718   struct list *lsa_to_refresh = list_new ();
03719 
03720   if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03721     zlog_debug ("LSA[Refresh]:ospf_lsa_refresh_walker(): start");
03722 
03723   
03724   i = ospf->lsa_refresh_queue.index;
03725   
03726   /* Note: if clock has jumped backwards, then time change could be negative,
03727      so we are careful to cast the expression to unsigned before taking
03728      modulus. */
03729   ospf->lsa_refresh_queue.index =
03730    ((unsigned long)(ospf->lsa_refresh_queue.index +
03731             (quagga_time (NULL) - ospf->lsa_refresher_started) /
03732             OSPF_LSA_REFRESHER_GRANULARITY)) % OSPF_LSA_REFRESHER_SLOTS;
03733 
03734   if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03735     zlog_debug ("LSA[Refresh]: ospf_lsa_refresh_walker(): next index %d",
03736            ospf->lsa_refresh_queue.index);
03737 
03738   for (;i != ospf->lsa_refresh_queue.index;
03739        i = (i + 1) % OSPF_LSA_REFRESHER_SLOTS)
03740     {
03741       if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03742     zlog_debug ("LSA[Refresh]: ospf_lsa_refresh_walker(): "
03743                "refresh index %d", i);
03744 
03745       refresh_list = ospf->lsa_refresh_queue.qs [i];
03746       
03747       ospf->lsa_refresh_queue.qs [i] = NULL;
03748 
03749       if (refresh_list)
03750     {
03751       for (ALL_LIST_ELEMENTS (refresh_list, node, nnode, lsa))
03752         {
03753           if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03754         zlog_debug ("LSA[Refresh:%s]: ospf_lsa_refresh_walker(): "
03755                    "refresh lsa %p (slot %d)", 
03756                    inet_ntoa (lsa->data->id), lsa, i);
03757           
03758           list_delete_node (refresh_list, node);
03759           ospf_lsa_unlock (&lsa); /* lsa_refresh_queue */
03760           lsa->refresh_list = -1;
03761           listnode_add (lsa_to_refresh, lsa);
03762         }
03763       list_free (refresh_list);
03764     }
03765     }
03766 
03767   ospf->t_lsa_refresher = thread_add_timer (master, ospf_lsa_refresh_walker,
03768                        ospf, ospf->lsa_refresh_interval);
03769   ospf->lsa_refresher_started = quagga_time (NULL);
03770 
03771   for (ALL_LIST_ELEMENTS (lsa_to_refresh, node, nnode, lsa))
03772     ospf_lsa_refresh (ospf, lsa);
03773   
03774   list_delete (lsa_to_refresh);
03775   
03776   if (IS_DEBUG_OSPF (lsa, LSA_REFRESH))
03777     zlog_debug ("LSA[Refresh]: ospf_lsa_refresh_walker(): end");
03778   
03779   return 0;
03780 }
