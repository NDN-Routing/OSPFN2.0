00152 #define VTY_GET_LONG(NAME,V,STR) \
00153 do { \
00154   char *endptr = NULL; \
00155   (V) = strtoul ((STR), &endptr, 10); \
00156   if (*endptr != '\0' || (V) == ULONG_MAX) \
00157     { \
00158       vty_out (vty, "%% Invalid %s value%s", NAME, VTY_NEWLINE); \
00159       return CMD_WARNING; \
00160     } \
00161 } while (0)

00163 #define VTY_GET_INTEGER_RANGE(NAME,V,STR,MIN,MAX) \
00164 do { \
00165   unsigned long tmpl; \
00166   VTY_GET_LONG(NAME, tmpl, STR); \
00167   if ( (tmpl < (MIN)) || (tmpl > (MAX))) \
00168     { \
00169       vty_out (vty, "%% Invalid %s value%s", NAME, VTY_NEWLINE); \
00170       return CMD_WARNING; \
00171     } \
