02267       if (varflag)
02268     argv[argc++] = vector_slot (vline, i);
02269       else
02270     {
02271       vector descvec = vector_slot (matched_element->strvec, i);
02272 
02273       if (vector_active (descvec) == 1)
02274         {
02275           struct desc *desc = vector_slot (descvec, 0);
02276 
02277           if (CMD_VARARG (desc->cmd))
02278         varflag = 1;
02279

02281         argv[argc++] = vector_slot (vline, i);
02282         }
02283       else
02284         argv[argc++] = vector_slot (vline, i);
02285     }
02286 
02287       if (argc >= CMD_ARGC_MAX)
02288     return CMD_ERR_EXEED_ARGC_MAX;
02289     }
02290 
02291   /* For vtysh execution. */
02292   if (cmd)
02293     *cmd = matched_element;
02294 
02295   if (matched_element->daemon)
02296     return CMD_SUCCESS_DAEMON;
02297 
02298   /* Now execute matched command */
02299   return (*matched_element->func) (matched_element, vty, argc, argv);

03192 static int
03193 set_log_file(struct vty *vty, const char *fname, int loglevel)
03194 {
03195   int ret;
03196   char *p = NULL;
03197   const char *fullpath;
03198   
03199   /* Path detection. */
03200   if (! IS_DIRECTORY_SEP (*fname))
03201     {
03202       char cwd[MAXPATHLEN+1];
03203       cwd[MAXPATHLEN] = '\0';
03204       
03205       if (getcwd (cwd, MAXPATHLEN) == NULL)
03206         {
03207           zlog_err ("config_log_file: Unable to alloc mem!");
03208           return CMD_WARNING;
03209         }
03210       
03211       if ( (p = XMALLOC (MTYPE_TMP, strlen (cwd) + strlen (fname) + 2))
03212           == NULL)
03213         {
03214           zlog_err ("config_log_file: Unable to alloc mem!");
03215           return CMD_WARNING;
03216         }
03217       sprintf (p, "%s/%s", cwd, fname);
03218       fullpath = p;
03219     }
03220   else
03221     fullpath = fname;
03222 
03223   ret = zlog_set_file (NULL, fullpath, loglevel);
03224 
03225   if (p)
03226     XFREE (MTYPE_TMP, p);
03227 
03228   if (!ret)
03229     {
03230       vty_out (vty, "can't open logfile %s\n", fname);
03231       return CMD_WARNING;
03232     }
03233 
03234   if (host.logfile)
03235     XFREE (MTYPE_HOST, host.logfile);
03236 
03237   host.logfile = XSTRDUP (MTYPE_HOST, fname);
03238 
03239   return CMD_SUCCESS;
03240 }
