00323           {
00324         fprintf  (fp, "%s\n", line);
00325         fflush (fp);
00326           }
00327         if (! NO_DELIMITER (i))
00328           {
00329         fprintf (fp, "!\n");
00330         fflush (fp);
00331           }
00332       }
00333     if (NO_DELIMITER (i))
00334       {
00335         fprintf (fp, "!\n");
00336         fflush (fp);
00337       }
00338       }
00339 
00340   for (i = 0; i < vector_active (configvec); i++)
00341     if ((master = vector_slot (configvec, i)) != NULL)
00342       {
00343     list_delete (master);
00344     vector_slot (configvec, i) = NULL;
00345       }
00346   list_delete_all_node (config_top);
