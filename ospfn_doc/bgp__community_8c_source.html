00163 community_uniq_sort (struct community *com)

00298 {

00304   /* Lookup community hash. */

00306 
00307   /* Arguemnt com is allocated temporary.  So when it is not used in

00310     community_free (com);

00313   find->refcnt++;

00317     find->str = community_com2str (find);

00411 {

00414 
00415   if (com1 == NULL && com2 == NULL)
00416     return 1;

00419     return 0;
00420 
00421   if (com1->size < com2->size)
00422     return 0;
00423 
00424   /* Every community on com2 needs to be on com1 for this to match */
00425   while (i < com1->size && j < com2->size)
00426     {
00427       if (memcmp (com1->val + i, com2->val + j, sizeof (u_int32_t)) == 0)
00428     j++;
00429       i++;

00431

00433     return 1;
00434   else
00435     return 0;

00480 /* Get next community token from string. */
00481 static const char *
00482 community_gettoken (const char *buf, enum community_token *token, 
00483                     u_int32_t *val)
00484 {
00485   const char *p = buf;
00486 
00487   /* Skip white space. */
00488   while (isspace ((int) *p))
00489     p++;
00490 
00491   /* Check the end of the line. */
00492   if (*p == '\0')
00493     return NULL;
00494 
00495   /* Well known community string check. */
00496   if (isalpha ((int) *p)) 
00497     {
00498       if (strncmp (p, "internet", strlen ("internet")) == 0)
00499     {
00500       *val = COMMUNITY_INTERNET;
00501       *token = community_token_no_export;
00502       p += strlen ("internet");
00503       return p;
00504     }
00505       if (strncmp (p, "no-export", strlen ("no-export")) == 0)
00506     {
00507       *val = COMMUNITY_NO_EXPORT;
00508       *token = community_token_no_export;
00509       p += strlen ("no-export");
00510       return p;
00511     }
00512       if (strncmp (p, "no-advertise", strlen ("no-advertise")) == 0)
00513     {
00514       *val = COMMUNITY_NO_ADVERTISE;
00515       *token = community_token_no_advertise;
00516       p += strlen ("no-advertise");
00517       return p;
00518     }
00519       if (strncmp (p, "local-AS", strlen ("local-AS")) == 0)
00520     {
00521       *val = COMMUNITY_LOCAL_AS;
00522       *token = community_token_local_as;
00523       p += strlen ("local-AS");
00524       return p;
00525     }
00526 
00527       /* Unknown string. */
00528       *token = community_token_unknown;
00529       return NULL;
00530     }
00531 
00532   /* Community value. */
00533   if (isdigit ((int) *p)) 
00534     {
00535       int separator = 0;
00536       int digit = 0;
00537       u_int32_t community_low = 0;
00538       u_int32_t community_high = 0;
00539 
00540       while (isdigit ((int) *p) || *p == ':') 
00541     {
00542       if (*p == ':') 
00543         {
00544           if (separator)
00545         {
00546           *token = community_token_unknown;
00547           return NULL;
00548         }
00549           else
00550         {
00551           separator = 1;
00552           digit = 0;
00553           community_high = community_low << 16;
00554           community_low = 0;
00555         }
00556         }
00557       else 
00558         {
00559           digit = 1;
00560           community_low *= 10;
00561           community_low += (*p - '0');
00562         }
00563       p++;
00564     }
00565       if (! digit)
00566     {
00567       *token = community_token_unknown;
00568       return NULL;
00569     }
00570       *val = community_high + community_low;
00571       *token = community_token_val;
00572       return p;
00573     }
00574   *token = community_token_unknown;
00575   return NULL;
00576 }
