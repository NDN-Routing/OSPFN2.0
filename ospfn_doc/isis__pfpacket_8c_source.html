00065 isis_multicast_join (int fd, int registerto, int if_num)
00066 {
00067   struct packet_mreq mreq;
00068 
00069   memset (&mreq, 0, sizeof (mreq));
00070   mreq.mr_ifindex = if_num;
00071   if (registerto)
00072     {
00073       mreq.mr_type = PACKET_MR_MULTICAST;
00074       mreq.mr_alen = ETH_ALEN;
00075       if (registerto == 1)
00076     memcpy (&mreq.mr_address, ALL_L1_ISS, ETH_ALEN);
00077       else if (registerto == 2)
00078     memcpy (&mreq.mr_address, ALL_L2_ISS, ETH_ALEN);
00079       else if (registerto == 3)
00080     memcpy (&mreq.mr_address, ALL_ISS, ETH_ALEN);
00081       else
00082     memcpy (&mreq.mr_address, ALL_ESS, ETH_ALEN);
00083 
00084     }
00085   else
00086     {
00087       mreq.mr_type = PACKET_MR_ALLMULTI;
00088     }
00089 #ifdef EXTREME_DEBUG
00090   zlog_debug ("isis_multicast_join(): fd=%d, reg_to=%d, if_num=%d, "
00091           "address = %02x:%02x:%02x:%02x:%02x:%02x",
00092           fd, registerto, if_num, mreq.mr_address[0], mreq.mr_address[1],
00093           mreq.mr_address[2], mreq.mr_address[3], mreq.mr_address[4],
00094           mreq.mr_address[5]);
00095 #endif /* EXTREME_DEBUG */
00096   if (setsockopt (fd, SOL_PACKET, PACKET_ADD_MEMBERSHIP, &mreq,
00097           sizeof (struct packet_mreq)))
00098     {
00099       zlog_warn ("isis_multicast_join(): setsockopt(): %s", safe_strerror (errno));

00119 
00120   /*
00121    * Bind to the physical interface
00122    */
00123   memset (&s_addr, 0, sizeof (struct sockaddr_ll));
00124   s_addr.sll_family = AF_PACKET;
00125   s_addr.sll_protocol = htons (ETH_P_ALL);
00126   s_addr.sll_ifindex = circuit->interface->ifindex;
00127 
00128   if (bind (fd, (struct sockaddr *) (&s_addr),
00129         sizeof (struct sockaddr_ll)) < 0)
00130     {
00131       zlog_warn ("open_packet_socket(): bind() failed: %s", safe_strerror (errno));
00132       return ISIS_WARNING;
00133     }
00134 
00135   circuit->fd = fd;
00136 
00137   if (circuit->circ_type == CIRCUIT_T_BROADCAST)
00138     {
00139       /*
00140        * Join to multicast groups
00141        * according to
00142        * 8.4.2 - Broadcast subnetwork IIH PDUs
00143        * FIXME: is there a case only one will fail??
00144        */
00145       if (circuit->circuit_is_type & IS_LEVEL_1)
00146     {
00147       /* joining ALL_L1_ISS */
00148       retval = isis_multicast_join (circuit->fd, 1,
00149                     circuit->interface->ifindex);
00150       /* joining ALL_ISS */
00151       retval = isis_multicast_join (circuit->fd, 3,
00152                     circuit->interface->ifindex);
00153     }
00154       if (circuit->circuit_is_type & IS_LEVEL_2)
00155     /* joining ALL_L2_ISS */
00156     retval = isis_multicast_join (circuit->fd, 2,
00157                       circuit->interface->ifindex);
00158     }
00159   else
00160     {
00161       retval =
00162     isis_multicast_join (circuit->fd, 0, circuit->interface->ifindex);
00163     }

00220 int
00221 isis_recv_pdu_bcast (struct isis_circuit *circuit, u_char * ssnpa)
00222 {
00223   int bytesread, addr_len;
00224   struct sockaddr_ll s_addr;
00225   u_char llc[LLC_LEN];
00226 
00227   addr_len = sizeof (s_addr);
00228 
00229   memset (&s_addr, 0, sizeof (struct sockaddr_ll));
00230 
00231   bytesread = recvfrom (circuit->fd, (void *) &llc,
00232             LLC_LEN, MSG_PEEK,
00233             (struct sockaddr *) &s_addr, (socklen_t *) &addr_len);
00234 
00235   if (bytesread < 0)
00236     {
00237       zlog_warn ("isis_recv_packet_bcast(): fd %d, recvfrom (): %s",
00238          circuit->fd, safe_strerror (errno));
00239       zlog_warn ("circuit is %s", circuit->interface->name);
00240       zlog_warn ("circuit fd %d", circuit->fd);
00241       zlog_warn ("bytesread %d", bytesread);
00242       /* get rid of the packet */
00243       bytesread = read (circuit->fd, discard_buff, sizeof (discard_buff));
00244       return ISIS_WARNING;
00245     }
00246   /*
00247    * Filtering by llc field, discard packets sent by this host (other circuit)
00248    */
00249   if (!llc_check (llc) || s_addr.sll_pkttype == PACKET_OUTGOING)
00250     {
00251       /*  Read the packet into discard buff */
00252       bytesread = read (circuit->fd, discard_buff, sizeof (discard_buff));
00253       if (bytesread < 0)
00254     zlog_warn ("isis_recv_pdu_bcast(): read() failed");
00255       return ISIS_WARNING;
00256     }
00257 
00258   /* on lan we have to read to the static buff first */
00259   bytesread = recvfrom (circuit->fd, sock_buff, circuit->interface->mtu, 0,
00260             (struct sockaddr *) &s_addr, (socklen_t *) &addr_len);
00261 
00262   /* then we lose the LLC */
00263   stream_write (circuit->rcv_stream, sock_buff + LLC_LEN, bytesread - LLC_LEN);
00264 
00265   memcpy (ssnpa, &s_addr.sll_addr, s_addr.sll_halen);
00266 
00267   return ISIS_OK;
00268 }

00280   bytesread = stream_recvfrom (circuit->rcv_stream, circuit->fd,
00281                                circuit->interface->mtu, 0,

00289       if (bytesread < 0)
00290     zlog_warn ("isis_recv_pdu_p2p(): read() failed");
00291       return ISIS_WARNING;

00293 
00294   /* If we don't have protocol type 0x00FE which is
00295    * ISO over GRE we exit with pain :)
00296    */
00297   if (ntohs (s_addr.sll_protocol) != 0x00FE)
00298     {
00299       zlog_warn ("isis_recv_pdu_p2p(): protocol mismatch(): %X",
00300          ntohs (s_addr.sll_protocol));
00301       return ISIS_WARNING;
00302     }
00303 
00304   memcpy (ssnpa, &s_addr.sll_addr, s_addr.sll_halen);
00305 
00306   return ISIS_OK;

00309 int
00310 isis_send_pdu_bcast (struct isis_circuit *circuit, int level)
00311 {
00312   /* we need to do the LLC in here because of P2P circuits, which will
00313    * not need it
00314    */
00315   int written = 1;
00316   struct sockaddr_ll sa;
00317 
00318   stream_set_getp (circuit->snd_stream, 0);
00319   memset (&sa, 0, sizeof (struct sockaddr_ll));
00320   sa.sll_family = AF_PACKET;
00321   sa.sll_protocol = htons (stream_get_endp (circuit->snd_stream) + LLC_LEN);
00322   sa.sll_ifindex = circuit->interface->ifindex;
00323   sa.sll_halen = ETH_ALEN;
00324   if (level == 1)
00325     memcpy (&sa.sll_addr, ALL_L1_ISS, ETH_ALEN);
00326   else
00327     memcpy (&sa.sll_addr, ALL_L2_ISS, ETH_ALEN);
00328 
00329   /* on a broadcast circuit */
00330   /* first we put the LLC in */
00331   sock_buff[0] = 0xFE;
00332   sock_buff[1] = 0xFE;
00333   sock_buff[2] = 0x03;
00334 
00335   /* then we copy the data */
00336   memcpy (sock_buff + LLC_LEN, circuit->snd_stream->data,
00337       stream_get_endp (circuit->snd_stream));
00338 
00339   /* now we can send this */
00340   written = sendto (circuit->fd, sock_buff,
00341             stream_get_endp(circuit->snd_stream) + LLC_LEN, 0,
00342             (struct sockaddr *) &sa, sizeof (struct sockaddr_ll));
00343 
00344   return ISIS_OK;
00345 }
