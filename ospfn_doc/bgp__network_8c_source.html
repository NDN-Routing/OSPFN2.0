00304 /* BGP try to connect to the peer.  */
00305 int
00306 bgp_connect (struct peer *peer)
00307 {
00308   unsigned int ifindex = 0;
00309 
00310   /* Make socket for the peer. */
00311   peer->fd = sockunion_socket (&peer->su);
00312   if (peer->fd < 0)
00313     return -1;
00314 
00315   /* If we can get socket for the peer, adjest TTL and make connection. */
00316   if (peer_sort (peer) == BGP_PEER_EBGP)
00317     sockopt_ttl (peer->su.sa.sa_family, peer->fd, peer->ttl);
00318 
00319   sockopt_reuseaddr (peer->fd);
00320   sockopt_reuseport (peer->fd);
00321   
00322 #ifdef IPTOS_PREC_INTERNETCONTROL
00323   if (sockunion_family (&peer->su) == AF_INET)
00324     setsockopt_ipv4_tos (peer->fd, IPTOS_PREC_INTERNETCONTROL);
00325 #endif
00326 
00327   if (peer->password)
00328     bgp_md5_set_connect (peer->fd, &peer->su, peer->password);
00329 
00330   /* Bind socket. */
00331   bgp_bind (peer);
00332 
00333   /* Update source bind. */
00334   bgp_update_source (peer);
00335 
00336 #ifdef HAVE_IPV6
00337   if (peer->ifname)
00338     ifindex = if_nametoindex (peer->ifname);
00339 #endif /* HAVE_IPV6 */
00340 
00341   if (BGP_DEBUG (events, EVENTS))
00342     plog_debug (peer->log, "%s [Event] Connect start to %s fd %d",
00343            peer->host, peer->host, peer->fd);
00344 
00345   /* Connect to the remote peer. */
00346   return sockunion_connect (peer->fd, &peer->su, htons (peer->port), ifindex);
00347 }

00374 {

00377

00380

00387   /* Want only IPV6 on ipv6 socket (not mapped addresses) */

00389     int on = 1;

00391         (void *) &on, sizeof (on));

00393 #endif
00394 
00395   if (bgpd_privs.change (ZPRIVS_RAISE) )
00396     zlog_err ("bgp_socket: could not raise privs");
00397 
00398   ret = bind (sock, sa, salen);
00399   en = errno;
00400   if (bgpd_privs.change (ZPRIVS_LOWER) )
00401     zlog_err ("bgp_bind_address: could not lower privs");
00402 
00403   if (ret < 0)
00404     {
00405       zlog_err ("bind: %s", safe_strerror (en));
00406       return ret;
00407     }
00408 
00409   ret = listen (sock, 3);
00410   if (ret < 0)
00411     {
00412       zlog_err ("listen: %s", safe_strerror (errno));
00413       return ret;
00414     }
00415 
00416   listener = XMALLOC (MTYPE_BGP_LISTENER, sizeof(*listener));
00417   listener->fd = sock;
00418   memcpy(&listener->su, sa, salen);
00419   listener->thread = thread_add_read (master, bgp_accept, listener, sock);
00420   listnode_add (bm->listen_sockets, listener);
00421 
00422   return 0;
