00103 {

00107 
00108   memset (&key, 0, sizeof (key));
00109   ospf6_lsdb_set_key (&key, &lsa->header->type, sizeof (lsa->header->type));
00110   ospf6_lsdb_set_key (&key, &lsa->header->adv_router,
00111                       sizeof (lsa->header->adv_router));

00113 
00114   current = route_node_get (lsdb->table, (struct prefix *) &key);
00115   old = current->info;
00116   current->info = lsa;
00117   ospf6_lsa_lock (lsa);
00118 
00119   if (old)
00120     {
00121       if (old->prev)
00122         old->prev->next = lsa;
00123       if (old->next)
00124         old->next->prev = lsa;

00126       lsa->prev = old->prev;
00127     }
00128   else
00129     {
00130       /* next link */
00131       nextnode = current;
00132       route_lock_node (nextnode);
00133       do {
00134         nextnode = route_next (nextnode);
00135       } while (nextnode && nextnode->info == NULL);
00136       if (nextnode == NULL)
00137         lsa->next = NULL;
00138       else
00139         {
00140           next = nextnode->info;
00141           lsa->next = next;
00142           next->prev = lsa;
00143           route_unlock_node (nextnode);
00144         }
00145 
00146       /* prev link */
00147       prevnode = current;
00148       route_lock_node (prevnode);
00149       do {
00150         prevnode = route_prev (prevnode);
00151       } while (prevnode && prevnode->info == NULL);
00152       if (prevnode == NULL)
00153         lsa->prev = NULL;
00154       else
00155         {
00156           prev = prevnode->info;
00157           lsa->prev = prev;
00158           prev->next = lsa;
00159           route_unlock_node (prevnode);
00160         }
00161 
00162       lsdb->count++;
00163     }
00164 
00165   if (old)
00166     {
00167       if (OSPF6_LSA_IS_CHANGED (old, lsa))
00168         {
00169           if (OSPF6_LSA_IS_MAXAGE (lsa))
00170             {
00171               if (lsdb->hook_remove)
00172                 {
00173                   (*lsdb->hook_remove) (old);
00174                   (*lsdb->hook_remove) (lsa);
00175                 }
00176             }
00177           else if (OSPF6_LSA_IS_MAXAGE (old))
00178             {
00179               if (lsdb->hook_add)
00180                 (*lsdb->hook_add) (lsa);
00181             }
00182           else
00183             {
00184               if (lsdb->hook_remove)
00185                 (*lsdb->hook_remove) (old);
00186               if (lsdb->hook_add)
00187                 (*lsdb->hook_add) (lsa);
00188             }
00189         }
00190     }
00191   else if (OSPF6_LSA_IS_MAXAGE (lsa))
00192     {
00193       if (lsdb->hook_remove)
00194         (*lsdb->hook_remove) (lsa);
00195     }
00196   else
00197     {
00198       if (lsdb->hook_add)
00199         (*lsdb->hook_add) (lsa);
00200     }
00201 
00202   if (old)
00203     ospf6_lsa_unlock (old);
00204 
00205   ospf6_lsdb_count_assert (lsdb);
00206 }

00288          prefix_match (&node->p, p))

00292     }

00305     return NULL;
00306

00308     {
00309       struct route_node *prev = node;
00310       struct ospf6_lsa *lsa_prev;
00311       struct ospf6_lsa *lsa_next;
00312 
00313       node = route_next (node);

00315         node = route_next (node);
00316 
00317       lsa_prev = prev->info;
00318       lsa_next = (node ? node->info : NULL);
00319       assert (lsa_prev);

00324     }
00325

00327     return NULL;
00328

00538 /* Decide new Link State ID to originate.
00539    note return value is network byte order */
00540 u_int32_t
00541 ospf6_new_ls_id (u_int16_t type, u_int32_t adv_router,
00542                  struct ospf6_lsdb *lsdb)
00543 {
00544   struct ospf6_lsa *lsa;
00545   u_int32_t id = 1;
00546 
00547   for (lsa = ospf6_lsdb_type_router_head (type, adv_router, lsdb); lsa;
00548        lsa = ospf6_lsdb_type_router_next (type, adv_router, lsa))
00549     {
00550       if (ntohl (lsa->header->id) < id)
00551         continue;
00552       if (ntohl (lsa->header->id) > id)
00553       {
00554         ospf6_lsa_unlock (lsa);
00555         break;
00556       }
00557       id++;
00558     }
00559 
00560   return ((u_int32_t) htonl (id));
00561 }
