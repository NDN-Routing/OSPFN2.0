00038 void
00039 rip_zebra_ipv4_add (struct prefix_ipv4 *p, struct in_addr *nexthop, 
00040             u_int32_t metric, u_char distance)
00041 {
00042   struct zapi_ipv4 api;
00043 
00044   if (zclient->redist[ZEBRA_ROUTE_RIP])
00045     {
00046       api.type = ZEBRA_ROUTE_RIP;
00047       api.flags = 0;
00048       api.message = 0;
00049       SET_FLAG (api.message, ZAPI_MESSAGE_NEXTHOP);
00050       api.nexthop_num = 1;
00051       api.nexthop = &nexthop;
00052       api.ifindex_num = 0;
00053       SET_FLAG (api.message, ZAPI_MESSAGE_METRIC);
00054       api.metric = metric;
00055 
00056       if (distance && distance != ZEBRA_RIP_DISTANCE_DEFAULT)
00057     {
00058       SET_FLAG (api.message, ZAPI_MESSAGE_DISTANCE);
00059       api.distance = distance;
00060     }
00061 
00062       zapi_ipv4_route (ZEBRA_IPV4_ROUTE_ADD, zclient, p, &api);
00063 
00064       rip_global_route_changes++;
00065     }
00066 }

00092 /* Zebra route add and delete treatment. */
00093 static int
00094 rip_zebra_read_ipv4 (int command, struct zclient *zclient, zebra_size_t length)
00095 {
00096   struct stream *s;
00097   struct zapi_ipv4 api;
00098   unsigned long ifindex;
00099   struct in_addr nexthop;
00100   struct prefix_ipv4 p;
00101   
00102   s = zclient->ibuf;
00103   ifindex = 0;
00104   nexthop.s_addr = 0;
00105 
00106   /* Type, flags, message. */
00107   api.type = stream_getc (s);
00108   api.flags = stream_getc (s);
00109   api.message = stream_getc (s);
00110 
00111   /* IPv4 prefix. */
00112   memset (&p, 0, sizeof (struct prefix_ipv4));
00113   p.family = AF_INET;
00114   p.prefixlen = stream_getc (s);
00115   stream_get (&p.prefix, s, PSIZE (p.prefixlen));
00116 
00117   /* Nexthop, ifindex, distance, metric. */
00118   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_NEXTHOP))
00119     {
00120       api.nexthop_num = stream_getc (s);
00121       nexthop.s_addr = stream_get_ipv4 (s);
00122     }
00123   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_IFINDEX))
00124     {
00125       api.ifindex_num = stream_getc (s);
00126       ifindex = stream_getl (s);
00127     }
00128   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_DISTANCE))
00129     api.distance = stream_getc (s);
00130   else
00131     api.distance = 255;
00132   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_METRIC))
00133     api.metric = stream_getl (s);
00134   else
00135     api.metric = 0;
00136 
00137   /* Then fetch IPv4 prefixes. */
00138   if (command == ZEBRA_IPV4_ROUTE_ADD)
00139     rip_redistribute_add (api.type, RIP_ROUTE_REDISTRIBUTE, &p, ifindex, 
00140                           &nexthop, api.metric, api.distance);
00141   else 
00142     rip_redistribute_delete (api.type, RIP_ROUTE_REDISTRIBUTE, &p, ifindex);
00143 
00144   return 0;
00145 }
