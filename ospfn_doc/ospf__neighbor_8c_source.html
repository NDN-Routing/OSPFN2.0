00046 /* Fill in the the 'key' as appropriate to retrieve the entry for nbr
00047  * from the ospf_interface's nbrs table. Indexed by interface address
00048  * for all cases except Virtual-link interfaces, where neighbours are
00049  * indexed by router-ID instead.
00050  */
00051 static void
00052 ospf_nbr_key (struct ospf_interface *oi, struct ospf_neighbor *nbr,
00053               struct prefix *key)
00054 {
00055   key->family = AF_INET;
00056   key->prefixlen = IPV4_MAX_BITLEN;
00057 
00058   /* vlinks are indexed by router-id */
00059   if (oi->type == OSPF_IFTYPE_VIRTUALLINK)
00060     key->u.prefix4 = nbr->router_id;
00061   else
00062     key->u.prefix4 = nbr->src;
00063   return;
00064 }

00147 /* Delete specified OSPF neighbor from interface. */
00148 void
00149 ospf_nbr_delete (struct ospf_neighbor *nbr)
00150 {
00151   struct ospf_interface *oi;
00152   struct route_node *rn;
00153   struct prefix p;
00154 
00155   oi = nbr->oi;
00156   
00157   /* get appropriate prefix 'key' */
00158   ospf_nbr_key (oi, nbr, &p);
00159 
00160   rn = route_node_lookup (oi->nbrs, &p);
00161   if (rn)
00162     {
00163       /* If lookup for a NBR succeeds, the leaf route_node could
00164        * only exist because there is (or was) a nbr there.
00165        * If the nbr was deleted, the leaf route_node should have
00166        * lost its last refcount too, and be deleted.
00167        * Therefore a looked-up leaf route_node in nbrs table
00168        * should never have NULL info.
00169        */
00170       assert (rn->info);
00171       
00172       if (rn->info)
00173     {
00174       rn->info = NULL;
00175       route_unlock_node (rn);
00176     }
00177       else
00178     zlog_info ("Can't find neighbor %s in the interface %s",
00179            inet_ntoa (nbr->src), IF_NAME (oi));
00180 
00181       route_unlock_node (rn);
00182     }
00183 
00184   /* Free ospf_neighbor structure. */
00185   ospf_nbr_free (nbr);
00186 }

00211 
00212   /* Initial state */
00213   oi->nbr_self->address = *oi->address;
00214   oi->nbr_self->priority = OSPF_IF_PARAM (oi, priority);

00227       case OSPF_AREA_NSSA:
00228         UNSET_FLAG (oi->nbr_self->options, OSPF_OPTION_E);
00229         SET_FLAG (oi->nbr_self->options, OSPF_OPTION_NP);
00230         break;
00231     }
00232   
00233   /* Add nbr_self to nbrs table */
00234   ospf_nbr_key (oi, oi->nbr_self, &p);
00235   
00236   rn = route_node_get (oi->nbrs, &p);
00237   if (rn->info)
00238     {
00239       /* There is already pseudo neighbor. */
00240       assert (oi->nbr_self == rn->info);
00241       route_unlock_node (rn);
00242     }
00243   else
00244     rn->info = oi->nbr_self;

00251 {
00252   struct ospf_neighbor *nbr;
00253   struct route_node *rn;
00254   int count = 0;
00255 
00256   for (rn = route_top (oi->nbrs); rn; rn = route_next (rn))
00257     if ((nbr = rn->info))
00258       if (!IPV4_ADDR_SAME (&nbr->router_id, &oi->ospf->router_id))
00259     if (state == 0 || nbr->state == state)
00260       count++;

00289 ospf_nbr_lookup_by_addr (struct route_table *nbrs,
00290              struct in_addr *addr)

00297   p.prefixlen = IPV4_MAX_BITLEN;

00299

00301   if (! rn)
00302     return NULL;

00304   /* See comment in ospf_nbr_delete */
00305   assert (rn->info);

00307   if (rn->info == NULL)

00310       return NULL;
00311     }

00337 void
00338 ospf_renegotiate_optional_capabilities (struct ospf *top)
00339 {
00340   struct listnode *node;
00341   struct ospf_interface *oi;
00342   struct route_table *nbrs;
00343   struct route_node *rn;
00344   struct ospf_neighbor *nbr;
00345 
00346   /* At first, flush self-originated LSAs from routing domain. */
00347   ospf_flush_self_originated_lsas_now (top);
00348 
00349   /* Revert all neighbor status to ExStart. */
00350   for (ALL_LIST_ELEMENTS_RO (top->oiflist, node, oi))
00351     {
00352       if ((nbrs = oi->nbrs) == NULL)
00353         continue;
00354 
00355       for (rn = route_top (nbrs); rn; rn = route_next (rn))
00356         {
00357           if ((nbr = rn->info) == NULL || nbr == oi->nbr_self)
00358             continue;
00359 
00360           if (nbr->state < NSM_ExStart)
00361             continue;
00362 
00363           if (IS_DEBUG_OSPF_EVENT)
00364             zlog_debug ("Renegotiate optional capabilities with neighbor(%s)", inet_ntoa (nbr->router_id));
00365 
00366           OSPF_NSM_EVENT_SCHEDULE (nbr, NSM_SeqNumberMismatch);
00367         }
00368     }
00369 
00370   return;
00371 }

00388   struct ospf_neighbor *nbr;
00389   
00390   nbr = ospf_nbr_new (oi);
00391   nbr->state = NSM_Down;
00392   nbr->src = p->u.prefix4;
00393   memcpy (&nbr->address, p, sizeof (struct prefix));
00394 
00395   nbr->nbr_nbma = NULL;
00396   if (oi->type == OSPF_IFTYPE_NBMA)
00397     {
00398       struct ospf_nbr_nbma *nbr_nbma;
00399       struct listnode *node;
00400 
00401       for (ALL_LIST_ELEMENTS_RO (oi->nbr_nbma, node, nbr_nbma))
00402         {
00403           if (IPV4_ADDR_SAME(&nbr_nbma->addr, &nbr->src))
00404             {
00405               nbr_nbma->nbr = nbr;
00406               nbr->nbr_nbma = nbr_nbma;
00407 
00408               if (nbr_nbma->t_poll)
00409                 OSPF_POLL_TIMER_OFF (nbr_nbma->t_poll);
00410 
00411               nbr->state_change = nbr_nbma->state_change + 1;
00412             }
00413         }
00414     }
00415       
00416   /* New nbr, save the crypto sequence number if necessary */
00417   if (ntohs (ospfh->auth_type) == OSPF_AUTH_CRYPTOGRAPHIC)
00418     nbr->crypt_seqnum = ospfh->u.crypt.crypt_seqnum;
00419   
00420   if (IS_DEBUG_OSPF_EVENT)
00421     zlog_debug ("NSM[%s:%s]: start", IF_NAME (nbr->oi),
00422                inet_ntoa (nbr->router_id));
