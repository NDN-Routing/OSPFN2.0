00081 ripng_aggregate_add (struct prefix *p)
00082 {
00083   struct route_node *top;
00084   struct route_node *rp;
00085   struct ripng_info *rinfo;
00086   struct ripng_aggregate *aggregate;
00087   struct ripng_aggregate *sub;
00088 
00089   /* Get top node for aggregation. */
00090   top = route_node_get (ripng->table, p);
00091 
00092   /* Allocate new aggregate. */
00093   aggregate = ripng_aggregate_new ();
00094   aggregate->metric = 1;
00095 
00096   top->aggregate = aggregate;
00097 
00098   /* Suppress routes match to the aggregate. */
00099   for (rp = route_lock_node (top); rp; rp = route_next_until (rp, top))
00100     {
00101       /* Suppress normal route. */
00102       if ((rinfo = rp->info) != NULL)
00103     {
00104       aggregate->count++;
00105       rinfo->suppress++;
00106     }
00107       /* Suppress aggregate route.  This may not need. */
00108       if (rp != top && (sub = rp->aggregate) != NULL)
00109     {
00110       aggregate->count++;
00111       sub->suppress++;
00112     }
00113     }
00114 
00115   return 0;
00116 }
