00055 typedef enum
00056   {
00057     /* An I/O error occurred.  The buffer should be destroyed and the
00058        file descriptor should be closed. */
00059     BUFFER_ERROR = -1,
00060 
00061     /* The data was written successfully, and the buffer is now empty
00062        (there is no pending data waiting to be flushed). */
00063     BUFFER_EMPTY = 0,
00064 
00065     /* There is pending data in the buffer waiting to be flushed.  Please
00066        try flushing the buffer when select indicates that the file descriptor
00067        is writeable. */
00068     BUFFER_PENDING = 1
00069   } buffer_status_t;

00088 /* Attempt to write enough data to the given fd to fill a window of the
