00045 /* Below is the structure linked to every
00046    route node. Note that for Network routing
00047    entries a single ospf_route is kept, while
00048    for ABRs and ASBRs (Router routing entries),
00049    we link an instance of ospf_router_route

00051 
00052    nr->info is a (struct ospf_route *) for OSPF_DESTINATION_NETWORK

00054    nr->info is a (struct ospf_router_route *) for OSPF_DESTINATION_ROUTER
00055 */

00057 struct route_standard
00058 {
00059   /* Link Sate Origin. */
00060   struct lsa_header *origin;

00062   /* Associated Area. */
00063   struct in_addr area_id;   /* The area the route belongs to */
00064 
00065   /*  Area Type */
00066   int external_routing;
00067 
00068   /* Optional Capability. */
00069   u_char options;       /* Get from LSA header. */

00073 };

00101   /* Destination ID. */     /* i.e. Link State ID. */
00102   struct in_addr id;
00103 
00104   /* Address Mask. */
00105   struct in_addr mask;      /* Only valid for networks. */
00106 
00107   /* Path Type. */
00108   u_char path_type;
00109 
00110   /* List of Paths. */
00111   struct list *paths;
00112 
00113   /* Link State Cost. */
00114   u_int32_t cost;       /* i.e. metric. */
00115 
00116   /* Route specific info. */
00117   union
00118   {
00119     struct route_standard std;
