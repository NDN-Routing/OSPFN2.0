00332 static pid_t
00333 run_background(const char *shell_cmd)
00334 {
00335   pid_t child;
00336 
00337   switch (child = fork())
00338     {
00339     case -1:
00340       zlog_err("fork failed, cannot run command [%s]: %s",
00341            shell_cmd,safe_strerror(errno));
00342       return -1;
00343     case 0:
00344       /* Child process. */
00345       /* Use separate process group so child processes can be killed easily. */
00346       if (setpgid(0,0) < 0)
00347         zlog_warn("warning: setpgid(0,0) failed: %s",safe_strerror(errno));
00348       {
00349         const char *argv[4] = { "sh", "-c", shell_cmd, NULL};
00350     execv("/bin/sh",(char *const *)argv);
00351     zlog_err("execv(/bin/sh -c '%s') failed: %s",
00352          shell_cmd,safe_strerror(errno));
00353     _exit(127);
00354       }
00355     default:
00356       /* Parent process: we will reap the child later. */
00357       zlog_err("Forked background command [pid %d]: %s",(int)child,shell_cmd);
00358       return child;
00359     }
00360 }

00364 {

00471 static int
00472 run_job(struct restart_info *restart, const char *cmdtype, const char *command,
00473     int force, int update_interval)
00474 {
00475   struct timeval delay;
00476 
00477   if (gs.loglevel > LOG_DEBUG+1)
00478     zlog_debug("attempting to %s %s",cmdtype,restart->name);
00479 
00480   if (restart->pid)
00481     {
00482       if (gs.loglevel > LOG_DEBUG+1)
00483         zlog_debug("cannot %s %s, previous pid %d still running",
00484            cmdtype,restart->name,(int)restart->pid);
00485       return -1;
00486     }
00487 
00488   /* Note: time_elapsed test must come before the force test, since we need
00489      to make sure that delay is initialized for use below in updating the
00490      restart interval. */
00491   if ((time_elapsed(&delay,&restart->time)->tv_sec < restart->interval) &&
00492       !force)
00493     {
00494       if (gs.loglevel > LOG_DEBUG+1)
00495         zlog_debug("postponing %s %s: "
00496            "elapsed time %ld < retry interval %ld",
00497            cmdtype,restart->name,(long)delay.tv_sec,restart->interval);
00498       return -1;
00499     }
00500 
00501   gettimeofday(&restart->time,NULL);
00502   restart->kills = 0;
00503   {
00504     char cmd[strlen(command)+strlen(restart->name)+1];
00505     snprintf(cmd,sizeof(cmd),command,restart->name);
00506     if ((restart->pid = run_background(cmd)) > 0)
00507       {
00508     restart->t_kill = thread_add_timer(master,restart_kill,restart,
00509                        gs.restart_timeout);
00510     restart->what = cmdtype;
00511     gs.numpids++;
00512       }
00513     else
00514       restart->pid = 0;
00515   }
00516 
00517   /* Calculate the new restart interval. */
00518   if (update_interval)
00519     {
00520       if (delay.tv_sec > 2*gs.max_restart_interval)
00521     restart->interval = gs.min_restart_interval;
00522       else if ((restart->interval *= 2) > gs.max_restart_interval)
00523     restart->interval = gs.max_restart_interval;
00524       if (gs.loglevel > LOG_DEBUG+1)
00525     zlog_debug("restart %s interval is now %ld",
00526            restart->name,restart->interval);
00527     }
00528   return restart->pid;
00529 }

00580   else if (gs.loglevel > LOG_DEBUG)
00581     zlog_debug("%s still down : %s",dmn->name,why);
00582   if (IS_UP(dmn))
00583     gs.numdown++;
00584   dmn->state = DAEMON_DOWN;
00585   if (dmn->fd >= 0)
00586     {
00587       close(dmn->fd);
00588       dmn->fd = -1;
00589     }
00590   THREAD_OFF(dmn->t_read);
00591   THREAD_OFF(dmn->t_write);
00592   THREAD_OFF(dmn->t_wakeup);
00593   if (try_connect(dmn) < 0)
00594     SET_WAKEUP_DOWN(dmn);
00595   phase_check();
00596 }

00598 static int
00599 handle_read(struct thread *t_read)
00600 {
00601   struct daemon *dmn = THREAD_ARG(t_read);
00602   static const char resp[sizeof(PING_TOKEN)+4] = PING_TOKEN "\n";
00603   char buf[sizeof(resp)+100];
00604   ssize_t rc;
00605   struct timeval delay;
00606 
00607   dmn->t_read = NULL;
00608   if ((rc = read(dmn->fd,buf,sizeof(buf))) < 0)
00609     {
00610       char why[100];
00611 
00612       if (ERRNO_IO_RETRY(errno))
00613     {
00614       /* Pretend it never happened. */
00615       SET_READ_HANDLER(dmn);
00616       return 0;
00617     }
00618       snprintf(why,sizeof(why),"unexpected read error: %s",
00619            safe_strerror(errno));
00620       daemon_down(dmn,why);
00621       return 0;
00622     }
00623   if (rc == 0)
00624     {
00625       daemon_down(dmn,"read returned EOF");
00626       return 0;
00627     }
00628   if (!dmn->echo_sent.tv_sec)
00629     {
00630       char why[sizeof(buf)+100];
00631       snprintf(why,sizeof(why),"unexpected read returns %d bytes: %.*s",
00632            (int)rc,(int)rc,buf);
00633       daemon_down(dmn,why);
00634       return 0;
00635     }
00636 
00637   /* We are expecting an echo response: is there any chance that the
00638      response would not be returned entirely in the first read?  That
00639      seems inconceivable... */
00640   if ((rc != sizeof(resp)) || memcmp(buf,resp,sizeof(resp)))
00641     {
00642       char why[100+sizeof(buf)];
00643       snprintf(why,sizeof(why),"read returned bad echo response of %d bytes "
00644                    "(expecting %u): %.*s",
00645            (int)rc,(u_int)sizeof(resp),(int)rc,buf);
00646       daemon_down(dmn,why);
00647       return 0;
00648     }
00649 
00650   time_elapsed(&delay,&dmn->echo_sent);
00651   dmn->echo_sent.tv_sec = 0;
00652   if (dmn->state == DAEMON_UNRESPONSIVE)
00653     {
00654       if (delay.tv_sec < gs.timeout)
00655     {
00656       dmn->state = DAEMON_UP;
00657       zlog_warn("%s state -> up : echo response received after %ld.%06ld "
00658             "seconds", dmn->name,delay.tv_sec,delay.tv_usec);
00659     }
00660       else
00661     zlog_warn("%s: slow echo response finally received after %ld.%06ld "
00662           "seconds", dmn->name,delay.tv_sec,delay.tv_usec);
00663     }
00664   else if (gs.loglevel > LOG_DEBUG+1)
00665     zlog_debug("%s: echo response received after %ld.%06ld seconds",
00666            dmn->name,delay.tv_sec,delay.tv_usec);
00667 
00668   SET_READ_HANDLER(dmn);
00669   if (dmn->t_wakeup)
00670     thread_cancel(dmn->t_wakeup);
00671   SET_WAKEUP_ECHO(dmn);
00672 
00673   return 0;
00674 }

00688 static int
00689 check_connect(struct thread *t_write)
00690 {
00691   struct daemon *dmn = THREAD_ARG(t_write);
00692   int sockerr;
00693   socklen_t reslen = sizeof(sockerr);
00694 
00695   dmn->t_write = NULL;

00697     {

00708            safe_strerror(sockerr));

00730 /* Making connection to protocol daemon. */
00731 static int
00732 try_connect(struct daemon *dmn)
00733 {
00734   int sock;
00735   struct sockaddr_un addr;
00736   socklen_t len;
00737 
00738   if (gs.loglevel > LOG_DEBUG+1)
00739     zlog_debug("%s: attempting to connect",dmn->name);
00740   dmn->connect_tries++;
00741 
00742   memset (&addr, 0, sizeof (struct sockaddr_un));
00743   addr.sun_family = AF_UNIX;
00744   snprintf(addr.sun_path, sizeof(addr.sun_path), "%s/%s.vty",
00745        gs.vtydir,dmn->name);
00746 #ifdef HAVE_STRUCT_SOCKADDR_UN_SUN_LEN
00747   len = addr.sun_len = SUN_LEN(&addr);
00748 #else
00749   len = sizeof (addr.sun_family) + strlen (addr.sun_path);
00750 #endif /* HAVE_STRUCT_SOCKADDR_UN_SUN_LEN */
00751 
00752   /* Quick check to see if we might succeed before we go to the trouble
00753      of creating a socket. */
00754   if (access(addr.sun_path, W_OK) < 0)
00755     {
00756       if (errno != ENOENT)
00757         zlog_err("%s: access to socket %s denied: %s",
00758         dmn->name,addr.sun_path,safe_strerror(errno));
00759       return -1;
00760     }
00761 
00762   if ((sock = socket (AF_UNIX, SOCK_STREAM, 0)) < 0)
00763     {
00764       zlog_err("%s(%s): cannot make socket: %s",
00765            __func__,addr.sun_path, safe_strerror(errno));
00766       return -1;
00767     }
00768 
00769   if (set_nonblocking(sock) < 0)
00770     {
00771       zlog_err("%s(%s): set_nonblocking(%d) failed",
00772            __func__, addr.sun_path, sock);
00773       close(sock);
00774       return -1;
00775     }
00776 
00777   if (connect (sock, (struct sockaddr *) &addr, len) < 0)
00778     {
00779       if ((errno != EINPROGRESS) && (errno != EWOULDBLOCK))
00780     {
00781       if (gs.loglevel > LOG_DEBUG)
00782         zlog_debug("%s(%s): connect failed: %s",
00783                __func__,addr.sun_path, safe_strerror(errno));
00784       close (sock);
00785       return -1;
00786     }
00787       if (gs.loglevel > LOG_DEBUG)
00788     zlog_debug("%s: connection in progress",dmn->name);
00789       dmn->state = DAEMON_CONNECTING;
00790       dmn->fd = sock;
00791       dmn->t_write = thread_add_write(master,check_connect,dmn,dmn->fd);
00792       dmn->t_wakeup = thread_add_timer(master,wakeup_connect_hanging,dmn,
00793                        gs.timeout);
00794       SET_READ_HANDLER(dmn);
00795       return 0;
00796     }
00797 
00798   dmn->fd = sock;
00799   SET_READ_HANDLER(dmn);
00800   daemon_up(dmn,"connect succeeded");
00801   return 1;
00802 }

00824 static void
00825 phase_check(void)
00826 {
00827   switch (gs.phase)
00828     {
00829     case PHASE_NONE:
00830       break;
00831     case PHASE_STOPS_PENDING:
00832       if (gs.numpids)
00833     break;
00834       zlog_info("Phased restart: all routing daemon stop jobs have completed.");
00835       set_phase(PHASE_WAITING_DOWN);
00836       /*FALLTHRU*/
00837     case PHASE_WAITING_DOWN:
00838       if (gs.numdown+IS_UP(gs.special) < gs.numdaemons)
00839         break;
00840       zlog_info("Phased restart: all routing daemons now down.");
00841       run_job(&gs.special->restart,"restart",gs.restart_command,1,1);
00842       set_phase(PHASE_ZEBRA_RESTART_PENDING);
00843       /*FALLTHRU*/
00844     case PHASE_ZEBRA_RESTART_PENDING:
00845       if (gs.special->restart.pid)
00846     break;
00847       zlog_info("Phased restart: %s restart job completed.",gs.special->name);
00848       set_phase(PHASE_WAITING_ZEBRA_UP);
00849       /*FALLTHRU*/
00850     case PHASE_WAITING_ZEBRA_UP:
00851       if (!IS_UP(gs.special))
00852         break;
00853       zlog_info("Phased restart: %s is now up.",gs.special->name);
00854       {
00855         struct daemon *dmn;
00856     for (dmn = gs.daemons; dmn; dmn = dmn->next)
00857       {
00858         if (dmn != gs.special)
00859           run_job(&dmn->restart,"start",gs.start_command,1,0);
00860       }
00861       }
00862       gs.phase = PHASE_NONE;
00863       THREAD_OFF(gs.t_phase_hanging);
00864       zlog_notice("Phased global restart has completed.");
00865       break;
00866     }
00867 }

00869 static void
00870 try_restart(struct daemon *dmn)
00871 {
00872   switch (gs.mode)
00873   {
00874   case MODE_MONITOR:
00875     return;
00876   case MODE_GLOBAL_RESTART:
00877     run_job(&gs.restart,"restart",gs.restart_command,0,1);
00878     break;
00879   case MODE_SEPARATE_RESTART:
00880     run_job(&dmn->restart,"restart",gs.restart_command,0,1);
00881     break;
00882   case MODE_PHASED_ZEBRA_RESTART:
00883     if (dmn != gs.special)
00884       {
00885         if ((gs.special->state == DAEMON_UP) && (gs.phase == PHASE_NONE))
00886       run_job(&dmn->restart,"restart",gs.restart_command,0,1);
00887     else
00888       zlog_debug("%s: postponing restart attempt because master %s daemon "
00889              "not up [%s], or phased restart in progress",
00890              dmn->name,gs.special->name,state_str[gs.special->state]);
00891     break;
00892       }
00893     /*FALLTHRU*/
00894   case MODE_PHASED_ALL_RESTART:
00895     if ((gs.phase != PHASE_NONE) || gs.numpids)
00896       {
00897     if (gs.loglevel > LOG_DEBUG+1)
00898       zlog_debug("postponing phased global restart: restart already in "
00899              "progress [%s], or outstanding child processes [%d]",
00900              phase_str[gs.phase],gs.numpids);
00901         break;
00902       }
00903     /* Is it too soon for a restart? */
00904     {
00905       struct timeval delay;
00906       if (time_elapsed(&delay,&gs.special->restart.time)->tv_sec <
00907           gs.special->restart.interval)
00908     {
00909       if (gs.loglevel > LOG_DEBUG+1)
00910         zlog_debug("postponing phased global restart: "
00911                "elapsed time %ld < retry interval %ld",
00912                (long)delay.tv_sec,gs.special->restart.interval);
00913       break;
00914     }
00915     }
00916     zlog_info("Phased restart: stopping all routing daemons.");
00917     /* First step: stop all other daemons. */
00918     for (dmn = gs.daemons; dmn; dmn = dmn->next)
00919       {
00920         if (dmn != gs.special)
00921       run_job(&dmn->restart,"stop",gs.stop_command,1,1);
00922       }
00923     set_phase(PHASE_STOPS_PENDING);
00924     break;
00925   default:
00926     zlog_err("error: unknown restart mode %d",gs.mode);
00927     break;
00928   }
00929 }

01005 /* This is an ugly hack to circumvent problems with passing command-line
01006    arguments that contain spaces.  The fix is to use a configuration file. */
01007 static char *
01008 translate_blanks(const char *cmd, const char *blankstr)
01009 {
01010   char *res;
01011   char *p;
01012   size_t bslen = strlen(blankstr);
01013 
01014   if (!(res = strdup(cmd)))
01015     {

01024     }
01025   return res;
