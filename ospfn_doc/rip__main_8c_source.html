00184 /* Main routine of ripd. */
00185 int
00186 main (int argc, char **argv)
00187 {
00188   char *p;
00189   int daemon_mode = 0;
00190   int dryrun = 0;
00191   char *progname;
00192   struct thread thread;
00193 
00194   /* Set umask before anything for security */
00195   umask (0027);
00196 
00197   /* Get program name. */
00198   progname = ((p = strrchr (argv[0], '/')) ? ++p : argv[0]);
00199 
00200   /* First of all we need logging init. */
00201   zlog_default = openzlog (progname, ZLOG_RIP,
00202                LOG_CONS|LOG_NDELAY|LOG_PID, LOG_DAEMON);
00203 
00204   /* Command line option parse. */
00205   while (1) 
00206     {
00207       int opt;
00208 
00209       opt = getopt_long (argc, argv, "df:i:hA:P:u:g:rvC", longopts, 0);
00210     
00211       if (opt == EOF)
00212     break;
00213 
00214       switch (opt) 
00215     {
00216     case 0:
00217       break;
00218     case 'd':
00219       daemon_mode = 1;
00220       break;
00221     case 'f':
00222       config_file = optarg;
00223       break;
00224     case 'A':
00225       vty_addr = optarg;
00226       break;
00227         case 'i':
00228           pid_file = optarg;
00229           break;
00230     case 'P':
00231           /* Deal with atoi() returning 0 on failure, and ripd not
00232              listening on rip port... */
00233           if (strcmp(optarg, "0") == 0) 
00234             {
00235               vty_port = 0;
00236               break;
00237             } 
00238           vty_port = atoi (optarg);
00239           if (vty_port <= 0 || vty_port > 0xffff)
00240             vty_port = RIP_VTY_PORT;
00241       break;
00242     case 'r':
00243       retain_mode = 1;
00244       break;
00245     case 'C':
00246       dryrun = 1;
00247       break;
00248     case 'u':
00249       ripd_privs.user = optarg;
00250       break;
00251     case 'g':
00252       ripd_privs.group = optarg;
00253       break;
00254     case 'v':
00255       print_version (progname);
00256       exit (0);
00257       break;
00258     case 'h':
00259       usage (progname, 0);
00260       break;
00261     default:
00262       usage (progname, 1);
00263       break;
00264     }
00265     }
00266 
00267   /* Prepare master thread. */
00268   master = thread_master_create ();
00269 
00270   /* Library initialization. */
00271   zprivs_init (&ripd_privs);
00272   signal_init (master, Q_SIGC(ripd_signals), ripd_signals);
00273   cmd_init (1);
00274   vty_init (master);
00275   memory_init ();
00276   keychain_init ();
00277 
00278   /* RIP related initialization. */
00279   rip_init ();
00280   rip_if_init ();
00281   rip_zclient_init ();
00282   rip_peer_init ();
00283 
00284   /* Sort all installed commands. */
00285   sort_node ();
00286 
00287   /* Get configuration file. */
00288   vty_read_config (config_file, config_default);
00289 
00290   /* Start execution only if not in dry-run mode */
00291   if(dryrun)
00292     return (0);
00293   
00294   /* Change to the daemon program. */
00295   if (daemon_mode && daemon (0, 0) < 0)
00296     {
00297       zlog_err("RIPd daemon failed: %s", strerror(errno));
00298       exit (1);
00299     }
00300 
00301   /* Pid file create. */
00302   pid_output (pid_file);
00303 
00304   /* Create VTY's socket */
00305   vty_serv_sock (vty_addr, vty_port, RIP_VTYSH_PATH);
00306 
00307   /* Print banner. */
00308   zlog_notice ("RIPd %s starting: vty@%d", QUAGGA_VERSION, vty_port);
00309 
00310   /* Execute each thread. */
00311   while (thread_fetch (master, &thread))
00312     thread_call (&thread);
00313 
00314   /* Not reached. */
00315   return (0);
00316 }
