00135 /* Match function return 1 if match is success else return zero. */
00136 static route_map_result_t
00137 route_match_metric (void *rule, struct prefix *prefix, 
00138             route_map_object_t type, void *object)
00139 {
00140   u_int32_t *metric;
00141   struct ripng_info *rinfo;
00142 
00143   if (type == RMAP_RIPNG)
00144     {
00145       metric = rule;
00146       rinfo = object;
00147     
00148       if (rinfo->metric == *metric)
00149     return RMAP_MATCH;
00150       else
00151     return RMAP_NOMATCH;
00152     }

00290 /* Set metric to attribute. */
00291 static route_map_result_t
00292 route_set_metric (void *rule, struct prefix *prefix, 
00293           route_map_object_t type, void *object)
00294 {
00295   if (type == RMAP_RIPNG)
00296     {
00297       struct rip_metric_modifier *mod;
00298       struct ripng_info *rinfo;
00299 
00300       mod = rule;
00301       rinfo = object;
00302 
00303       if (mod->type == metric_increment)
00304     rinfo->metric_out += mod->metric;
00305       else if (mod->type == metric_decrement)
00306     rinfo->metric_out-= mod->metric;
00307       else if (mod->type == metric_absolute)
00308     rinfo->metric_out = mod->metric;
00309 
00310       if (rinfo->metric_out < 1)
00311     rinfo->metric_out = 1;
00312       if (rinfo->metric_out > RIPNG_METRIC_INFINITY)
00313     rinfo->metric_out = RIPNG_METRIC_INFINITY;
00314 
00315       rinfo->metric_set = 1;
00316     }
00317   return RMAP_OKAY;
00318 }

00320 /* set metric compilation. */
00321 static void *
00322 route_set_metric_compile (const char *arg)
00323 {
00324   int len;
00325   const char *pnt;
00326   int type;
00327   long metric;
00328   char *endptr = NULL;
00329   struct rip_metric_modifier *mod;
00330 
00331   len = strlen (arg);
00332   pnt = arg;
00333 
00334   if (len == 0)
00335     return NULL;
00336 
00337   /* Examine first character. */
00338   if (arg[0] == '+')
00339     {
00340       type = metric_increment;
00341       pnt++;
00342     }
00343   else if (arg[0] == '-')
00344     {
00345       type = metric_decrement;
00346       pnt++;
00347     }
00348   else
00349     type = metric_absolute;
00350 
00351   /* Check beginning with digit string. */
00352   if (*pnt < '0' || *pnt > '9')
00353     return NULL;
00354 
00355   /* Convert string to integer. */
00356   metric = strtol (pnt, &endptr, 10);
00357 
00358   if (metric == LONG_MAX || *endptr != '\0')
00359     return NULL;
00360   /* Commented out by Hasso Tepper, to avoid problems in vtysh. */
00361   /* if (metric < 0 || metric > RIPNG_METRIC_INFINITY) */
00362   if (metric < 0)
00363     return NULL;
00364 
00365   mod = XMALLOC (MTYPE_ROUTE_MAP_COMPILED, 
00366          sizeof (struct rip_metric_modifier));
00367   mod->type = type;
00368   mod->metric = metric;
00369 
00370   return mod;
00371 }
