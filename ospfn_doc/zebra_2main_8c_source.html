00221   /* preserve my name */

00223

00225                LOG_CONS|LOG_NDELAY|LOG_PID, LOG_DAEMON);

00242     case 0:

00247       daemon_mode = 1;

00257       break;
00258     case 'A':
00259       vty_addr = optarg;
00260       break;
00261         case 'i':
00262           pid_file = optarg;
00263           break;
00264     case 'P':
00265       /* Deal with atoi() returning 0 on failure, and zebra not
00266          listening on zebra port... */
00267       if (strcmp(optarg, "0") == 0) 
00268         {
00269           vty_port = 0;
00270           break;
00271         }

00273       if (vty_port <= 0 || vty_port > 0xffff)

00277       retain_mode = 1;
00278       break;
00279 #ifdef HAVE_NETLINK
00280     case 's':
00281       nl_rcvbufsize = atoi (optarg);
00282       break;
00283 #endif /* HAVE_NETLINK */
00284     case 'u':
00285       zserv_privs.user = optarg;
00286       break;
00287     case 'g':
00288       zserv_privs.group = optarg;
00289       break;
00290     case 'v':

00293       break;
00294     case 'h':
00295       usage (progname, 0);
00296       break;

00300     }

00344   /* Process the configuration file. Among other configuration

00359 
00360   /* Exit when zebra is working in batch mode. */
00361   if (batch_mode)
00362     exit (0);
00363 
00364   /* Daemonize. */

00374   /* After we have successfully acquired the pidfile, we can be sure

00383     rib_sweep_route ();
00384 
00385   /* Needed for BSD routing socket. */
00386   pid = getpid ();
00387 
00388   /* This must be done only after locking pidfile (bug #403). */
00389   zebra_zserv_socket_init ();

00391   /* Make vty server socket. */
00392   vty_serv_sock (vty_addr, vty_port, ZEBRA_VTYSH_PATH);
00393 
00394   /* Print banner. */
00395   zlog_notice ("Zebra %s starting: vty@%d", QUAGGA_VERSION, vty_port);
00396 
00397   while (thread_fetch (zebrad.master, &thread))
00398     thread_call (&thread);
00399 
00400   /* Not reached... */
