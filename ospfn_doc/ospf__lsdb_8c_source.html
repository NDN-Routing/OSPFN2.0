00084 static void
00085 ospf_lsdb_delete_entry (struct ospf_lsdb *lsdb, struct route_node *rn)
00086 {
00087   struct ospf_lsa *lsa = rn->info;
00088   
00089   if (!lsa)
00090     return;
00091   
00092   assert (rn->table == lsdb->type[lsa->data->type].db);
00093   
00094   if (IS_LSA_SELF (lsa))
00095     lsdb->type[lsa->data->type].count_self--;
00096   lsdb->type[lsa->data->type].count--;
00097   lsdb->type[lsa->data->type].checksum -= ntohs(lsa->data->checksum);
00098   lsdb->total--;
00099   rn->info = NULL;
00100   route_unlock_node (rn);
00101 #ifdef MONITOR_LSDB_CHANGE
00102   if (lsdb->del_lsa_hook != NULL)
00103     (* lsdb->del_lsa_hook)(lsa);
00104 #endif /* MONITOR_LSDB_CHANGE */
00105   ospf_lsa_unlock (&lsa); /* lsdb */
00106   return;
