00069 /*
00070  * Perform a ``left rotation'' adjustment on the tree.  The given node P and
00071  * its right child C are rearranged so that the P instead becomes the left
00072  * child of C.   The left subtree of C is inherited as the new right subtree
00073  * for P.  The ordering of the keys within the tree is thus preserved.
00074  */
00075 
00076 static void rotate_left(dnode_t *upper)
00077 {
00078     dnode_t *lower, *lowleft, *upparent;
00079 
00080     lower = upper->right;
00081     upper->right = lowleft = lower->left;
00082     lowleft->parent = upper;
00083 
00084     lower->parent = upparent = upper->parent;
00085 
00086     /* don't need to check for root node here because root->parent is
00087        the sentinel nil node, and root->parent->left points back to root */
00088 
00089     if (upper == upparent->left) {
00090     upparent->left = lower;
00091     } else {
00092     assert (upper == upparent->right);
00093     upparent->right = lower;
00094     }
00095 
00096     lower->left = upper;
00097     upper->parent = lower;
00098 }

00140 /*
00141  * This procedure performs a verification that the given subtree is a binary
00142  * search tree. It performs an inorder traversal of the tree using the
00143  * dict_next() successor function, verifying that the key of each node is
00144  * strictly lower than that of its successor, if duplicates are not allowed,
00145  * or lower or equal if duplicates are allowed.  This function is used for
00146  * debugging purposes. 
00147  */
00148 
00149 static int verify_bintree(dict_t *dict)
00150 {
00151     dnode_t *first, *next;
00152 
00153     first = dict_first(dict);
00154 
00155     if (dict->dupes) {
00156     while (first && (next = dict_next(dict, first))) {
00157         if (dict->compare(first->key, next->key) > 0)
00158         return 0;
00159         first = next;
00160     }
00161     } else {
00162     while (first && (next = dict_next(dict, first))) {
00163         if (dict->compare(first->key, next->key) >= 0)
00164         return 0;
00165         first = next;
00166     }
00167     }
00168     return 1;
00169 }

00172 /*
00173  * This function recursively verifies that the given binary subtree satisfies
00174  * three of the red black properties. It checks that every red node has only
00175  * black children. It makes sure that each node is either red or black. And it
00176  * checks that every path has the same count of black nodes from root to leaf.
00177  * It returns the blackheight of the given subtree; this allows blackheights to
00178  * be computed recursively and compared for left and right siblings for
00179  * mismatches. It does not check for every nil node being black, because there
00180  * is only one sentinel nil node. The return value of this function is the
00181  * black height of the subtree rooted at the node ``root'', or zero if the
00182  * subtree is not red-black.
00183  */
00184 
00185 static unsigned int verify_redblack(dnode_t *nil, dnode_t *root)
00186 {
00187     unsigned height_left, height_right;
00188 
00189     if (root != nil) {
00190     height_left = verify_redblack(nil, root->left);
00191     height_right = verify_redblack(nil, root->right);
00192     if (height_left == 0 || height_right == 0)
00193         return 0;
00194     if (height_left != height_right)
00195         return 0;
00196     if (root->color == dnode_red) {
00197         if (root->left->color != dnode_black)
00198         return 0;
00199         if (root->right->color != dnode_black)
00200         return 0;
00201         return height_left;
00202     }
00203     if (root->color != dnode_black)
00204         return 0;
00205     return height_left + 1;
00206     } 
00207     return 1;
00208 }

00374 /*
00375  * Verify the integrity of the dictionary structure.  This is provided for
00376  * debugging purposes, and should be placed in assert statements.   Just because
00377  * this function succeeds doesn't mean that the tree is not corrupt. Certain
00378  * corruptions in the tree may simply cause undefined behavior.
00379  */ 
00380 
00381 int dict_verify(dict_t *dict)
00382 {
00383     dnode_t *nil = dict_nil(dict), *root = dict_root(dict);
00384 
00385     /* check that the sentinel node and root node are black */
00386     if (root->color != dnode_black)
00387     return 0;
00388     if (nil->color != dnode_black)
00389     return 0;
00390     if (nil->right != nil)
00391     return 0;
00392     /* nil->left is the root node; check that its parent pointer is nil */
00393     if (nil->left->parent != nil)
00394     return 0;
00395     /* perform a weak test that the tree is a binary search tree */
00396     if (!verify_bintree(dict))
00397     return 0;
00398     /* verify that the tree is a red-black tree */
00399     if (!verify_redblack(nil, root))
00400     return 0;
00401     if (verify_node_count(nil, root) != dict_count(dict))
00402     return 0;
00403     return 1;
00404 }

00431 /*
00432  * Locate a node in the dictionary having the given key.
00433  * If the node is not found, a null a pointer is returned (rather than 
00434  * a pointer that dictionary's nil sentinel node), otherwise a pointer to the
00435  * located node is returned.
00436  */
00437 
00438 dnode_t *dict_lookup(dict_t *dict, const void *key)
00439 {
00440     dnode_t *root = dict_root(dict);
00441     dnode_t *nil = dict_nil(dict);
00442     dnode_t *saved;
00443     int result;
00444 
00445     /* simple binary search adapted for trees that contain duplicate keys */
00446 
00447     while (root != nil) {
00448     result = dict->compare(key, root->key);
00449     if (result < 0)
00450         root = root->left;
00451     else if (result > 0)
00452         root = root->right;
00453     else {
00454         if (!dict->dupes) { /* no duplicates, return match      */
00455         return root;
00456         } else {        /* could be dupes, find leftmost one    */
00457         do {
00458             saved = root;
00459             root = root->left;
00460             while (root != nil && dict->compare(key, root->key))
00461             root = root->right;
00462         } while (root != nil);
00463         return saved;
00464         }
00465     }
00466     }
00467 
00468     return NULL;
00469 }

00535 /*
00536  * Insert a node into the dictionary. The node should have been
00537  * initialized with a data field. All other fields are ignored.
00538  * The behavior is undefined if the user attempts to insert into
00539  * a dictionary that is already full (for which the dict_isfull()
00540  * function returns true).
00541  */
00542 
00543 void dict_insert(dict_t *dict, dnode_t *node, const void *key)
00544 {
00545     dnode_t *where = dict_root(dict), *nil = dict_nil(dict);
00546     dnode_t *parent = nil, *uncle, *grandpa;
00547     int result = -1;
00548 
00549     node->key = key;
00550 
00551     assert (!dict_isfull(dict));
00552     assert (!dict_contains(dict, node));
00553     assert (!dnode_is_in_a_dict(node));
00554 
00555     /* basic binary tree insert */
00556 
00557     while (where != nil) {
00558     parent = where;
00559     result = dict->compare(key, where->key);
00560     /* trap attempts at duplicate key insertion unless it's explicitly allowed */
00561     assert (dict->dupes || result != 0);
00562     if (result < 0)
00563         where = where->left;
00564     else
00565         where = where->right;
00566     }
00567 
00568     assert (where == nil);
00569 
00570     if (result < 0)
00571     parent->left = node;
00572     else
00573     parent->right = node;
00574 
00575     node->parent = parent;
00576     node->left = nil;
00577     node->right = nil;
00578 
00579     dict->nodecount++;
00580 
00581     /* red black adjustments */
00582 
00583     node->color = dnode_red;
00584 
00585     while (parent->color == dnode_red) {
00586     grandpa = parent->parent;
00587     if (parent == grandpa->left) {
00588         uncle = grandpa->right;
00589         if (uncle->color == dnode_red) {    /* red parent, red uncle */
00590         parent->color = dnode_black;
00591         uncle->color = dnode_black;
00592         grandpa->color = dnode_red;
00593         node = grandpa;
00594         parent = grandpa->parent;
00595         } else {                /* red parent, black uncle */
00596             if (node == parent->right) {
00597             rotate_left(parent);
00598             parent = node;
00599             assert (grandpa == parent->parent);
00600             /* rotation between parent and child preserves grandpa */
00601         }
00602         parent->color = dnode_black;
00603         grandpa->color = dnode_red;
00604         rotate_right(grandpa);
00605         break;
00606         }
00607     } else {    /* symmetric cases: parent == parent->parent->right */
00608         uncle = grandpa->left;
00609         if (uncle->color == dnode_red) {
00610         parent->color = dnode_black;
00611         uncle->color = dnode_black;
00612         grandpa->color = dnode_red;
00613         node = grandpa;
00614         parent = grandpa->parent;
00615         } else {
00616             if (node == parent->left) {
00617             rotate_right(parent);
00618             parent = node;
00619             assert (grandpa == parent->parent);
00620         }
00621         parent->color = dnode_black;
00622         grandpa->color = dnode_red;
00623         rotate_left(grandpa);
00624         break;
00625         }
00626     }
00627     }
00628 
00629     dict_root(dict)->color = dnode_black;
00630 
00631     assert (dict_verify(dict));
00632 }

00634 /*
00635  * Delete the given node from the dictionary. If the given node does not belong
00636  * to the given dictionary, undefined behavior results.  A pointer to the
00637  * deleted node is returned.
00638  */
00639 
00640 dnode_t *dict_delete(dict_t *dict, dnode_t *delete)
00641 {
00642     dnode_t *nil = dict_nil(dict), *child, *delparent = delete->parent;
00643 
00644     /* basic deletion */
00645 
00646     assert (!dict_isempty(dict));
00647     assert (dict_contains(dict, delete));
00648 
00649     /*
00650      * If the node being deleted has two children, then we replace it with its
00651      * successor (i.e. the leftmost node in the right subtree.) By doing this,
00652      * we avoid the traditional algorithm under which the successor's key and
00653      * value *only* move to the deleted node and the successor is spliced out
00654      * from the tree. We cannot use this approach because the user may hold
00655      * pointers to the successor, or nodes may be inextricably tied to some
00656      * other structures by way of embedding, etc. So we must splice out the
00657      * node we are given, not some other node, and must not move contents from
00658      * one node to another behind the user's back.
00659      */
00660 
00661     if (delete->left != nil && delete->right != nil) {
00662     dnode_t *next = dict_next(dict, delete);
00663     dnode_t *nextparent = next->parent;
00664     dnode_color_t nextcolor = next->color;
00665 
00666     assert (next != nil);
00667     assert (next->parent != nil);
00668     assert (next->left == nil);
00669 
00670     /*
00671      * First, splice out the successor from the tree completely, by
00672      * moving up its right child into its place.
00673      */
00674 
00675     child = next->right;
00676     child->parent = nextparent;
00677 
00678     if (nextparent->left == next) {
00679         nextparent->left = child;
00680     } else {
00681         assert (nextparent->right == next);
00682         nextparent->right = child;
00683     }
00684 
00685     /*
00686      * Now that the successor has been extricated from the tree, install it
00687      * in place of the node that we want deleted.
00688      */
00689 
00690     next->parent = delparent;
00691     next->left = delete->left;
00692     next->right = delete->right;
00693     next->left->parent = next;
00694     next->right->parent = next;
00695     next->color = delete->color;
00696     delete->color = nextcolor;
00697 
00698     if (delparent->left == delete) {

00704

00715     } else {
00716         assert (delete == delparent->right);
00717         delparent->right = child;
00718     }
00719     }

00723     delete->left = NULL;

00746             assert (sister != nil);

00749             && sister->right->color == dnode_black) {
00750             sister->color = dnode_red;
00751             child = parent;
00752         } else {
00753             if (sister->right->color == dnode_black) {
00754             assert (sister->left->color == dnode_red);
00755             sister->left->color = dnode_black;
00756             sister->color = dnode_red;
00757             rotate_right(sister);
00758             sister = parent->right;
00759             assert (sister != nil);
00760             }
00761             sister->color = parent->color;
00762             sister->right->color = dnode_black;
00763             parent->color = dnode_black;
00764             rotate_left(parent);
00765             break;
00766         }
00767         } else {    /* symmetric case: child == child->parent->right */
00768         assert (child == parent->right);
00769         sister = parent->left;
00770         assert (sister != nil);
00771         if (sister->color == dnode_red) {
00772             sister->color = dnode_black;
00773             parent->color = dnode_red;
00774             rotate_right(parent);
00775             sister = parent->left;
00776             assert (sister != nil);
00777         }
00778         if (sister->right->color == dnode_black
00779             && sister->left->color == dnode_black) {
00780             sister->color = dnode_red;
00781             child = parent;
00782         } else {
00783             if (sister->left->color == dnode_black) {
00784             assert (sister->right->color == dnode_red);
00785             sister->right->color = dnode_black;
00786             sister->color = dnode_red;

00788             sister = parent->left;

00793             parent->color = dnode_black;

00802     }
00803 
00804     assert (dict_verify(dict));

01058 void dict_load_end(dict_load_t *load)
01059 {
01060     dict_t *dict = load->dictptr;
01061     dnode_t *tree[DICT_DEPTH_MAX] = { 0 };
01062     dnode_t *curr, *dictnil = dict_nil(dict), *loadnil = &load->nilnode, *next;
01063     dnode_t *complete = 0;
01064     dictcount_t fullcount = DICTCOUNT_T_MAX, nodecount = dict->nodecount;
01065     dictcount_t botrowcount;
01066     unsigned baselevel = 0, level = 0, i;
01067 
01068     assert (dnode_red == 0 && dnode_black == 1);
01069 
01070     while (fullcount >= nodecount && fullcount)
01071     fullcount >>= 1;
01072 
01073     botrowcount = nodecount - fullcount;
01074 
01075     for (curr = loadnil->left; curr != loadnil; curr = next) {
01076     next = curr->left;
01077 
01078     if (complete == NULL && botrowcount-- == 0) {
01079         assert (baselevel == 0);
01080         assert (level == 0);
01081         baselevel = level = 1;
01082         complete = tree[0];
01083 
01084         if (complete != 0) {
01085         tree[0] = 0;
01086         complete->right = dictnil;
01087         while (tree[level] != 0) {
01088             tree[level]->right = complete;
01089             complete->parent = tree[level];
01090             complete = tree[level];
01091             tree[level++] = 0;
01092         }
01093         }
01094     }
01095 
01096     if (complete == NULL) {
01097         curr->left = dictnil;
01098         curr->right = dictnil;
01099         curr->color = level % 2;
01100         complete = curr;
01101 
01102         assert (level == baselevel);
01103         while (tree[level] != 0) {
01104         tree[level]->right = complete;
01105         complete->parent = tree[level];
01106         complete = tree[level];
01107         tree[level++] = 0;
01108         }
01109     } else {
01110         curr->left = complete;
01111         curr->color = (level + 1) % 2;
01112         complete->parent = curr;
01113         tree[level] = curr;
01114         complete = 0;
01115         level = baselevel;
01116     }
01117     }
01118 
01119     if (complete == NULL)
01120     complete = dictnil;
01121 
01122     for (i = 0; i < DICT_DEPTH_MAX; i++) {
01123     if (tree[i] != 0) {
01124         tree[i]->right = complete;
01125         complete->parent = tree[i];
01126         complete = tree[i];
01127     }
01128     }
01129 
01130     dictnil->color = dnode_black;
01131     dictnil->right = dictnil;
01132     complete->parent = dictnil;
01133     complete->color = dnode_black;
01134     dict_root(dict) = complete;
01135 
01136     assert (dict_verify(dict));
01137 }

01139 void dict_merge(dict_t *dest, dict_t *source)
01140 {
01141     dict_load_t load;
01142     dnode_t *leftnode = dict_first(dest), *rightnode = dict_first(source);
01143 
01144     assert (dict_similar(dest, source));    
01145 
01146     if (source == dest)
01147     return;
01148 
01149     dest->nodecount = 0;
01150     load_begin_internal(&load, dest);
01151 
01152     for (;;) {
01153     if (leftnode != NULL && rightnode != NULL) {
01154         if (dest->compare(leftnode->key, rightnode->key) < 0)
01155         goto copyleft;
01156         else
01157         goto copyright;
01158     } else if (leftnode != NULL) {
01159         goto copyleft;
01160     } else if (rightnode != NULL) {
01161         goto copyright;
01162     } else {
01163         assert (leftnode == NULL && rightnode == NULL);
01164         break;
01165     }
01166 
01167     copyleft:
01168     {
01169         dnode_t *next = dict_next(dest, leftnode);
01170         #ifndef NDEBUG
01171         leftnode->left = NULL;  /* suppress assertion in dict_load_next */
01172         #endif
01173         dict_load_next(&load, leftnode, leftnode->key);
01174         leftnode = next;
01175         continue;
01176     }
01177     
01178     copyright:
01179     {
01180         dnode_t *next = dict_next(source, rightnode);
01181         #ifndef NDEBUG
01182         rightnode->left = NULL;
01183         #endif
01184         dict_load_next(&load, rightnode, rightnode->key);
01185         rightnode = next;
01186         continue;
01187     }
01188     }
01189 
01190     dict_clear(source);
01191     dict_load_end(&load);
01192 }
