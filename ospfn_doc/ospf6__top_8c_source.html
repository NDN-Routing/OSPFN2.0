00202 static int
00203 ospf6_maxage_remover (struct thread *thread)
00204 {
00205   struct ospf6 *o = (struct ospf6 *) THREAD_ARG (thread);
00206   struct ospf6_area *oa;
00207   struct ospf6_interface *oi;
00208   struct ospf6_neighbor *on;
00209   struct listnode *i, *j, *k;
00210 
00211   o->maxage_remover = (struct thread *) NULL;
00212 
00213   for (ALL_LIST_ELEMENTS_RO (o->area_list, i, oa))
00214     {
00215       for (ALL_LIST_ELEMENTS_RO (oa->if_list, j, oi))
00216         {
00217           for (ALL_LIST_ELEMENTS_RO (oi->neighbor_list, k, on))
00218             {
00219               if (on->state != OSPF6_NEIGHBOR_EXCHANGE &&
00220                   on->state != OSPF6_NEIGHBOR_LOADING)
00221                 continue;
00222 
00223               return 0;
00224             }
00225         }
00226     }
00227 
00228   for (ALL_LIST_ELEMENTS_RO (o->area_list, i, oa))
00229     {
00230       for (ALL_LIST_ELEMENTS_RO (oa->if_list, j, oi))
00231         OSPF6_LSDB_MAXAGE_REMOVER (oi->lsdb);
00232       
00233       OSPF6_LSDB_MAXAGE_REMOVER (oa->lsdb);
00234     }
00235   OSPF6_LSDB_MAXAGE_REMOVER (o->lsdb);
00236 
00237   return 0;
00238 }
