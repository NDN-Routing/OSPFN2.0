00022 /* This programme shows the effects of 'heavy' long-running functions
00023  * on the cooperative threading model, as demonstrated by heavy.c, and how
00024  * they can be mitigated using a background thread.
00025  *
00026  * Run it with a config file containing 'password whatever', telnet to it
00027  * (it defaults to port 4000) and enter the 'clear foo string' command.
00028  * then type whatever and observe that, unlike heavy.c, the vty interface
00029  * remains responsive.
00030  */

00051 struct work_state {
00052   struct vty *vty;
00053   char *str;
00054   int i;
00055 };

00057 static void
00058 slow_func (struct vty *vty, const char *str, const int i)
00059 {
00060   double x = 1;
00061   int j;
00062   
00063   for (j = 0; j < 300; j++)
00064     x += sin(x)*j;
00065   
00066   if ((i % ITERS_LATER) == 0)
00067     printf ("%s: %d, temporary error, save this somehow and do it later..\n",
00068             __func__, i);
00069   
00070   if ((i % ITERS_ERR) == 0)
00071     printf ("%s: hard error\n", __func__);
00072   
00073   if ((i % ITERS_PRINT) == 0)
00074     printf ("%s did %d, x = %g\n", str, i, x);
00075 }

00077 static int
00078 clear_something (struct thread *thread)
00079 {
00080   struct work_state *ws = THREAD_ARG(thread);
00081   
00082   /* this could be like iterating through 150k of route_table 
00083    * or worse, iterating through a list of peers, to bgp_stop them with
00084    * each having 150k route tables to process...
00085    */
00086   while (ws->i < ITERS_MAX)
00087     {
00088       slow_func(ws->vty, ws->str, ws->i);
00089       ws->i++;
00090       if (thread_should_yield(thread))
00091         {
00092       thread_add_background(master, clear_something, ws, 0);
00093       return 0;
00094         }
00095     }
00096   
00097   /* All done! */
00098   XFREE (MTYPE_TMP, ws->str);
00099   XFREE (MTYPE_TMP, ws);
00100   return 0;
00101 }
