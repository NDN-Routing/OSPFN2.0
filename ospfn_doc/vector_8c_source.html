00078 /* Check assigned index, and if it runs short double index pointer */
00079 void
00080 vector_ensure (vector v, unsigned int num)
00081 {
00082   if (v->alloced > num)
00083     return;
00084 
00085   v->index = XREALLOC (MTYPE_VECTOR_INDEX, 
00086                v->index, sizeof (void *) * (v->alloced * 2));
00087   memset (&v->index[v->alloced], 0, sizeof (void *) * v->alloced);
00088   v->alloced *= 2;
00089   
00090   if (v->alloced <= num)
00091     vector_ensure (v, num);
00092 }

00094 /* This function only returns next empty slot index.  It dose not mean
00095    the slot's index memory is assigned, please call vector_ensure()

00097 int
00098 vector_empty_slot (vector v)
00099 {
00100   unsigned int i;
00101 
00102   if (v->active == 0)
00103     return 0;

00105   for (i = 0; i < v->active; i++)
