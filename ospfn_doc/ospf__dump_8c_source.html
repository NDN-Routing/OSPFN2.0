00232 const char *
00233 ospf_timeval_dump (struct timeval *t, char *buf, size_t size)
00234 {
00235   /* Making formatted timer strings. */
00236 #define MINUTE_IN_SECONDS   60
00237 #define HOUR_IN_SECONDS     (60*MINUTE_IN_SECONDS)
00238 #define DAY_IN_SECONDS      (24*HOUR_IN_SECONDS)
00239 #define WEEK_IN_SECONDS     (7*DAY_IN_SECONDS)
00240   unsigned long w, d, h, m, s, ms;
00241   
00242   if (!t)
00243     return "inactive";
00244   
00245   w = d = h = m = s = ms = 0;
00246   memset (buf, 0, size);
00247   
00248   ms = t->tv_usec / 1000;
00249   
00250   if (ms >= 1000)
00251     {
00252       t->tv_sec += ms / 1000;
00253       ms %= 1000;
00254     }
00255   
00256   if (t->tv_sec > WEEK_IN_SECONDS)
00257     {
00258       w = t->tv_sec / WEEK_IN_SECONDS;
00259       t->tv_sec -= w * WEEK_IN_SECONDS;
00260     }
00261   
00262   if (t->tv_sec > DAY_IN_SECONDS)
00263     {
00264       d = t->tv_sec / DAY_IN_SECONDS;
00265       t->tv_sec -= d * DAY_IN_SECONDS;
00266     }
00267   
00268   if (t->tv_sec >= HOUR_IN_SECONDS)
00269     {
00270       h = t->tv_sec / HOUR_IN_SECONDS;
00271       t->tv_sec -= h * HOUR_IN_SECONDS;
00272     }
00273   
00274   if (t->tv_sec >= MINUTE_IN_SECONDS)
00275     {
00276       m = t->tv_sec / MINUTE_IN_SECONDS;
00277       t->tv_sec -= m * MINUTE_IN_SECONDS;
00278     }
00279   
00280   if (w > 99)
00281     snprintf (buf, size, "%ldw%1ldd", w, d);
00282   else if (w)
00283     snprintf (buf, size, "%ldw%1ldd%02ldh", w, d, h);
00284   else if (d)
00285     snprintf (buf, size, "%1ldd%02ldh%02ldm", d, h, m);
00286   else if (h)
00287     snprintf (buf, size, "%ldh%02ldm%02lds", h, m, t->tv_sec);
00288   else if (m)
00289     snprintf (buf, size, "%ldm%02lds", m, t->tv_sec);
00290   else
00291     snprintf (buf, size, "%ld.%03lds", t->tv_sec, ms);
00292   
00293   return buf;
00294 }
