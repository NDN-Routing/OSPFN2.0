00035 /* Interface looking up using infamous SIOCGIFCONF. */
00036 static int
00037 interface_list_ioctl (void)
00038 {
00039   int ret;
00040   int sock;
00041 #define IFNUM_BASE 32
00042   int ifnum;
00043   struct ifreq *ifreq;
00044   struct ifconf ifconf;
00045   struct interface *ifp;
00046   int n;
00047   int lastlen;
00048 
00049   /* Normally SIOCGIFCONF works with AF_INET socket. */
00050   sock = socket (AF_INET, SOCK_DGRAM, 0);
00051   if (sock < 0) 
00052     {
00053       zlog_warn ("Can't make AF_INET socket stream: %s", safe_strerror (errno));
00054       return -1;
00055     }
00056 
00057   /* Set initial ifreq count.  This will be double when SIOCGIFCONF
00058      fail.  Solaris has SIOCGIFNUM. */
00059 #ifdef SIOCGIFNUM
00060   ret = ioctl (sock, SIOCGIFNUM, &ifnum);
00061   if (ret < 0)
00062     ifnum = IFNUM_BASE;
00063   else
00064     ifnum++;
00065 #else
00066   ifnum = IFNUM_BASE;
00067 #endif /* SIOCGIFNUM */
00068 
00069   ifconf.ifc_buf = NULL;
00070 
00071   lastlen = 0;
00072   /* Loop until SIOCGIFCONF success. */
00073   for (;;) 
00074     {
00075       ifconf.ifc_len = sizeof (struct ifreq) * ifnum;
00076       ifconf.ifc_buf = XREALLOC(MTYPE_TMP, ifconf.ifc_buf, ifconf.ifc_len);
00077 
00078       ret = ioctl(sock, SIOCGIFCONF, &ifconf);
00079 
00080       if (ret < 0) 
00081     {
00082       zlog_warn ("SIOCGIFCONF: %s", safe_strerror(errno));
00083       goto end;
00084     }
00085       /* Repeatedly get info til buffer fails to grow. */
00086       if (ifconf.ifc_len > lastlen)
00087     {
00088           lastlen = ifconf.ifc_len;
00089       ifnum += 10;
00090       continue;
00091     }
00092       /* Success. */
00093       break;
00094     }
00095 
00096   /* Allocate interface. */
00097   ifreq = ifconf.ifc_req;
00098 
00099 #ifdef OPEN_BSD
00100   for (n = 0; n < ifconf.ifc_len; )
00101     {
00102       int size;
00103 
00104       ifreq = (struct ifreq *)((caddr_t) ifconf.ifc_req + n);
00105       ifp = if_get_by_name_len(ifreq->ifr_name,
00106                    strnlen(ifreq->ifr_name,
00107                        sizeof(ifreq->ifr_name)));
00108       if_add_update (ifp);
00109       size = ifreq->ifr_addr.sa_len;
00110       if (size < sizeof (ifreq->ifr_addr))
00111     size = sizeof (ifreq->ifr_addr);
00112       size += sizeof (ifreq->ifr_name);
00113       n += size;
00114     }
00115 #else
00116   for (n = 0; n < ifconf.ifc_len; n += sizeof(struct ifreq))
00117     {
00118       ifp = if_get_by_name_len(ifreq->ifr_name,
00119                    strnlen(ifreq->ifr_name,
00120                        sizeof(ifreq->ifr_name)));
00121       if_add_update (ifp);
00122       ifreq++;
00123     }
00124 #endif /* OPEN_BSD */
00125 
00126  end:
00127   close (sock);
00128   XFREE (MTYPE_TMP, ifconf.ifc_buf);
00129 
00130   return ret;
00131 }

00135 if_get_index (struct interface *ifp)
00136 {
00137 #if defined(HAVE_IF_NAMETOINDEX)
00138   /* Modern systems should have if_nametoindex(3). */
00139   ifp->ifindex = if_nametoindex(ifp->name);
00140 #elif defined(SIOCGIFINDEX) && !defined(HAVE_BROKEN_ALIASES)
00141   /* Fall-back for older linuxes. */
00142   int ret;
00143   struct ifreq ifreq;
00144   static int if_fake_index;
00145 
00146   ifreq_set_name (&ifreq, ifp);
00147 
00148   ret = if_ioctl (SIOCGIFINDEX, (caddr_t) &ifreq);
00149   if (ret < 0)
00150     {
00151       /* Linux 2.0.X does not have interface index. */
00152       ifp->ifindex = if_fake_index++;
00153       return ifp->ifindex;
00154     }
00155 
00156   /* OK we got interface index. */
00157 #ifdef ifr_ifindex
00158   ifp->ifindex = ifreq.ifr_ifindex;
00159 #else
00160   ifp->ifindex = ifreq.ifr_index;
00161 #endif
00162 
00163 #else
00164 /* Linux 2.2.X does not provide individual interface index 
00165    for aliases and we know it. For others issue a warning. */
00166 #if !defined(HAVE_BROKEN_ALIASES)
00167 #warning "Using if_fake_index. You may want to add appropriate"
00168 #warning "mapping from ifname to ifindex for your system..."
00169 #endif
00170   /* This branch probably won't provide usable results, but anyway... */
00171   static int if_fake_index = 1;
00172   ifp->ifindex = if_fake_index++;
00173 #endif
00174 
00175   return ifp->ifindex;
00176 }

00335    IPv4 address. */
00336 int
00337 if_get_addr (struct interface *ifp)
00338 {
00339   int ret;
00340   struct ifreq ifreq;
00341   struct sockaddr_in addr;
00342   struct sockaddr_in mask;
00343   struct sockaddr_in dest;
00344   struct in_addr *dest_pnt;
00345   u_char prefixlen;
00346   int flags = 0;
00347 
00348   /* Interface's name and address family. */
00349   strncpy (ifreq.ifr_name, ifp->name, IFNAMSIZ);
00350   ifreq.ifr_addr.sa_family = AF_INET;
00351 
00352   /* Interface's address. */
00353   ret = if_ioctl (SIOCGIFADDR, (caddr_t) &ifreq);
00354   if (ret < 0) 
00355     {
00356       if (errno != EADDRNOTAVAIL)
00357     {
00358       zlog_warn ("SIOCGIFADDR fail: %s", safe_strerror (errno));
00359       return ret;
00360     }
00361       return 0;
00362     }
00363   memcpy (&addr, &ifreq.ifr_addr, sizeof (struct sockaddr_in));
00364 
00365   /* Interface's network mask. */
00366   ret = if_ioctl (SIOCGIFNETMASK, (caddr_t) &ifreq);
00367   if (ret < 0) 
00368     {
00369       if (errno != EADDRNOTAVAIL) 
00370     {
00371       zlog_warn ("SIOCGIFNETMASK fail: %s", safe_strerror (errno));
00372       return ret;
00373     }
00374       return 0;
00375     }
00376 #ifdef ifr_netmask
00377   memcpy (&mask, &ifreq.ifr_netmask, sizeof (struct sockaddr_in));
00378 #else
00379   memcpy (&mask, &ifreq.ifr_addr, sizeof (struct sockaddr_in));
00380 #endif /* ifr_netmask */
00381   prefixlen = ip_masklen (mask.sin_addr);
00382 
00383   /* Point to point or borad cast address pointer init. */
00384   dest_pnt = NULL;
00385 
00386   ret = if_ioctl (SIOCGIFDSTADDR, (caddr_t) &ifreq);
00387   if (ret < 0) 
00388     {
00389       if (errno != EADDRNOTAVAIL) 
00390     zlog_warn ("SIOCGIFDSTADDR fail: %s", safe_strerror (errno));
00391     }
00392   else if (!IPV4_ADDR_SAME(&addr.sin_addr, &ifreq.ifr_dstaddr.sin_addr))
00393     {
00394       memcpy (&dest, &ifreq.ifr_dstaddr, sizeof (struct sockaddr_in));
00395       dest_pnt = &dest.sin_addr;
00396       flags = ZEBRA_IFA_PEER;
00397     }
00398   if (!dest_pnt)
00399     {
00400       ret = if_ioctl (SIOCGIFBRDADDR, (caddr_t) &ifreq);
00401       if (ret < 0) 
00402     {
00403       if (errno != EADDRNOTAVAIL) 
00404         zlog_warn ("SIOCGIFBRDADDR fail: %s", safe_strerror (errno));
00405     }
00406       else if (!IPV4_ADDR_SAME(&addr.sin_addr, &ifreq.ifr_broadaddr.sin_addr))
00407         {
00408       memcpy (&dest, &ifreq.ifr_broadaddr, sizeof (struct sockaddr_in));
00409       dest_pnt = &dest.sin_addr;
00410         }
00411     }
00412 
00413 
00414   /* Set address to the interface. */
00415   connected_add_ipv4 (ifp, flags, &addr.sin_addr, prefixlen, dest_pnt, NULL);
00416
