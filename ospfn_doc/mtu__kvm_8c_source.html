00030 /* get interface MTU to use kvm_read */
00031 void
00032 if_kvm_get_mtu (struct interface *ifp)
00033 {
00034   kvm_t *kvmd;
00035   struct ifnet ifnet;
00036   unsigned long ifnetaddr;
00037   int len;
00038  
00039   char ifname[IFNAMSIZ];
00040   char tname[INTERFACE_NAMSIZ + 1];
00041   char buf[_POSIX2_LINE_MAX];
00042  
00043   struct nlist nl[] = 
00044   {
00045     {"_ifnet"},
00046     {""}
00047   };
00048 
00049   ifp->mtu6 = ifp->mtu = -1;
00050   
00051   kvmd = kvm_openfiles (NULL, NULL, NULL, O_RDONLY, buf);
00052 
00053   if (kvmd == NULL) 
00054     return ;
00055   
00056   kvm_nlist(kvmd, nl);
00057  
00058   ifnetaddr = nl[0].n_value;
00059  
00060   if (kvm_read(kvmd, ifnetaddr, (char *)&ifnetaddr, sizeof ifnetaddr) < 0) 
00061     {
00062       kvm_close (kvmd);
00063       return ;
00064     }
00065  
00066   while(ifnetaddr != 0) 
00067     {
00068       if (kvm_read (kvmd, ifnetaddr, (char *)&ifnet, sizeof ifnet) < 0) 
00069     {
00070       kvm_close (kvmd);
00071       return ;
00072     }
00073 
00074       if (kvm_read (kvmd, (u_long)ifnet.if_name, ifname, IFNAMSIZ) < 0) 
00075     {
00076       kvm_close (kvmd);
00077       return ;
00078     }
00079 
00080       len = snprintf (tname, INTERFACE_NAMSIZ + 1, 
00081               "%s%d", ifname, ifnet.if_unit);
00082 
00083       if (strncmp (tname, ifp->name, len) == 0)
00084     break;
00085 
00086       ifnetaddr = (u_long)ifnet.if_next;
00087     }
00088 
00089   kvm_close (kvmd);
00090 
00091   if (ifnetaddr == 0) 
00092     {
00093       return ;
00094     }
00095 
00096   ifp->mtu6 = ifp->mtu = ifnet.if_mtu;
00097 }
