00095 /* Join to the add/leave multicast group. */
00096 static int
00097 if_group (struct interface *ifp, 
00098       int sock, 
00099       u_int32_t group, 
00100       int add_leave)
00101 {
00102   struct zebra_if *zi;
00103   struct ip_mreq m;
00104   struct prefix *p;
00105   int ret;
00106   char b1[INET_ADDRSTRLEN];
00107 
00108   zi = ifp->info;
00109 
00110   memset (&m, 0, sizeof (m));
00111   m.imr_multiaddr.s_addr = htonl (group);
00112   p = irdp_get_prefix(ifp);
00113 
00114   if(!p) {
00115         zlog_warn ("IRDP: can't get address for %s", ifp->name);
00116     return 1;
00117   }
00118 
00119   m.imr_interface = p->u.prefix4;
00120 
00121   ret = setsockopt (sock, IPPROTO_IP, add_leave,
00122             (char *) &m, sizeof (struct ip_mreq));
00123   if (ret < 0)
00124     zlog_warn ("IRDP: %s can't setsockopt %s: %s",
00125            add_leave == IP_ADD_MEMBERSHIP? "join group":"leave group", 
00126            inet_2a(group, b1),
00127            safe_strerror (errno));
00128 
00129   return ret;
00130 }

00195 static void
00196 irdp_if_start(struct interface *ifp, int multicast, int set_defaults)
00197 {
00198   struct zebra_if *zi= ifp->info;
00199   struct irdp_interface *irdp = &zi->irdp;
00200   struct listnode *node;
00201   struct connected *ifc;
00202   u_int32_t timer, seed;
00203 
00204   if (irdp->flags & IF_ACTIVE ) {
00205     zlog_warn("IRDP: Interface is already active %s", ifp->name);
00206     return;
00207   }
00208   if ((irdp_sock < 0) && ((irdp_sock = irdp_sock_init()) < 0)) {
00209     zlog_warn("IRDP: Cannot activate interface %s (cannot create "
00210           "IRDP socket)", ifp->name);
00211     return;
00212   }
00213   irdp->flags |= IF_ACTIVE;
00214 
00215   if(!multicast) 
00216     irdp->flags |= IF_BROADCAST;
00217 
00218   if_add_update(ifp);
00219 
00220   if (! (ifp->flags & IFF_UP)) {
00221     zlog_warn("IRDP: Interface is down %s", ifp->name);
00222   }
00223 
00224   /* Shall we cancel if_start if if_add_group fails? */
00225 
00226   if( multicast) {
00227     if_add_group(ifp);
00228     
00229     if (! (ifp->flags & (IFF_MULTICAST|IFF_ALLMULTI))) {
00230       zlog_warn("IRDP: Interface not multicast enabled %s", ifp->name);
00231     }
00232   }
00233 
00234   if(set_defaults) 
00235     if_set_defaults(ifp);
00236 
00237   irdp->irdp_sent = 0;
00238 
00239   /* The spec suggests this for randomness */
00240 
00241   seed = 0;
00242   if( ifp->connected)
00243     for (ALL_LIST_ELEMENTS_RO (ifp->connected, node, ifc))
00244       {
00245         seed = ifc->address->u.prefix4.s_addr;
00246         break;
00247       }
00248   
00249   srandom(seed);
00250   timer =  (random () % IRDP_DEFAULT_INTERVAL) + 1; 
00251 
00252   irdp->AdvPrefList = list_new();
00253   irdp->AdvPrefList->del =  (void (*)(void *)) Adv_free; /* Destructor */
00254 
00255 
00256   /* And this for startup. Speed limit from 1991 :-). But it's OK*/
00257 
00258   if(irdp->irdp_sent < MAX_INITIAL_ADVERTISEMENTS &&
00259      timer > MAX_INITIAL_ADVERT_INTERVAL ) 
00260       timer= MAX_INITIAL_ADVERT_INTERVAL;
00261 
00262   
00263   if(irdp->flags & IF_DEBUG_MISC)
00264     zlog_debug("IRDP: Init timer for %s set to %u", 
00265            ifp->name, 
00266            timer);
00267 
00268   irdp->t_advertise = thread_add_timer(zebrad.master, 
00269                        irdp_send_thread, 
00270                        ifp, 
00271                        timer);
00272 }

00275 irdp_if_stop(struct interface *ifp)
00276 {
00277   struct zebra_if *zi=ifp->info;
00278   struct irdp_interface *irdp=&zi->irdp;
00279   
00280   if (irdp == NULL) {
00281     zlog_warn ("Interface %s structure is NULL", ifp->name);
00282     return;
00283   }
00284 
00285   if (! (irdp->flags & IF_ACTIVE )) {
00286     zlog_warn("Interface is not active %s", ifp->name);
00287     return;
00288   }
00289 
00290   if(! (irdp->flags & IF_BROADCAST)) 
00291     if_drop_group(ifp);
00292 
00293   irdp_advert_off(ifp);
00294 
00295   list_delete(irdp->AdvPrefList);
00296   irdp->AdvPrefList=NULL;
00297 
00298   irdp->flags = 0;
00299 }

00314   irdp->flags &= ~IF_ACTIVE;
00315
