00065 /* SIGTSTP handler.  This function care user's ^Z input. */
00066 void
00067 sigtstp (int sig)
00068 {
00069   /* Execute "end" command. */
00070   vtysh_execute ("end");
00071   
00072   /* Initialize readline. */
00073   rl_initialize ();
00074   printf ("\n");
00075 
00076   /* Check jmpflag for duplicate siglongjmp(). */
00077   if (! jmpflag)
00078     return;
00079 
00080   jmpflag = 0;
00081 
00082   /* Back to main command loop. */
00083   siglongjmp (jmpbuf, 1);
00084 }

00170 /* Read a string, and return a pointer to it.  Returns NULL on EOF. */
00171 char *
00172 vtysh_rl_gets ()
00173 {
00174   HIST_ENTRY *last;
00175   /* If the buffer has already been allocated, return the memory
00176    * to the free pool. */
00177   if (line_read)
00178     {
00179       free (line_read);
00180       line_read = NULL;
00181     }
00182      
00183   /* Get a line from the user.  Change prompt according to node.  XXX. */
00184   line_read = readline (vtysh_prompt ());
00185      
00186   /* If the line has any text in it, save it on the history. But only if
00187    * last command in history isn't the same one. */
00188   if (line_read && *line_read)
00189     {
00190       using_history();
00191       last = previous_history();
00192       if (!last || strcmp (last->line, line_read) != 0) {
00193     add_history (line_read);
00194     append_history(1,history_file);
00195       }
00196     }
00197      
00198   return (line_read);
00199 }
