00145 /* convert zebras privileges to system capabilities */
00146 static pset_t *
00147 zcaps2sys (zebra_capabilities_t *zcaps, int num)
00148 {
00149   pset_t *syscaps;
00150   int i, j = 0, count = 0;
00151   
00152   if (!num)
00153     return NULL;
00154   
00155   /* first count up how many system caps we have */
00156   for (i= 0; i < num; i++)
00157     count += cap_map[zcaps[i]].num;
00158   
00159   if ( (syscaps = XCALLOC (MTYPE_PRIVS, (sizeof(pset_t) * num))) == NULL)
00160     {
00161       fprintf (stderr, "%s: could not allocate syscaps!", __func__);
00162       return NULL;
00163     }
00164   
00165   syscaps->caps = XCALLOC (MTYPE_PRIVS, (sizeof (pvalue_t) * count));
00166   
00167   if (!syscaps->caps)
00168     {
00169       fprintf (stderr, "%s: could not XCALLOC caps!", __func__);
00170       return NULL;
00171     }
00172   
00173   /* copy the capabilities over */
00174   count = 0;
00175   for (i=0; i < num; i++)
00176     for (j = 0; j < cap_map[zcaps[i]].num; j++)
00177       syscaps->caps[count++] = cap_map[zcaps[i]].system_caps[j];
00178   
00179   /* iterations above should be exact same as previous count, obviously.. */
00180   syscaps->num = count;
00181   
00182   return syscaps;
00183 }

00186 int 
00187 zprivs_change_caps (zebra_privs_ops_t op)
00188 {
00189   cap_flag_value_t cflag;
00190   
00191   /* should be no possibility of being called without valid caps */
00192   assert (zprivs_state.syscaps_p && zprivs_state.caps);
00193   if (! (zprivs_state.syscaps_p && zprivs_state.caps))
00194     exit (1);
00195     
00196   if (op == ZPRIVS_RAISE)
00197     cflag = CAP_SET;
00198   else if (op == ZPRIVS_LOWER)
00199     cflag = CAP_CLEAR;
00200   else
00201     return -1;
00202 
00203   if ( !cap_set_flag (zprivs_state.caps, CAP_EFFECTIVE,
00204                        zprivs_state.syscaps_p->num, 
00205                        zprivs_state.syscaps_p->caps, 
00206                        cflag))
00207     return cap_set_proc (zprivs_state.caps);
00208   return -1;
00209 }

00238 zprivs_caps_init (struct zebra_privs_t *zprivs)
00239 {
00240   zprivs_state.syscaps_p = zcaps2sys (zprivs->caps_p, zprivs->cap_num_p);
00241   zprivs_state.syscaps_i = zcaps2sys (zprivs->caps_i, zprivs->cap_num_i);
00242 
00243   /* Tell kernel we want caps maintained across uid changes */
00244   if ( prctl(PR_SET_KEEPCAPS, 1, 0, 0, 0) == -1 )
00245     {
00246       fprintf (stderr, "privs_init: could not set PR_SET_KEEPCAPS, %s\n",
00247                 safe_strerror (errno) );
00248       exit(1);
00249     }
00250 
00251   if ( !zprivs_state.syscaps_p )
00252     {
00253       fprintf (stderr, "privs_init: capabilities enabled, "
00254                        "but no capabilities supplied\n");
00255     }
00256 
00257   /* we have caps, we have no need to ever change back the original user */
00258   if (zprivs_state.zuid)
00259     {
00260       if ( setreuid (zprivs_state.zuid, zprivs_state.zuid) )
00261         {
00262           fprintf (stderr, "zprivs_init (cap): could not setreuid, %s\n", 
00263                      safe_strerror (errno));
00264           exit (1);
00265         }
00266     }
00267   
00268   if ( !(zprivs_state.caps = cap_init()) )
00269     {
00270       fprintf (stderr, "privs_init: failed to cap_init, %s\n", 
00271                safe_strerror (errno));
00272       exit (1);
00273     }
00274 
00275   if ( cap_clear (zprivs_state.caps) )
00276     {
00277       fprintf (stderr, "privs_init: failed to cap_clear, %s\n", 
00278                safe_strerror (errno));
00279       exit (1);
00280     }
00281   
00282   /* set permitted caps */
00283   cap_set_flag(zprivs_state.caps, CAP_PERMITTED, 
00284                zprivs_state.syscaps_p->num,
00285                zprivs_state.syscaps_p->caps,
00286                CAP_SET);
00287     
00288   /* set inheritable caps, if any */
00289   if (zprivs_state.syscaps_i && zprivs_state.syscaps_i->num)
00290     {
00291       cap_set_flag(zprivs_state.caps, CAP_INHERITABLE, 
00292                    zprivs_state.syscaps_i->num, 
00293                    zprivs_state.syscaps_i->caps, 
00294                    CAP_SET);
00295     }
00296   
00297   /* apply caps. CAP_EFFECTIVE is cleared. we'll raise the caps as 
00298    * and when, and only when, they are needed.
00299    */
00300   if ( cap_set_proc (zprivs_state.caps) ) 
00301     {
00302       fprintf (stderr, "privs_init: initial cap_set_proc failed\n");
00303       exit (1);
00304     }
00305   
00306   /* set methods for the caller to use */
00307   zprivs->change = zprivs_change_caps;
00308   zprivs->current_state = zprivs_state_caps;
00309 }

00373 /* callback exported to users to RAISE and LOWER effective privileges
00374  * from nothing to the given permitted set and back down
00375  */
00376 int 
00377 zprivs_change_caps (zebra_privs_ops_t op)
00378 {
00379   
00380   /* should be no possibility of being called without valid caps */
00381   assert (zprivs_state.syscaps_p);
00382   if (!zprivs_state.syscaps_p)
00383     {
00384       fprintf (stderr, "%s: Eek, missing caps!", __func__);
00385       exit (1);
00386     }
00387   
00388   /* to raise: copy original permitted into our working effective set
00389    * to lower: just clear the working effective set
00390    */
00391   if (op == ZPRIVS_RAISE)
00392     priv_copyset (zprivs_state.syscaps_p, zprivs_state.caps);
00393   else if (op == ZPRIVS_LOWER)
00394     priv_emptyset (zprivs_state.caps);
00395   else
00396     return -1;
00397   
00398   if (setppriv (PRIV_SET, PRIV_EFFECTIVE, zprivs_state.caps) != 0)
00399     return -1;
00400   
00401   return 0;
00402 }

00404 /* Retrieve current privilege state, is it RAISED or LOWERED? */
00405 zebra_privs_current_t 
00406 zprivs_state_caps (void)
00407 {
00408   zebra_privs_current_t result;
00409   pset_t *effective;
00410   
00411   if ( (effective = priv_allocset()) == NULL)
00412     {
00413       fprintf (stderr, "%s: failed to get priv_allocset! %s\n", __func__,
00414                safe_strerror (errno));
00415       return ZPRIVS_UNKNOWN;
00416     }
00417   
00418   if (getppriv (PRIV_EFFECTIVE, effective))
00419     {
00420       fprintf (stderr, "%s: failed to get state! %s\n", __func__,
00421                safe_strerror (errno));
00422       result = ZPRIVS_UNKNOWN;
00423     }
00424   else
00425     {
00426       if (priv_isemptyset (effective) == B_TRUE)
00427         result = ZPRIVS_LOWERED;
00428       else
00429         result = ZPRIVS_RAISED;
00430     }
00431   
00432   if (effective)
00433     priv_freeset (effective);
00434   
00435   return result;

00449   if (!zprivs_state.syscaps_p)

00461     {

00473       fprintf (stderr, "%s: couldn't get empty set!\n", __func__);
00474       exit (1);

00477   
00478   /* Hey kernel, we know about privileges! 
00479    * this isn't strictly required, use of setppriv should have same effect

00485       exit (1);

00488   /* need either valid or empty sets for both p and i.. */
00489   assert (zprivs_state.syscaps_i && zprivs_state.syscaps_p);
00490

00493    */

00498           fprintf (stderr, "%s: could not setreuid, %s\n",

00500           exit (1);

00502     }
00503

00590 void
00591 zprivs_init(struct zebra_privs_t *zprivs)
00592 {
00593   struct passwd *pwentry = NULL;
00594   struct group *grentry = NULL;
00595 
00596   if (!zprivs)
00597     {
00598       fprintf (stderr, "zprivs_init: called with NULL arg!\n");
00599       exit (1);
00600     }
00601 
00602   /* NULL privs */
00603   if (! (zprivs->user || zprivs->group 
00604          || zprivs->cap_num_p || zprivs->cap_num_i) )
00605     {
00606       zprivs->change = zprivs_change_null;
00607       zprivs->current_state = zprivs_state_null;
00608       return;
00609     }
00610 
00611   if (zprivs->user)
00612     {
00613       if ( (pwentry = getpwnam (zprivs->user)) )
00614         {
00615           zprivs_state.zuid = pwentry->pw_uid;
00616         }
00617       else
00618         {
00619           /* cant use log.h here as it depends on vty */
00620           fprintf (stderr, "privs_init: could not lookup user %s\n",
00621                    zprivs->user);
00622           exit (1);
00623         }
00624     }
00625 
00626   grentry = NULL;
00627 
00628   if (zprivs->vty_group)
00629     /* Add the vty_group to the supplementary groups so it can be chowned to */
00630     {
00631       if ( (grentry = getgrnam (zprivs->vty_group)) )
00632         {
00633           zprivs_state.vtygrp = grentry->gr_gid;
00634           if ( setgroups (1, &zprivs_state.vtygrp) )
00635             {
00636               fprintf (stderr, "privs_init: could not setgroups, %s\n",
00637                          safe_strerror (errno) );
00638               exit (1);
00639             }       
00640         }
00641       else
00642         {
00643           fprintf (stderr, "privs_init: could not lookup vty group %s\n",
00644                    zprivs->vty_group);
00645           exit (1);
00646         }
00647     }
00648   
00649   if (zprivs->group)
00650     {
00651       if ( (grentry = getgrnam (zprivs->group)) )
00652         {
00653           zprivs_state.zgid = grentry->gr_gid;
00654         }
00655       else
00656         {
00657           fprintf (stderr, "privs_init: could not lookup group %s\n",
00658                    zprivs->group);
00659           exit (1);
00660         }
00661       /* change group now, forever. uid we do later */
00662       if ( setregid (zprivs_state.zgid, zprivs_state.zgid) )
00663         {
00664           fprintf (stderr, "zprivs_init: could not setregid, %s\n",
00665                     safe_strerror (errno) );
00666           exit (1);
00667         }
00668     }
00669   
00670 #ifdef HAVE_CAPABILITIES
00671   zprivs_caps_init (zprivs);
00672 #else /* !HAVE_CAPABILITIES */
00673   /* we dont have caps. we'll need to maintain rid and saved uid
00674    * and change euid back to saved uid (who we presume has all neccessary
00675    * privileges) whenever we are asked to raise our privileges.
00676    *
00677    * This is not worth that much security wise, but all we can do.
00678    */
00679   zprivs_state.zsuid = geteuid();  
00680   if ( zprivs_state.zuid )
00681     {
00682       if ( setreuid (-1, zprivs_state.zuid) )
00683         {
00684           fprintf (stderr, "privs_init (uid): could not setreuid, %s\n", 
00685                    safe_strerror (errno));
00686           exit (1);
00687         }
00688     }
00689   
00690   zprivs->change = zprivs_change_uid;
00691   zprivs->current_state = zprivs_state_uid;
00692 #endif /* HAVE_CAPABILITIES */
00693 }
