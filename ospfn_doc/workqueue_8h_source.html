00030 /* action value, for use by item processor and item error handlers */
00031 typedef enum
00032 {
00033   WQ_SUCCESS = 0,
00034   WQ_ERROR,             /* Error, run error handler if provided */
00035   WQ_RETRY_NOW,         /* retry immediately */
00036   WQ_RETRY_LATER,       /* retry later, cease processing work queue */
00037   WQ_REQUEUE,       /* requeue item, continue processing work queue */

00039                          * Similar to WQ_RETRY_LATER, but doesn't penalise
