00233 /* timer thread to process a work queue
00234  * will reschedule itself if required,
00235  * otherwise work_queue_item_add 
00236  */
00237 int
00238 work_queue_run (struct thread *thread)
00239 {
00240   struct work_queue *wq;
00241   struct work_queue_item *item;
00242   wq_item_status ret;
00243   unsigned int cycles = 0;
00244   struct listnode *node, *nnode;
00245   char yielded = 0;
00246 
00247   wq = THREAD_ARG (thread);
00248   wq->thread = NULL;
00249 
00250   assert (wq && wq->items);
00251 
00252   /* calculate cycle granularity:
00253    * list iteration == 1 cycle
00254    * granularity == # cycles between checks whether we should yield.
00255    *
00256    * granularity should be > 0, and can increase slowly after each run to
00257    * provide some hysteris, but not past cycles.best or 2*cycles.
00258    *
00259    * Best: starts low, can only increase
00260    *
00261    * Granularity: starts at WORK_QUEUE_MIN_GRANULARITY, can be decreased 
00262    *              if we run to end of time slot, can increase otherwise 
00263    *              by a small factor.
00264    *
00265    * We could use just the average and save some work, however we want to be
00266    * able to adjust quickly to CPU pressure. Average wont shift much if
00267    * daemon has been running a long time.
00268    */
00269    if (wq->cycles.granularity == 0)
00270      wq->cycles.granularity = WORK_QUEUE_MIN_GRANULARITY;
00271 
00272   for (ALL_LIST_ELEMENTS (wq->items, node, nnode, item))
00273   {
00274     assert (item && item->data);
00275     
00276     /* dont run items which are past their allowed retries */
00277     if (item->ran > wq->spec.max_retries)
00278       {
00279         /* run error handler, if any */
00280     if (wq->spec.errorfunc)
00281       wq->spec.errorfunc (wq, item->data);
00282     work_queue_item_remove (wq, node);
00283     continue;
00284       }
00285 
00286     /* run and take care of items that want to be retried immediately */
00287     do
00288       {
00289         ret = wq->spec.workfunc (wq, item->data);
00290         item->ran++;
00291       }
00292     while ((ret == WQ_RETRY_NOW) 
00293            && (item->ran < wq->spec.max_retries));
00294 
00295     switch (ret)
00296       {
00297       case WQ_QUEUE_BLOCKED:
00298         {
00299           /* decrement item->ran again, cause this isn't an item
00300            * specific error, and fall through to WQ_RETRY_LATER
00301            */
00302           item->ran--;
00303         }
00304       case WQ_RETRY_LATER:
00305     {
00306       goto stats;
00307     }
00308       case WQ_REQUEUE:
00309     {
00310       item->ran--;
00311       work_queue_item_requeue (wq, node);
00312       break;
00313     }
00314       case WQ_RETRY_NOW:
00315         /* a RETRY_NOW that gets here has exceeded max_tries, same as ERROR */
00316       case WQ_ERROR:
00317     {
00318       if (wq->spec.errorfunc)
00319         wq->spec.errorfunc (wq, item);
00320     }
00321     /* fall through here is deliberate */
00322       case WQ_SUCCESS:
00323       default:
00324     {
00325       work_queue_item_remove (wq, node);
00326       break;
00327     }
00328       }
00329 
00330     /* completed cycle */
00331     cycles++;
00332 
00333     /* test if we should yield */
00334     if ( !(cycles % wq->cycles.granularity) 
00335         && thread_should_yield (thread))
00336       {
00337         yielded = 1;
00338         goto stats;
00339       }
00340   }
00341 
00342 stats:
00343 
00344 #define WQ_HYSTERIS_FACTOR 2
00345 
00346   /* we yielded, check whether granularity should be reduced */
00347   if (yielded && (cycles < wq->cycles.granularity))
00348     {
00349       wq->cycles.granularity = ((cycles > 0) ? cycles 
00350                                              : WORK_QUEUE_MIN_GRANULARITY);
00351     }
00352   
00353   if (cycles >= (wq->cycles.granularity))
00354     {
00355       if (cycles > wq->cycles.best)
00356         wq->cycles.best = cycles;
00357       
00358       /* along with yielded check, provides hysteris for granularity */
00359       if (cycles > (wq->cycles.granularity * WQ_HYSTERIS_FACTOR * 2))
00360         wq->cycles.granularity *= WQ_HYSTERIS_FACTOR; /* quick ramp-up */
00361       else if (cycles > (wq->cycles.granularity * WQ_HYSTERIS_FACTOR))
00362         wq->cycles.granularity += WQ_HYSTERIS_FACTOR;
00363     }
00364 #undef WQ_HYSTERIS_FACTOR
00365   
00366   wq->runs++;
00367   wq->cycles.total += cycles;
00368 
00369 #if 0
00370   printf ("%s: cycles %d, new: best %d, worst %d\n",
00371             __func__, cycles, wq->cycles.best, wq->cycles.granularity);
00372 #endif
00373   
00374   /* Is the queue done yet? If it is, call the completion callback. */
00375   if (listcount (wq->items) > 0)
00376     work_queue_schedule (wq, 0);
00377   else if (wq->spec.completion_func)
00378     wq->spec.completion_func (wq);
00379   
00380   return 0;
00381 }
