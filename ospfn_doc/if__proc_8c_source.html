00040 /* Return statistics data pointer. */
00041 static char *
00042 interface_name_cut (char *buf, char **name)
00043 {
00044   char *stat;
00045 
00046   /* Skip white space.  Line will include header spaces. */
00047   while (*buf == ' ')
00048     buf++;
00049   *name = buf;
00050 
00051   /* Cut interface name. */
00052   stat = strrchr (buf, ':');
00053   *stat++ = '\0';
00054 
00055   return stat;
00056 }

00125 /* Update interface's statistics. */
00126 void
00127 ifstat_update_proc (void)
00128 {
00129   FILE *fp;
00130   char buf[PROCBUFSIZ];
00131   int version;
00132   struct interface *ifp;
00133   char *stat;
00134   char *name;
00135 
00136   /* Open /proc/net/dev. */
00137   fp = fopen (_PATH_PROC_NET_DEV, "r");
00138   if (fp == NULL)
00139     {
00140       zlog_warn ("Can't open proc file %s: %s",
00141          _PATH_PROC_NET_DEV, safe_strerror (errno));
00142       return;
00143     }
00144 
00145   /* Drop header lines. */
00146   fgets (buf, PROCBUFSIZ, fp);
00147   fgets (buf, PROCBUFSIZ, fp);
00148 
00149   /* To detect proc format veresion, parse second line. */
00150   if (strstr (buf, "compressed"))
00151     version = 3;
00152   else if (strstr (buf, "bytes"))
00153     version = 2;
00154   else
00155     version = 1;
00156 
00157   /* Update each interface's statistics. */
00158   while (fgets (buf, PROCBUFSIZ, fp) != NULL)
00159     {
00160       stat = interface_name_cut (buf, &name);
00161       ifp = if_get_by_name (name);
00162       ifstat_dev_fields (version, stat, ifp);
00163     }
00164   fclose(fp);
00165   return;
00166 }
