00050 /* Generic signal handler 
00051  * Schedules signal event thread
00052  */
00053 static void
00054 quagga_signal_handler (int signo)
00055 {
00056   int i;
00057   struct quagga_signal_t *sig;
00058   
00059   for (i = 0; i < sigmaster.sigc; i++)
00060     {
00061       sig = &(sigmaster.signals[i]);
00062       
00063       if (sig->signal == signo)
00064         sig->caught = 1;
00065     }
00066   
00067   sigmaster.caught = 1;

00070 /* check if signals have been caught and run appropriate handlers */
00071 int
00072 quagga_sigevent_process (void)
00073 {
00074   struct quagga_signal_t *sig;
00075   int i;
00076 #ifdef SIGEVENT_BLOCK_SIGNALS
00077   /* shouldnt need to block signals, but potentially may be needed */
00078   sigset_t newmask, oldmask;
00079 
00080   /*
00081    * Block most signals, but be careful not to defer SIGTRAP because
00082    * doing so breaks gdb, at least on NetBSD 2.0.  Avoid asking to
00083    * block SIGKILL, just because we shouldn't be able to do so.
00084    */
00085   sigfillset (&newmask);
00086   sigdelset (&newmask, SIGTRAP);
00087   sigdelset (&newmask, SIGKILL);
00088    
00089   if ( (sigprocmask (SIG_BLOCK, &newmask, &oldmask)) < 0)
00090     {
00091       zlog_err ("quagga_signal_timer: couldnt block signals!");
00092       return -1;
00093     }
00094 #endif /* SIGEVENT_BLOCK_SIGNALS */
00095 
00096   if (sigmaster.caught > 0)
00097     {
00098       sigmaster.caught = 0;
00099       /* must not read or set sigmaster.caught after here,
00100        * race condition with per-sig caught flags if one does
00101        */
00102       
00103       for (i = 0; i < sigmaster.sigc; i++)
00104         {
00105           sig = &(sigmaster.signals[i]);
00106 
00107           if (sig->caught > 0)
00108             {
00109               sig->caught = 0;
00110               sig->handler ();
00111             }
00112         }
00113     }
00114 
00115 #ifdef SIGEVENT_BLOCK_SIGNALS
00116   if ( sigprocmask (SIG_UNBLOCK, &oldmask, NULL) < 0 );
00117     return -1;
00118 #endif /* SIGEVENT_BLOCK_SIGNALS */
00119 
00120   return 0;
00121 }

00141 static int
00142 signal_set (int signo)
00143 {
00144   int ret;
00145   struct sigaction sig;
00146   struct sigaction osig;
00147 
00148   sig.sa_handler = &quagga_signal_handler;
00149   sigfillset (&sig.sa_mask);
00150   sig.sa_flags = 0;
00151   if (signo == SIGALRM) {
00152 #ifdef SA_INTERRUPT
00153       sig.sa_flags |= SA_INTERRUPT; /* SunOS */
00154 #endif
00155   } else {
00156 #ifdef SA_RESTART
00157       sig.sa_flags |= SA_RESTART;
00158 #endif /* SA_RESTART */
00159   }
00160 
00161   ret = sigaction (signo, &sig, &osig);
00162   if (ret < 0) 
00163     return ret;
00164   else
00165     return 0;

00264 #ifdef SA_SIGINFO
00265             , siginfo_t *info, void *context
00266 #endif
00267            );
00268   } sigmap[] = {
00269     { core_signals, sizeof(core_signals)/sizeof(core_signals[0]), core_handler},
00270     { exit_signals, sizeof(exit_signals)/sizeof(exit_signals[0]), exit_handler},
00271     { ignore_signals, sizeof(ignore_signals)/sizeof(ignore_signals[0]), NULL},
00272   };
00273   u_int i;
00274 
00275   for (i = 0; i < sizeof(sigmap)/sizeof(sigmap[0]); i++)
00276     {
00277       u_int j;
00278 
00279       for (j = 0; j < sigmap[i].nsigs; j++)
00280         {
00281       struct sigaction oact;
00282       if ((sigaction(sigmap[i].sigs[j],NULL,&oact) == 0) &&
00283           (oact.sa_handler == SIG_DFL))
00284         {
00285           struct sigaction act;
00286           sigfillset (&act.sa_mask);
00287           if (sigmap[i].handler == NULL)
00288             {
00289           act.sa_handler = SIG_IGN;
00290           act.sa_flags = 0;
00291             }
00292           else
00293             {
00294 #ifdef SA_SIGINFO
00295           /* Request extra arguments to signal handler. */
00296           act.sa_sigaction = sigmap[i].handler;
00297           act.sa_flags = SA_SIGINFO;
00298 #else
00299           act.sa_handler = sigmap[i].handler;
00300           act.sa_flags = 0;
00301 #endif
00302             }
00303           if (sigaction(sigmap[i].sigs[j],&act,NULL) < 0)
00304             zlog_warn("Unable to set signal handler for signal %d: %s",
00305               sigmap[i].sigs[j],safe_strerror(errno));
00306 
00307         }
00308         }
00309     }
