00312              struct route_node *top)
00313 {
00314   struct prefix_ipv4 p;
00315   struct route_table *table;
00316   struct route_node *rn;
00317   struct rib *match;
00318   struct nexthop *newhop;
00319 
00320   if (nexthop->type == NEXTHOP_TYPE_IPV4)
00321     nexthop->ifindex = 0;
00322 
00323   if (set)
00324     UNSET_FLAG (nexthop->flags, NEXTHOP_FLAG_RECURSIVE);
00325 
00326   /* Make lookup prefix. */
00327   memset (&p, 0, sizeof (struct prefix_ipv4));
00328   p.family = AF_INET;
00329   p.prefixlen = IPV4_MAX_PREFIXLEN;
00330   p.prefix = nexthop->gate.ipv4;
00331 
00332   /* Lookup table.  */
00333   table = vrf_table (AFI_IP, SAFI_UNICAST, 0);
00334   if (! table)
00335     return 0;
00336 
00337   rn = route_node_match (table, (struct prefix *) &p);
00338   while (rn)
00339     {
00340       route_unlock_node (rn);
00341       
00342       /* If lookup self prefix return immediately. */
00343       if (rn == top)
00344     return 0;
00345 
00346       /* Pick up selected route. */
00347       for (match = rn->info; match; match = match->next)
00348     {
00349       if (CHECK_FLAG (match->status, RIB_ENTRY_REMOVED))
00350         continue;
00351       if (CHECK_FLAG (match->flags, ZEBRA_FLAG_SELECTED))
00352         break;
00353     }
00354 
00355       /* If there is no selected route or matched route is EGP, go up
00356          tree. */
00357       if (! match 
00358       || match->type == ZEBRA_ROUTE_BGP)
00359     {
00360       do {
00361         rn = rn->parent;
00362       } while (rn && rn->info == NULL);
00363       if (rn)
00364         route_lock_node (rn);
00365     }
00366       else
00367     {
00368       if (match->type == ZEBRA_ROUTE_CONNECT)
00369         {
00370           /* Directly point connected route. */
00371           newhop = match->nexthop;
00372           if (newhop && nexthop->type == NEXTHOP_TYPE_IPV4)
00373         nexthop->ifindex = newhop->ifindex;
00374

00872 /* Iterate over all nexthops of the given RIB entry and refresh their
00873  * ACTIVE flag. rib->nexthop_active_num is updated accordingly. If any
00874  * nexthop is found to toggle the ACTIVE flag, the whole rib structure
00875  * is flagged with ZEBRA_FLAG_CHANGED. The 4th 'set' argument is
00876  * transparently passed to nexthop_active_check().
00877  *
00878  * Return value is the new number of active nexthops.
00879  */

00886

00971 /* Core function for processing routing information base. */
00972 static void
00973 rib_process (struct route_node *rn)
00974 {
00975   struct rib *rib;
00976   struct rib *next;
00977   struct rib *fib = NULL;
00978   struct rib *select = NULL;
00979   struct rib *del = NULL;
00980   int installed = 0;
00981   struct nexthop *nexthop = NULL;
00982   char buf[INET6_ADDRSTRLEN];
00983   
00984   assert (rn);
00985   
00986   if (IS_ZEBRA_DEBUG_RIB || IS_ZEBRA_DEBUG_RIB_Q)
00987     inet_ntop (rn->p.family, &rn->p.u.prefix, buf, INET6_ADDRSTRLEN);
00988 
00989   for (rib = rn->info; rib; rib = next)
00990     {
00991       /* The next pointer is saved, because current pointer
00992        * may be passed to rib_unlink() in the middle of iteration.
00993        */
00994       next = rib->next;
00995       
00996       /* Currently installed rib. */
00997       if (CHECK_FLAG (rib->flags, ZEBRA_FLAG_SELECTED))
00998         {
00999           assert (fib == NULL);
01000           fib = rib;
01001         }
01002       
01003       /* Unlock removed routes, so they'll be freed, bar the FIB entry,
01004        * which we need to do do further work with below.
01005        */
01006       if (CHECK_FLAG (rib->status, RIB_ENTRY_REMOVED))
01007         {
01008           if (rib != fib)
01009             {
01010               if (IS_ZEBRA_DEBUG_RIB)
01011                 zlog_debug ("%s: %s/%d: rn %p, removing rib %p", __func__,
01012                   buf, rn->p.prefixlen, rn, rib);
01013                 rib_unlink (rn, rib);
01014             }
01015           else
01016             del = rib;
01017           
01018           continue;
01019         }
01020       
01021       /* Skip unreachable nexthop. */
01022       if (! nexthop_active_update (rn, rib, 0))
01023         continue;
01024 
01025       /* Infinit distance. */
01026       if (rib->distance == DISTANCE_INFINITY)
01027         continue;
01028 
01029       /* Newly selected rib, the common case. */
01030       if (!select)
01031         {
01032           select = rib;
01033           continue;
01034         }
01035       
01036       /* filter route selection in following order:
01037        * - connected beats other types
01038        * - lower distance beats higher
01039        * - lower metric beats higher for equal distance
01040        * - last, hence oldest, route wins tie break.
01041        */
01042       
01043       /* Connected routes. Pick the last connected
01044        * route of the set of lowest metric connected routes.
01045        */
01046       if (rib->type == ZEBRA_ROUTE_CONNECT)
01047         {
01048           if (select->type != ZEBRA_ROUTE_CONNECT
01049               || rib->metric <= select->metric)
01050             select = rib;
01051           continue;
01052         }
01053       else if (select->type == ZEBRA_ROUTE_CONNECT)
01054         continue;
01055       
01056       /* higher distance loses */
01057       if (rib->distance > select->distance)
01058         continue;
01059       
01060       /* lower wins */
01061       if (rib->distance < select->distance)
01062         {
01063           select = rib;
01064           continue;
01065         }
01066       
01067       /* metric tie-breaks equal distance */
01068       if (rib->metric <= select->metric)
01069         select = rib;
01070     } /* for (rib = rn->info; rib; rib = next) */
01071 
01072   /* After the cycle is finished, the following pointers will be set:
01073    * select --- the winner RIB entry, if any was found, otherwise NULL
01074    * fib    --- the SELECTED RIB entry, if any, otherwise NULL
01075    * del    --- equal to fib, if fib is queued for deletion, NULL otherwise
01076    * rib    --- NULL
01077    */
01078 
01079   /* Same RIB entry is selected. Update FIB and finish. */
01080   if (select && select == fib)
01081     {
01082       if (IS_ZEBRA_DEBUG_RIB)
01083         zlog_debug ("%s: %s/%d: Updating existing route, select %p, fib %p",
01084                      __func__, buf, rn->p.prefixlen, select, fib);
01085       if (CHECK_FLAG (select->flags, ZEBRA_FLAG_CHANGED))
01086         {
01087           redistribute_delete (&rn->p, select);
01088           if (! RIB_SYSTEM_ROUTE (select))
01089             rib_uninstall_kernel (rn, select);
01090 
01091           /* Set real nexthop. */
01092           nexthop_active_update (rn, select, 1);
01093   
01094           if (! RIB_SYSTEM_ROUTE (select))
01095             rib_install_kernel (rn, select);
01096           redistribute_add (&rn->p, select);
01097         }
01098       else if (! RIB_SYSTEM_ROUTE (select))
01099         {
01100           /* Housekeeping code to deal with 
01101              race conditions in kernel with linux
01102              netlink reporting interface up before IPv4 or IPv6 protocol
01103              is ready to add routes.
01104              This makes sure the routes are IN the kernel.
01105            */
01106 
01107           for (nexthop = select->nexthop; nexthop; nexthop = nexthop->next)
01108             if (CHECK_FLAG (nexthop->flags, NEXTHOP_FLAG_FIB))
01109             {
01110               installed = 1;
01111               break;
01112             }
01113           if (! installed) 
01114             rib_install_kernel (rn, select);
01115         }
01116       goto end;
01117     }
01118 
01119   /* At this point we either haven't found the best RIB entry or it is
01120    * different from what we currently intend to flag with SELECTED. In both
01121    * cases, if a RIB block is present in FIB, it should be withdrawn.
01122    */
01123   if (fib)
01124     {
01125       if (IS_ZEBRA_DEBUG_RIB)
01126         zlog_debug ("%s: %s/%d: Removing existing route, fib %p", __func__,
01127           buf, rn->p.prefixlen, fib);
01128       redistribute_delete (&rn->p, fib);
01129       if (! RIB_SYSTEM_ROUTE (fib))
01130     rib_uninstall_kernel (rn, fib);
01131       UNSET_FLAG (fib->flags, ZEBRA_FLAG_SELECTED);
01132 
01133       /* Set real nexthop. */
01134       nexthop_active_update (rn, fib, 1);
01135     }
01136 
01137   /* Regardless of some RIB entry being SELECTED or not before, now we can
01138    * tell, that if a new winner exists, FIB is still not updated with this
01139    * data, but ready to be.
01140    */
01141   if (select)
01142     {
01143       if (IS_ZEBRA_DEBUG_RIB)
01144         zlog_debug ("%s: %s/%d: Adding route, select %p", __func__, buf,
01145           rn->p.prefixlen, select);
01146       /* Set real nexthop. */
01147       nexthop_active_update (rn, select, 1);
01148 
01149       if (! RIB_SYSTEM_ROUTE (select))
01150         rib_install_kernel (rn, select);
01151       SET_FLAG (select->flags, ZEBRA_FLAG_SELECTED);
01152       redistribute_add (&rn->p, select);
01153     }
01154 
01155   /* FIB route was removed, should be deleted */
01156   if (del)
01157     {
01158       if (IS_ZEBRA_DEBUG_RIB)
01159         zlog_debug ("%s: %s/%d: Deleting fib %p, rn %p", __func__, buf,
01160           rn->p.prefixlen, del, rn);
01161       rib_unlink (rn, del);
01162     }
01163 
01164 end:
01165   if (IS_ZEBRA_DEBUG_RIB_Q)
01166     zlog_debug ("%s: %s/%d: rn %p dequeued", __func__, buf, rn->p.prefixlen, rn);
01167 }

01341 /* RIB updates are processed via a queue of pointers to route_nodes.
01342  *
01343  * The queue length is bounded by the maximal size of the routing table,
01344  * as a route_node will not be requeued, if already queued.
01345  *
01346  * RIBs are submitted via rib_addnode or rib_delnode which set minimal
01347  * state, or static_install_ipv{4,6} (when an existing RIB is updated)
01348  * and then submit route_node to queue for best-path selection later.
01349  * Order of add/delete state changes are preserved for any given RIB.
01350  *
01351  * Deleted RIBs are reaped during best-path selection.
01352  *
01353  * rib_addnode
01354  * |-> rib_link or unset RIB_ENTRY_REMOVE        |->Update kernel with
01355  *       |-------->|                             |  best RIB, if required
01356  *                 |                             |
01357  * static_install->|->rib_addqueue...... -> rib_process
01358  *                 |                             |
01359  *       |-------->|                             |-> rib_unlink
01360  * |-> set RIB_ENTRY_REMOVE                           |
01361  * rib_delnode                                  (RIB freed)
01362  *
01363  *
01364  * Queueing state for a route_node is kept in the head RIB entry, this
01365  * state must be preserved as and when the head RIB entry of a
01366  * route_node is changed by rib_unlink / rib_link. A small complication,
01367  * but saves having to allocate a dedicated object for this.
01368  * 
01369  * Refcounting (aka "locking" throughout the GNU Zebra and Quagga code):
01370  *
01371  * - route_nodes: refcounted by:
01372  *   - RIBs attached to route_node:
01373  *     - managed by: rib_link/unlink
01374  *   - route_node processing queue
01375  *     - managed by: rib_addqueue, rib_process.
01376  *
01377  */
01378  
01379 /* Add RIB to head of the route node. */
01380 static void
01381 rib_link (struct route_node *rn, struct rib *rib)
01382 {
01383   struct rib *head;
01384   char buf[INET6_ADDRSTRLEN];
01385   
01386   assert (rib && rn);
01387   
01388   route_lock_node (rn); /* rn route table reference */
01389 
01390   if (IS_ZEBRA_DEBUG_RIB)
01391   {
01392     inet_ntop (rn->p.family, &rn->p.u.prefix, buf, INET6_ADDRSTRLEN);
01393     zlog_debug ("%s: %s/%d: rn %p, rib %p", __func__,
01394       buf, rn->p.prefixlen, rn, rib);
01395   }
01396 
01397   head = rn->info;
01398   if (head)
01399     {
01400       if (IS_ZEBRA_DEBUG_RIB)
01401         zlog_debug ("%s: %s/%d: new head, rn_status copied over", __func__,
01402           buf, rn->p.prefixlen);
01403       head->prev = rib;
01404       /* Transfer the rn status flags to the new head RIB */
01405       rib->rn_status = head->rn_status;
01406     }
01407   rib->next = head;
01408   rn->info = rib;
01409   rib_queue_add (&zebrad, rn);
01410 }

01412 static void
01413 rib_addnode (struct route_node *rn, struct rib *rib)
01414 {
01415   /* RIB node has been un-removed before route-node is processed. 
01416    * route_node must hence already be on the queue for processing.. 
01417    */
01418   if (CHECK_FLAG (rib->status, RIB_ENTRY_REMOVED))
01419     {
01420       if (IS_ZEBRA_DEBUG_RIB)

01425                     __func__, buf, rn->p.prefixlen, rn, rib);

01433 static void
01434 rib_unlink (struct route_node *rn, struct rib *rib)
01435 {
01436   struct nexthop *nexthop, *next;
01437   char buf[INET6_ADDRSTRLEN];
01438 
01439   assert (rn && rib);
01440 
01441   if (IS_ZEBRA_DEBUG_RIB)
01442   {
01443     inet_ntop (rn->p.family, &rn->p.u.prefix, buf, INET6_ADDRSTRLEN);
01444     zlog_debug ("%s: %s/%d: rn %p, rib %p",
01445                 __func__, buf, rn->p.prefixlen, rn, rib);
01446   }
01447 
01448   if (rib->next)
01449     rib->next->prev = rib->prev;
01450 
01451   if (rib->prev)
01452     rib->prev->next = rib->next;
01453   else
01454     {
01455       rn->info = rib->next;
01456       
01457       if (rn->info)
01458         {
01459           if (IS_ZEBRA_DEBUG_RIB)
01460             zlog_debug ("%s: %s/%d: rn %p, rib %p, new head copy",
01461                         __func__, buf, rn->p.prefixlen, rn, rib);
01462           rib->next->rn_status = rib->rn_status;
01463         }
01464     }
01465 
01466   /* free RIB and nexthops */
01467   for (nexthop = rib->nexthop; nexthop; nexthop = next)
01468     {
01469       next = nexthop->next;
01470       nexthop_free (nexthop);
01471     }
01472   XFREE (MTYPE_RIB, rib);
01473 
01474   route_unlock_node (rn); /* rn route table reference */
01475 }

01491 int
01492 rib_add_ipv4 (int type, int flags, struct prefix_ipv4 *p, 
01493           struct in_addr *gate, struct in_addr *src,
01494           unsigned int ifindex, u_int32_t vrf_id,
01495           u_int32_t metric, u_char distance)
01496 {
01497   struct rib *rib;
01498   struct rib *same = NULL;
01499   struct route_table *table;
01500   struct route_node *rn;
01501   struct nexthop *nexthop;
01502 
01503   /* Lookup table.  */
01504   table = vrf_table (AFI_IP, SAFI_UNICAST, 0);
01505   if (! table)
01506     return 0;
01507 
01508   /* Make it sure prefixlen is applied to the prefix. */
01509   apply_mask_ipv4 (p);
01510 
01511   /* Set default distance by route type. */
01512   if (distance == 0)
01513     {
01514       distance = route_info[type].distance;
01515 
01516       /* iBGP distance is 200. */
01517       if (type == ZEBRA_ROUTE_BGP && CHECK_FLAG (flags, ZEBRA_FLAG_IBGP))
01518     distance = 200;
01519     }
01520 
01521   /* Lookup route node.*/
01522   rn = route_node_get (table, (struct prefix *) p);
01523 
01524   /* If same type of route are installed, treat it as a implicit
01525      withdraw. */
01526   for (rib = rn->info; rib; rib = rib->next)
01527     {
01528       if (CHECK_FLAG (rib->status, RIB_ENTRY_REMOVED))
01529         continue;
01530       
01531       if (rib->type != type)
01532     continue;
01533       if (rib->type != ZEBRA_ROUTE_CONNECT)
01534         {
01535           same = rib;
01536           break;
01537         }
01538       /* Duplicate connected route comes in. */
01539       else if ((nexthop = rib->nexthop) &&
01540            nexthop->type == NEXTHOP_TYPE_IFINDEX &&
01541            nexthop->ifindex == ifindex &&
01542            !CHECK_FLAG (rib->status, RIB_ENTRY_REMOVED))
01543     {
01544       rib->refcnt++;
01545       return 0 ;
01546     }
01547     }
01548 
01549   /* Allocate new rib structure. */
01550   rib = XCALLOC (MTYPE_RIB, sizeof (struct rib));
01551   rib->type = type;
01552   rib->distance = distance;
01553   rib->flags = flags;
01554   rib->metric = metric;
01555   rib->table = vrf_id;
01556   rib->nexthop_num = 0;
01557   rib->uptime = time (NULL);
01558 
01559   /* Nexthop settings. */
01560   if (gate)
01561     {
01562       if (ifindex)
01563     nexthop_ipv4_ifindex_add (rib, gate, src, ifindex);
01564       else
01565     nexthop_ipv4_add (rib, gate, src);
01566     }
01567   else
01568     nexthop_ifindex_add (rib, ifindex);
01569 
01570   /* If this route is kernel route, set FIB flag to the route. */
01571   if (type == ZEBRA_ROUTE_KERNEL || type == ZEBRA_ROUTE_CONNECT)
01572     for (nexthop = rib->nexthop; nexthop; nexthop = nexthop->next)
01573       SET_FLAG (nexthop->flags, NEXTHOP_FLAG_FIB);
01574 
01575   /* Link new rib to node.*/
01576   if (IS_ZEBRA_DEBUG_RIB)
01577     zlog_debug ("%s: calling rib_addnode (%p, %p)", __func__, rn, rib);
01578   rib_addnode (rn, rib);
01579   
01580   /* Free implicit route.*/
01581   if (same)
01582   {
01583     if (IS_ZEBRA_DEBUG_RIB)
01584       zlog_debug ("%s: calling rib_delnode (%p, %p)", __func__, rn, rib);
01585     rib_delnode (rn, same);
01586   }
01587   
01588   route_unlock_node (rn);
01589   return 0;
01590 }

01697 /* Check if requested address assignment will fail due to another
01698  * route being installed by zebra in FIB already. Take necessary
01699  * actions, if needed: remove such a route from FIB and deSELECT
01700  * corresponding RIB entry. Then put affected RN into RIBQ head.
01701  */
01702 void rib_lookup_and_pushup (struct prefix_ipv4 * p)
01703 {
01704   struct route_table *table;
01705   struct route_node *rn;
01706   struct rib *rib;
01707   unsigned changed = 0;
01708 
01709   if (NULL == (table = vrf_table (AFI_IP, SAFI_UNICAST, 0)))
01710   {
01711     zlog_err ("%s: vrf_table() returned NULL", __func__);
01712     return;
01713   }
01714 
01715   /* No matches would be the simplest case. */
01716   if (NULL == (rn = route_node_lookup (table, (struct prefix *) p)))
01717     return;
01718 
01719   /* Unlock node. */
01720   route_unlock_node (rn);
01721 
01722   /* Check all RIB entries. In case any changes have to be done, requeue
01723    * the RN into RIBQ head. If the routing message about the new connected
01724    * route (generated by the IP address we are going to assign very soon)
01725    * comes before the RIBQ is processed, the new RIB entry will join
01726    * RIBQ record already on head. This is necessary for proper revalidation
01727    * of the rest of the RIB.
01728    */
01729   for (rib = rn->info; rib; rib = rib->next)
01730   {
01731     if (CHECK_FLAG (rib->flags, ZEBRA_FLAG_SELECTED) &&
01732       ! RIB_SYSTEM_ROUTE (rib))
01733     {
01734       changed = 1;
01735       if (IS_ZEBRA_DEBUG_RIB)
01736       {
01737         char buf[INET_ADDRSTRLEN];
01738         inet_ntop (rn->p.family, &p->prefix, buf, INET_ADDRSTRLEN);
01739         zlog_debug ("%s: freeing way for connected prefix %s/%d", __func__, buf, p->prefixlen);
01740         rib_dump (__func__, (struct prefix_ipv4 *)&rn->p, rib);
01741       }
01742       rib_uninstall (rn, rib);
01743     }
01744   }
01745   if (changed)
01746     rib_queue_add (&zebrad, rn);
01747 }

02101 static_add_ipv4 (struct prefix *p, struct in_addr *gate, const char *ifname,
02102          u_char flags, u_char distance, u_int32_t vrf_id)
02103 {
02104   u_char type = 0;
02105   struct route_node *rn;
02106   struct static_ipv4 *si;
02107   struct static_ipv4 *pp;
02108   struct static_ipv4 *cp;
02109   struct static_ipv4 *update = NULL;
02110   struct route_table *stable;
02111 
02112   /* Lookup table.  */
02113   stable = vrf_static_table (AFI_IP, SAFI_UNICAST, vrf_id);
02114   if (! stable)
02115     return -1;
02116   
02117   /* Lookup static route prefix. */
02118   rn = route_node_get (stable, p);
02119 
02120   /* Make flags. */
02121   if (gate)
02122     type = STATIC_IPV4_GATEWAY;
02123   else if (ifname)
02124     type = STATIC_IPV4_IFNAME;
02125   else
02126     type = STATIC_IPV4_BLACKHOLE;
02127 
02128   /* Do nothing if there is a same static route.  */
02129   for (si = rn->info; si; si = si->next)
02130     {
02131       if (type == si->type
02132       && (! gate || IPV4_ADDR_SAME (gate, &si->gate.ipv4))
02133       && (! ifname || strcmp (ifname, si->gate.ifname) == 0))
02134     {
02135       if (distance == si->distance)
02136         {
02137           route_unlock_node (rn);
02138           return 0;
02139         }
02140       else
02141         update = si;
02142     }
02143     }
02144 
02145   /* Distance changed.  */
02146   if (update)
02147     static_delete_ipv4 (p, gate, ifname, update->distance, vrf_id);
02148 
02149   /* Make new static route structure. */
02150   si = XCALLOC (MTYPE_STATIC_IPV4, sizeof (struct static_ipv4));
02151 
02152   si->type = type;
02153   si->distance = distance;
02154   si->flags = flags;
02155 
02156   if (gate)
02157     si->gate.ipv4 = *gate;
02158   if (ifname)
02159     si->gate.ifname = XSTRDUP (0, ifname);
02160 
02161   /* Add new static route information to the tree with sort by
02162      distance value and gateway address. */
02163   for (pp = NULL, cp = rn->info; cp; pp = cp, cp = cp->next)
02164     {
02165       if (si->distance < cp->distance)
02166     break;
02167       if (si->distance > cp->distance)
02168     continue;
02169       if (si->type == STATIC_IPV4_GATEWAY && cp->type == STATIC_IPV4_GATEWAY)
02170     {
02171       if (ntohl (si->gate.ipv4.s_addr) < ntohl (cp->gate.ipv4.s_addr))
02172         break;
02173       if (ntohl (si->gate.ipv4.s_addr) > ntohl (cp->gate.ipv4.s_addr))
02174         continue;
02175     }
02176     }
02177 
02178   /* Make linked list. */
02179   if (pp)
02180     pp->next = si;
02181   else
02182     rn->info = si;
02183   if (cp)
02184     cp->prev = si;
02185   si->prev = pp;
02186   si->next = cp;
02187 
02188   /* Install into rib. */
02189   static_install_ipv4 (p, si);
02190 
02191   return 1;
02192 }

02652 int
02653 static_add_ipv6 (struct prefix *p, u_char type, struct in6_addr *gate,
02654          const char *ifname, u_char flags, u_char distance,
02655          u_int32_t vrf_id)
02656 {
02657   struct route_node *rn;
02658   struct static_ipv6 *si;
02659   struct static_ipv6 *pp;
02660   struct static_ipv6 *cp;
02661   struct route_table *stable;
02662 
02663   /* Lookup table.  */
02664   stable = vrf_static_table (AFI_IP6, SAFI_UNICAST, vrf_id);
02665   if (! stable)
02666     return -1;
02667     
02668   if (!gate &&
02669       (type == STATIC_IPV6_GATEWAY || type == STATIC_IPV6_GATEWAY_IFNAME))
02670     return -1;
02671   
02672   if (!ifname && 
02673       (type == STATIC_IPV6_GATEWAY_IFNAME || type == STATIC_IPV6_IFNAME))
02674     return -1;
02675 
02676   /* Lookup static route prefix. */
02677   rn = route_node_get (stable, p);
02678 
02679   /* Do nothing if there is a same static route.  */
02680   for (si = rn->info; si; si = si->next)
02681     {
02682       if (distance == si->distance 
02683       && type == si->type
02684       && (! gate || IPV6_ADDR_SAME (gate, &si->ipv6))
02685       && (! ifname || strcmp (ifname, si->ifname) == 0))
02686     {
02687       route_unlock_node (rn);
02688       return 0;
02689     }
02690     }
02691 
02692   /* Make new static route structure. */
02693   si = XCALLOC (MTYPE_STATIC_IPV6, sizeof (struct static_ipv6));
02694 
02695   si->type = type;
02696   si->distance = distance;
02697   si->flags = flags;
02698 
02699   switch (type)
02700     {
02701     case STATIC_IPV6_GATEWAY:
02702       si->ipv6 = *gate;
02703       break;
02704     case STATIC_IPV6_IFNAME:
02705       si->ifname = XSTRDUP (0, ifname);
02706       break;
02707     case STATIC_IPV6_GATEWAY_IFNAME:
02708       si->ipv6 = *gate;
02709       si->ifname = XSTRDUP (0, ifname);
02710       break;
02711     }
02712 
02713   /* Add new static route information to the tree with sort by
02714      distance value and gateway address. */
02715   for (pp = NULL, cp = rn->info; cp; pp = cp, cp = cp->next)
02716     {
02717       if (si->distance < cp->distance)
02718     break;
02719       if (si->distance > cp->distance)
02720     continue;
02721     }
02722 
02723   /* Make linked list. */
02724   if (pp)
02725     pp->next = si;
02726   else
02727     rn->info = si;
02728   if (cp)
02729     cp->prev = si;
02730   si->prev = pp;
02731   si->next = cp;
02732 
02733   /* Install into rib. */
02734   static_install_ipv6 (p, si);
02735 
02736   return 1;
02737 }
