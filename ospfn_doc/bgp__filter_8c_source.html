00192 /* Insert new AS list to list of as_list.  Each as_list is sorted by
00193    the name. */
00194 static struct as_list *
00195 as_list_insert (const char *name)
00196 {
00197   size_t i;
00198   long number;
00199   struct as_list *aslist;
00200   struct as_list *point;
00201   struct as_list_list *list;
00202 
00203   /* Allocate new access_list and copy given name. */
00204   aslist = as_list_new ();
00205   aslist->name = strdup (name);
00206   assert (aslist->name);
00207 
00208   /* If name is made by all digit character.  We treat it as
00209      number. */
00210   for (number = 0, i = 0; i < strlen (name); i++)
00211     {
00212       if (isdigit ((int) name[i]))
00213     number = (number * 10) + (name[i] - '0');
00214       else
00215     break;
00216     }
00217 
00218   /* In case of name is all digit character */
00219   if (i == strlen (name))
00220     {
00221       aslist->type = ACCESS_TYPE_NUMBER;
00222 
00223       /* Set access_list to number list. */
00224       list = &as_list_master.num;
00225 
00226       for (point = list->head; point; point = point->next)
00227     if (atol (point->name) >= number)
00228       break;
00229     }
00230   else
00231     {
00232       aslist->type = ACCESS_TYPE_STRING;
00233 
00234       /* Set access_list to string list. */
00235       list = &as_list_master.str;
00236   
00237       /* Set point to insertion point. */
00238       for (point = list->head; point; point = point->next)
00239     if (strcmp (point->name, name) >= 0)
00240       break;
00241     }
00242 
00243   /* In case of this is the first element of master. */
00244   if (list->head == NULL)
00245     {
00246       list->head = list->tail = aslist;
00247       return aslist;
00248     }
00249 
00250   /* In case of insertion is made at the tail of access_list. */
00251   if (point == NULL)
00252     {
00253       aslist->prev = list->tail;
00254       list->tail->next = aslist;
00255       list->tail = aslist;
00256       return aslist;
00257     }
00258 
00259   /* In case of insertion is made at the head of access_list. */
00260   if (point == list->head)
00261     {
00262       aslist->next = list->head;
00263       list->head->prev = aslist;
00264       list->head = aslist;
00265       return aslist;
00266     }
00267 
00268   /* Insertion is made at middle of the access_list. */
00269   aslist->next = point;
00270   aslist->prev = point->prev;
00271 
00272   if (point->prev)
00273     point->prev->next = aslist;
00274   point->prev = aslist;
00275 
00276   return aslist;
00277 }
