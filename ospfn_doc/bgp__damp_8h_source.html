00065 /* Specified parameter set configuration. */
00066 struct bgp_damp_config
00067 {
00068   /* Value over which routes suppressed.  */
00069   unsigned int suppress_value;
00070 
00071   /* Value below which suppressed routes reused.  */
00072   unsigned int reuse_limit;
00073 
00074   /* Max time a route can be suppressed.  */
00075   time_t max_suppress_time;      
00076 
00077   /* Time during which accumulated penalty reduces by half.  */
00078   time_t half_life;
00079 
00080   /* Non-configurable parameters but fixed at implementation time.
00081    * To change this values, init_bgp_damp() should be modified.
00082    */
00083   time_t tmax;           /* Max time previous instability retained */
00084   unsigned int reuse_list_size;  /* Number of reuse lists */
00085   unsigned int reuse_index_size; /* Size of reuse index array */
00086 
00087   /* Non-configurable parameters.  Most of these are calculated from
00088    * the configurable parameters above.
00089    */
00090   unsigned int ceiling;         /* Max value a penalty can attain */
00091   unsigned int decay_rate_per_tick; /* Calculated from half-life */
00092   unsigned int decay_array_size; /* Calculated using config parameters */
00093   double scale_factor;
00094   unsigned int reuse_scale_factor; 
00095          
00096   /* Decay array per-set based. */ 
00097   double *decay_array;  
00098 
00099   /* Reuse index array per-set based. */ 
00100   int *reuse_index;
00101 
00102   /* Reuse list array per-set based. */  
00103   struct bgp_damp_info **reuse_list;
00104   int reuse_offset;
00105         
00106   /* All dampening information which is not on reuse list.  */
00107   struct bgp_damp_info *no_reuse_list;
00108 
00109   /* Reuse timer thread per-set base. */
00110   struct thread* t_reuse;
00111 };
