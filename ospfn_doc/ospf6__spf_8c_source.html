00046 static int
00047 ospf6_vertex_cmp (void *a, void *b)
00048 {
00049   struct ospf6_vertex *va = (struct ospf6_vertex *) a;
00050   struct ospf6_vertex *vb = (struct ospf6_vertex *) b;
00051 
00052   /* ascending order */
00053   return (va->cost - vb->cost);
00054 }

00056 static int
00057 ospf6_vertex_id_cmp (void *a, void *b)
00058 {
00059   struct ospf6_vertex *va = (struct ospf6_vertex *) a;
00060   struct ospf6_vertex *vb = (struct ospf6_vertex *) b;
00061   int ret = 0;
00062 
00063   ret = ntohl (ospf6_linkstate_prefix_adv_router (&va->vertex_id)) -
00064         ntohl (ospf6_linkstate_prefix_adv_router (&vb->vertex_id));
00065   if (ret)
00066     return ret;
00067 
00068   ret = ntohl (ospf6_linkstate_prefix_id (&va->vertex_id)) -
00069         ntohl (ospf6_linkstate_prefix_id (&vb->vertex_id));
00070   return ret;
00071 }

00073 static struct ospf6_vertex *
00074 ospf6_vertex_create (struct ospf6_lsa *lsa)
00075 {
00076   struct ospf6_vertex *v;
00077   int i;
00078 
00079   v = (struct ospf6_vertex *)
00080     XMALLOC (MTYPE_OSPF6_VERTEX, sizeof (struct ospf6_vertex));
00081 
00082   /* type */
00083   if (ntohs (lsa->header->type) == OSPF6_LSTYPE_ROUTER)
00084     v->type = OSPF6_VERTEX_TYPE_ROUTER;
00085   else if (ntohs (lsa->header->type) == OSPF6_LSTYPE_NETWORK)
00086     v->type = OSPF6_VERTEX_TYPE_NETWORK;
00087   else
00088     assert (0);
00089 
00090   /* vertex_id */
00091   ospf6_linkstate_prefix (lsa->header->adv_router, lsa->header->id,
00092                           &v->vertex_id);
00093 
00094   /* name */
00095   ospf6_linkstate_prefix2str (&v->vertex_id, v->name, sizeof (v->name));
00096 
00097   /* Associated LSA */
00098   v->lsa = lsa;
00099 
00100   /* capability bits + options */
00101   v->capability = *(u_char *)(OSPF6_LSA_HEADER_END (lsa->header));
00102   v->options[0] = *(u_char *)(OSPF6_LSA_HEADER_END (lsa->header) + 1);
00103   v->options[1] = *(u_char *)(OSPF6_LSA_HEADER_END (lsa->header) + 2);
00104   v->options[2] = *(u_char *)(OSPF6_LSA_HEADER_END (lsa->header) + 3);
00105 
00106   for (i = 0; i < OSPF6_MULTI_PATH_LIMIT; i++)
00107     ospf6_nexthop_clear (&v->nexthop[i]);
00108 
00109   v->parent = NULL;
00110   v->child_list = list_new ();
00111   v->child_list->cmp = ospf6_vertex_id_cmp;
00112 
00113   return v;
00114 }

00169 static char *
00170 ospf6_lsdesc_backlink (struct ospf6_lsa *lsa,
00171                        caddr_t lsdesc, struct ospf6_vertex *v)
00172 {
00173   caddr_t backlink, found = NULL;
00174   int size;
00175 
00176   size = (OSPF6_LSA_IS_TYPE (ROUTER, lsa) ?
00177           sizeof (struct ospf6_router_lsdesc) :
00178           sizeof (struct ospf6_network_lsdesc));
00179   for (backlink = OSPF6_LSA_HEADER_END (lsa->header) + 4;
00180        backlink + size <= OSPF6_LSA_END (lsa->header); backlink += size)
00181     {
00182       assert (! (OSPF6_LSA_IS_TYPE (NETWORK, lsa) &&
00183                  VERTEX_IS_TYPE (NETWORK, v)));
00184 
00185       if (OSPF6_LSA_IS_TYPE (NETWORK, lsa) &&
00186           NETWORK_LSDESC_GET_NBR_ROUTERID (backlink)
00187             == v->lsa->header->adv_router)
00188         found = backlink;
00189       else if (VERTEX_IS_TYPE (NETWORK, v) &&
00190           ROUTER_LSDESC_IS_TYPE (TRANSIT_NETWORK, backlink) &&
00191           ROUTER_LSDESC_GET_NBR_ROUTERID (backlink)
00192             == v->lsa->header->adv_router &&
00193           ROUTER_LSDESC_GET_NBR_IFID (backlink)
00194             == ntohl (v->lsa->header->id))
00195         found = backlink;
00196       else
00197         {
00198           if (! ROUTER_LSDESC_IS_TYPE (POINTTOPOINT, backlink) ||
00199               ! ROUTER_LSDESC_IS_TYPE (POINTTOPOINT, lsdesc))
00200             continue;
00201           if (ROUTER_LSDESC_GET_NBR_IFID (backlink) !=
00202               ROUTER_LSDESC_GET_IFID (lsdesc) ||
00203               ROUTER_LSDESC_GET_NBR_IFID (lsdesc) !=
00204               ROUTER_LSDESC_GET_IFID (backlink))
00205             continue;
00206           if (ROUTER_LSDESC_GET_NBR_ROUTERID (backlink) !=
00207               v->lsa->header->adv_router ||
00208               ROUTER_LSDESC_GET_NBR_ROUTERID (lsdesc) !=
00209               lsa->header->adv_router)
00210             continue;
00211           found = backlink;
00212         }
00213     }
00214 
00215   if (IS_OSPF6_DEBUG_SPF (PROCESS))
00216     zlog_debug ("  Backlink %s", (found ? "OK" : "FAIL"));
00217 
00218   return found;
00219 }

00221 static void
00222 ospf6_nexthop_calc (struct ospf6_vertex *w, struct ospf6_vertex *v,
00223                     caddr_t lsdesc)
00224 {
00225   int i, ifindex;
00226   struct ospf6_interface *oi;
00227   u_int16_t type;
00228   u_int32_t adv_router;
00229   struct ospf6_lsa *lsa;
00230   struct ospf6_link_lsa *link_lsa;
00231   char buf[64];
00232 
00233   assert (VERTEX_IS_TYPE (ROUTER, w));
00234   ifindex = (VERTEX_IS_TYPE (NETWORK, v) ? v->nexthop[0].ifindex :
00235              ROUTER_LSDESC_GET_IFID (lsdesc));
00236   oi = ospf6_interface_lookup_by_ifindex (ifindex);
00237   if (oi == NULL)
00238     {
00239       if (IS_OSPF6_DEBUG_SPF (PROCESS))
00240         zlog_debug ("Can't find interface in SPF: ifindex %d", ifindex);
00241       return;
00242     }
00243 
00244   type = htons (OSPF6_LSTYPE_LINK);
00245   adv_router = (VERTEX_IS_TYPE (NETWORK, v) ?
00246                 NETWORK_LSDESC_GET_NBR_ROUTERID (lsdesc) :
00247                 ROUTER_LSDESC_GET_NBR_ROUTERID (lsdesc));
00248 
00249   i = 0;
00250   for (lsa = ospf6_lsdb_type_router_head (type, adv_router, oi->lsdb); lsa;
00251        lsa = ospf6_lsdb_type_router_next (type, adv_router, lsa))
00252     {
00253       if (VERTEX_IS_TYPE (ROUTER, v) &&
00254           htonl (ROUTER_LSDESC_GET_NBR_IFID (lsdesc)) != lsa->header->id)
00255         continue;
00256 
00257       link_lsa = (struct ospf6_link_lsa *) OSPF6_LSA_HEADER_END (lsa->header);
00258       if (IS_OSPF6_DEBUG_SPF (PROCESS))
00259         {
00260           inet_ntop (AF_INET6, &link_lsa->linklocal_addr, buf, sizeof (buf));
00261           zlog_debug ("  nexthop %s from %s", buf, lsa->name);
00262         }
00263 
00264       if (i < OSPF6_MULTI_PATH_LIMIT)
00265         {
00266           memcpy (&w->nexthop[i].address, &link_lsa->linklocal_addr,
00267                   sizeof (struct in6_addr));
00268           w->nexthop[i].ifindex = ifindex;
00269           i++;
00270         }
00271     }
00272 
00273   if (i == 0 && IS_OSPF6_DEBUG_SPF (PROCESS))
00274     zlog_debug ("No nexthop for %s found", w->name);
00275 }

00277 static int
00278 ospf6_spf_install (struct ospf6_vertex *v,
00279                    struct ospf6_route_table *result_table)
00280 {
00281   struct ospf6_route *route;
00282   int i, j;
00283   struct ospf6_vertex *prev, *w;
00284   struct listnode *node, *nnode;
00285 
00286   if (IS_OSPF6_DEBUG_SPF (PROCESS))
00287     zlog_debug ("SPF install %s hops %d cost %d",
00288         v->name, v->hops, v->cost);
00289 
00290   route = ospf6_route_lookup (&v->vertex_id, result_table);
00291   if (route && route->path.cost < v->cost)
00292     {
00293       if (IS_OSPF6_DEBUG_SPF (PROCESS))
00294         zlog_debug ("  already installed with lower cost (%d), ignore",
00295             route->path.cost);
00296       ospf6_vertex_delete (v);
00297       return -1;
00298     }
00299   else if (route && route->path.cost == v->cost)
00300     {
00301       if (IS_OSPF6_DEBUG_SPF (PROCESS))
00302         zlog_debug ("  another path found, merge");
00303 
00304       for (i = 0; ospf6_nexthop_is_set (&v->nexthop[i]) &&
00305            i < OSPF6_MULTI_PATH_LIMIT; i++)
00306         {
00307           for (j = 0; j < OSPF6_MULTI_PATH_LIMIT; j++)
00308             {
00309               if (ospf6_nexthop_is_set (&route->nexthop[j]))
00310                 {
00311                   if (ospf6_nexthop_is_same (&route->nexthop[j],
00312                                              &v->nexthop[i]))
00313                     break;
00314                   else
00315                     continue;
00316                 }
00317               ospf6_nexthop_copy (&route->nexthop[j], &v->nexthop[i]);
00318               break;
00319             }
00320         }
00321 
00322       prev = (struct ospf6_vertex *) route->route_option;
00323       if (prev->hops > v->hops)
00324         {
00325           for (ALL_LIST_ELEMENTS (prev->child_list, node, nnode, w))
00326             {
00327               assert (w->parent == prev);
00328               w->parent = v;
00329               listnode_add_sort (v->child_list, w);
00330             }
00331           listnode_delete (prev->parent->child_list, prev);
00332           listnode_add_sort (v->parent->child_list, v);
00333 
00334           ospf6_vertex_delete (prev);
00335           route->route_option = v;
00336         }
00337       else
00338         ospf6_vertex_delete (v);
00339 
00340       return -1;
00341     }
00342 
00343   /* There should be no case where candidate being installed (variable
00344      "v") is closer than the one in the SPF tree (variable "route").
00345      In the case something has gone wrong with the behavior of
00346      Priority-Queue. */
00347 
00348   /* the case where the route exists already is handled and returned
00349      up to here. */
00350   assert (route == NULL);
00351 
00352   route = ospf6_route_create ();
00353   memcpy (&route->prefix, &v->vertex_id, sizeof (struct prefix));
00354   route->type = OSPF6_DEST_TYPE_LINKSTATE;
00355   route->path.type = OSPF6_PATH_TYPE_INTRA;
00356   route->path.origin.type = v->lsa->header->type;
00357   route->path.origin.id = v->lsa->header->id;
00358   route->path.origin.adv_router = v->lsa->header->adv_router;
00359   route->path.metric_type = 1;
00360   route->path.cost = v->cost;
00361   route->path.cost_e2 = v->hops;
00362   route->path.router_bits = v->capability;
00363   route->path.options[0] = v->options[0];
00364   route->path.options[1] = v->options[1];
00365   route->path.options[2] = v->options[2];
00366 
00367   for (i = 0; ospf6_nexthop_is_set (&v->nexthop[i]) &&
00368        i < OSPF6_MULTI_PATH_LIMIT; i++)
00369     ospf6_nexthop_copy (&route->nexthop[i], &v->nexthop[i]);
00370 
00371   if (v->parent)
00372     listnode_add_sort (v->parent->child_list, v);
00373   route->route_option = v;
00374 
00375   ospf6_route_add (route, result_table);
00376   return 0;
00377 }

00393 /* RFC2328 16.1.  Calculating the shortest-path tree for an area */
00394 /* RFC2740 3.8.1.  Calculating the shortest path tree for an area */
00395 void
00396 ospf6_spf_calculation (u_int32_t router_id,
00397                        struct ospf6_route_table *result_table,
00398                        struct ospf6_area *oa)
00399 {
00400   struct pqueue *candidate_list;
00401   struct ospf6_vertex *root, *v, *w;
00402   int i;
00403   int size;
00404   caddr_t lsdesc;
00405   struct ospf6_lsa *lsa;
00406 
00407   /* initialize */
00408   candidate_list = pqueue_create ();
00409   candidate_list->cmp = ospf6_vertex_cmp;
00410 
00411   ospf6_spf_table_finish (result_table);
00412 
00413   /* Install the calculating router itself as the root of the SPF tree */
00414   /* construct root vertex */
00415   lsa = ospf6_lsdb_lookup (htons (OSPF6_LSTYPE_ROUTER), htonl (0),
00416                            router_id, oa->lsdb);
00417   if (lsa == NULL)
00418     return;
00419   root = ospf6_vertex_create (lsa);
00420   root->area = oa;
00421   root->cost = 0;
00422   root->hops = 0;
00423   root->nexthop[0].ifindex = 0; /* loopbak I/F is better ... */
00424   inet_pton (AF_INET6, "::1", &root->nexthop[0].address);
00425 
00426   /* Actually insert root to the candidate-list as the only candidate */
00427   pqueue_enqueue (root, candidate_list);
00428 
00429   /* Iterate until candidate-list becomes empty */
00430   while (candidate_list->size)
00431     {
00432       /* get closest candidate from priority queue */
00433       v = pqueue_dequeue (candidate_list);
00434 
00435       /* installing may result in merging or rejecting of the vertex */
00436       if (ospf6_spf_install (v, result_table) < 0)
00437         continue;
00438 
00439       /* For each LS description in the just-added vertex V's LSA */
00440       size = (VERTEX_IS_TYPE (ROUTER, v) ?
00441               sizeof (struct ospf6_router_lsdesc) :
00442               sizeof (struct ospf6_network_lsdesc));
00443       for (lsdesc = OSPF6_LSA_HEADER_END (v->lsa->header) + 4;
00444            lsdesc + size <= OSPF6_LSA_END (v->lsa->header); lsdesc += size)
00445         {
00446           lsa = ospf6_lsdesc_lsa (lsdesc, v);
00447           if (lsa == NULL)
00448             continue;
00449 
00450           if (! ospf6_lsdesc_backlink (lsa, lsdesc, v))
00451             continue;
00452 
00453           w = ospf6_vertex_create (lsa);
00454           w->area = oa;
00455           w->parent = v;
00456           if (VERTEX_IS_TYPE (ROUTER, v))
00457             {
00458               w->cost = v->cost + ROUTER_LSDESC_GET_METRIC (lsdesc);
00459               w->hops = v->hops + (VERTEX_IS_TYPE (NETWORK, w) ? 0 : 1);
00460             }
00461           else /* NETWORK */
00462             {
00463               w->cost = v->cost;
00464               w->hops = v->hops + 1;
00465             }
00466 
00467           /* nexthop calculation */
00468           if (w->hops == 0)
00469             w->nexthop[0].ifindex = ROUTER_LSDESC_GET_IFID (lsdesc);
00470           else if (w->hops == 1 && v->hops == 0)
00471             ospf6_nexthop_calc (w, v, lsdesc);
00472           else
00473             {
00474               for (i = 0; ospf6_nexthop_is_set (&v->nexthop[i]) &&
00475                    i < OSPF6_MULTI_PATH_LIMIT; i++)
00476                 ospf6_nexthop_copy (&w->nexthop[i], &v->nexthop[i]);
00477             }
00478 
00479           /* add new candidate to the candidate_list */
00480           if (IS_OSPF6_DEBUG_SPF (PROCESS))
00481             zlog_debug ("  New candidate: %s hops %d cost %d",
00482             w->name, w->hops, w->cost);
00483           pqueue_enqueue (w, candidate_list);
00484         }
00485     }
00486 
00487   pqueue_delete (candidate_list);
00488 }

00540   ospf6_intra_brouter_calculation (oa);
