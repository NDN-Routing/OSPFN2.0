00188   /* Set umask before anything for security */
00189   umask (0027);
00190 
00191   /* get program name */
00192   progname = ((p = strrchr (argv[0], '/')) ? ++p : argv[0]);
00193 
00194   /* Invoked by a priviledged user? -- endo. */
00195   if (geteuid () != 0)
00196     {
00197       errno = EPERM;
00198       perror (progname);
00199       exit (1);
00200     }
00201 
00202   zlog_default = openzlog (progname, ZLOG_OSPF,
00203                LOG_CONS|LOG_NDELAY|LOG_PID, LOG_DAEMON);
00204 
00205   /* OSPF master init. */
00206   ospf_master_init ();
00207 
00208 #ifdef SUPPORT_OSPF_API
00209   /* OSPF apiserver is disabled by default. */
00210   ospf_apiserver_enable = 0;
00211 #endif /* SUPPORT_OSPF_API */
00212 
00213   while (1) 
00214     {
00215       int opt;
00216 
00217       opt = getopt_long (argc, argv, "df:i:hA:P:u:g:avC", longopts, 0);
00218     
00219       if (opt == EOF)
00220     break;
00221 
00222       switch (opt) 
00223     {
00224     case 0:
00225       break;
00226     case 'd':
00227       daemon_mode = 1;
00228       break;
00229     case 'f':
00230       config_file = optarg;
00231       break;
00232     case 'A':
00233       vty_addr = optarg;
00234       break;
00235         case 'i':
00236           pid_file = optarg;
00237           break;
00238     case 'P':
00239           /* Deal with atoi() returning 0 on failure, and ospfd not

00241           if (strcmp(optarg, "0") == 0) 
00242             {
00243               vty_port = 0;
00244               break;
00245             } 
00246           vty_port = atoi (optarg);
00247           if (vty_port <= 0 || vty_port > 0xffff)
00248             vty_port = OSPF_VTY_PORT;
00249       break;
00250     case 'u':
00251       ospfd_privs.user = optarg;
00252       break;
00253     case 'g':
00254       ospfd_privs.group = optarg;
00255       break;
00256 #ifdef SUPPORT_OSPF_API
00257     case 'a':
00258       ospf_apiserver_enable = 1;
00259       break;
00260 #endif /* SUPPORT_OSPF_API */
00261     case 'v':
00262       print_version (progname);
00263       exit (0);
00264       break;
00265     case 'C':
00266       dryrun = 1;
00267       break;
00268     case 'h':
00269       usage (progname, 0);

00272       usage (progname, 1);

00274     }
00275     }
