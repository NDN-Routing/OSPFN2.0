00251 /* normalise the segment.
00252  * In particular, merge runs of AS_SEQUENCEs into one segment
00253  * Internally, we do not care about the wire segment length limit, and
00254  * we want each distinct AS_PATHs to have the exact same internal
00255  * representation - eg, so that our hashing actually works..
00256  */
00257 static struct assegment *
00258 assegment_normalise (struct assegment *head)
00259 {
00260   struct assegment *seg = head, *pin;
00261   struct assegment *tmp;
00262   
00263   if (!head)
00264     return head;
00265   
00266   while (seg)
00267     {
00268       pin = seg;
00269       
00270       /* Sort values SET segments, for determinism in paths to aid
00271        * creation of hash values / path comparisons
00272        * and because it helps other lesser implementations ;)
00273        */
00274       if (seg->type == AS_SET || seg->type == AS_CONFED_SET)
00275         {
00276       int tail = 0;
00277       int i;
00278       
00279       qsort (seg->as, seg->length, sizeof(as_t), int_cmp);
00280       
00281       /* weed out dupes */
00282       for (i=1; i < seg->length; i++)
00283         {
00284           if (seg->as[tail] == seg->as[i])
00285             continue;
00286           
00287           tail++;
00288           if (tail < i)
00289             seg->as[tail] = seg->as[i];
00290         }
00291       /* seg->length can be 0.. */
00292       if (seg->length)
00293         seg->length = tail + 1;
00294     }
00295 
00296       /* read ahead from the current, pinned segment while the segments
00297        * are packable/mergeable. Append all following packable segments
00298        * to the segment we have pinned and remove these appended
00299        * segments.
00300        */      
00301       while (pin->next && ASSEGMENT_TYPES_PACKABLE(pin, pin->next))
00302         {
00303           tmp = pin->next;
00304           seg = pin->next;
00305           
00306           /* append the next sequence to the pinned sequence */
00307           pin = assegment_append_asns (pin, seg->as, seg->length);
00308           
00309           /* bypass the next sequence */
00310           pin->next = seg->next;
00311           
00312           /* get rid of the now referenceless segment */
00313           assegment_free (tmp);
00314           
00315         }
00316 
00317       seg = pin->next;
00318     }
00319   return head;
00320 }

00716             case AS_SET:

00718 
00719             case AS_CONFED_SEQUENCE:
00720             case AS_CONFED_SET:
00721               if (!as4_path)
00722                 break ;
00723               /* RFC4893 3: "invalid for the AS4_PATH attribute"            */
00724               /* fall through */
00725 
00726             default:    /* reject unknown or invalid AS_PATH segment types  */
00727               seg_size = 0 ;
00728           } ;
00729         }
00730       else
00731         seg_size = 0 ;
00732 
00733      /* Stop now if segment is not valid (discarding anything collected to date)
00734       *
00735       * RFC4271 4.3, Path Attributes, b) AS_PATH:
00736       *
00737       *   "path segment value field contains one or more AS numbers"
00738            */
00739       if ((seg_size == 0) || (seg_size > length) || (segh.length == 0))
00740         {
00741             assegment_free_all (head);
00742           return NULL;
00743         }
00744       
00745       length -= seg_size ;
00746       
00747       /* now its safe to trust lengths */
00748       seg = assegment_new (segh.type, segh.length);
00749       
00750       if (head)
00751         prev->next = seg;
00752       else /* it's the first segment */
00753         head = prev = seg;
00754       
00755       for (i = 0; i < segh.length; i++)
00756     seg->as[i] = (use32bit) ? stream_getl (s) : stream_getw (s);
00757 
00758       if (BGP_DEBUG (as4, AS4_SEGMENT))
00759     zlog_debug ("[AS4SEG] Parse aspath segment: length left: %lu",
00760                 (unsigned long) length);
00761       
00762       prev = seg;
00763     }
00764  
00765   return assegment_normalise (head);
00766 }

00768 /* AS path parse function -- parses AS_PATH and AS4_PATH attributes
00769  *
00770  * Requires: s        -- stream, currently positioned before first segment
00771  *                       of AS_PATH or AS4_PATH (ie after attribute header)
00772  *           length   -- length of the value of the AS_PATH or AS4_PATH
00773  *           use32bit -- true <=> 4Byte ASN, otherwise 2Byte ASN
00774  *           as4_path -- true <=> AS4_PATH, otherwise AS_PATH
00775  *
00776  * Returns: if valid: address of struct aspath in the hash of known aspaths,
00777  *                    with reference count incremented.
00778  *              else: NULL
00779  *
00780  * NB: empty AS path (length == 0) is valid.  The returned struct aspath will
00781  *     have segments == NULL and str == zero length string (unique).
00782  */
00783 struct aspath *
00784 aspath_parse (struct stream *s, size_t length, int use32bit, int as4_path)
00785 {
00786   struct aspath as;
00787   struct aspath *find;
00788 
00789   /* Parse each segment and construct normalised list of struct assegment */
00790   memset (&as, 0, sizeof (struct aspath));
00791   if (length != 0)
00792     {
00793       as.segments = assegments_parse (s, length, use32bit, as4_path);
00794 
00795       if (as.segments == NULL)
00796         return NULL ;   /* Invalid AS_PATH or AS4_PATH  */
00797     } ;
00798   
00799   /* If already same aspath exist then return it. */
00800   find = hash_get (ashash, &as, aspath_hash_alloc);
00801   
00802   assert(find) ;        /* valid aspath, so must find or create */
00803   
00804   /* aspath_hash_alloc dupes segments too. that probably could be
00805    * optimised out.
00806    */
00807   assegment_free_all (as.segments);
00808   if (as.str)
00809     XFREE (MTYPE_AS_STR, as.str);
00810   
00811   find->refcnt++;
00812 
00813   return find;
00814 }

00845 /* write aspath data to stream */
00846 size_t
00847 aspath_put (struct stream *s, struct aspath *as, int use32bit )
00848 {
00849   struct assegment *seg = as->segments;
00850   size_t bytes = 0;
00851   
00852   if (!seg || seg->length == 0)
00853     return 0;
00854   
00855   if (seg)
00856     {
00857       /*
00858        * Hey, what do we do when we have > STREAM_WRITABLE(s) here?
00859        * At the moment, we would write out a partial aspath, and our peer
00860        * will complain and drop the session :-/
00861        *
00862        * The general assumption here is that many things tested will
00863        * never happen.  And, in real live, up to now, they have not.
00864        */
00865       while (seg && (ASSEGMENT_LEN(seg, use32bit) <= STREAM_WRITEABLE(s)))
00866         {
00867           struct assegment *next = seg->next;
00868           int written = 0;
00869           int asns_packed = 0;
00870           size_t lenp;
00871           
00872           /* Overlength segments have to be split up */
00873           while ( (seg->length - written) > AS_SEGMENT_MAX)
00874             {
00875               assegment_header_put (s, seg->type, AS_SEGMENT_MAX);
00876               assegment_data_put (s, seg->as, AS_SEGMENT_MAX, use32bit);
00877               written += AS_SEGMENT_MAX;
00878               bytes += ASSEGMENT_SIZE (written, use32bit);
00879             }
00880           
00881           /* write the final segment, probably is also the first */
00882           lenp = assegment_header_put (s, seg->type, seg->length - written);
00883           assegment_data_put (s, (seg->as + written), seg->length - written, 
00884                               use32bit);
00885           
00886           /* Sequence-type segments can be 'packed' together
00887            * Case of a segment which was overlength and split up
00888            * will be missed here, but that doesn't matter.
00889            */
00890           while (next && ASSEGMENTS_PACKABLE (seg, next))
00891             {
00892               /* NB: We should never normally get here given we
00893                * normalise aspath data when parse them. However, better
00894                * safe than sorry. We potentially could call
00895                * assegment_normalise here instead, but it's cheaper and
00896                * easier to do it on the fly here rather than go through
00897                * the segment list twice every time we write out
00898                * aspath's.
00899                */
00900               
00901               /* Next segment's data can fit in this one */
00902               assegment_data_put (s, next->as, next->length, use32bit);
00903               
00904               /* update the length of the segment header */
00905           stream_putc_at (s, lenp, seg->length - written + next->length);
00906               asns_packed += next->length;
00907                
00908           next = next->next;
00909         }
00910           
00911           bytes += ASSEGMENT_SIZE (seg->length - written + asns_packed, 
00912                    use32bit);
00913           seg = next;
00914         }
00915     }
00916   return bytes;
00917 }

00986 /* Modify as1 using as2 for aggregation. */
00987 struct aspath *
00988 aspath_aggregate (struct aspath *as1, struct aspath *as2)
00989 {
00990   int i;
00991   int minlen;
00992   int match;
00993   int from;
00994   struct assegment *seg1 = as1->segments;
00995   struct assegment *seg2 = as2->segments;
00996   struct aspath *aspath = NULL;
00997   struct assegment *asset;
00998   struct assegment *prevseg = NULL;
00999 
01000   match = 0;
01001   minlen = 0;
01002   aspath = NULL;
01003   asset = NULL;
01004 
01005   /* First of all check common leading sequence. */
01006   while (seg1 && seg2)
01007     {      
01008       /* Check segment type. */
01009       if (seg1->type != seg2->type)
01010     break;
01011 
01012       /* Minimum segment length. */
01013       minlen = min (seg1->length, seg2->length);
01014 
01015       for (match = 0; match < minlen; match++)
01016     if (seg1->as[match] != seg2->as[match])
01017       break;
01018 
01019       if (match)
01020     {
01021       struct assegment *seg = assegment_new (seg1->type, 0);
01022       
01023       seg = assegment_append_asns (seg, seg1->as, match);
01024 
01025       if (! aspath)
01026         {
01027           aspath = aspath_new ();
01028           aspath->segments = seg;
01029          }
01030       else
01031         prevseg->next = seg;
01032       
01033       prevseg = seg;
01034     }
01035 
01036       if (match != minlen || match != seg1->length 
01037       || seg1->length != seg2->length)
01038     break;
01039       
01040       seg1 = seg1->next;
01041       seg2 = seg2->next;
01042     }
01043 
01044   if (! aspath)
01045     aspath = aspath_new();
01046 
01047   /* Make as-set using rest of all information. */
01048   from = match;
01049   while (seg1)
01050     {
01051       for (i = from; i < seg1->length; i++)
01052     asset = aspath_aggregate_as_set_add (aspath, asset, seg1->as[i]);
01053       
01054       from = 0;
01055       seg1 = seg1->next;
01056     }
01057 
01058   from = match;
01059   while (seg2)
01060     {
01061       for (i = from; i < seg2->length; i++)
01062     asset = aspath_aggregate_as_set_add (aspath, asset, seg2->as[i]);
01063 
01064       from = 0;
01065       seg2 = seg2->next;
01066     }
01067   
01068   assegment_normalise (aspath->segments);
01069   aspath_str_update (aspath);
01070   return aspath;
01071 }

01198 /* Prepend as1 to as2.  as2 should be uninterned aspath. */
01199 struct aspath *
01200 aspath_prepend (struct aspath *as1, struct aspath *as2)
01201 {
01202   struct assegment *seg1;
01203   struct assegment *seg2;
01204 
01205   if (! as1 || ! as2)
01206     return NULL;
01207   
01208   seg1 = as1->segments;
01209   seg2 = as2->segments;
01210   
01211   /* If as2 is empty, only need to dupe as1's chain onto as2 */
01212   if (seg2 == NULL)
01213     {
01214       as2->segments = assegment_dup_all (as1->segments);
01215       aspath_str_update (as2);
01216       return as2;
01217     }
01218   
01219   /* If as1 is empty AS, no prepending to do. */
01220   if (seg1 == NULL)
01221     return as2;
01222   
01223   /* find the tail as1's segment chain. */
01224   while (seg1 && seg1->next)
01225     seg1 = seg1->next;
01226 
01227   /* Delete any AS_CONFED_SEQUENCE segment from as2. */
01228   if (seg1->type == AS_SEQUENCE && seg2->type == AS_CONFED_SEQUENCE)
01229     as2 = aspath_delete_confed_seq (as2);
01230 
01231   /* Compare last segment type of as1 and first segment type of as2. */
01232   if (seg1->type != seg2->type)
01233     return aspath_merge (as1, as2);
01234 
01235   if (seg1->type == AS_SEQUENCE)
01236     {
01237       /* We have two chains of segments, as1->segments and seg2, 
01238        * and we have to attach them together, merging the attaching
01239        * segments together into one.
01240        * 
01241        * 1. dupe as1->segments onto head of as2
01242        * 2. merge seg2's asns onto last segment of this new chain
01243        * 3. attach chain after seg2
01244        */
01245       
01246       /* dupe as1 onto as2's head */
01247       seg1 = as2->segments = assegment_dup_all (as1->segments);
01248       
01249       /* refind the tail of as2, reusing seg1 */
01250       while (seg1 && seg1->next)
01251         seg1 = seg1->next;
01252       
01253       /* merge the old head, seg2, into tail, seg1 */
01254       seg1 = assegment_append_asns (seg1, seg2->as, seg2->length);
01255       
01256       /* bypass the merged seg2, and attach any chain after it to
01257        * chain descending from as2's head
01258        */
01259       seg1->next = seg2->next;
01260       
01261       /* seg2 is now referenceless and useless*/
01262       assegment_free (seg2);
01263       
01264       /* we've now prepended as1's segment chain to as2, merging
01265        * the inbetween AS_SEQUENCE of seg2 in the process 
01266        */
01267       aspath_str_update (as2);
01268       return as2;
01269     }
01270   else
01271     {
01272       /* AS_SET merge code is needed at here. */
01273       return aspath_merge (as1, as2);
01274     }
01275   /* XXX: Ermmm, what if as1 has multiple segments?? */
01276   
01277   /* Not reached */
01278 }

01280 /* Iterate over AS_PATH segments and wipe all occurences of the
01281  * listed AS numbers. Hence some segments may lose some or even
01282  * all data on the way, the operation is implemented as a smarter
01283  * version of aspath_dup(), which allocates memory to hold the new
01284  * data, not the original. The new AS path is returned.
01285  */
01286 struct aspath *
01287 aspath_filter_exclude (struct aspath * source, struct aspath * exclude_list)
01288 {
01289   struct assegment * srcseg, * exclseg, * lastseg;
01290   struct aspath * newpath;
01291 
01292   newpath = aspath_new();
01293   lastseg = NULL;
01294 
01295   for (srcseg = source->segments; srcseg; srcseg = srcseg->next)
01296   {
01297     unsigned i, y, newlen = 0, done = 0, skip_as;
01298     struct assegment * newseg;
01299 
01300     /* Find out, how much ASns are we going to pick from this segment.
01301      * We can't perform filtering right inline, because the size of
01302      * the new segment isn't known at the moment yet.
01303      */
01304     for (i = 0; i < srcseg->length; i++)
01305     {
01306       skip_as = 0;
01307       for (exclseg = exclude_list->segments; exclseg && !skip_as; exclseg = exclseg->next)
01308         for (y = 0; y < exclseg->length; y++)
01309           if (srcseg->as[i] == exclseg->as[y])
01310           {
01311             skip_as = 1;
01312             // There's no sense in testing the rest of exclusion list, bail out.
01313             break;
01314           }
01315       if (!skip_as)
01316         newlen++;
01317     }
01318     /* newlen is now the number of ASns to copy */
01319     if (!newlen)
01320       continue;
01321 
01322     /* Actual copying. Allocate memory and iterate once more, performing filtering. */
01323     newseg = assegment_new (srcseg->type, newlen);
01324     for (i = 0; i < srcseg->length; i++)
01325     {
01326       skip_as = 0;
01327       for (exclseg = exclude_list->segments; exclseg && !skip_as; exclseg = exclseg->next)
01328         for (y = 0; y < exclseg->length; y++)
01329           if (srcseg->as[i] == exclseg->as[y])
01330           {
01331             skip_as = 1;
01332             break;
01333           }
01334       if (skip_as)
01335         continue;
01336       newseg->as[done++] = srcseg->as[i];
01337     }
01338     /* At his point newlen must be equal to done, and both must be positive. Append
01339      * the filtered segment to the gross result. */
01340     if (!lastseg)
01341       newpath->segments = newseg;
01342     else
01343       lastseg->next = newseg;
01344     lastseg = newseg;
01345   }
01346   aspath_str_update (newpath);
01347   /* We are happy returning even an empty AS_PATH, because the administrator
01348    * might expect this very behaviour. There's a mean to avoid this, if necessary,
01349    * by having a match rule against certain AS_PATH regexps in the route-map index.
01350    */
01351   aspath_free (source);
01352   return newpath;
01353 }

01433 /* Truncate an aspath after a number of hops, and put the hops remaining
01434  * at the front of another aspath.  Needed for AS4 compat.
01435  *
01436  * Returned aspath is a /new/ aspath, which should either by free'd or
01437  * interned by the caller, as desired.
01438  */
01439 struct aspath *
01440 aspath_reconcile_as4 ( struct aspath *aspath, struct aspath *as4path)
01441 {
01442   struct assegment *seg, *newseg, *prevseg = NULL;
01443   struct aspath *newpath = NULL, *mergedpath;
01444   int hops, cpasns = 0;
01445   
01446   if (!aspath)
01447     return NULL;
01448   
01449   seg = aspath->segments;
01450   
01451   /* CONFEDs should get reconciled too.. */
01452   hops = (aspath_count_hops (aspath) + aspath_count_confeds (aspath))
01453          - aspath_count_hops (as4path);
01454   
01455   if (hops < 0)
01456     {
01457       if (BGP_DEBUG (as4, AS4))
01458         zlog_warn ("[AS4] Fewer hops in AS_PATH than NEW_AS_PATH");
01459       /* Something's gone wrong. The RFC says we should now ignore AS4_PATH,
01460        * which is daft behaviour - it contains vital loop-detection
01461        * information which must have been removed from AS_PATH.
01462        */
01463        hops = aspath_count_hops (aspath);
01464     }
01465   
01466   if (!hops)
01467    return aspath_dup (as4path);
01468   
01469   if ( BGP_DEBUG(as4, AS4))
01470     zlog_debug("[AS4] got AS_PATH %s and AS4_PATH %s synthesizing now",
01471                aspath->str, as4path->str);
01472 
01473   while (seg && hops > 0)
01474     {
01475       switch (seg->type)
01476         {
01477           case AS_SET:
01478           case AS_CONFED_SET:
01479             hops--;
01480             cpasns = seg->length;
01481             break;
01482           case AS_CONFED_SEQUENCE:
01483         /* Should never split a confed-sequence, if hop-count
01484          * suggests we must then something's gone wrong somewhere.
01485          *
01486          * Most important goal is to preserve AS_PATHs prime function
01487          * as loop-detector, so we fudge the numbers so that the entire
01488          * confed-sequence is merged in.
01489          */
01490         if (hops < seg->length)
01491           {
01492             if (BGP_DEBUG (as4, AS4))
01493               zlog_debug ("[AS4] AS4PATHmangle: AS_CONFED_SEQUENCE falls"
01494                           " across 2/4 ASN boundary somewhere, broken..");
01495             hops = seg->length;
01496           }
01497       case AS_SEQUENCE:
01498         cpasns = MIN(seg->length, hops);
01499         hops -= seg->length;
01500     }
01501       
01502       assert (cpasns <= seg->length);
01503       
01504       newseg = assegment_new (seg->type, 0);
01505       newseg = assegment_append_asns (newseg, seg->as, cpasns);
01506 
01507       if (!newpath)
01508         {
01509           newpath = aspath_new ();
01510           newpath->segments = newseg;
01511         }
01512       else
01513         prevseg->next = newseg;
01514 
01515       prevseg = newseg;
01516       seg = seg->next;
01517     }
01518     
01519   /* We may be able to join some segments here, and we must
01520    * do this because... we want normalised aspaths in out hash
01521    * and we do not want to stumble in aspath_put.
01522    */
01523   mergedpath = aspath_merge (newpath, aspath_dup(as4path));
01524   aspath_free (newpath);
01525   mergedpath->segments = assegment_normalise (mergedpath->segments);
01526   aspath_str_update (mergedpath);
01527   
01528   if ( BGP_DEBUG(as4, AS4))
01529     zlog_debug ("[AS4] result of synthesizing is %s",
01530                 mergedpath->str);
01531   
01532   return mergedpath;
01533 }

01675 /* Return next token and point for string parse. */
01676 static const char *
01677 aspath_gettoken (const char *buf, enum as_token *token, u_long *asno)
01678 {
01679   const char *p = buf;
01680 
01681   /* Skip seperators (space for sequences, ',' for sets). */
01682   while (isspace ((int) *p) || *p == ',')
01683     p++;
01684 
01685   /* Check the end of the string and type specify characters
01686      (e.g. {}()). */
01687   switch (*p)
01688     {
01689     case '\0':
01690       return NULL;
01691     case '{':
01692       *token = as_token_set_start;
01693       p++;
01694       return p;
01695     case '}':
01696       *token = as_token_set_end;
01697       p++;
01698       return p;
01699     case '(':
01700       *token = as_token_confed_seq_start;
01701       p++;
01702       return p;
01703     case ')':
01704       *token = as_token_confed_seq_end;
01705       p++;
01706       return p;
01707     case '[':
01708       *token = as_token_confed_set_start;
01709       p++;
01710       return p;
01711     case ']':
01712       *token = as_token_confed_set_end;
01713       p++;
01714       return p;
01715     }
01716 
01717   /* Check actual AS value. */
01718   if (isdigit ((int) *p)) 
01719     {
01720       as_t asval;
01721       
01722       *token = as_token_asval;
01723       asval = (*p - '0');
01724       p++;
01725       
01726       while (isdigit ((int) *p)) 
01727         {
01728           asval *= 10;
01729           asval += (*p - '0');
01730           p++;
01731         }
01732       *asno = asval;
01733       return p;
01734     }
01735   
01736   /* There is no match then return unknown token. */
01737   *token = as_token_unknown;
01738   return  p++;
01739 }
