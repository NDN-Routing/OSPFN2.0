  while (1)

      opt = getopt_long (argc, argv, "df:i:hp:l:A:P:rnu:g:vC", longopts, 0);
    
      if (opt == EOF)
    break;

      switch (opt) 
    {
    case 0:
      break;

      break;

      break;
        case 'i':

    case 'p':
      tmp_port = atoi (optarg);

        bm->port = BGP_PORT_DEFAULT;
      else
        bm->port = tmp_port;
      break;
    case 'A':
      vty_addr = optarg;
      break;
    case 'P':
          /* Deal with atoi() returning 0 on failure, and bgpd not
             listening on bgp port... */
          if (strcmp(optarg, "0") == 0) 
            {
              vty_port = 0;
              break;
            } 
          vty_port = atoi (optarg);
      if (vty_port <= 0 || vty_port > 0xffff)
        vty_port = BGP_VTY_PORT;
      break;
    case 'r':
      retain_mode = 1;
      break;
    case 'l':
      bm->address = optarg;
      /* listenon implies -n */
    case 'n':
      bgp_option_set (BGP_OPT_NO_FIB);
      break;
    case 'u':
      bgpd_privs.user = optarg;
      break;
    case 'g':
      bgpd_privs.group = optarg;
      break;
    case 'v':
      print_version (progname);
      exit (0);
      break;
    case 'C':
      dryrun = 1;
      break;
    case 'h':
      usage (progname, 0);
      break;
    default:
      usage (progname, 1);
      break;
    }
    }

  master = bm->master;

  srand (time (NULL));
