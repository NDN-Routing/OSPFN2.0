00229 
00230   /* Get the programname without the preceding path. */

00232 
00233   zlog_default = openzlog (progname, ZLOG_ISIS,
00234                LOG_CONS | LOG_NDELAY | LOG_PID, LOG_DAEMON);
00235

00238   _argv = argv;
00239   _envp = envp;

00241   if (*argv[0] == '.')
00242     snprintf (_progpath, sizeof (_progpath), "%s/%s", _cwd, _argv[0]);
00243   else
00244     snprintf (_progpath, sizeof (_progpath), "%s", argv[0]);
00245 
00246   /* Command line argument treatment. */
00247   while (1)
00248     {
00249       opt = getopt_long (argc, argv, "df:i:hA:p:P:u:g:vC", longopts, 0);
00250 
00251       if (opt == EOF)
00252     break;
00253 
00254       switch (opt)
00255     {
00256     case 0:
00257       break;
00258     case 'd':
00259       daemon_mode = 1;
00260       break;
00261     case 'f':
00262       config_file = optarg;
00263       break;
00264     case 'i':
00265       pid_file = optarg;
00266       break;
00267     case 'A':
00268       vty_addr = optarg;
00269       break;
00270     case 'P':
00271       /* Deal with atoi() returning 0 on failure, and isisd not
00272          listening on isisd port... */
00273       if (strcmp (optarg, "0") == 0)
00274         {
00275           vty_port = 0;
00276           break;
00277         }
00278       vty_port = atoi (optarg);
00279       vty_port = (vty_port ? vty_port : ISISD_VTY_PORT);
00280       break;
00281     case 'u':
00282       isisd_privs.user = optarg;
00283       break;
00284     case 'g':
00285       isisd_privs.group = optarg;
00286       break;
00287     case 'v':
00288       printf ("ISISd version %s\n", ISISD_VERSION);
00289       printf ("Copyright (c) 2001-2002 Sampo Saaristo,"
00290           " Ofer Wald and Hannes Gredler\n");

00292       exit (0);
00293       break;
00294     case 'C':
00295       dryrun = 1;
00296       break;
00297     case 'h':
00298       usage (0);
00299       break;
00300     default:
00301       usage (1);
00302       break;
00303     }
00304     }
00305

00308 
00309   /* random seed from time */
00310   srand (time (NULL));
00311 
00312   /*
00313    *  initializations
00314    */

00317   cmd_init (1);
00318   vty_init (master);

00322   dyn_cache_init ();

00324 
00325   /* parse config file */
00326   /* this is needed three times! because we have interfaces before the areas */
00327   vty_read_config (config_file, config_default);
00328   vty_read_config (config_file, config_default);
00329   vty_read_config (config_file, config_default);
00330 
00331   /* Start execution only if not in dry-run mode */
00332   if (dryrun)
00333     return(0);
00334   
00335   /* demonize */
00336   if (daemon_mode && daemon (0, 0) < 0)
00337     {
00338       zlog_err("ISISd daemon failed: %s", strerror(errno));
00339       exit (1);

00341 
00342   /* Process ID file creation. */
00343   pid_output (pid_file);
00344 
00345   /* Make isis vty socket. */
00346   vty_serv_sock (vty_addr, vty_port, ISIS_VTYSH_PATH);
00347 
00348   /* Print banner. */
00349   zlog_notice ("Quagga-ISISd %s starting: vty@%d", QUAGGA_VERSION, vty_port);
00350 
00351   /* Start finite state machine. */
00352   while (thread_fetch (master, &thread))
00353     thread_call (&thread);
00354 
00355   /* Not reached. */
00356   exit (0);
