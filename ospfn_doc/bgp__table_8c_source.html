00102 /* Free route table. */
00103 static void
00104 bgp_table_free (struct bgp_table *rt)
00105 {
00106   struct bgp_node *tmp_node;
00107   struct bgp_node *node;
00108  
00109   if (rt == NULL)
00110     return;
00111 
00112   node = rt->top;
00113 
00114   /* Bulk deletion of nodes remaining in this table.  This function is not
00115      called until workers have completed their dependency on this table.
00116      A final bgp_unlock_node() will not be called for these nodes. */
00117   while (node)
00118     {
00119       if (node->l_left)
00120     {
00121       node = node->l_left;
00122       continue;
00123     }
00124 
00125       if (node->l_right)
00126     {
00127       node = node->l_right;
00128       continue;
00129     }
00130 
00131       tmp_node = node;
00132       node = node->parent;
00133 
00134       tmp_node->table->count--;
00135       tmp_node->lock = 0;  /* to cause assert if unlocked after this */
00136       bgp_node_free (tmp_node);
00137 
00138       if (node != NULL)
00139     {
00140       if (node->l_left == tmp_node)
00141         node->l_left = NULL;
00142       else
00143         node->l_right = NULL;
00144     }
00145       else
00146     {
00147       break;
00148     }
00149     }
00150  
00151   assert (rt->count == 0);
00152 
00153   if (rt->owner)
00154     {
00155       peer_unlock (rt->owner);
00156       rt->owner = NULL;
00157     }
00158 
00159   XFREE (MTYPE_BGP_TABLE, rt);
00160   return;
00161 }

00169 /* Common prefix route genaration. */
00170 static void
00171 route_common (struct prefix *n, struct prefix *p, struct prefix *new)
00172 {
00173   int i;
00174   u_char diff;
00175   u_char mask;
00176 
00177   u_char *np = (u_char *)&n->u.prefix;
00178   u_char *pp = (u_char *)&p->u.prefix;
00179   u_char *newp = (u_char *)&new->u.prefix;
00180 
00181   for (i = 0; i < p->prefixlen / 8; i++)
00182     {
00183       if (np[i] == pp[i])
00184     newp[i] = np[i];
00185       else
00186     break;
00187     }
00188 
00189   new->prefixlen = i * 8;
00190 
00191   if (new->prefixlen != p->prefixlen)
00192     {
00193       diff = np[i] ^ pp[i];
00194       mask = 0x80;
00195       while (new->prefixlen < p->prefixlen && !(mask & diff))
00196     {
00197       mask >>= 1;
00198       new->prefixlen++;
00199     }
00200       newp[i] = np[i] & maskbit[new->prefixlen % 8];
00201     }
00202 }

00232 /* Find matched prefix. */
00233 struct bgp_node *
00234 bgp_node_match (const struct bgp_table *table, struct prefix *p)
00235 {
00236   struct bgp_node *node;
00237   struct bgp_node *matched;
00238 
00239   matched = NULL;
00240   node = table->top;
00241 
00242   /* Walk down tree.  If there is matched route then store it to
00243      matched. */
00244   while (node && node->p.prefixlen <= p->prefixlen && 
00245      prefix_match (&node->p, p))
00246     {
00247       if (node->info)
00248     matched = node;
00249       node = node->link[prefix_bit(&p->u.prefix, node->p.prefixlen)];
00250     }
00251 
00252   /* If matched route found, return it. */
00253   if (matched)
00254     return bgp_lock_node (matched);
00255 
00256   return NULL;
00257 }

00365 /* Delete node from the routing table. */
00366 static void
00367 bgp_node_delete (struct bgp_node *node)
00368 {
00369   struct bgp_node *child;
00370   struct bgp_node *parent;
00371 
00372   assert (node->lock == 0);
00373   assert (node->info == NULL);
00374 
00375   if (node->l_left && node->l_right)
00376     return;
00377 
00378   if (node->l_left)
00379     child = node->l_left;
00380   else
00381     child = node->l_right;
00382 
00383   parent = node->parent;
00384 
00385   if (child)
00386     child->parent = parent;
00387 
00388   if (parent)
00389     {
00390       if (parent->l_left == node)
00391     parent->l_left = child;
00392       else
00393     parent->l_right = child;
00394     }
00395   else
00396     node->table->top = child;
00397   
00398   node->table->count--;
00399   
00400   bgp_node_free (node);
00401 
00402   /* If parent node is stub then delete it also. */
00403   if (parent && parent->lock == 0)
00404     bgp_node_delete (parent);
