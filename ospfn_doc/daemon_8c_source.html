00029 int
00030 daemon (int nochdir, int noclose)
00031 {
00032   pid_t pid;
00033 
00034   pid = fork ();
00035 
00036   /* In case of fork is error. */
00037   if (pid < 0)
00038     {
00039       zlog_err ("fork failed: %s", safe_strerror(errno));
00040       return -1;
00041     }
00042 
00043   /* In case of this is parent process. */
00044   if (pid != 0)
00045     exit (0);
00046 
00047   /* Become session leader and get pid. */
00048   pid = setsid();
00049 
00050   if (pid == -1)
00051     {
00052       zlog_err ("setsid failed: %s", safe_strerror(errno));
00053       return -1;
00054     }
00055 
00056   /* Change directory to root. */
00057   if (! nochdir)
00058     chdir ("/");
00059 
00060   /* File descriptor close. */
00061   if (! noclose)
00062     {
00063       int fd;
00064 
00065       fd = open ("/dev/null", O_RDWR, 0);
00066       if (fd != -1)
00067     {
00068       dup2 (fd, STDIN_FILENO);
00069       dup2 (fd, STDOUT_FILENO);
00070       dup2 (fd, STDERR_FILENO);
00071       if (fd > 2)
00072         close (fd);
00073     }
00074     }
00075 
00076   umask (0027);
00077 
00078   return 0;
00079 }
