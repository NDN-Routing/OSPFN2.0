00089 /* Get RIP peer.  At the same time update timeout thread. */
00090 static struct rip_peer *
00091 rip_peer_get (struct in_addr *addr)
00092 {
00093   struct rip_peer *peer;
00094 
00095   peer = rip_peer_lookup (addr);
00096 
00097   if (peer)
00098     {
00099       if (peer->t_timeout)
00100     thread_cancel (peer->t_timeout);
00101     }
00102   else
00103     {
00104       peer = rip_peer_new ();
00105       peer->addr = *addr;
00106       listnode_add_sort (peer_list, peer);
00107     }
00108 
00109   /* Update timeout thread. */
00110   peer->t_timeout = thread_add_timer (master, rip_peer_timeout, peer,
00111                       RIP_PEER_TIMER_DEFAULT);
00112 
00113   /* Last update time set. */
00114   time (&peer->uptime);
00115   
00116   return peer;
00117 }

00143 /* Display peer uptime. */
00144 static char *
00145 rip_peer_uptime (struct rip_peer *peer, char *buf, size_t len)
00146 {
00147   time_t uptime;
00148   struct tm *tm;
00149 
00150   /* If there is no connection has been done before print `never'. */
00151   if (peer->uptime == 0)
00152     {
00153       snprintf (buf, len, "never   ");
00154       return buf;
00155     }
00156 
00157   /* Get current time. */
00158   uptime = time (NULL);
00159   uptime -= peer->uptime;
00160   tm = gmtime (&uptime);
00161 
00162   /* Making formatted timer strings. */
00163 #define ONE_DAY_SECOND 60*60*24
00164 #define ONE_WEEK_SECOND 60*60*24*7
00165 
00166   if (uptime < ONE_DAY_SECOND)
00167     snprintf (buf, len, "%02d:%02d:%02d", 
00168           tm->tm_hour, tm->tm_min, tm->tm_sec);
00169   else if (uptime < ONE_WEEK_SECOND)
00170     snprintf (buf, len, "%dd%02dh%02dm", 
00171           tm->tm_yday, tm->tm_hour, tm->tm_min);
00172   else
00173     snprintf (buf, len, "%02dw%dd%02dh", 
00174           tm->tm_yday/7, tm->tm_yday - ((tm->tm_yday/7) * 7), tm->tm_hour);
00175   return buf;
00176 }
