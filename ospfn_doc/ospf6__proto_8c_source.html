00028 void
00029 ospf6_prefix_apply_mask (struct ospf6_prefix *op)
00030 {
00031   u_char *pnt, mask;
00032   int index, offset;
00033 
00034   pnt = (u_char *)((caddr_t) op + sizeof (struct ospf6_prefix));
00035   index = op->prefix_length / 8;
00036   offset = op->prefix_length % 8;
00037   mask = 0xff << (8 - offset);
00038 
00039   if (index > 16)
00040     {
00041       zlog_warn ("Prefix length too long: %d", op->prefix_length);
00042       return;
00043     }
00044 
00045   if (index == 16)
00046     return;
00047 
00048   pnt[index] &= mask;
00049   index ++;
00050 
00051   while (index < OSPF6_PREFIX_SPACE (op->prefix_length))
00052     pnt[index++] = 0;
00053 }
