00069 static void 
00070 handle_route_entry (mib2_ipRouteEntry_t *routeEntry)
00071 {
00072     struct prefix_ipv4  prefix;
00073     struct in_addr      tmpaddr, gateway;
00074     u_char          zebra_flags = 0;
00075 
00076     if (routeEntry->ipRouteInfo.re_ire_type & IRE_CACHETABLE)
00077         return;
00078 
00079     if (routeEntry->ipRouteInfo.re_ire_type & IRE_HOST_REDIRECT)
00080         zebra_flags |= ZEBRA_FLAG_SELFROUTE;
00081 
00082     prefix.family = AF_INET;
00083 
00084     tmpaddr.s_addr = routeEntry->ipRouteDest;
00085     prefix.prefix = tmpaddr;
00086 
00087     tmpaddr.s_addr = routeEntry->ipRouteMask;
00088     prefix.prefixlen = ip_masklen (tmpaddr);
00089 
00090     gateway.s_addr = routeEntry->ipRouteNextHop;
00091 
00092     rib_add_ipv4 (ZEBRA_ROUTE_KERNEL, zebra_flags, &prefix,
00093               &gateway, NULL, 0, 0, 0, 0);
00094 }

00096 void
00097 route_read (void)
00098 {
00099     char            storage[RT_BUFSIZ];
00100 
00101     struct T_optmgmt_req    *TLIreq = (struct T_optmgmt_req *) storage;
00102     struct T_optmgmt_ack    *TLIack = (struct T_optmgmt_ack *) storage;
00103     struct T_error_ack  *TLIerr = (struct T_error_ack *) storage;
00104 
00105     struct opthdr       *MIB2hdr;
00106 
00107     mib2_ipRouteEntry_t *routeEntry, *lastRouteEntry;
00108 
00109     struct strbuf       msgdata;
00110     int         flags, dev, retval, process;
00111 
00112     if ((dev = open (_PATH_GETMSG_ROUTE, O_RDWR)) == -1) {
00113         zlog_warn ("can't open %s: %s", _PATH_GETMSG_ROUTE,
00114             safe_strerror (errno));
00115         return;
00116     }
00117 
00118     TLIreq->PRIM_type = T_OPTMGMT_REQ;
00119     TLIreq->OPT_offset = sizeof (struct T_optmgmt_req);
00120     TLIreq->OPT_length = sizeof (struct opthdr);
00121     TLIreq->MGMT_flags = T_CURRENT;
00122 
00123     MIB2hdr = (struct opthdr *) &TLIreq[1];
00124 
00125     MIB2hdr->level = MIB2_IP;
00126     MIB2hdr->name = 0;
00127     MIB2hdr->len = 0;
00128     
00129     msgdata.buf = storage;
00130     msgdata.len = sizeof (struct T_optmgmt_req) + sizeof (struct opthdr);
00131 
00132     flags = 0;
00133 
00134     if (putmsg (dev, &msgdata, NULL, flags) == -1) {
00135         zlog_warn ("putmsg failed: %s", safe_strerror (errno));
00136         goto exit;
00137     }
00138 
00139     MIB2hdr = (struct opthdr *) &TLIack[1];
00140     msgdata.maxlen = sizeof (storage);
00141 
00142     while (1) {
00143         flags = 0;
00144         retval = getmsg (dev, &msgdata, NULL, &flags);
00145 
00146         if (retval == -1) {
00147             zlog_warn ("getmsg(ctl) failed: %s", safe_strerror (errno));
00148             goto exit;
00149         }
00150 
00151         /* This is normal loop termination */
00152         if (retval == 0 &&
00153             msgdata.len >= sizeof (struct T_optmgmt_ack) &&
00154             TLIack->PRIM_type == T_OPTMGMT_ACK &&
00155             TLIack->MGMT_flags == T_SUCCESS &&
00156             MIB2hdr->len == 0)
00157             break;
00158 
00159         if (msgdata.len >= sizeof (struct T_error_ack) &&
00160             TLIerr->PRIM_type == T_ERROR_ACK) {
00161             zlog_warn ("getmsg(ctl) returned T_ERROR_ACK: %s",
00162                 safe_strerror ((TLIerr->TLI_error == TSYSERR)
00163                 ? TLIerr->UNIX_error : EPROTO));
00164             break;
00165         }
00166 
00167         /* should dump more debugging info to the log statement,
00168            like what GateD does in this instance, but not
00169            critical yet. */
00170         if (retval != MOREDATA ||
00171             msgdata.len < sizeof (struct T_optmgmt_ack) ||
00172             TLIack->PRIM_type != T_OPTMGMT_ACK ||
00173             TLIack->MGMT_flags != T_SUCCESS) {
00174             errno = ENOMSG;
00175             zlog_warn ("getmsg(ctl) returned bizarreness");
00176             break;
00177         }
00178 
00179         /* MIB2_IP_21 is the the pseudo-MIB2 ipRouteTable
00180            entry, see <inet/mib2.h>. "This isn't the MIB data
00181            you're looking for." */
00182         process = (MIB2hdr->level == MIB2_IP &&
00183             MIB2hdr->name == MIB2_IP_21) ? 1 : 0;
00184 
00185         /* getmsg writes the data buffer out completely, not
00186            to the closest smaller multiple. Unless reassembling
00187            data structures across buffer boundaries is your idea
00188            of a good time, set maxlen to the closest smaller
00189            multiple of the size of the datastructure you're
00190            retrieving. */
00191         msgdata.maxlen = sizeof (storage) - (sizeof (storage)
00192             % sizeof (mib2_ipRouteEntry_t));
00193 
00194         msgdata.len = 0;
00195         flags = 0;
00196 
00197         do {
00198             retval = getmsg (dev, NULL, &msgdata, &flags);
00199 
00200             if (retval == -1) {
00201                 zlog_warn ("getmsg(data) failed: %s",
00202                     safe_strerror (errno));
00203                 goto exit;
00204             }
00205 
00206             if (!(retval == 0 || retval == MOREDATA)) {
00207                 zlog_warn ("getmsg(data) returned %d", retval);
00208                 goto exit;
00209             }
00210 
00211             if (process) {
00212                 if (msgdata.len %
00213                     sizeof (mib2_ipRouteEntry_t) != 0) {
00214                     zlog_warn ("getmsg(data) returned "
00215 "msgdata.len = %d (%% sizeof (mib2_ipRouteEntry_t) != 0)", msgdata.len);
00216                     goto exit;
00217                 }
00218 
00219                 routeEntry = (mib2_ipRouteEntry_t *)
00220                     msgdata.buf;
00221                 lastRouteEntry = (mib2_ipRouteEntry_t *)
00222                     (msgdata.buf + msgdata.len);
00223                 do {
00224                     handle_route_entry (routeEntry);
00225                 } while (++routeEntry < lastRouteEntry);
00226             }
00227         } while (retval == MOREDATA);
00228     }
00229 
00230 exit:
00231     close (dev);
00232 }
