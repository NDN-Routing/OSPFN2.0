00232 {
00233   struct stream *s;
00234   struct zapi_ipv4 api;
00235   unsigned long ifindex;
00236   struct in_addr nexthop;
00237   struct prefix_ipv4 p;
00238

00247 
00248   /* IPv4 prefix. */

00250   p.family = AF_INET;

00253 
00254   /* Nexthop, ifindex, distance, metric. */
00255   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_NEXTHOP))
00256     {
00257       api.nexthop_num = stream_getc (s);

00264     }

00266     api.distance = stream_getc (s);
00267   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_METRIC))
00268     api.metric = stream_getl (s);
00269   else
00270     api.metric = 0;
00271 
00272   if (command == ZEBRA_IPV4_ROUTE_ADD)
00273     {
00274       if (BGP_DEBUG(zebra, ZEBRA))
00275     {
00276       char buf[2][INET_ADDRSTRLEN];
00277       zlog_debug("Zebra rcvd: IPv4 route add %s %s/%d nexthop %s metric %u",
00278              zebra_route_string(api.type),
00279              inet_ntop(AF_INET, &p.prefix, buf[0], sizeof(buf[0])),
00280              p.prefixlen,
00281              inet_ntop(AF_INET, &nexthop, buf[1], sizeof(buf[1])),
00282              api.metric);
00283     }
00284       bgp_redistribute_add((struct prefix *)&p, &nexthop, api.metric, api.type);
00285     }
00286   else
00287     {
00288       if (BGP_DEBUG(zebra, ZEBRA))
00289     {
00290       char buf[2][INET_ADDRSTRLEN];

00292              "nexthop %s metric %u",
00293              zebra_route_string(api.type),
00294              inet_ntop(AF_INET, &p.prefix, buf[0], sizeof(buf[0])),
00295              p.prefixlen,
00296              inet_ntop(AF_INET, &nexthop, buf[1], sizeof(buf[1])),
00297              api.metric);

00299       bgp_redistribute_delete((struct prefix *)&p, api.type);
00300     }

00530 int
00531 bgp_nexthop_set (union sockunion *local, union sockunion *remote, 
00532          struct bgp_nexthop *nexthop, struct peer *peer)
00533 {
00534   int ret = 0;
00535   struct interface *ifp = NULL;
00536 
00537   memset (nexthop, 0, sizeof (struct bgp_nexthop));
00538 
00539   if (!local)
00540     return -1;
00541   if (!remote)
00542     return -1;
00543 
00544   if (local->sa.sa_family == AF_INET)
00545     {
00546       nexthop->v4 = local->sin.sin_addr;
00547       ifp = if_lookup_by_ipv4 (&local->sin.sin_addr);
00548     }
00549 #ifdef HAVE_IPV6
00550   if (local->sa.sa_family == AF_INET6)
00551     {
00552       if (IN6_IS_ADDR_LINKLOCAL (&local->sin6.sin6_addr))
00553     {
00554       if (peer->ifname)
00555         ifp = if_lookup_by_index (if_nametoindex (peer->ifname));
00556     }
00557       else
00558     ifp = if_lookup_by_ipv6 (&local->sin6.sin6_addr);
00559     }
00560 #endif /* HAVE_IPV6 */
00561 
00562   if (!ifp)
00563     return -1;
00564 
00565   nexthop->ifp = ifp;
00566 
00567   /* IPv4 connection. */
00568   if (local->sa.sa_family == AF_INET)
00569     {
00570 #ifdef HAVE_IPV6
00571       /* IPv6 nexthop*/
00572       ret = if_get_ipv6_global (ifp, &nexthop->v6_global);
00573 
00574       /* There is no global nexthop. */
00575       if (!ret)
00576     if_get_ipv6_local (ifp, &nexthop->v6_global);
00577       else
00578     if_get_ipv6_local (ifp, &nexthop->v6_local);
00579 #endif /* HAVE_IPV6 */
00580     }
00581 
00582 #ifdef HAVE_IPV6
00583   /* IPv6 connection. */
00584   if (local->sa.sa_family == AF_INET6)
00585     {
00586       struct interface *direct = NULL;
00587 
00588       /* IPv4 nexthop.  I don't care about it. */
00589       if (peer->local_id.s_addr)
00590     nexthop->v4 = peer->local_id;
00591 
00592       /* Global address*/
00593       if (! IN6_IS_ADDR_LINKLOCAL (&local->sin6.sin6_addr))
00594     {
00595       memcpy (&nexthop->v6_global, &local->sin6.sin6_addr, 
00596           IPV6_MAX_BYTELEN);
00597 
00598       /* If directory connected set link-local address. */
00599       direct = if_lookup_by_ipv6 (&remote->sin6.sin6_addr);
00600       if (direct)
00601         if_get_ipv6_local (ifp, &nexthop->v6_local);
00602     }
00603       else
00604     /* Link-local address. */
00605     {
00606       ret = if_get_ipv6_global (ifp, &nexthop->v6_global);
00607 
00608       /* If there is no global address.  Set link-local address as
00609              global.  I know this break RFC specification... */
00610       if (!ret)
00611         memcpy (&nexthop->v6_global, &local->sin6.sin6_addr, 
00612             IPV6_MAX_BYTELEN);
00613       else
00614         memcpy (&nexthop->v6_local, &local->sin6.sin6_addr, 
00615             IPV6_MAX_BYTELEN);
00616     }
00617     }
00618 
00619   if (IN6_IS_ADDR_LINKLOCAL (&local->sin6.sin6_addr) ||
00620       if_lookup_by_ipv6 (&remote->sin6.sin6_addr))
00621     peer->shared_network = 1;
00622   else
00623     peer->shared_network = 0;
00624 
00625   /* KAME stack specific treatment.  */
00626 #ifdef KAME
00627   if (IN6_IS_ADDR_LINKLOCAL (&nexthop->v6_global)
00628       && IN6_LINKLOCAL_IFINDEX (nexthop->v6_global))
00629     {
00630       SET_IN6_LINKLOCAL_IFINDEX (nexthop->v6_global, 0);
00631     }
00632   if (IN6_IS_ADDR_LINKLOCAL (&nexthop->v6_local)
00633       && IN6_LINKLOCAL_IFINDEX (nexthop->v6_local))
00634     {
00635       SET_IN6_LINKLOCAL_IFINDEX (nexthop->v6_local, 0);
00636     }
00637 #endif /* KAME */
00638 #endif /* HAVE_IPV6 */
00639   return ret;
00640 }
