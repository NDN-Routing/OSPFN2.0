00302 /* Main routine of bgpd. Treatment of argument and start bgp finite
00303    state machine is handled at here. */
00304 int
00305 main (int argc, char **argv)
00306 {
00307   char *p;
00308   int opt;
00309   int daemon_mode = 0;
00310   int dryrun = 0;
00311   char *progname;
00312   struct thread thread;
00313   int tmp_port;
00314 
00315   /* Set umask before anything for security */
00316   umask (0027);
00317 
00318   /* Preserve name of myself. */
00319   progname = ((p = strrchr (argv[0], '/')) ? ++p : argv[0]);
00320 
00321   zlog_default = openzlog (progname, ZLOG_BGP,
00322                LOG_CONS|LOG_NDELAY|LOG_PID, LOG_DAEMON);
00323 
00324   /* BGP master init. */
00325   bgp_master_init ();
00326 
00327   /* Command line argument treatment. */
00328   while (1) 
00329     {
00330       opt = getopt_long (argc, argv, "df:i:hp:l:A:P:rnu:g:vC", longopts, 0);
00331     
00332       if (opt == EOF)
00333     break;
00334 
00335       switch (opt) 
00336     {
00337     case 0:
00338       break;
00339     case 'd':
00340       daemon_mode = 1;
00341       break;
00342     case 'f':
00343       config_file = optarg;
00344       break;
00345         case 'i':
00346           pid_file = optarg;
00347           break;
00348     case 'p':
00349       tmp_port = atoi (optarg);
00350       if (tmp_port <= 0 || tmp_port > 0xffff)
00351         bm->port = BGP_PORT_DEFAULT;
00352       else
00353         bm->port = tmp_port;
00354       break;
00355     case 'A':
00356       vty_addr = optarg;
00357       break;
00358     case 'P':
00359           /* Deal with atoi() returning 0 on failure, and bgpd not
00360              listening on bgp port... */
00361           if (strcmp(optarg, "0") == 0) 
00362             {
00363               vty_port = 0;
00364               break;
00365             } 
00366           vty_port = atoi (optarg);
00367       if (vty_port <= 0 || vty_port > 0xffff)
00368         vty_port = BGP_VTY_PORT;
00369       break;
00370     case 'r':
00371       retain_mode = 1;
00372       break;
00373     case 'l':
00374       bm->address = optarg;
00375       /* listenon implies -n */
00376     case 'n':
00377       bgp_option_set (BGP_OPT_NO_FIB);
00378       break;
00379     case 'u':
00380       bgpd_privs.user = optarg;
00381       break;
00382     case 'g':
00383       bgpd_privs.group = optarg;
00384       break;
00385     case 'v':
00386       print_version (progname);
00387       exit (0);
00388       break;
00389     case 'C':
00390       dryrun = 1;
00391       break;
00392     case 'h':
00393       usage (progname, 0);
00394       break;
00395     default:
00396       usage (progname, 1);
00397       break;
00398     }
00399     }
00400 
00401   /* Make thread master. */
00402   master = bm->master;
00403 
00404   /* Initializations. */
00405   srand (time (NULL));
00406   signal_init (master, Q_SIGC(bgp_signals), bgp_signals);
00407   zprivs_init (&bgpd_privs);
00408   cmd_init (1);
00409   vty_init (master);
00410   memory_init ();
00411 
00412   /* BGP related initialization.  */
00413   bgp_init ();
00414 
00415   /* Sort CLI commands. */
00416   sort_node ();
00417 
00418   /* Parse config file. */
00419   vty_read_config (config_file, config_default);
00420 
00421   /* Start execution only if not in dry-run mode */
00422   if(dryrun)
00423     return(0);
00424   
00425   /* Turn into daemon if daemon_mode is set. */
00426   if (daemon_mode && daemon (0, 0) < 0)
00427     {
00428       zlog_err("BGPd daemon failed: %s", strerror(errno));
00429       return (1);
00430     }
00431 
00432 
00433   /* Process ID file creation. */
00434   pid_output (pid_file);
00435 
00436   /* Make bgp vty socket. */
00437   vty_serv_sock (vty_addr, vty_port, BGP_VTYSH_PATH);
00438 
00439   /* Print banner. */
00440   zlog_notice ("BGPd %s starting: vty@%d, bgp@%s:%d", QUAGGA_VERSION,
00441            vty_port, 
00442            (bm->address ? bm->address : "<all>"),
00443            bm->port);
00444 
00445   /* Start finite state machine, here we go! */
00446   while (thread_fetch (master, &thread))
00447     thread_call (&thread);
00448 
00449   /* Not reached. */
00450   return (0);
00451 }
