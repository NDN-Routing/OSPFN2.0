00047 /* The following bits are used to determine the regexp syntax we
00048    recognize.  The set/not-set meanings are chosen so that Emacs syntax
00049    remains the value 0.  The bits are given in alphabetical order, and
00050    the definitions shifted by one from the previous bit; thus, when we
00051    add or remove a bit, only one other definition need change.  */

00054 /* If this bit is not set, then \ inside a bracket expression is literal.
00055    If set, then such a \ quotes the following character.  */
00056 #define RE_BACKSLASH_ESCAPE_IN_LISTS ((unsigned long int) 1)
00057 
00058 /* If this bit is not set, then + and ? are operators, and \+ and \? are
00059      literals.
00060    If set, then \+ and \? are operators and + and ? are literals.  */
00061 #define RE_BK_PLUS_QM (RE_BACKSLASH_ESCAPE_IN_LISTS << 1)
00062 
00063 /* If this bit is set, then character classes are supported.  They are:
00064      [:alpha:], [:upper:], [:lower:],  [:digit:], [:alnum:], [:xdigit:],
00065      [:space:], [:print:], [:punct:], [:graph:], and [:cntrl:].
00066    If not set, then character classes are not supported.  */
00067 #define RE_CHAR_CLASSES (RE_BK_PLUS_QM << 1)
00068 
00069 /* If this bit is set, then ^ and $ are always anchors (outside bracket
00070      expressions, of course).
00071    If this bit is not set, then it depends:
00072         ^  is an anchor if it is at the beginning of a regular
00073            expression or after an open-group or an alternation operator;
00074         $  is an anchor if it is at the end of a regular expression, or
00075            before a close-group or an alternation operator.
00076 
00077    This bit could be (re)combined with RE_CONTEXT_INDEP_OPS, because
00078    POSIX draft 11.2 says that * etc. in leading positions is undefined.
00079    We already implemented a previous draft which made those constructs
00080    invalid, though, so we haven't changed the code back.  */
00081 #define RE_CONTEXT_INDEP_ANCHORS (RE_CHAR_CLASSES << 1)
00082 
00083 /* If this bit is set, then special characters are always special
00084      regardless of where they are in the pattern.
00085    If this bit is not set, then special characters are special only in
00086      some contexts; otherwise they are ordinary.  Specifically,
00087      * + ? and intervals are only special when not after the beginning,
00088      open-group, or alternation operator.  */
00089 #define RE_CONTEXT_INDEP_OPS (RE_CONTEXT_INDEP_ANCHORS << 1)
00090 
00091 /* If this bit is set, then *, +, ?, and { cannot be first in an re or
00092      immediately after an alternation or begin-group operator.  */
00093 #define RE_CONTEXT_INVALID_OPS (RE_CONTEXT_INDEP_OPS << 1)
00094 
00095 /* If this bit is set, then . matches newline.
00096    If not set, then it doesn't.  */
00097 #define RE_DOT_NEWLINE (RE_CONTEXT_INVALID_OPS << 1)
00098 
00099 /* If this bit is set, then . doesn't match NUL.
00100    If not set, then it does.  */
00101 #define RE_DOT_NOT_NULL (RE_DOT_NEWLINE << 1)
00102 
00103 /* If this bit is set, nonmatching lists [^...] do not match newline.
00104    If not set, they do.  */
00105 #define RE_HAT_LISTS_NOT_NEWLINE (RE_DOT_NOT_NULL << 1)
00106 
00107 /* If this bit is set, either \{...\} or {...} defines an
00108      interval, depending on RE_NO_BK_BRACES.
00109    If not set, \{, \}, {, and } are literals.  */
00110 #define RE_INTERVALS (RE_HAT_LISTS_NOT_NEWLINE << 1)
00111 
00112 /* If this bit is set, +, ? and | aren't recognized as operators.
00113    If not set, they are.  */
00114 #define RE_LIMITED_OPS (RE_INTERVALS << 1)
00115 
00116 /* If this bit is set, newline is an alternation operator.
00117    If not set, newline is literal.  */
00118 #define RE_NEWLINE_ALT (RE_LIMITED_OPS << 1)
00119 
00120 /* If this bit is set, then `{...}' defines an interval, and \{ and \}
00121      are literals.
00122   If not set, then `\{...\}' defines an interval.  */
00123 #define RE_NO_BK_BRACES (RE_NEWLINE_ALT << 1)
00124 
00125 /* If this bit is set, (...) defines a group, and \( and \) are literals.
00126    If not set, \(...\) defines a group, and ( and ) are literals.  */
00127 #define RE_NO_BK_PARENS (RE_NO_BK_BRACES << 1)
00128 
00129 /* If this bit is set, then <digit> matches <digit>.
00130    If not set, then <digit> is a back-reference.  */
00131 #define RE_NO_BK_REFS (RE_NO_BK_PARENS << 1)
00132 
00133 /* If this bit is set, then | is an alternation operator, and \| is literal.
00134    If not set, then \| is an alternation operator, and | is literal.  */
00135 #define RE_NO_BK_VBAR (RE_NO_BK_REFS << 1)
00136 
00137 /* If this bit is set, then an ending range point collating higher
00138      than the starting range point, as in [z-a], is invalid.
00139    If not set, then when ending range point collates higher than the
00140      starting range point, the range is ignored.  */
00141 #define RE_NO_EMPTY_RANGES (RE_NO_BK_VBAR << 1)
00142 
00143 /* If this bit is set, then an unmatched ) is ordinary.
00144    If not set, then an unmatched ) is invalid.  */
00145 #define RE_UNMATCHED_RIGHT_PAREN_ORD (RE_NO_EMPTY_RANGES << 1)
00146 
00147 /* If this bit is set, succeed as soon as we match the whole pattern,
00148    without further backtracking.  */
00149 #define RE_NO_POSIX_BACKTRACKING (RE_UNMATCHED_RIGHT_PAREN_ORD << 1)
00150 
00151 /* If this bit is set, do not process the GNU regex operators.
00152    If not set, then the GNU regex operators are recognized. */
00153 #define RE_NO_GNU_OPS (RE_NO_POSIX_BACKTRACKING << 1)
00154 
00155 /* If this bit is set, turn on internal regex debugging.
00156    If not set, and debugging was on, turn it off.
00157    This only works if regex.c is compiled -DDEBUG.
00158    We define this bit always, so that all that's needed to turn on
00159    debugging is to recompile regex.c; the calling code can always have
00160    this bit set, and it won't affect anything in the normal case. */
00161 #define RE_DEBUG (RE_NO_GNU_OPS << 1)

00323 struct re_pattern_buffer
00324 {
00325 /* [[[begin pattern_buffer]]] */
00326     /* Space that holds the compiled pattern.  It is declared as
00327           `unsigned char *' because its elements are
00328            sometimes used as array indexes.  */
00329   unsigned char *buffer;
00330 
00331     /* Number of bytes to which `buffer' points.  */
00332   unsigned long int allocated;
00333 
00334     /* Number of bytes actually used in `buffer'.  */
00335   unsigned long int used;
00336 
00337         /* Syntax setting with which the pattern was compiled.  */

00339

00343   char *fastmap;
00344 
00345         /* Either a translate table to apply to all characters before
00346            comparing them, or zero for no translation.  The translation

00349   RE_TRANSLATE_TYPE translate;
00350

00353

00360 
00361         /* If REGS_UNALLOCATED, allocate space in the `regs' structure
