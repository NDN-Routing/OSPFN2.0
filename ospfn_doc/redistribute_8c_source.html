00043 int
00044 zebra_check_addr (struct prefix *p)
00045 {
00046   if (p->family == AF_INET)
00047     {
00048       u_int32_t addr;
00049 
00050       addr = p->u.prefix4.s_addr;
00051       addr = ntohl (addr);
00052 
00053       if (IPV4_NET127 (addr)
00054           || IN_CLASSD (addr)
00055           || IPV4_LINKLOCAL(addr))
00056     return 0;
00057     }
00058 #ifdef HAVE_IPV6
00059   if (p->family == AF_INET6)
00060     {
00061       if (IN6_IS_ADDR_LOOPBACK (&p->u.prefix6))
00062     return 0;
00063       if (IN6_IS_ADDR_LINKLOCAL(&p->u.prefix6))
00064     return 0;
00065     }
00066 #endif /* HAVE_IPV6 */
00067   return 1;
00068 }

00070 static int
00071 is_default (struct prefix *p)
00072 {
00073   if (p->family == AF_INET)
00074     if (p->u.prefix4.s_addr == 0 && p->prefixlen == 0)
00075       return 1;
00076 #ifdef HAVE_IPV6
00077 #if 0  /* IPv6 default separation is now pending until protocol daemon
00078           can handle that. */
00079   if (p->family == AF_INET6)
00080     if (IN6_IS_ADDR_UNSPECIFIED (&p->u.prefix6) && p->prefixlen == 0)
00081       return 1;
00082 #endif /* 0 */
00083 #endif /* HAVE_IPV6 */
00084   return 0;
00085 }

00099   /* Lookup default route. */
00100   memset (&p, 0, sizeof (struct prefix_ipv4));
00101   p.family = AF_INET;
00102 
00103   /* Lookup table.  */
00104   table = vrf_table (AFI_IP, SAFI_UNICAST, 0);
00105   if (table)
00106     {
00107       rn = route_node_lookup (table, (struct prefix *)&p);
00108       if (rn)
00109     {
00110       for (newrib = rn->info; newrib; newrib = newrib->next)
00111         if (CHECK_FLAG (newrib->flags, ZEBRA_FLAG_SELECTED)
00112         && newrib->distance != DISTANCE_INFINITY)
00113           zsend_route_multipath (ZEBRA_IPV4_ROUTE_ADD, client, &rn->p, newrib);

00140 /* Redistribute routes. */
00141 static void
00142 zebra_redistribute (struct zserv *client, int type)
00143 {
00144   struct rib *newrib;
00145   struct route_table *table;
00146   struct route_node *rn;
00147 
00148   table = vrf_table (AFI_IP, SAFI_UNICAST, 0);
00149   if (table)
00150     for (rn = route_top (table); rn; rn = route_next (rn))
00151       for (newrib = rn->info; newrib; newrib = newrib->next)
00152     if (CHECK_FLAG (newrib->flags, ZEBRA_FLAG_SELECTED) 
00153         && newrib->type == type 
00154         && newrib->distance != DISTANCE_INFINITY
00155         && zebra_check_addr (&rn->p))
00156       zsend_route_multipath (ZEBRA_IPV4_ROUTE_ADD, client, &rn->p, newrib);
00157   
00158 #ifdef HAVE_IPV6
00159   table = vrf_table (AFI_IP6, SAFI_UNICAST, 0);
00160   if (table)
00161     for (rn = route_top (table); rn; rn = route_next (rn))
00162       for (newrib = rn->info; newrib; newrib = newrib->next)
00163     if (CHECK_FLAG (newrib->flags, ZEBRA_FLAG_SELECTED)
00164         && newrib->type == type 
00165         && newrib->distance != DISTANCE_INFINITY
00166         && zebra_check_addr (&rn->p))
00167       zsend_route_multipath (ZEBRA_IPV6_ROUTE_ADD, client, &rn->p, newrib);

00172 redistribute_add (struct prefix *p, struct rib *rib)
00173 {
00174   struct listnode *node, *nnode;
00175   struct zserv *client;
00176 
00177   for (ALL_LIST_ELEMENTS (zebrad.client_list, node, nnode, client))
00178     {
00179       if (is_default (p))
00180         {
00181           if (client->redist_default || client->redist[rib->type])
00182             {
00183               if (p->family == AF_INET)
00184                 zsend_route_multipath (ZEBRA_IPV4_ROUTE_ADD, client, p, rib);
00185 #ifdef HAVE_IPV6

00187                 zsend_route_multipath (ZEBRA_IPV6_ROUTE_ADD, client, p, rib);
00188 #endif /* HAVE_IPV6 */

00255     case ZEBRA_ROUTE_OSPF:
00256     case ZEBRA_ROUTE_OSPF6:
00257     case ZEBRA_ROUTE_BGP:
00258       if (! client->redist[type])
00259     {
00260       client->redist[type] = 1;
00261       zebra_redistribute (client, type);
