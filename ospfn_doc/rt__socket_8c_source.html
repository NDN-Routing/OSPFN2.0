00044 /* Adjust netmask socket length. Return value is a adjusted sin_len
00045    value. */

00047 sin_masklen (struct in_addr mask)
00048 {
00049   char *p, *lim;
00050   int len;
00051   struct sockaddr_in sin;
00052 
00053   if (mask.s_addr == 0) 
00054     return sizeof (long);
00055 
00056   sin.sin_addr = mask;
00057   len = sizeof (struct sockaddr_in);
00058 
00059   lim = (char *) &sin.sin_addr;
00060   p = lim + sizeof (sin.sin_addr);
00061 
00062   while (*--p == 0 && p >= lim) 
00063     len--;
00064   return len;
