             VTY_NEWLINE);
      else
        vty_out (vty, "debug zebra packet recv%s%s",
             IS_ZEBRA_DEBUG_DETAIL ? " detail" : "",
             VTY_NEWLINE);
      write++;
    }
    }

  if (IS_ZEBRA_DEBUG_RIB_Q)

      write++;
    }
  return write;
}
