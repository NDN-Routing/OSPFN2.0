00031   int nleft;

00034   nleft = nbytes;
00035 
00036   while (nleft > 0)

00041     return (nread);

00043     if (nread == 0)

00075 int
00076 set_nonblocking(int fd)
00077 {
00078   int flags;
00079 
00080   /* According to the Single UNIX Spec, the return value for F_GETFL should
00081      never be negative. */
00082   if ((flags = fcntl(fd, F_GETFL)) < 0)
00083     {
00084       zlog_warn("fcntl(F_GETFL) failed for fd %d: %s",
00085             fd, safe_strerror(errno));
00086       return -1;
00087     }
00088   if (fcntl(fd, F_SETFL, (flags | O_NONBLOCK)) < 0)
00089     {
00090       zlog_warn("fcntl failed setting fd %d non-blocking: %s",
00091             fd, safe_strerror(errno));
00092       return -1;
00093     }
00094   return 0;
