00095 static int
00096 zebra_server_send_message(struct zserv *client)
00097 {
00098   if (client->t_suicide)
00099     return -1;
00100   switch (buffer_write(client->wb, client->sock, STREAM_DATA(client->obuf),
00101                stream_get_endp(client->obuf)))
00102     {
00103     case BUFFER_ERROR:
00104       zlog_warn("%s: buffer_write failed to zserv client fd %d, closing",
00105              __func__, client->sock);
00106       /* Schedule a delayed close since many of the functions that call this
00107          one do not check the return code.  They do not allow for the
00108      possibility that an I/O error may have caused the client to be
00109      deleted. */
00110       client->t_suicide = thread_add_event(zebrad.master, zserv_delayed_close,
00111                        client, 0);
00112       return -1;
00113     case BUFFER_EMPTY:
00114       THREAD_OFF(client->t_write);
00115       break;
00116     case BUFFER_PENDING:
00117       THREAD_WRITE_ON(zebrad.master, client->t_write,
00118               zserv_flush_data, client, client->sock);
00119       break;
00120     }
00121   return 0;
00122 }

00214 /* Interface address is added/deleted. Send ZEBRA_INTERFACE_ADDRESS_ADD or
00215  * ZEBRA_INTERFACE_ADDRESS_DELETE to the client. 
00216  *
00217  * A ZEBRA_INTERFACE_ADDRESS_ADD is sent in the following situations:
00218  * - in response to a 3-byte ZEBRA_INTERFACE_ADD request
00219  *   from the client, after the ZEBRA_INTERFACE_ADD has been
00220  *   sent from zebra to the client
00221  * - redistribute new address info to all clients in the following situations
00222  *    - at startup, when zebra figures out the available interfaces
00223  *    - when an interface is added (where support for
00224  *      RTM_IFANNOUNCE or AF_NETLINK sockets is available), or when
00225  *      an interface is marked IFF_UP (i.e., an RTM_IFINFO message is
00226  *      received)
00227  *    - for the vty commands "ip address A.B.C.D/M [<secondary>|<label LINE>]"
00228  *      and "no bandwidth <1-10000000>", "ipv6 address X:X::X:X/M"
00229  *    - when an RTM_NEWADDR message is received from the kernel,
00230  * 
00231  * The call tree that triggers ZEBRA_INTERFACE_ADDRESS_DELETE: 
00232  *
00233  *                   zsend_interface_address(DELETE)
00234  *                           ^                         
00235  *                           |                        
00236  *          zebra_interface_address_delete_update    
00237  *             ^                        ^      ^
00238  *             |                        |      if_delete_update
00239  *             |                        |
00240  *         ip_address_uninstall        connected_delete_ipv4
00241  *         [ipv6_addresss_uninstall]   [connected_delete_ipv6]
00242  *             ^                        ^
00243  *             |                        |
00244  *             |                  RTM_NEWADDR on routing/netlink socket
00245  *             |
00246  *         vty commands:
00247  *     "no ip address A.B.C.D/M [label LINE]"
00248  *     "no ip address A.B.C.D/M secondary"
00249  *     ["no ipv6 address X:X::X:X/M"]
00250  *
00251  */

00253 zsend_interface_address (int cmd, struct zserv *client, 
00254                          struct interface *ifp, struct connected *ifc)

00263 
00264   s = client->obuf;

00266   
00267   zserv_create_header (s, cmd);
00268   stream_putl (s, ifp->ifindex);

00271   stream_putc (s, ifc->flags);

00340  * The zebra server sends the clients  a ZEBRA_IPV4_ROUTE_ADD or a
00341  * ZEBRA_IPV6_ROUTE_ADD via zsend_route_multipath in the following
00342  * situations:

00344  *   by sending a ZEBRA_REDISTRIBUTE_DEFAULT_ADD to the zebra server, in the

00346  *   ZEBRA_REDISTRIBUTE_ADD as a result of the "redistribute" vty cmd,

00350  * ZEBRA_IPV6_ROUTE_DELETE via zsend_route_multipath when:
00351  * - a "ip route"  or "ipv6 route" vty command is issued, a prefix is

00353  *   has to be redistributed to the clients 
00354  * 
00355  * XXX The ZEBRA_IPV*_ROUTE_ADD message is also sent by the client to the
00356  * zebra server when the client wants to tell the zebra server to add a
00357  * route to the kernel (zapi_ipv4_add etc. ).  Since it's essentially the

00364                        struct rib *rib)

00367   struct stream *s;

00370   int nhnum = 0;
00371   u_char zapi_flags = 0;
00372   
00373   s = client->obuf;
00374   stream_reset (s);
00375   
00376   zserv_create_header (s, cmd);
00377   
00378   /* Put type and nexthop. */
00379   stream_putc (s, rib->type);
00380   stream_putc (s, rib->flags);
00381   
00382   /* marker for message flags field */

00384   stream_putc (s, 0);
00385 
00386   /* Prefix. */
00387   psize = PSIZE (p->prefixlen);

00390 
00391   /* 
00392    * XXX The message format sent by zebra below does not match the format
00393    * of the corresponding message expected by the zebra server
00394    * itself (e.g., see zread_ipv4_add). The nexthop_num is not set correctly,
00395    * (is there a bug on the client side if more than one segment is sent?)

00397    * is hard-coded.

00400

00402     {

00406           SET_FLAG (zapi_flags, ZAPI_MESSAGE_IFINDEX);
00407           
00408           if (nhnummark == 0)
00409             {
00410               nhnummark = stream_get_endp (s);
00411               stream_putc (s, 1); /* placeholder */
00412             }
00413           
00414           nhnum++;
00415 
00416           switch(nexthop->type) 
00417             {
00418               case NEXTHOP_TYPE_IPV4:
00419               case NEXTHOP_TYPE_IPV4_IFINDEX:
00420                 stream_put_in_addr (s, &nexthop->gate.ipv4);
00421                 break;
00422 #ifdef HAVE_IPV6
00423               case NEXTHOP_TYPE_IPV6:
00424               case NEXTHOP_TYPE_IPV6_IFINDEX:
00425               case NEXTHOP_TYPE_IPV6_IFNAME:
00426                 stream_write (s, (u_char *) &nexthop->gate.ipv6, 16);
00427                 break;
00428 #endif
00429               default:
00430                 if (cmd == ZEBRA_IPV4_ROUTE_ADD 
00431                     || cmd == ZEBRA_IPV4_ROUTE_DELETE)
00432                   {
00433                     struct in_addr empty;
00434                     memset (&empty, 0, sizeof (struct in_addr));
00435                     stream_write (s, (u_char *) &empty, IPV4_MAX_BYTELEN);
00436                   }
00437                 else
00438                   {
00439                     struct in6_addr empty;
00440                     memset (&empty, 0, sizeof (struct in6_addr));
00441                     stream_write (s, (u_char *) &empty, IPV6_MAX_BYTELEN);
00442                   }
00443               }
00444 
00445           /* Interface index. */
00446           stream_putc (s, 1);
00447           stream_putl (s, nexthop->ifindex);
00448 
00449           break;
00450         }
00451     }
00452 
00453   /* Metric */
00454   if (cmd == ZEBRA_IPV4_ROUTE_ADD || ZEBRA_IPV6_ROUTE_ADD)
00455     {
00456       SET_FLAG (zapi_flags, ZAPI_MESSAGE_DISTANCE);
00457       stream_putc (s, rib->distance);

00459       stream_putl (s, rib->metric);

00468

00470   stream_putw_at (s, 0, stream_get_endp (s));
00471
