00164 /* Write data to buffer. */
00165 void
00166 buffer_put(struct buffer *b, const void *p, size_t size)
00167 {
00168   struct buffer_data *data = b->tail;
00169   const char *ptr = p;
00170 
00171   /* We use even last one byte of data buffer. */
00172   while (size)    
00173     {
00174       size_t chunk;
00175 
00176       /* If there is no data buffer add it. */
00177       if (data == NULL || data->cp == b->size)
00178     data = buffer_add (b);
00179 
00180       chunk = ((size <= (b->size - data->cp)) ? size : (b->size - data->cp));
00181       memcpy ((data->data + data->cp), ptr, chunk);
00182       size -= chunk;
00183       ptr += chunk;
00184       data->cp += chunk;
00185     }
00186 }

00202 /* Keep flushing data to the fd until the buffer is empty or an error is
00203    encountered or the operation would block. */
00204 buffer_status_t
00205 buffer_flush_all (struct buffer *b, int fd)
00206 {
00207   buffer_status_t ret;
00208   struct buffer_data *head;
00209   size_t head_sp;
00210 
00211   if (!b->head)
00212     return BUFFER_EMPTY;
00213   head_sp = (head = b->head)->sp;
00214   /* Flush all data. */
00215   while ((ret = buffer_flush_available(b, fd)) == BUFFER_PENDING)
00216     {
00217       if ((b->head == head) && (head_sp == head->sp) && (errno != EINTR))
00218         /* No data was flushed, so kernel buffer must be full. */
00219     return ret;
00220       head_sp = (head = b->head)->sp;
00221     }
00222 
00223   return ret;
00224 }

00387 /* This function (unlike other buffer_flush* functions above) is designed
00388 to work with non-blocking sockets.  It does not attempt to write out
00389 all of the queued data, just a "big" chunk.  It returns 0 if it was
00390 able to empty out the buffers completely, 1 if more flushing is
00391 required later, or -1 on a fatal write error. */
00392 buffer_status_t
00393 buffer_flush_available(struct buffer *b, int fd)
00394 {
00395 
00396 /* These are just reasonable values to make sure a significant amount of
00397 data is written.  There's no need to go crazy and try to write it all
00398 in one shot. */
00399 #ifdef IOV_MAX
00400 #define MAX_CHUNKS ((IOV_MAX >= 16) ? 16 : IOV_MAX)
00401 #else
00402 #define MAX_CHUNKS 16
00403 #endif
00404 #define MAX_FLUSH 131072
00405 
00406   struct buffer_data *d;
00407   size_t written;
00408   struct iovec iov[MAX_CHUNKS];
00409   size_t iovcnt = 0;
00410   size_t nbyte = 0;
00411 
00412   for (d = b->head; d && (iovcnt < MAX_CHUNKS) && (nbyte < MAX_FLUSH);
00413        d = d->next, iovcnt++)
00414     {
00415       iov[iovcnt].iov_base = d->data+d->sp;
00416       nbyte += (iov[iovcnt].iov_len = d->cp-d->sp);
00417     }
00418 
00419   if (!nbyte)
00420     /* No data to flush: should we issue a warning message? */
00421     return BUFFER_EMPTY;
00422 
00423   /* only place where written should be sign compared */
00424   if ((ssize_t)(written = writev(fd,iov,iovcnt)) < 0)
00425     {
00426       if (ERRNO_IO_RETRY(errno))
00427     /* Calling code should try again later. */
00428         return BUFFER_PENDING;
00429       zlog_warn("%s: write error on fd %d: %s",
00430         __func__, fd, safe_strerror(errno));
00431       return BUFFER_ERROR;
00432     }
00433 
00434   /* Free printed buffer data. */
00435   while (written > 0)
00436     {
00437       struct buffer_data *d;
00438       if (!(d = b->head))
00439         {
00440           zlog_err("%s: corruption detected: buffer queue empty, "
00441            "but written is %lu", __func__, (u_long)written);
00442       break;
00443         }
00444       if (written < d->cp-d->sp)
00445         {
00446       d->sp += written;
00447       return BUFFER_PENDING;
00448     }
00449 
00450       written -= (d->cp-d->sp);
00451       if (!(b->head = d->next))
00452         b->tail = NULL;
00453       BUFFER_DATA_FREE(d);
00454     }
00455 
00456   return b->head ? BUFFER_PENDING : BUFFER_EMPTY;
