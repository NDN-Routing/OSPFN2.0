00043 interface_list_ioctl (int af)
00044 {
00045   int ret;
00046   int sock;
00047 #define IFNUM_BASE 32
00048   struct lifnum lifn;
00049   int ifnum;
00050   struct lifreq *lifreq;
00051   struct lifconf lifconf;
00052   struct interface *ifp;
00053   int n;
00054   int save_errno;
00055   size_t needed, lastneeded = 0;
00056   char *buf = NULL;
00057 
00058   if (zserv_privs.change(ZPRIVS_RAISE))
00059     zlog (NULL, LOG_ERR, "Can't raise privileges");
00060   
00061   sock = socket (af, SOCK_DGRAM, 0);
00062   if (sock < 0)
00063     {
00064       zlog_warn ("Can't make %s socket stream: %s",
00065                  (af == AF_INET ? "AF_INET" : "AF_INET6"), safe_strerror (errno));
00066                  
00067       if (zserv_privs.change(ZPRIVS_LOWER))
00068         zlog (NULL, LOG_ERR, "Can't lower privileges");
00069         
00070       return -1;
00071     }
00072 
00073 calculate_lifc_len:     /* must hold privileges to enter here */
00074   lifn.lifn_family = af;
00075   lifn.lifn_flags = LIFC_NOXMIT; /* we want NOXMIT interfaces too */
00076   ret = ioctl (sock, SIOCGLIFNUM, &lifn);
00077   save_errno = errno;
00078   
00079   if (zserv_privs.change(ZPRIVS_LOWER))
00080     zlog (NULL, LOG_ERR, "Can't lower privileges");
00081  
00082   if (ret < 0)
00083     {
00084       zlog_warn ("interface_list_ioctl: SIOCGLIFNUM failed %s",
00085                  safe_strerror (save_errno));
00086       close (sock);
00087       return -1;
00088     }
00089   ifnum = lifn.lifn_count;
00090 
00091   /*
00092    * When calculating the buffer size needed, add a small number
00093    * of interfaces to those we counted.  We do this to capture
00094    * the interface status of potential interfaces which may have
00095    * been plumbed between the SIOCGLIFNUM and the SIOCGLIFCONF.
00096    */
00097   needed = (ifnum + 4) * sizeof (struct lifreq);
00098   if (needed > lastneeded || needed < lastneeded / 2)
00099     {
00100       if (buf != NULL)
00101         XFREE (MTYPE_TMP, buf);
00102       if ((buf = XMALLOC (MTYPE_TMP, needed)) == NULL)
00103         {
00104           zlog_warn ("interface_list_ioctl: malloc failed");
00105           close (sock);
00106           return -1;
00107         }
00108     }
00109   lastneeded = needed;
00110 
00111   lifconf.lifc_family = af;
00112   lifconf.lifc_flags = LIFC_NOXMIT;
00113   lifconf.lifc_len = needed;
00114   lifconf.lifc_buf = buf;
00115 
00116   if (zserv_privs.change(ZPRIVS_RAISE))
00117     zlog (NULL, LOG_ERR, "Can't raise privileges");
00118     
00119   ret = ioctl (sock, SIOCGLIFCONF, &lifconf);
00120 
00121   if (ret < 0)
00122     {
00123       if (errno == EINVAL)
00124         goto calculate_lifc_len; /* deliberately hold privileges */
00125 
00126       zlog_warn ("SIOCGLIFCONF: %s", safe_strerror (errno));
00127 
00128       if (zserv_privs.change(ZPRIVS_LOWER))
00129         zlog (NULL, LOG_ERR, "Can't lower privileges");
00130 
00131       goto end;
00132     }
00133 
00134   if (zserv_privs.change(ZPRIVS_LOWER))
00135     zlog (NULL, LOG_ERR, "Can't lower privileges");
00136     
00137   /* Allocate interface. */
00138   lifreq = lifconf.lifc_req;
00139 
00140   for (n = 0; n < lifconf.lifc_len; n += sizeof (struct lifreq))
00141     {
00142       /* we treat Solaris logical interfaces as addresses, because that is
00143        * how PF_ROUTE on Solaris treats them. Hence we can not directly use
00144        * the lifreq_name to get the ifp.  We need to normalise the name
00145        * before attempting get.
00146        *
00147        * Solaris logical interface names are in the form of:
00148        * <interface name>:<logical interface id>
00149        */
00150       unsigned int normallen = 0;
00151       uint64_t lifflags;
00152       
00153       /* We should exclude ~IFF_UP interfaces, as we'll find out about them
00154        * coming up later through RTM_NEWADDR message on the route socket.
00155        */
00156       if (if_get_flags_direct (lifreq->lifr_name, &lifflags,
00157                            lifreq->lifr_addr.ss_family)
00158           || !CHECK_FLAG (lifflags, IFF_UP))
00159         {
00160           lifreq++;
00161           continue;
00162         }
00163       
00164       /* Find the normalised name */
00165       while ( (normallen < sizeof(lifreq->lifr_name))
00166              && ( *(lifreq->lifr_name + normallen) != '\0')
00167              && ( *(lifreq->lifr_name + normallen) != ':') )
00168         normallen++;
00169       
00170       ifp = if_get_by_name_len(lifreq->lifr_name, normallen);
00171 
00172       if (lifreq->lifr_addr.ss_family == AF_INET)
00173         ifp->flags |= IFF_IPV4;
00174 
00175       if (lifreq->lifr_addr.ss_family == AF_INET6)
00176         {
00177 #ifdef HAVE_IPV6
00178           ifp->flags |= IFF_IPV6;
00179 #else
00180           lifreq++;
00181           continue;
00182 #endif /* HAVE_IPV6 */
00183         }
00184         
00185       if_add_update (ifp);
00186 
00187       interface_info_ioctl (ifp);
00188       
00189       /* If a logical interface pass the full name so it can be
00190        * as a label on the address
00191        */
00192       if ( *(lifreq->lifr_name + normallen) != '\0')
00193         if_get_addr (ifp, (struct sockaddr *) &lifreq->lifr_addr,
00194                      lifreq->lifr_name);
00195       else
00196         if_get_addr (ifp, (struct sockaddr *) &lifreq->lifr_addr, NULL);
00197         
00198       /* Poke the interface flags. Lets IFF_UP mangling kick in */
00199       if_flags_update (ifp, ifp->flags);
00200       
00201       lifreq++;
00202     }
00203 
00204 end:
00205   close (sock);
00206   XFREE (MTYPE_TMP, lifconf.lifc_buf);
00207   return ret;
00208 }

00251 /* Retrieve address information for the given ifp */
00252 static int
00253 if_get_addr (struct interface *ifp, struct sockaddr *addr, const char *label)
00254 {
00255   int ret;
00256   struct lifreq lifreq;
00257   struct sockaddr_storage mask, dest;
00258   char *dest_pnt = NULL;
00259   u_char prefixlen = 0;
00260   afi_t af;
00261   int flags = 0;
00262 
00263   /* Interface's name and address family.
00264    * We need to use the logical interface name / label, if we've been
00265    * given one, in order to get the right address
00266    */
00267   strncpy (lifreq.lifr_name, (label ? label : ifp->name), IFNAMSIZ);
00268 
00269   /* Interface's address. */
00270   memcpy (&lifreq.lifr_addr, addr, ADDRLEN (addr));
00271   af = addr->sa_family;
00272 
00273   /* Point to point or broad cast address pointer init. */
00274   dest_pnt = NULL;
00275 
00276   if (AF_IOCTL (af, SIOCGLIFDSTADDR, (caddr_t) & lifreq) >= 0)
00277     {
00278       memcpy (&dest, &lifreq.lifr_dstaddr, ADDRLEN (addr));
00279       if (af == AF_INET)
00280         dest_pnt = (char *) &(SIN (&dest)->sin_addr);
00281       else
00282         dest_pnt = (char *) &(SIN6 (&dest)->sin6_addr);
00283       flags = ZEBRA_IFA_PEER;
00284     }
00285 
00286   if (af == AF_INET)
00287     {
00288       ret = if_ioctl (SIOCGLIFNETMASK, (caddr_t) & lifreq);
00289       
00290       if (ret < 0)
00291         {
00292           if (errno != EADDRNOTAVAIL)
00293             {
00294               zlog_warn ("SIOCGLIFNETMASK (%s) fail: %s", ifp->name,
00295                          safe_strerror (errno));
00296               return ret;
00297             }
00298           return 0;
00299         }
00300       memcpy (&mask, &lifreq.lifr_addr, ADDRLEN (addr));
00301 
00302       prefixlen = ip_masklen (SIN (&mask)->sin_addr);
00303       if (!dest_pnt && (if_ioctl (SIOCGLIFBRDADDR, (caddr_t) & lifreq) >= 0))
00304     {
00305           memcpy (&dest, &lifreq.lifr_broadaddr, sizeof (struct sockaddr_in));
00306           dest_pnt = (char *) &SIN (&dest)->sin_addr;
00307         }
00308     }
00309 #ifdef HAVE_IPV6
00310   else if (af == AF_INET6)
00311     {
00312       if (if_ioctl_ipv6 (SIOCGLIFSUBNET, (caddr_t) & lifreq) < 0)
00313     {
00314       if (ifp->flags & IFF_POINTOPOINT)
00315         prefixlen = IPV6_MAX_BITLEN;
00316       else
00317         zlog_warn ("SIOCGLIFSUBNET (%s) fail: %s",
00318                ifp->name, safe_strerror (errno));
00319     }
00320       else
00321     {
00322       prefixlen = lifreq.lifr_addrlen;
00323     }
00324     }
00325 #endif /* HAVE_IPV6 */
00326 
00327   /* Set address to the interface. */
00328   if (af == AF_INET)
00329     connected_add_ipv4 (ifp, flags, &SIN (addr)->sin_addr, prefixlen,
00330                         (struct in_addr *) dest_pnt, label);
00331 #ifdef HAVE_IPV6
00332   else if (af == AF_INET6)
00333     connected_add_ipv6 (ifp, flags, &SIN6 (addr)->sin6_addr, prefixlen,
00334                         (struct in6_addr *) dest_pnt, label);
00335 #endif /* HAVE_IPV6 */
00336 
00337   return 0;
00338 }
