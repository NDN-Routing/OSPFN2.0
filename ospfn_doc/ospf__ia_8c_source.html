00050 ospf_find_abr_route (struct route_table *rtrs, 
00051                      struct prefix_ipv4 *abr,
00052                      struct ospf_area *area)
00053 {
00054   struct route_node *rn;
00055   struct ospf_route *or;
00056   struct listnode *node;
00057 
00058   if ((rn = route_node_lookup (rtrs, (struct prefix *) abr)) == NULL)
00059     return NULL;
00060 
00061   route_unlock_node (rn);
00062 
00063   for (ALL_LIST_ELEMENTS_RO ((struct list *) rn->info, node, or))
00064     if (IPV4_ADDR_SAME (&or->u.std.area_id, &area->area_id)

00066       return or;
00067 
00068   return NULL;

00078

00080     zlog_debug ("ospf_ia_network_route(): processing summary route to %s/%d", 
00081            inet_ntoa (p->prefix), p->prefixlen);
00082 
00083   /* Find a route to the same dest */
00084   if ((rn1 = route_node_lookup (rt, (struct prefix *) p)))
00085     {
00086       int res;
00087 
00088       route_unlock_node (rn1);
00089 
00090       if ((or = rn1->info))
00091     {
00092       if (IS_DEBUG_OSPF_EVENT)
00093         zlog_debug ("ospf_ia_network_route(): "
00094                "Found a route to the same network");
00095       /* Check the existing route. */
00096       if ((res = ospf_route_cmp (ospf, new_or, or)) < 0)
00097         {
00098           /* New route is better, so replace old one. */
00099           ospf_route_subst (rn1, new_or, abr_or);
00100         }
00101       else if (res == 0)
00102         {
00103           /* New and old route are equal, so next hops can be added. */
00104           route_lock_node (rn1);
00105           ospf_route_copy_nexthops (or, abr_or->paths);
00106           route_unlock_node (rn1);
00107 
00108           /* new route can be deleted, because existing route has been updated. */
00109           ospf_route_free (new_or);
00110         }
00111       else
00112         {
00113           /* New route is worse, so free it. */
00114           ospf_route_free (new_or);
00115           return;
00116         }
00117     } /* if (or)*/
00118     } /*if (rn1)*/
00119   else
00120     { /* no route */
00121       if (IS_DEBUG_OSPF_EVENT)
00122     zlog_debug ("ospf_ia_network_route(): add new route to %s/%d",

00131                       struct ospf_route *new_or, struct ospf_route *abr_or)

00133   struct ospf_route *or = NULL;
00134   struct route_node *rn;
00135   int ret;
00136 
00137   if (IS_DEBUG_OSPF_EVENT)
00138     zlog_debug ("ospf_ia_router_route(): considering %s/%d", 
00139            inet_ntoa (p->prefix), p->prefixlen);
00140   /* Find a route to the same dest */
00141   rn = route_node_get (rtrs, (struct prefix *) p);
00142    
00143   if (rn->info == NULL)
00144     /* This is a new route */
00145     rn->info = list_new ();
00146   else
00147     {
00148       struct ospf_area *or_area;
00149       or_area = ospf_area_lookup_by_area_id (ospf, new_or->u.std.area_id);
00150       assert (or_area);
00151       /* This is an additional route */
00152       route_unlock_node (rn);
00153       or = ospf_find_asbr_route_through_area (rtrs, p, or_area);
00154     }
00155 
00156   if (or)
00157     {
00158       if (IS_DEBUG_OSPF_EVENT)
00159     zlog_debug ("ospf_ia_router_route(): "
00160            "a route to the same ABR through the same area exists");
00161       /* New route is better */
00162       if ((ret = ospf_route_cmp (ospf, new_or, or)) < 0)
00163     {
00164       listnode_delete (rn->info, or);
00165       ospf_route_free (or);
00166       /* proceed down */
00167     }
00168       /* Routes are the same */
00169       else if (ret == 0)
00170     {
00171       if (IS_DEBUG_OSPF_EVENT)
00172         zlog_debug ("ospf_ia_router_route(): merging the new route");
00173 
00174       ospf_route_copy_nexthops (or, abr_or->paths);
00175       ospf_route_free (new_or);
00176       return;
00177     }
00178       /* New route is worse */
00179       else
00180     {
00181       if (IS_DEBUG_OSPF_EVENT)
00182         zlog_debug ("ospf_ia_router_route(): skipping the new route");
00183       ospf_route_free (new_or);
00184       return;
00185     }
00186     }
00187 
00188   ospf_route_copy_nexthops (new_or, abr_or->paths);
00189 
00190   if (IS_DEBUG_OSPF_EVENT)
00191     zlog_debug ("ospf_ia_router_route(): adding the new route");

00356 
00357       return;

00359   else
00360     {
00361       route_unlock_node (rn);
00362       if (rn->info == NULL)
00363         return;
00364     }
00365 
00366   or = rn->info;
00367 
00368   if (or->path_type != OSPF_PATH_INTRA_AREA &&
00369       or->path_type != OSPF_PATH_INTER_AREA)

00371       if (IS_DEBUG_OSPF_EVENT)
00372     zlog_debug ("ospf_update_network_route(): ERR: path type is wrong");
00373       return;
00374     }

00376   if (ospf->abr_type == OSPF_ABR_SHORTCUT)
00377     {
00378       if (or->path_type == OSPF_PATH_INTRA_AREA &&

00381       if (IS_DEBUG_OSPF_EVENT)
00382         zlog_debug ("ospf_update_network_route(): Shortcut: "
00383                "this intra-area path is not backbone");
00384       return;
00385     }
00386     }
00387   else   /* Not Shortcut ABR */
00388     {
00389       if (!OSPF_IS_AREA_ID_BACKBONE (or->u.std.area_id))

00412

00422       !OSPF_IS_AREA_ID_BACKBONE (or->u.std.area_id))
00423     {
00424       or->path_type = OSPF_PATH_INTER_AREA;
00425       or->u.std.area_id = area->area_id;
00426       or->u.std.external_routing = area->external_routing;
00427           /* Note that we can do this only in Shortcut ABR mode,
00428              because standard ABR must leave the route type and area
00429              unchanged
00430           */
00431         }
00432     }
