00055 /* AS External LSA origination */
00056 static void
00057 ospf6_as_external_lsa_originate (struct ospf6_route *route)
00058 {
00059   char buffer[OSPF6_MAX_LSASIZE];
00060   struct ospf6_lsa_header *lsa_header;
00061   struct ospf6_lsa *old, *lsa;
00062   struct ospf6_external_info *info = route->route_option;
00063 
00064   struct ospf6_as_external_lsa *as_external_lsa;
00065   char buf[64];
00066   caddr_t p;
00067 
00068   /* find previous LSA */
00069   old = ospf6_lsdb_lookup (htons (OSPF6_LSTYPE_AS_EXTERNAL),
00070                            route->path.origin.id, ospf6->router_id,
00071                            ospf6->lsdb);
00072 
00073   if (IS_OSPF6_DEBUG_ASBR || IS_OSPF6_DEBUG_ORIGINATE (AS_EXTERNAL))
00074     {
00075       prefix2str (&route->prefix, buf, sizeof (buf));
00076       zlog_debug ("Originate AS-External-LSA for %s", buf);
00077     }
00078 
00079   /* prepare buffer */
00080   memset (buffer, 0, sizeof (buffer));
00081   lsa_header = (struct ospf6_lsa_header *) buffer;
00082   as_external_lsa = (struct ospf6_as_external_lsa *)
00083     ((caddr_t) lsa_header + sizeof (struct ospf6_lsa_header));
00084   p = (caddr_t)
00085     ((caddr_t) as_external_lsa + sizeof (struct ospf6_as_external_lsa));
00086 
00087   /* Fill AS-External-LSA */
00088   /* Metric type */
00089   if (route->path.metric_type == 2)
00090     SET_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_E);
00091   else
00092     UNSET_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_E);
00093 
00094   /* forwarding address */
00095   if (! IN6_IS_ADDR_UNSPECIFIED (&info->forwarding))
00096     SET_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_F);
00097   else
00098     UNSET_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_F);
00099 
00100   /* external route tag */
00101   UNSET_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_T);
00102 
00103   /* Set metric */
00104   OSPF6_ASBR_METRIC_SET (as_external_lsa, route->path.cost);
00105 
00106   /* prefixlen */
00107   as_external_lsa->prefix.prefix_length = route->prefix.prefixlen;
00108 
00109   /* PrefixOptions */
00110   as_external_lsa->prefix.prefix_options = route->path.prefix_options;
00111 
00112   /* don't use refer LS-type */
00113   as_external_lsa->prefix.prefix_refer_lstype = htons (0);
00114 
00115   /* set Prefix */
00116   memcpy (p, &route->prefix.u.prefix6,
00117           OSPF6_PREFIX_SPACE (route->prefix.prefixlen));
00118   ospf6_prefix_apply_mask (&as_external_lsa->prefix);
00119   p += OSPF6_PREFIX_SPACE (route->prefix.prefixlen);
00120 
00121   /* Forwarding address */
00122   if (CHECK_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_F))
00123     {
00124       memcpy (p, &info->forwarding, sizeof (struct in6_addr));
00125       p += sizeof (struct in6_addr);
00126     }
00127 
00128   /* External Route Tag */
00129   if (CHECK_FLAG (as_external_lsa->bits_metric, OSPF6_ASBR_BIT_T))
00130     {
00131       /* xxx */
00132     }
00133 
00134   /* Fill LSA Header */
00135   lsa_header->age = 0;
00136   lsa_header->type = htons (OSPF6_LSTYPE_AS_EXTERNAL);
00137   lsa_header->id = route->path.origin.id;
00138   lsa_header->adv_router = ospf6->router_id;
00139   lsa_header->seqnum =
00140     ospf6_new_ls_seqnum (lsa_header->type, lsa_header->id,
00141                          lsa_header->adv_router, ospf6->lsdb);
00142   lsa_header->length = htons ((caddr_t) p - (caddr_t) lsa_header);
00143 
00144   /* LSA checksum */
00145   ospf6_lsa_checksum (lsa_header);
00146 
00147   /* create LSA */
00148   lsa = ospf6_lsa_create (lsa_header);
00149 
00150   /* Originate */
00151   ospf6_lsa_originate_process (lsa, ospf6);
00152 }

00263   if (route == NULL)
00264     {
00265       if (IS_OSPF6_DEBUG_EXAMIN (AS_EXTERNAL))

00270       return;
00271     }
00272 
00273   for (ospf6_route_lock (route);
00274        route && ospf6_route_is_prefix (&prefix, route);
00275        route = ospf6_route_next (route))
00276     {
00277       if (route->type != OSPF6_DEST_TYPE_NETWORK)
00278         continue;
00279       if (route->path.origin.type != lsa->header->type)
00280         continue;
00281       if (route->path.origin.id != lsa->header->id)
00282         continue;
00283       if (route->path.origin.adv_router != lsa->header->adv_router)
00284         continue;
00285 
00286       if (IS_OSPF6_DEBUG_EXAMIN (AS_EXTERNAL))

00567       if (IS_OSPF6_DEBUG_ASBR)
00568         {
00569           prefix2str (prefix, pbuf, sizeof (pbuf));
00570           zlog_debug ("No such route %s to withdraw", pbuf);
00571         }
00572       return;
00573     }
00574 
00575   info = match->route_option;
00576   assert (info);
00577 
00578   if (info->type != type)
00579     {
00580       if (IS_OSPF6_DEBUG_ASBR)
00581         {
00582           prefix2str (prefix, pbuf, sizeof (pbuf));
00583           zlog_debug ("Original protocol mismatch: %s", pbuf);
00584         }
00585       return;
00586     }
00587 
00588   if (IS_OSPF6_DEBUG_ASBR)
00589     {
00590       prefix2str (prefix, pbuf, sizeof (pbuf));
00591       inet_ntop (AF_INET, &prefix_id.u.prefix4, ibuf, sizeof (ibuf));
00592       zlog_debug ("Withdraw %s (AS-External Id:%s)", pbuf, ibuf);
00593     }
00594 
00595   lsa = ospf6_lsdb_lookup (htons (OSPF6_LSTYPE_AS_EXTERNAL),
00596                            htonl (info->id), ospf6->router_id, ospf6->lsdb);
00597   if (lsa)
00598     ospf6_lsa_purge (lsa);
00599 
00600   /* remove binding in external_id_table */
00601   prefix_id.family = AF_INET;
00602   prefix_id.prefixlen = 32;
00603   prefix_id.u.prefix4.s_addr = htonl (info->id);
00604   node = route_node_lookup (ospf6->external_id_table, &prefix_id);
00605   assert (node);
00606   node->info = NULL;
00607   route_unlock_node (node);
00608 
00609   ospf6_route_remove (match, ospf6->external_table);
00610   XFREE (MTYPE_OSPF6_EXTERNAL_INFO, info);
00611 
00612   /* Router-Bit (ASBR Flag) may have to be updated */
00613   for (ALL_LIST_ELEMENTS (ospf6->area_list, lnode, lnnode, oa))
00614     OSPF6_ROUTER_LSA_SCHEDULE (oa);
00615 }
