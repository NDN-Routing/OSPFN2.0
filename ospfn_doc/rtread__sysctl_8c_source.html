00043     CTL_NET,
00044     PF_ROUTE,

00052   if (sysctl (mib, MIBSIZ, NULL, &bufsiz, NULL, 0) < 0)

00058   /* Allocate buffer. */
00059   ref = buf = XMALLOC (MTYPE_TMP, bufsiz);
00060   
00061   /* Read routing table information by calling sysctl(). */
00062   if (sysctl (mib, MIBSIZ, buf, &bufsiz, NULL, 0) < 0) 
00063     {
00064       zlog_warn ("sysctl() fail by %s", safe_strerror (errno));
00065       return;
00066     }
00067 
00068   for (end = buf + bufsiz; buf < end; buf += rtm->rtm_msglen) 
00069     {
00070       rtm = (struct rt_msghdr *) buf;
00071       /* We must set RTF_DONE here, so rtm_read() doesn't ignore the message. */
00072       SET_FLAG (rtm->rtm_flags, RTF_DONE);
00073       rtm_read (rtm);
00074     }
00075 
00076   /* Free buffer. */
00077   XFREE (MTYPE_TMP, ref);
