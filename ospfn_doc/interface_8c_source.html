00116 /* Tie an interface address to its derived subnet list of addresses. */
00117 int
00118 if_subnet_add (struct interface *ifp, struct connected *ifc)
00119 {
00120   struct route_node *rn;
00121   struct zebra_if *zebra_if;
00122   struct prefix cp;
00123   struct list *addr_list;
00124 
00125   assert (ifp && ifp->info && ifc);
00126   zebra_if = ifp->info;
00127 
00128   /* Get address derived subnet node and associated address list, while marking
00129      address secondary attribute appropriately. */
00130   cp = *ifc->address;
00131   apply_mask (&cp);
00132   rn = route_node_get (zebra_if->ipv4_subnets, &cp);
00133 
00134   if ((addr_list = rn->info))
00135     SET_FLAG (ifc->flags, ZEBRA_IFA_SECONDARY);
00136   else
00137     {
00138       UNSET_FLAG (ifc->flags, ZEBRA_IFA_SECONDARY);
00139       rn->info = addr_list = list_new ();
00140       route_lock_node (rn);
00141     }
00142 
00143   /* Tie address at the tail of address list. */
00144   listnode_add (addr_list, ifc);
00145   
00146   /* Return list element count. */
00147   return (addr_list->count);
00148 }

00150 /* Untie an interface address from its derived subnet list of addresses. */
00151 int
00152 if_subnet_delete (struct interface *ifp, struct connected *ifc)
00153 {
00154   struct route_node *rn;
00155   struct zebra_if *zebra_if;
00156   struct list *addr_list;
00157 
00158   assert (ifp && ifp->info && ifc);
00159   zebra_if = ifp->info;
00160 
00161   /* Get address derived subnet node. */
00162   rn = route_node_lookup (zebra_if->ipv4_subnets, ifc->address);
00163   if (! (rn && rn->info))
00164     return -1;
00165   route_unlock_node (rn);
00166   
00167   /* Untie address from subnet's address list. */
00168   addr_list = rn->info;
00169   listnode_delete (addr_list, ifc);
00170   route_unlock_node (rn);
00171 
00172   /* Return list element count, if not empty. */
00173   if (addr_list->count)
00174     {
00175       /* If deleted address is primary, mark subsequent one as such and distribute. */
00176       if (! CHECK_FLAG (ifc->flags, ZEBRA_IFA_SECONDARY))
00177     {
00178       ifc = listgetdata (listhead (addr_list));
00179       zebra_interface_address_delete_update (ifp, ifc);
00180       UNSET_FLAG (ifc->flags, ZEBRA_IFA_SECONDARY);
00181       zebra_interface_address_add_update (ifp, ifc);
00182     }
00183       
00184       return addr_list->count;
00185     }
00186   
00187   /* Otherwise, free list and route node. */
00188   list_free (addr_list);
00189   rn->info = NULL;
00190   route_unlock_node (rn);
00191 
00192   return 0;
00193 }

00195 /* if_flags_mangle: A place for hacks that require mangling
00196  * or tweaking the interface flags.
00197  *
00198  * ******************** Solaris flags hacks **************************
00199  *
00200  * Solaris IFF_UP flag reflects only the primary interface as the
00201  * routing socket only sends IFINFO for the primary interface.  Hence  
00202  * ~IFF_UP does not per se imply all the logical interfaces are also   
00203  * down - which we only know of as addresses. Instead we must determine
00204  * whether the interface really is up or not according to how many   
00205  * addresses are still attached. (Solaris always sends RTM_DELADDR if
00206  * an interface, logical or not, goes ~IFF_UP).
00207  *
00208  * Ie, we mangle IFF_UP to *additionally* reflect whether or not there
00209  * are addresses left in struct connected, not just the actual underlying
00210  * IFF_UP flag.
00211  *
00212  * We must hence remember the real state of IFF_UP, which we do in
00213  * struct zebra_if.primary_state.
00214  *
00215  * Setting IFF_UP within zebra to administratively shutdown the
00216  * interface will affect only the primary interface/address on Solaris.
00217  ************************End Solaris flags hacks ***********************
00218  */
00219 static inline void
00220 if_flags_mangle (struct interface *ifp, uint64_t *newflags)
00221 {
00222 #ifdef SUNOS_5
00223   struct zebra_if *zif = ifp->info;
00224   
00225   zif->primary_state = *newflags & (IFF_UP & 0xff);
00226   
00227   if (CHECK_FLAG (zif->primary_state, IFF_UP)
00228       || listcount(ifp->connected) > 0)
00229     SET_FLAG (*newflags, IFF_UP);
00230   else
00231     UNSET_FLAG (*newflags, IFF_UP);
00232 #endif /* SUNOS_5 */
00233 }
00234 
00235 /* Update the flags field of the ifp with the new flag set provided.
00236  * Take whatever actions are required for any changes in flags we care
00237  * about.
00238  *
00239  * newflags should be the raw value, as obtained from the OS.
00240  */
00241 void
00242 if_flags_update (struct interface *ifp, uint64_t newflags)
00243 {
00244   if_flags_mangle (ifp, &newflags);
00245     
00246   if (if_is_operative (ifp))
00247     {
00248       /* operative -> inoperative? */
00249       ifp->flags = newflags;
00250       if (!if_is_operative (ifp))
00251         if_down (ifp);
00252     }
00253   else
00254     {
00255       /* inoperative -> operative? */
00256       ifp->flags = newflags;
00257       if (if_is_operative (ifp))
00258         if_up (ifp);
00259     }
00260 }

00378 if_delete_update (struct interface *ifp)
00379 {
00380   struct connected *ifc;
00381   struct prefix *p;
00382   struct route_node *rn;
00383   struct zebra_if *zebra_if;
00384 
00385   zebra_if = ifp->info;
00386 
00387   if (if_is_up(ifp))
00388     {
00389       zlog_err ("interface %s index %d is still up while being deleted.",
00390         ifp->name, ifp->ifindex);
00391       return;
00392     }
00393 
00394   /* Mark interface as inactive */
00395   UNSET_FLAG (ifp->status, ZEBRA_INTERFACE_ACTIVE);
00396   
00397   if (IS_ZEBRA_DEBUG_KERNEL)
00398     zlog_debug ("interface %s index %d is now inactive.",
00399            ifp->name, ifp->ifindex);
00400 
00401   /* Delete connected routes from the kernel. */
00402   if (ifp->connected)
00403     {
00404       struct listnode *node;
00405       struct listnode *last = NULL;
00406 
00407       while ((node = (last ? last->next : listhead (ifp->connected))))
00408     {
00409       ifc = listgetdata (node);
00410       p = ifc->address;
00411       
00412       if (p->family == AF_INET
00413           && (rn = route_node_lookup (zebra_if->ipv4_subnets, p)))
00414         {
00415           struct listnode *anode;
00416           struct listnode *next;
00417           struct listnode *first;
00418           struct list *addr_list;
00419           
00420           route_unlock_node (rn);
00421           addr_list = (struct list *) rn->info;
00422           
00423           /* Remove addresses, secondaries first. */
00424           first = listhead (addr_list);
00425           for (anode = first->next; anode || first; anode = next)
00426         {
00427           if (!anode)
00428             {
00429               anode = first;
00430               first = NULL;
00431             }
00432           next = anode->next;
00433 
00434           ifc = listgetdata (anode);
00435           p = ifc->address;
00436 
00437           connected_down_ipv4 (ifp, ifc);
00438 
00439           zebra_interface_address_delete_update (ifp, ifc);
00440 
00441           UNSET_FLAG (ifc->conf, ZEBRA_IFC_REAL);
00442 
00443           /* Remove from subnet chain. */
00444           list_delete_node (addr_list, anode);
00445           route_unlock_node (rn);
00446           
00447           /* Remove from interface address list (unconditionally). */
00448           if (!CHECK_FLAG (ifc->conf, ZEBRA_IFC_CONFIGURED))
00449             {
00450               listnode_delete (ifp->connected, ifc);
00451               connected_free (ifc);
00452                     }
00453                   else
00454                     last = node;
00455         }
00456 
00457           /* Free chain list and respective route node. */
00458           list_delete (addr_list);
00459           rn->info = NULL;
00460           route_unlock_node (rn);
00461         }
00462 #ifdef HAVE_IPV6
00463       else if (p->family == AF_INET6)
00464         {
00465           connected_down_ipv6 (ifp, ifc);
00466 
00467           zebra_interface_address_delete_update (ifp, ifc);
00468 
00469           UNSET_FLAG (ifc->conf, ZEBRA_IFC_REAL);
00470 
00471           if (CHECK_FLAG (ifc->conf, ZEBRA_IFC_CONFIGURED))
00472         last = node;
00473           else
00474         {
00475           listnode_delete (ifp->connected, ifc);
00476           connected_free (ifc);
00477         }
00478         }
00479 #endif /* HAVE_IPV6 */
00480       else
00481         {
00482           last = node;
00483         }
00484     }
00485     }
00486   zebra_interface_delete_update (ifp);
00487 
00488   /* Update ifindex after distributing the delete message.  This is in
00489      case any client needs to have the old value of ifindex available
00490      while processing the deletion.  Each client daemon is responsible
00491      for setting ifindex to IFINDEX_INTERNAL after processing the
00492      interface deletion message. */
00493   ifp->ifindex = IFINDEX_INTERNAL;
00494 }

01196     }

01200     SET_FLAG (ifc->conf, ZEBRA_IFC_CONFIGURED);

01222       if_subnet_add (ifp, ifc);
01223 
01224       /* IP address propery set. */

01358   struct prefix_ipv6 *p;

01360

01512 if_config_write (struct vty *vty)
01513 {
01514   struct listnode *node;
01515   struct interface *ifp;
01516 
01517   for (ALL_LIST_ELEMENTS_RO (iflist, node, ifp))
01518     {
01519       struct zebra_if *if_data;

01524       if_data = ifp->info;

01551

01553           vty_out (vty, " label %s", ifc->label);
01554 
01555         vty_out (vty, "%s", VTY_NEWLINE);
01556           }
01557       }

01559       if (if_data)
01560     {
01561       if (if_data->shutdown == IF_ZEBRA_SHUTDOWN_ON)
