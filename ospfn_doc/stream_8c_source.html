00033 #define GETP_VALID(S,G) \
00034   ((G) <= (S)->endp)
00035 #define PUT_AT_VALID(S,G) GETP_VALID(S,G)
00036 #define ENDP_VALID(S,E) \
00037   ((E) <= (S)->size)

00039 /* asserting sanity checks. Following must be true before
00040  * stream functions are called:
00041  *
00042  * Following must always be true of stream elements
00043  * before and after calls to stream functions:
00044  *
00045  * getp <= endp <= size
00046  *
00047  * Note that after a stream function is called following may be true:
00048  * if (getp == endp) then stream is no longer readable
00049  * if (endp == size) then stream is no longer writeable
00050  *
00051  * It is valid to put to anywhere within the size of the stream, but only
00052  * using stream_put..._at() functions.
00053  */
00054 #define STREAM_WARN_OFFSETS(S) \
00055   zlog_warn ("&(struct stream): %p, size: %lu, endp: %lu, getp: %lu\n", \
00056              (S), \
00057              (unsigned long) (S)->size, \
00058              (unsigned long) (S)->getp, \
00059              (unsigned long) (S)->endp)\
00060 
00061 #define STREAM_VERIFY_SANE(S) \
00062   do { \
00063     if ( !(GETP_VALID(S, (S)->getp)) && ENDP_VALID(S, (S)->endp) ) \
00064       STREAM_WARN_OFFSETS(S); \
00065     assert ( GETP_VALID(S, (S)->getp) ); \
00066     assert ( ENDP_VALID(S, (S)->endp) ); \
00067   } while (0)
00068 
00069 #define STREAM_BOUND_WARN(S, WHAT) \
00070   do { \
00071     zlog_warn ("%s: Attempt to %s out of bounds", __func__, (WHAT)); \
00072     STREAM_WARN_OFFSETS(S); \
00073     assert (0); \
00074   } while (0)
00075 
00076 /* XXX: Deprecated macro: do not use */
00077 #define CHECK_SIZE(S, Z) \
00078   do { \
00079     if (((S)->endp + (Z)) > (S)->size) \
00080       { \
00081         zlog_warn ("CHECK_SIZE: truncating requested size %lu\n", \
00082                    (unsigned long) (Z)); \
00083         STREAM_WARN_OFFSETS(S); \
00084         (Z) = (S)->size - (S)->endp; \
00085       } \
00086   } while (0);
