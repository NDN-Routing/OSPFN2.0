00191 static int
00192 ospf6_zebra_read_ipv6 (int command, struct zclient *zclient,
00193                        zebra_size_t length)
00194 {
00195   struct stream *s;
00196   struct zapi_ipv6 api;
00197   unsigned long ifindex;
00198   struct prefix_ipv6 p;
00199   struct in6_addr *nexthop;
00200 
00201   s = zclient->ibuf;
00202   ifindex = 0;
00203   nexthop = NULL;
00204   memset (&api, 0, sizeof (api));
00205 
00206   /* Type, flags, message. */
00207   api.type = stream_getc (s);
00208   api.flags = stream_getc (s);
00209   api.message = stream_getc (s);
00210 
00211   /* IPv6 prefix. */
00212   memset (&p, 0, sizeof (struct prefix_ipv6));
00213   p.family = AF_INET6;
00214   p.prefixlen = stream_getc (s);
00215   stream_get (&p.prefix, s, PSIZE (p.prefixlen));
00216 
00217   /* Nexthop, ifindex, distance, metric. */
00218   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_NEXTHOP))
00219     {
00220       api.nexthop_num = stream_getc (s);
00221       nexthop = (struct in6_addr *)
00222         malloc (api.nexthop_num * sizeof (struct in6_addr));
00223       stream_get (nexthop, s, api.nexthop_num * sizeof (struct in6_addr));
00224     }
00225   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_IFINDEX))
00226     {
00227       api.ifindex_num = stream_getc (s);
00228       ifindex = stream_getl (s);
00229     }
00230   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_DISTANCE))
00231     api.distance = stream_getc (s);
00232   else
00233     api.distance = 0;
00234   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_METRIC))
00235     api.metric = stream_getl (s);
00236   else
00237     api.metric = 0;
00238 
00239   if (IS_OSPF6_DEBUG_ZEBRA (RECV))
00240     {
00241       char prefixstr[128], nexthopstr[128];
00242       prefix2str ((struct prefix *)&p, prefixstr, sizeof (prefixstr));
00243       if (nexthop)
00244         inet_ntop (AF_INET6, nexthop, nexthopstr, sizeof (nexthopstr));
00245       else
00246         snprintf (nexthopstr, sizeof (nexthopstr), "::");
00247 
00248       zlog_debug ("Zebra Receive route %s: %s %s nexthop %s ifindex %ld",
00249           (command == ZEBRA_IPV6_ROUTE_ADD ? "add" : "delete"),
00250           zebra_route_string(api.type), prefixstr, nexthopstr, ifindex);
00251     }
00252  
00253   if (command == ZEBRA_IPV6_ROUTE_ADD)
00254     ospf6_asbr_redistribute_add (api.type, ifindex, (struct prefix *) &p,
00255                                  api.nexthop_num, nexthop);
00256   else
00257     ospf6_asbr_redistribute_remove (api.type, ifindex, (struct prefix *) &p);
00258 
00259   if (CHECK_FLAG (api.message, ZAPI_MESSAGE_NEXTHOP))
00260     free (nexthop);
00261 
00262   return 0;
00263 }

00345 #define ADD    0
00346 #define REM    1
00347 static void
00348 ospf6_zebra_route_update (int type, struct ospf6_route *request)
00349 {
00350   struct zapi_ipv6 api;
00351   char buf[64];
00352   int nhcount;
00353   struct in6_addr **nexthops;
00354   unsigned int *ifindexes;
00355   int i, ret = 0;
00356   struct prefix_ipv6 *dest;
00357 
00358   if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00359     {
00360       prefix2str (&request->prefix, buf, sizeof (buf));
00361       zlog_debug ("Send %s route: %s",
00362           (type == REM ? "remove" : "add"), buf);
00363     }
00364 
00365   if (zclient->sock < 0)
00366     {
00367       if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00368         zlog_debug ("  Not connected to Zebra");
00369       return;
00370     }
00371 
00372   if (request->path.origin.adv_router == ospf6->router_id &&
00373       (request->path.type == OSPF6_PATH_TYPE_EXTERNAL1 ||
00374        request->path.type == OSPF6_PATH_TYPE_EXTERNAL2))
00375     {
00376       if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00377         zlog_debug ("  Ignore self-originated external route");
00378       return;
00379     }
00380 
00381   /* If removing is the best path and if there's another path,
00382      treat this request as add the secondary path */
00383   if (type == REM && ospf6_route_is_best (request) &&
00384       request->next && ospf6_route_is_same (request, request->next))
00385     {
00386       if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00387         zlog_debug ("  Best-path removal resulted Sencondary addition");
00388       type = ADD;
00389       request = request->next;
00390     }
00391 
00392   /* Only the best path will be sent to zebra. */
00393   if (! ospf6_route_is_best (request))
00394     {
00395       /* this is not preferred best route, ignore */
00396       if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00397         zlog_debug ("  Ignore non-best route");
00398       return;
00399     }
00400 
00401   nhcount = 0;
00402   for (i = 0; i < OSPF6_MULTI_PATH_LIMIT; i++)
00403     if (ospf6_nexthop_is_set (&request->nexthop[i]))
00404       nhcount++;
00405 
00406   if (nhcount == 0)
00407     {
00408       if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00409         zlog_debug ("  No nexthop, ignore");
00410       return;
00411     }
00412 
00413   /* allocate memory for nexthop_list */
00414   nexthops = XCALLOC (MTYPE_OSPF6_OTHER,
00415                       nhcount * sizeof (struct in6_addr *));
00416   if (nexthops == NULL)
00417     {
00418       zlog_warn ("Can't send route to zebra: malloc failed");
00419       return;
00420     }
00421 
00422   /* allocate memory for ifindex_list */
00423   ifindexes = XCALLOC (MTYPE_OSPF6_OTHER,
00424                        nhcount * sizeof (unsigned int));
00425   if (ifindexes == NULL)
00426     {
00427       zlog_warn ("Can't send route to zebra: malloc failed");
00428       XFREE (MTYPE_OSPF6_OTHER, nexthops);
00429       return;
00430     }
00431 
00432   for (i = 0; i < nhcount; i++)
00433     {
00434       if (IS_OSPF6_DEBUG_ZEBRA (SEND))
00435     {
00436       char ifname[IFNAMSIZ];
00437       inet_ntop (AF_INET6, &request->nexthop[i].address,
00438              buf, sizeof (buf));
00439       if (!if_indextoname(request->nexthop[i].ifindex, ifname))
00440         strlcpy(ifname, "unknown", sizeof(ifname));
00441       zlog_debug ("  nexthop: %s%%%.*s(%d)", buf, IFNAMSIZ, ifname,
00442               request->nexthop[i].ifindex);
00443     }
00444       nexthops[i] = &request->nexthop[i].address;
00445       ifindexes[i] = request->nexthop[i].ifindex;
00446     }
00447 
00448   api.type = ZEBRA_ROUTE_OSPF6;
00449   api.flags = 0;
00450   api.message = 0;
00451   SET_FLAG (api.message, ZAPI_MESSAGE_NEXTHOP);
00452   api.nexthop_num = nhcount;
00453   api.nexthop = nexthops;
00454   SET_FLAG (api.message, ZAPI_MESSAGE_IFINDEX);
00455   api.ifindex_num = nhcount;
00456   api.ifindex = ifindexes;
00457   SET_FLAG (api.message, ZAPI_MESSAGE_METRIC);
00458   api.metric = (request->path.metric_type == 2 ?
00459                 request->path.cost_e2 : request->path.cost);
00460 
00461   dest = (struct prefix_ipv6 *) &request->prefix;
00462   if (type == REM)
00463     ret = zapi_ipv6_route (ZEBRA_IPV6_ROUTE_DELETE, zclient, dest, &api);
00464   else
00465     ret = zapi_ipv6_route (ZEBRA_IPV6_ROUTE_ADD, zclient, dest, &api);
00466 
00467   if (ret < 0)
00468     zlog_err ("zapi_ipv6_route() %s failed: %s",
00469               (type == REM ? "delete" : "add"), safe_strerror (errno));
00470 
00471   XFREE (MTYPE_OSPF6_OTHER, nexthops);
00472   XFREE (MTYPE_OSPF6_OTHER, ifindexes);
00473 
00474   return;
00475 }
