00063 #define timersub(a, b, res)                           \
00064   do {                                                \
00065     (res)->tv_sec = (a)->tv_sec - (b)->tv_sec;        \
00066     (res)->tv_usec = (a)->tv_usec - (b)->tv_usec;     \
00067     if ((res)->tv_usec < 0)                           \
00068       {                                               \
00069         (res)->tv_sec--;                              \
