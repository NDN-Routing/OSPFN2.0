00354 /* Convert HH:MM:SS MON DAY YEAR to time_t value.  -1 is returned when
00355    given string is malformed. */
00356 static time_t 
00357 key_str2time (const char *time_str, const char *day_str, const char *month_str,
00358           const char *year_str)
00359 {
00360   int i = 0;
00361   char *colon;
00362   struct tm tm;
00363   time_t time;
00364   unsigned int sec, min, hour;
00365   unsigned int day, month, year;
00366 
00367   const char *month_name[] = 
00368   {
00369     "January",
00370     "February",
00371     "March",
00372     "April",
00373     "May",
00374     "June",
00375     "July",
00376     "August",
00377     "September",
00378     "October",
00379     "November",
00380     "December",
00381     NULL
00382   };
00383 
00384 #define GET_LONG_RANGE(V,STR,MIN,MAX) \
00385 { \
00386   unsigned long tmpl; \
00387   char *endptr = NULL; \
00388   tmpl = strtoul ((STR), &endptr, 10); \
00389   if (*endptr != '\0' || tmpl == ULONG_MAX) \
00390     return -1; \
00391   if ( tmpl < (MIN) || tmpl > (MAX)) \
00392     return -1; \
00393   (V) = tmpl; \
00394 }
00395       
00396   /* Check hour field of time_str. */
00397   colon = strchr (time_str, ':');
00398   if (colon == NULL)
00399     return -1;
00400   *colon = '\0';
00401 
00402   /* Hour must be between 0 and 23. */
00403   GET_LONG_RANGE (hour, time_str, 0, 23);
00404 
00405   /* Check min field of time_str. */
00406   time_str = colon + 1;
00407   colon = strchr (time_str, ':');
00408   if (*time_str == '\0' || colon == NULL)
00409     return -1;
00410   *colon = '\0';
00411 
00412   /* Min must be between 0 and 59. */
00413   GET_LONG_RANGE (min, time_str, 0, 59);
00414 
00415   /* Check sec field of time_str. */
00416   time_str = colon + 1;
00417   if (*time_str == '\0')
00418     return -1;
00419   
00420   /* Sec must be between 0 and 59. */
00421   GET_LONG_RANGE (sec, time_str, 0, 59);
00422   
00423   /* Check day_str.  Day must be <1-31>. */
00424   GET_LONG_RANGE (day, day_str, 1, 31);
00425 
00426   /* Check month_str.  Month must match month_name. */
00427   month = 0;
00428   if (strlen (month_str) >= 3)
00429     for (i = 0; month_name[i]; i++)
00430       if (strncmp (month_str, month_name[i], strlen (month_str)) == 0)
00431     {
00432       month = i;
00433       break;
00434     }
00435   if (! month_name[i])
00436     return -1;
00437 
00438   /* Check year_str.  Year must be <1993-2035>. */
00439   GET_LONG_RANGE (year, year_str, 1993, 2035);
00440   
00441   memset (&tm, 0, sizeof (struct tm));
00442   tm.tm_sec = sec;
00443   tm.tm_min = min;
00444   tm.tm_hour = hour;
00445   tm.tm_mon = month;
00446   tm.tm_mday = day;
00447   tm.tm_year = year - 1900;
00448     
00449   time = mktime (&tm);
00450   
00451   return time;
00452 #undef GET_LONG_RANGE
00453 }

00455 static int
00456 key_lifetime_set (struct vty *vty, struct key_range *krange,
00457           const char *stime_str, const char *sday_str,
00458           const char *smonth_str, const char *syear_str,
00459           const char *etime_str, const char *eday_str,
00460           const char *emonth_str, const char *eyear_str)
00461 {
00462   time_t time_start;
00463   time_t time_end;
00464   
00465   time_start = key_str2time (stime_str, sday_str, smonth_str, syear_str);
00466   if (time_start < 0)
00467     {
00468       vty_out (vty, "Malformed time value%s", VTY_NEWLINE);
00469       return CMD_WARNING;

00471   time_end = key_str2time (etime_str, eday_str, emonth_str, eyear_str);
00472 
00473   if (time_end < 0)
00474     {
00475       vty_out (vty, "Malformed time value%s", VTY_NEWLINE);
00476       return CMD_WARNING;
00477     }
00478 
00479   if (time_end <= time_start)
00480     {
00481       vty_out (vty, "Expire time is not later than start time%s", VTY_NEWLINE);
00482       return CMD_WARNING;
00483     }
00484 
00485   krange->start = time_start;
00486   krange->end = time_end;
00487 
00488   return CMD_SUCCESS;
00489 }
