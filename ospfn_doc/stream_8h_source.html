00028 /*
00029  * A stream is an arbitrary buffer, whose contents generally are assumed to
00030  * be in network order.
00031  *
00032  * A stream has the following attributes associated with it:
00033  *
00034  * - size: the allocated, invariant size of the buffer.
00035  *
00036  * - getp: the get position marker, denoting the offset in the stream where
00037  *         the next read (or 'get') will be from. This getp marker is
00038  *         automatically adjusted when data is read from the stream, the
00039  *         user may also manipulate this offset as they wish, within limits
00040  *         (see below)
00041  *
00042  * - endp: the end position marker, denoting the offset in the stream where
00043  *         valid data ends, and if the user attempted to write (or
00044  *         'put') data where that data would be written (or 'put') to.
00045  *
00046  * These attributes are all size_t values.
00047  *
00048  * Constraints:
00049  *
00050  * 1. getp can never exceed endp
00051  *
00052  * - hence if getp is equal to endp, there is no more valid data that can be
00053  *   gotten from the stream (though, the user may reposition getp to earlier in
00054  *   the stream, if they wish).
00055  *
00056  * 2. endp can never exceed size
00057  *
00058  * - hence, if endp is equal to size, then the stream is full, and no more
00059  *   data can be written to the stream.
00060  *
00061  * In other words the following must always be true, and the stream
00062  * abstraction is allowed internally to assert that the following property
00063  * holds true for a stream, as and when it wishes:
00064  *
00065  *        getp <= endp <= size
00066  *
00067  * It is the users responsibility to ensure this property is never violated.
00068  *
00069  * A stream therefore can be thought of like this:
00070  *
00071  *  ---------------------------------------------------
00072  *  |XXXXXXXXXXXXXXXXXXXXXXXX                         |
00073  *  ---------------------------------------------------
00074  *               ^               ^                        ^
00075  *               getp            endp                     size
00076  *
00077  * This shows a stream containing data (shown as 'X') up to the endp offset.
00078  * The stream is empty from endp to size. Without adjusting getp, there are
00079  * still endp-getp bytes of valid data to be read from the stream.
00080  *
00081  * Methods are provided to get and put to/from the stream, as well as 
00082  * retrieve the values of the 3 markers and manipulate the getp marker.
00083  *
00084  * Note:
00085  * At the moment, newly allocated streams are zero filled. Hence, one can
00086  * use stream_forward_endp() to effectively create arbitrary zero-fill
00087  * padding. However, note that stream_reset() does *not* zero-out the
00088  * stream. This property should **not** be relied upon.
00089  *
00090  * Best practice is to use stream_put (<stream *>, NULL, <size>) to zero out
00091  * any part of a stream which isn't otherwise written to.
00092  */

00094 /* Stream buffer. */
00095 struct stream
00096 {
00097   struct stream *next;
00098 
00099   /* Remainder is ***private*** to stream
00100    * direct access is frowned upon!

00103   size_t getp;      /* next get position */
00104   size_t endp;      /* last valid data position */
00105   size_t size;      /* size of data segment */
