00042 /* Hook function for updating route_map assignment. */
00043 static void
00044 ospf_route_map_update (const char *name)
00045 {
00046   struct ospf *ospf;
00047   int type;
00048 
00049   /* If OSPF instatnce does not exist, return right now. */
00050   ospf = ospf_lookup ();
00051   if (ospf == NULL)
00052     return;
00053 
00054   /* Update route-map */
00055   for (type = 0; type <= ZEBRA_ROUTE_MAX; type++)
00056     {
00057       if (ROUTEMAP_NAME (ospf, type)
00058       && strcmp (ROUTEMAP_NAME (ospf, type), name) == 0)
00059     {
00060       /* Keep old route-map. */
00061       struct route_map *old = ROUTEMAP (ospf, type);
00062 
00063       /* Update route-map. */
00064       ROUTEMAP (ospf, type) =
00065         route_map_lookup_by_name (ROUTEMAP_NAME (ospf, type));
00066 
00067       /* No update for this distribute type. */
00068       if (old == NULL && ROUTEMAP (ospf, type) == NULL)
00069         continue;
00070 
00071       ospf_distribute_list_update (ospf, type);
00072     }
00073     }
00074 }

00092         {
00093           ospf_distribute_list_update (ospf, type);
00094         }

00192 /* `match ip netxthop ' */
00193 /* Match function return 1 if match is success else return zero. */
00194 static route_map_result_t
00195 route_match_ip_nexthop (void *rule, struct prefix *prefix,
00196             route_map_object_t type, void *object)
00197 {
00198   struct access_list *alist;
00199   struct external_info *ei = object;
00200   struct prefix_ipv4 p;
00201 
00202   if (type == RMAP_OSPF)
00203     {
00204       p.family = AF_INET;
00205       p.prefix = ei->nexthop;
00206       p.prefixlen = IPV4_MAX_BITLEN;
00207 
00208       alist = access_list_lookup (AFI_IP, (char *) rule);
00209       if (alist == NULL)
00210         return RMAP_NOMATCH;
00211 
00212       return (access_list_apply (alist, &p) == FILTER_DENY ?
00213               RMAP_NOMATCH : RMAP_MATCH);
00214     }
00215   return RMAP_NOMATCH;
00216 }
