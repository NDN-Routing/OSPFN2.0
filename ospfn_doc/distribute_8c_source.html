00172 static int
00173 distribute_list_unset (const char *ifname, enum distribute_type type, 
00174                const char *alist_name)

00176   struct distribute *dist;

00178   dist = distribute_lookup (ifname);
00179   if (!dist)
00180     return 0;
00181 
00182   if (type == DISTRIBUTE_IN)
00183     {
00184       if (!dist->list[DISTRIBUTE_IN])
00185     return 0;
00186       if (strcmp (dist->list[DISTRIBUTE_IN], alist_name) != 0)
00187     return 0;
00188 
00189       free (dist->list[DISTRIBUTE_IN]);
00190       dist->list[DISTRIBUTE_IN] = NULL;      
00191     }
00192 
00193   if (type == DISTRIBUTE_OUT)
00194     {
00195       if (!dist->list[DISTRIBUTE_OUT])
00196     return 0;
00197       if (strcmp (dist->list[DISTRIBUTE_OUT], alist_name) != 0)
00198     return 0;
00199 
00200       free (dist->list[DISTRIBUTE_OUT]);
00201       dist->list[DISTRIBUTE_OUT] = NULL;      
00202     }
00203 
00204   /* Apply this distribute-list to the interface. */
00205   (*distribute_delete_hook) (dist);
00206 
00207   /* If both out and in is NULL then free distribute list. */
00208   if (dist->list[DISTRIBUTE_IN] == NULL &&
00209       dist->list[DISTRIBUTE_OUT] == NULL &&
00210       dist->prefix[DISTRIBUTE_IN] == NULL &&
00211       dist->prefix[DISTRIBUTE_OUT] == NULL)
00212     {
00213       hash_release (disthash, dist);
00214       distribute_free (dist);
