00129       vty_out (vty, "Invalid direction: %s%s", direct_str, VTY_NEWLINE);

00135   if (metric < 0 || metric > 16)
00136     {
00137       vty_out (vty, "Invalid metric: %s%s", metric_str, VTY_NEWLINE);
00138       return CMD_WARNING;
00139     }
00140 
00141   /* Get offset-list structure with interface name. */
00142   offset = ripng_offset_list_get (ifname);
00143 
00144   if (offset->direct[direct].alist_name)
00145     free (offset->direct[direct].alist_name);
00146   offset->direct[direct].alist_name = strdup (alist);
00147   offset->direct[direct].metric = metric;
