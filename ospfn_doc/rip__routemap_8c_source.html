00182                                          rinfo->metric;

00513 /* Set metric to attribute. */
00514 static route_map_result_t
00515 route_set_metric (void *rule, struct prefix *prefix, 
00516           route_map_object_t type, void *object)
00517 {
00518   if (type == RMAP_RIP)
00519     {
00520       struct rip_metric_modifier *mod;
00521       struct rip_info *rinfo;
00522 
00523       mod = rule;
00524       rinfo = object;
00525 
00526       if (mod->type == metric_increment)
00527     rinfo->metric_out += mod->metric;
00528       else if (mod->type == metric_decrement)
00529     rinfo->metric_out -= mod->metric;
00530       else if (mod->type == metric_absolute)
00531     rinfo->metric_out = mod->metric;
00532 
00533       if ((signed int)rinfo->metric_out < 1)
00534     rinfo->metric_out = 1;
00535       if (rinfo->metric_out > RIP_METRIC_INFINITY)
00536     rinfo->metric_out = RIP_METRIC_INFINITY;
00537 
00538       rinfo->metric_set = 1;
00539     }
00540   return RMAP_OKAY;
00541 }

00543 /* set metric compilation. */
00544 static void *
00545 route_set_metric_compile (const char *arg)
00546 {
00547   int len;
00548   const char *pnt;
00549   int type;
00550   long metric;
00551   char *endptr = NULL;
00552   struct rip_metric_modifier *mod;
00553 
00554   len = strlen (arg);
00555   pnt = arg;
00556 
00557   if (len == 0)
00558     return NULL;
00559 
00560   /* Examine first character. */
00561   if (arg[0] == '+')
00562     {
00563       type = metric_increment;
00564       pnt++;
00565     }
00566   else if (arg[0] == '-')
00567     {
00568       type = metric_decrement;
00569       pnt++;
00570     }
00571   else
00572     type = metric_absolute;
00573 
00574   /* Check beginning with digit string. */
00575   if (*pnt < '0' || *pnt > '9')
00576     return NULL;
00577 
00578   /* Convert string to integer. */
00579   metric = strtol (pnt, &endptr, 10);
00580 
00581   if (metric == LONG_MAX || *endptr != '\0')
00582     return NULL;
00583   if (metric < 0 || metric > RIP_METRIC_INFINITY)
00584     return NULL;
00585 
00586   mod = XMALLOC (MTYPE_ROUTE_MAP_COMPILED, 
00587          sizeof (struct rip_metric_modifier));
00588   mod->type = type;
00589   mod->metric = metric;
00590 
00591   return mod;
00592 }
