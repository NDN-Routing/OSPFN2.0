00059 pid_t
00060 pid_output (const char *path)
00061 {
00062   int tmp;
00063   int fd;
00064   pid_t pid;
00065   char buf[16];
00066   struct flock lock;  
00067   mode_t oldumask;
00068 
00069   pid = getpid ();
00070 
00071   oldumask = umask(0777 & ~PIDFILE_MASK);
00072   fd = open (path, O_RDWR | O_CREAT, PIDFILE_MASK);
00073   if (fd < 0)
00074     {
00075       zlog_err("Can't create pid lock file %s (%s), exiting",
00076            path, safe_strerror(errno));
00077       umask(oldumask);
00078       exit(1);
00079     }
00080   else
00081     {
00082       size_t pidsize;
00083 
00084       umask(oldumask);
00085       memset (&lock, 0, sizeof(lock));
00086 
00087       lock.l_type = F_WRLCK;
00088       lock.l_whence = SEEK_SET;
00089 
00090       if (fcntl(fd, F_SETLK, &lock) < 0)
00091         {
00092           zlog_err("Could not lock pid_file %s, exiting", path);
00093           exit(1);
00094         }
00095 
00096       sprintf (buf, "%d\n", (int) pid);
00097       pidsize = strlen(buf);
00098       if ((tmp = write (fd, buf, pidsize)) != (int)pidsize)
00099         zlog_err("Could not write pid %d to pid_file %s, rc was %d: %s",
00100              (int)pid,path,tmp,safe_strerror(errno));
00101       else if (ftruncate(fd, pidsize) < 0)
00102         zlog_err("Could not truncate pid_file %s to %u bytes: %s",
00103              path,(u_int)pidsize,safe_strerror(errno));
00104     }
00105   return pid;
00106 }
