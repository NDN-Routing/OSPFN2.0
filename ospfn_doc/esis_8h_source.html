00074 struct esis_fixed {
00075     u_char          esis_proto_id;  /* network layer protocol identifier */
00076     u_char          esis_hdr_len;   /* length indicator (octets) */
00077     u_char          esis_vers;  /* version/protocol identifier
00078                      * extension */
00079     u_char          esis_res1;  /* reserved */
00080     u_char          esis_type;  /* type code */
00081     /* technically, type should be &='d 0x1f */
00082 #define ESIS_ESH    0x02    /* End System Hello */
00083 #define ESIS_ISH    0x04    /* Intermediate System Hello */
00084 #define ESIS_RD     0x06    /* Redirect */
00085     u_char          esis_ht_msb;    /* holding time (seconds) high byte */
00086     u_char          esis_ht_lsb;    /* holding time (seconds) low byte */
00087     u_char          esis_cksum_msb; /* checksum high byte */
00088     u_char          esis_cksum_lsb; /* checksum low byte */
00089 } __attribute__((packed));
